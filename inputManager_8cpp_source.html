174std::vector<std::string> InputManager::getTaggedValues(const std::vector<std::string> &values,
175 const std::unordered_map<std::string, std::vector<std::string>> &tagMap,
176 const std::vector<std::string> &tags, std::string checkType) const {
177
178 // check that checkType is valid
179 if(
180 checkType != "all" &&
181 checkType != "any" &&
182 checkType != "exact"
183 )
184 {
185 MACH3LOG_ERROR("Invalid tag check type specified: {}. Will instead use the default type: 'all'", checkType);
186 checkType = "all";
187 }
188
189 // If no tags were specified, take this to mean that anything should be a match
190 if (tags.size() == 0) return values;
191
192 std::vector<std::string> retVec;
193
194 MACH3LOG_DEBUG("Getting tagged values using checkType {}", checkType);
195 for ( std::string val: values )
196 {
197 MACH3LOG_DEBUG("Checking tags of {}", val);
198 // get the tags that this value has
199 const std::vector<std::string> &valTags = tagMap.at(val);
200
201 // we can skip it if it has no tags
202 if (valTags.size() == 0) continue;
203
204 // count how many of the specified tags match the tags of the current value
205 unsigned int tagCount = 0;
206 for ( const std::string &tag: tags )
207 {
208 if ( std::find( valTags.begin(), valTags.end(), tag ) != valTags.end() )
209 {
210 MACH3LOG_DEBUG(" - Matched tag {} !", tag);
211 tagCount ++;
212 }
213 }
214
215 // now decide if we include the current value based on the check type
216 if ( checkType == "all" )
217 {
218 if ( tagCount == tags.size() ) retVec.push_back(val);
219 }
220 else if ( checkType == "any" )
221 {
222 if ( tagCount > 0 ) retVec.push_back(val);
223 }
224 else if ( checkType == "exect" )
225 {
226 // EM: note that this will break if duplicate tags are specified in either vector... so please don't do that
227 if ( tagCount == valTags.size() ) retVec.push_back(val);
228 }
229 }
230
231 MACH3LOG_DEBUG("Found {} values matching the specified tags", retVec.size());
232 return retVec;
233

236std::vector<std::string> InputManager::parseLocation(const std::string &rawLocationString, std::string &fitter,
237 fileTypeEnum fileType, const std::string &parameter,
238 const std::string &sample, const std::string &parameter2) const {
239
240 std::string locationString(rawLocationString);
241 std::vector<std::string> tokens;
242
243 std::size_t pos = 0;
244 std::string toReplace = "";
245
246 // loop through the raw location string and replace any instance of {PARAMETER} with the name of
247 // the parameter
248 toReplace = "{PARAMETER}";
249 while ((pos = locationString.find(toReplace)) != std::string::npos)
250 {
251 locationString.replace(pos, toReplace.length(),
252 getFitterSpecificParamName(fitter, fileType, parameter));
253 }
254
255 // same again with {SAMPLE}
256 toReplace = "{SAMPLE}";
257 while ((pos = locationString.find(toReplace)) != std::string::npos)
258 {
259 locationString.replace(pos, toReplace.length(),
260 getFitterSpecificSampleName(fitter, fileType, sample));
261 }
262
263 // loop through the raw location string and replace any instance of {PARAMETER2} with the name of
264 // the second specified parameter
265 toReplace = "{PARAMETER2}";
266 while ((pos = locationString.find(toReplace)) != std::string::npos)
267 {
268 locationString.replace(pos, toReplace.length(),
269 getFitterSpecificParamName(fitter, fileType, parameter2));
270 }
271
272 // Now we go through and look for ":" and split the location into two parts
273 // first part should be the directory to look for the parameter
274 // lats part should be the end of the name of the object to look for
275 std::string token;
276 std::string delimiter = ":";
277 while ((pos = locationString.find(delimiter)) != std::string::npos)
278 {
279 token = locationString.substr(0, pos);
280 tokens.push_back(token);
281 locationString.erase(0, pos + delimiter.length());
282 }
283 tokens.push_back(locationString);
284
285 // should only have found 2 parts, anything more means there was too many
286 // ":" in the location string
287 if (tokens.size() > 2)
288 {
289 throw MaCh3Exception(__FILE__ , __LINE__, "Too many : tokens in location string: " + rawLocationString);
290 }
291
292 return tokens;
293}

295std::shared_ptr<TObject> InputManager::findRootObject(const InputFile &fileDef,
296 const std::vector<std::string> &locationVec) const {
297
298 std::shared_ptr<TObject> object = nullptr;
299
300 // if vector only has one element, just interpret it as the absolute path to the object
301 if (locationVec.size() == 1)
302 {
303 object = std::shared_ptr<TObject>(fileDef.file->Get(locationVec[0].c_str()));
304 }
305
306 // if vector has two elements, interpret the first as the directory and the second as a string to
307 // match with the end of the objects in the file
308 else if (locationVec.size() == 2)
309 {
310 TDirectoryFile *directory = (TDirectoryFile*)fileDef.file->Get(locationVec[0].c_str());
311 size_t nMatchingObjects = 0;
312
313 // let's make sure that the directory itself exists
314 if (directory == NULL)
315 {
316 object = nullptr;
317 }
318
319 else
320 {
321 // loop through the keys in the directory and find objects whose name matches the specified
322 // pattern
323 TIter next(directory->GetListOfKeys());
324 while (TKey *key = (TKey*)next())
325 {
326 if (strEndsWith(std::string(key->GetName()), locationVec[1]))
327 {
328 object = std::shared_ptr<TObject>(directory->Get(key->GetName()));
329 nMatchingObjects++;
330 }
331 }
332 }
333
334 // check that only one object matched the pattern
335 if (nMatchingObjects > 1)
336 {
337 MACH3LOG_CRITICAL("Too many objects match the pattern specified by {} {}", locationVec[0], locationVec.size()==2 ? locationVec[1] : "");
338 MACH3LOG_CRITICAL("Found {} matching objects, should just be one", nMatchingObjects);
339
340 throw MaCh3Exception(__FILE__ , __LINE__ );
341 }
342
343 }
344
345 // Vector too big!!
346 else
347 {
348 MACH3LOG_CRITICAL("Invalid object location vector");
349 MACH3LOG_CRITICAL("Should have two elements: [ (directory to look in), (end of the name of the object) ]");
350 MACH3LOG_CRITICAL("Or one element that is just the absolute path to the object");
351
352 throw MaCh3Exception(__FILE__ , __LINE__ );
353 }
354
355 return object;
356}

410bool InputManager::findRawChainSteps(InputFile &inputFileDef, const std::string &parameter, std::string &fitter, bool setInputBranch) const {
411 // we'll assume for now that the chain is in the form of a TTree and the branch names are parameter names
412
413 bool wasFound = false;
414
415 // make sure that the filedef object has all the necessary stuff to read from the posterior tree
416 if ( (inputFileDef.mcmcProc != nullptr) && (inputFileDef.posteriorTree != nullptr) )
417 {
418
419 const std::vector<TString> branchNames = inputFileDef.mcmcProc->GetBranchNames();
420
421 std::string specificName = getFitterSpecificParamName(fitter, kMCMC, parameter);
422
423 // loop over possible parameters and compare names
424 for ( int paramIdx = 0; paramIdx < inputFileDef.mcmcProc->GetNParams() -1 ; paramIdx ++ )
425 {
426 TString title;
427 double prior, priorError; // <- will be discarded
428 inputFileDef.mcmcProc->GetNthParameter(paramIdx, prior, priorError, title);
429
430 if ( strEndsWith(title.Data(), specificName) )
431 {
432 wasFound = true;
433 if ( setInputBranch )
434 {
435 // EM: should probably use MCMCProcessor for this so we can use caching, gpu etc.
436 inputFileDef.MCMCstepParamsMap[parameter] = new double( _BAD_DOUBLE_ ); // <- initialise the parameter step values
437 inputFileDef.posteriorTree->SetBranchAddress( branchNames[paramIdx], inputFileDef.MCMCstepParamsMap.at(parameter) );
438 }
439
440 break;
441 }
442 }
443 }
444
445 return wasFound;

480bool InputManager::findPostFitParamError(InputFile &inputFileDef, const std::string &parameter,
481 std::string &fitter, const std::string &errorType,
482 bool setInputFileError) {
483 std::string specificName = getFitterSpecificParamName(fitter, kPostFit, parameter);
484 YAML::Node thisFitterSpec_config = _fitterSpecConfig[fitter];
485
486 // EM: Get which hist this parameter lives in from the config
487 YAML::Node postFitErrorTypes = thisFitterSpec_config["postFitErrorTypes"];
488 YAML::Node specificErrorType = postFitErrorTypes[errorType];
489 std::vector<std::string> postFitLocations =
490 specificErrorType["location"].as<std::vector<std::string>>();
491
492 // EM: If the parameter has a specified list of locations then override the default one
493 std::vector<std::string> postFitLocations_override;
494 if (getFitterSpecificParamOption<std::vector<std::string>>(fitter, "postFitLoc",
495 postFitLocations_override, parameter))
496 {
497 postFitLocations = postFitLocations_override;
498 }
499
500 for (std::string postFitLoc : postFitLocations)
501 {
502 std::shared_ptr<TH1D> postFitErrors = std::static_pointer_cast<TH1D>(findRootObject(inputFileDef, parseLocation(postFitLoc, fitter, kPostFit, parameter)));
503
504 // EM: the postfit hist for this parameter isn't in this file
505 if (postFitErrors == nullptr)
506 continue;
507
508 // EM: Loop through the hist to see if it contains the parameter we're looking for
509 for (int binIdx = 0; binIdx <= postFitErrors->GetNbinsX(); binIdx++)
510 {
511 std::string binLabel = std::string(postFitErrors->GetXaxis()->GetBinLabel(binIdx));
512 if (strEndsWith(binLabel, specificName))
513 {
514 if (setInputFileError)
515 {
516 // EM: if specified, we fill the postfit error TH1D in the provided InputFile object
517 inputFileDef.postFitErrors[errorType][parameter] = postFitErrors->GetBinError(binIdx);
518 inputFileDef.postFitValues[errorType][parameter] = postFitErrors->GetBinContent(binIdx);
519 }
520
521 return true;
522 }
523 }
524 }
525 // EM: Didn't find the parameter in any of the specified locations
526 return false;
527}

529// EM: Lots of room for improvement in fillFileInfo and fillFileData, should be split up into more
530// methods, currently a lot of copy pasting
531void InputManager::fillFileInfo(InputFile &inputFileDef, bool printThoughts) {
536
537 // use the contents of the file to decide which fitter it came from and what type of file it is
538 if (printThoughts)
539 MACH3LOG_INFO("Checking contents of file {}", inputFileDef.fileName);
540
541 for (std::string fitter: _knownFitters)
542 {
543
544 // flag for whether or not the current fitter is the correct one
545 bool foundFitter = false;
546 if (printThoughts)
547 MACH3LOG_INFO("Checking if this is a {} file", fitter);
548
549 // EM: get the configuration specifying what the output of this fitter looks like
550 if (!_fitterSpecConfig[fitter])
551 {
552 throw MaCh3Exception(__FILE__ , __LINE__, "translation config doesnt contain a definition for fitter " + fitter);
553 }
554
555 YAML::Node thisFitterSpec_config = _fitterSpecConfig[fitter];
556
557 size_t numLLHParams;
558
559 // ##### Look for LLH scans in the input #####
560 // check for all 3 LLH directory types
561 for (std::string LLHType : {"sample", "penalty", "total"})
562 {
563
564 if (printThoughts)
565 MACH3LOG_INFO(".... searching for {} LLH scans... ", LLHType);
566
567 // vector of all the possible locations that we might find LLH scans for this type of LLH
568 YAML::Node testLLHConfig = thisFitterSpec_config[LLHType + "_LLH"];
569 std::vector<std::string> testLLHRawLocations =
570 testLLHConfig["location"].as<std::vector<std::string>>();
571
572 // counter for the total number of parameters we find scans for
573 numLLHParams = 0;
574
575 std::vector<std::string> enabledLLHParams;
576
577 for (const std::string &parameter : _knownParameters)
578 {
579 MACH3LOG_DEBUG(" - for {}", parameter);
580 inputFileDef.availableParams_map_LLH[LLHType][parameter] = false;
581
582 // check if we find the parameter at any of the locations we think it should be at
583 for (const std::string &rawLocation : testLLHRawLocations)
584 {
585 if (findRootObject(inputFileDef, parseLocation(rawLocation, fitter, kLLH, parameter)) !=
586 nullptr)
587 {
588 numLLHParams++;
589 enabledLLHParams.push_back(parameter);
590 inputFileDef.availableParams_map_LLH[LLHType][parameter] = true;
591 MACH3LOG_DEBUG(" FOUND!");
592 break; // <- we've found it, no point checking the rest of the locations
593 }
594 }
595 }
596
597 if (printThoughts)
598 MACH3LOG_INFO(".... Found {}\n", numLLHParams);
599
600 if (numLLHParams > 0)
601 {
602 foundFitter = true;
603 inputFileDef.hasLLHScans = true;
604 inputFileDef.availableParams_LLH = enabledLLHParams;
605 inputFileDef.hasLLHScans_map[LLHType] = true;
606 }
607
610 }
611
612 if (printThoughts)
613 MACH3LOG_INFO("");
614
615 // ##### now look for processed post fit errors #####
616 if (printThoughts)
617 {
618 MACH3LOG_INFO("....searching for Post Fit Parameters");
619
620 if (thisFitterSpec_config["defaultPostFitErrorType"])
621 {
622 MACH3LOG_DEBUG(" Default type specified with possible locations: ");
623 YAML::Node postFitErrorSpec = thisFitterSpec_config["postFitErrorTypes"];
624 YAML::Node defaultErrorType =
625 postFitErrorSpec[thisFitterSpec_config["defaultPostFitErrorType"].as<std::string>()];
626 std::vector<std::string> locations = defaultErrorType["location"].as<std::vector<std::string>>();
627 for (std::string loc : locations)
628 MACH3LOG_DEBUG(loc);
629 }
630 else
631 {
632 MACH3LOG_DEBUG(" No default location specified");
633 }
634
635 }
636
637 int numPostFitParams = 0;
638 std::vector<std::string> enabledPostFitParams;
639
640 std::string defaultErrorType =
641 thisFitterSpec_config["defaultPostFitErrorType"].as<std::string>();
642 for (std::string parameter : _knownParameters)
643 {
644 if (findPostFitParamError(inputFileDef, parameter, fitter, defaultErrorType))
645 {
646 numPostFitParams++;
647 enabledPostFitParams.push_back(parameter);
648 }
649 }
650
651 if (printThoughts)
652 MACH3LOG_INFO(".... Found {}\n", numPostFitParams);
653
654 if (numPostFitParams > 0)
655 {
656 foundFitter = true;
657 inputFileDef.hasPostFitErrors = true;
658 inputFileDef.availableParams_postFitErrors = enabledPostFitParams;
659 }
660
661 // ######### Now look for LLH scans broken down by sample #########
662 // This is really just a check to see if the number of by sample LLH scans is the same as normal
663 // LLH scans
664 if (printThoughts)
665 MACH3LOG_INFO("....Searching for LLH scans broken down by sample");
666
667 for (const std::string &sample : _knownSamples)
668 {
669 size_t numLLHBySampleParams = 0;
670 for (const std::string &parameter : _knownParameters)
671 {
672 inputFileDef.availableParams_map_LLHBySample[sample][parameter] = false;
673 if (findBySampleLLH(inputFileDef, parameter, fitter, sample))
674 {
675 inputFileDef.availableParams_map_LLHBySample[sample][parameter] = true;
676 numLLHBySampleParams++;
677 }
678 }
679
680 if (numLLHBySampleParams != 0)
681 {
682 inputFileDef.availableSamples_LLH.push_back(sample);
683
684 if (printThoughts)
685 MACH3LOG_INFO("........ Found {} LLH scans for sample {}", numLLHBySampleParams, sample);
686
687 if ((numLLHParams != numLLHBySampleParams))
688 {
689 MACH3LOG_ERROR("hmmmmm something weird is happening here");
690 MACH3LOG_ERROR(" I have {} LLH scans for sample {}", numLLHBySampleParams, sample);
691 MACH3LOG_ERROR(" But {} parameters with Total_LLH scals", numLLHParams);
692 }
693 }
694 }
695
696 // ######### Now for the main event: Look for MCMC chains and processed posteriors ###########
697
698 // EM: if "MCMCsteps" was defined for this fitter, we assume that it is a MaCh3 raw MCMC file
699 // thus it needs an MCMCProcessor to read from it. This isn't super general and it would probably
700 // be good to have some additional "isMaCh3" option that can decide whether or not to use MCMCProcessor
701 // but hey ho it's good enough for now
702 if ( thisFitterSpec_config["MCMCsteps"] )
703 {
704 MACH3LOG_DEBUG("Initialising MCMCProcessor for the input file");
705 std::vector<std::string> posteriorTreeRawLocations = thisFitterSpec_config["MCMCsteps"]["location"].as<std::vector<std::string>>();
706
707 TTree *postTree = NULL;
708 for ( const std::string &rawLoc: posteriorTreeRawLocations )
709 {
710 MACH3LOG_DEBUG(" - Looking for MCMC chain parameter values at: {}", rawLoc);
711
712 TObject *postTreeObj = inputFileDef.file->Get(rawLoc.c_str());
713
714 if ( postTreeObj != NULL )
715 {
716 postTree = (TTree *) postTreeObj;
717 inputFileDef.mcmcProc = new MCMCProcessor(inputFileDef.fileName);
718 inputFileDef.mcmcProc->Initialise();
719
720 MACH3LOG_DEBUG(" - FOUND!");
721 break;
722 }
723 }
724
725 if ( postTree != NULL )
726 {
727 inputFileDef.posteriorTree = postTree;
728 inputFileDef.nMCMCentries = postTree->GetEntries();
729 }
730 }
731
732 if (printThoughts)
733 {
734 MACH3LOG_INFO("....Searching for MCMC related things");
735 }
736
737 size_t num1dPosteriors = 0;
738 std::vector<std::string> enabled1dPosteriorParams;
739
740 size_t numMCMCchainParams = 0;
741 std::vector<std::string> enabledMCMCchainParams;
742
743 for ( const std::string &parameter : _knownParameters )
744 {
745 MACH3LOG_DEBUG(" - for {}", parameter);
746 // check for 1d post processing posterior.q
747 inputFileDef.availableParams_map_1dPosteriors[parameter] = false;
748 if ( thisFitterSpec_config["1dPosteriors"] && find1dPosterior(inputFileDef, parameter, fitter) )
749 {
750 MACH3LOG_DEBUG(" Found 1d processed posterior!");
751 enabled1dPosteriorParams.push_back(parameter);
752 num1dPosteriors++;
753 inputFileDef.availableParams_map_1dPosteriors[parameter] = true;
754 }
755 // now check for parameters in chain
756 inputFileDef.availableParams_map_MCMCchain[parameter] = false;
757 if ( thisFitterSpec_config["MCMCsteps"] && findRawChainSteps(inputFileDef, parameter, fitter) )
758 {
759 MACH3LOG_DEBUG(" Found raw MCMC steps!");
760 enabledMCMCchainParams.push_back(parameter);
761 numMCMCchainParams++;
762 inputFileDef.availableParams_map_MCMCchain[parameter] = true;
763 }
764 }
765
766 if (num1dPosteriors > 0 )
767 {
768 foundFitter = true;
769 inputFileDef.has1dPosteriors = true;
770 inputFileDef.availableParams_1dPosteriors = enabled1dPosteriorParams;
771
772 if (printThoughts)
773 MACH3LOG_INFO("........ Found {} 1d processed posteriors", num1dPosteriors);
774 }
775 if ( numMCMCchainParams > 0 )
776 {
777 foundFitter = true;
778 inputFileDef.hasMCMCchain = true;
779 inputFileDef.availableParams_MCMCchain = enabledMCMCchainParams;
780
781 if (printThoughts)
782 MACH3LOG_INFO("........ Found {} parameters in MCMC chain", numMCMCchainParams);
783 }
784
785 /*
786 switch (i) {
787 // any other weird fitter specific conditions/ edge cases should go in here
788 }
789 */
790
791 if (foundFitter)
792 {
793 if (printThoughts)
794 {
795 MACH3LOG_INFO("This is a {} file!\n", fitter);
796 }
797 inputFileDef.fitter = fitter;
798 return;
799 }
800 }
801
802 // if we didn't return above then the fitter type wasn't found
803 if (printThoughts)
804 MACH3LOG_WARN("I don't know what kinda fitter this came from, will proceed with caution");
805 inputFileDef.fitter = "UNKNOWN_FITTER";
806}

808void InputManager::fillFileData(InputFile &inputFileDef, bool printThoughts) {
809 // load in the data in the file using the info gotten above
810 // EM: a lot of this is copy paste from above, could be better organised
811 if (printThoughts)
812 MACH3LOG_INFO("....getting data from file {}", inputFileDef.fileName);
813
814 YAML::Node thisFitterSpec_config = _fitterSpecConfig[inputFileDef.fitter];
815
816 // set the default post fit error type so we can read it as default later
817 inputFileDef.defaultErrorType =
818 thisFitterSpec_config["defaultPostFitErrorType"].as<std::string>();
819
820 // ########### First fill up the LLH vectors ############
821 for (const std::string LLHType : {"sample", "penalty", "total"})
822 {
823 if (!inputFileDef.hasLLHScans_map.at(LLHType))
824 continue;
825
826 // vector of all the possible locations that we might find LLH scans for this type of LLH
827 YAML::Node testLLHConfig = thisFitterSpec_config[LLHType + "_LLH"];
828 const std::vector<std::string> testLLHRawLocations =
829 testLLHConfig["location"].as<std::vector<std::string>>();
830
831 // get the expected root object type of the llh scans
832 std::string LLHObjType = thisFitterSpec_config["LLHObjectType"].as<std::string>();
833
834 // EM: now get the objects from the file
835 for (const std::string &parameter : inputFileDef.availableParams_LLH)
836 {
837
838 std::shared_ptr<TObject> LLHObj = nullptr;
839
840 // check the locations for the object
841 for (const std::string &rawLocation : testLLHRawLocations)
842 {
843 LLHObj = findRootObject(inputFileDef,
844 parseLocation(rawLocation, inputFileDef.fitter, kLLH, parameter));
845 // if we found it then break out of the loop
846 if (LLHObj != nullptr)
847 break;
848 }
849
850 // double check that we actually found it.. just in case
851 if (LLHObj == nullptr)
852 {
853 MACH3LOG_ERROR("Hmmm, something seems to have gone wrong and I couldnt find the {} LLH scan for {} when attempting to read the data for it", LLHType, parameter);
854 MACH3LOG_ERROR("This will very likely cause segfaults and sadness");
855 }
856
857 // now convert it to a TGraph
858 std::shared_ptr<TGraph> LLHGraph = std::make_shared<TGraph>();
859
860 // EM: maybe have the type of the LLH object specified in the config to know what to downcast
861 // it to??
862 if (LLHObjType == "TH1D")
863 {
864 LLHGraph = std::make_shared<TGraph>((TH1D*)LLHObj.get());
865 }
866
867 else if (LLHObjType == "TGraph")
868 {
869 LLHGraph = std::shared_ptr<TGraph>((TGraph*)LLHObj.get()->Clone());
870 }
871
872 else
873 {
874 MACH3LOG_CRITICAL("ERROR: uknown type of LLH object specified: {}", LLHObjType);
875 throw MaCh3Exception(__FILE__ , __LINE__ );
876 }
877
878 inputFileDef.LLHScans_map[LLHType][parameter] = LLHGraph;
879 }
880 }
881
882 // ####### Get the by sample LLH scans #######
883 for (const std::string &parameter : inputFileDef.availableParams_LLH)
884 {
885 for (const std::string &sample : _knownSamples)
886 {
887 findBySampleLLH(inputFileDef, parameter, inputFileDef.fitter, sample, true);
888 }
889 }
890
891 // ####### Get the processed post fit errors #######
892 for (const std::string &parameter : inputFileDef.availableParams_postFitErrors)
893 {
894 std::vector<std::string> availableErrorTypes =
895 thisFitterSpec_config["AvailablePostFitErrorTypes"].as<std::vector<std::string>>();
896 for (const std::string &errorType : availableErrorTypes)
897 {
898 findPostFitParamError(inputFileDef, parameter, inputFileDef.fitter, errorType, true);
899 }
900 }
901
902 // ########## Get the MCMC related posteriors ###########
903 for (const std::string &parameter : inputFileDef.availableParams_MCMCchain)
904 {
905 findRawChainSteps(inputFileDef, parameter, inputFileDef.fitter, true);
906 }
907 for (const std::string &parameter : inputFileDef.availableParams_1dPosteriors)
908 {
909 find1dPosterior(inputFileDef, parameter, inputFileDef.fitter, true);
910 }
911}
