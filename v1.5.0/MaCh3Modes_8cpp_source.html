112 void MaCh3Modes::PrepareMap() {
113 // *******************
114 
115  int maxElement = 0;
116  for(int i = 0; i < NModes; ++i) {
117  for(size_t j = 0; j < fMode[i].GeneratorMaping.size(); j++) {
118  maxElement = std::max(maxElement, fMode[i].GeneratorMaping[j]);
119  }
120  }
121  ModeMap.resize(maxElement+1, Mode["UNKNOWN_BAD"]);
122 
123  for(int m = 0; m < NModes; ++m) {
124  for(size_t i = 0; i < fMode[m].GeneratorMaping.size(); ++i) {
125  if(fMode[m].GeneratorMaping[i] < 0) {
126  MACH3LOG_ERROR("Negative value of Mode {} for mode {}", fMode[m].GeneratorMaping[i], fMode[m].Name);
127  throw MaCh3Exception(__FILE__ , __LINE__ );
128  }
129  if(ModeMap[fMode[m].GeneratorMaping[i]] != Mode["UNKNOWN_BAD"])
130  {
131  MACH3LOG_ERROR("Generator mode {} already defined in mode {} can't do this for mode {}", fMode[m].GeneratorMaping[i], fMode[ModeMap[fMode[m].GeneratorMaping[i]]].Name, fMode[m].Name);
132  throw MaCh3Exception(__FILE__ , __LINE__ );
133  }
134  ModeMap[fMode[m].GeneratorMaping[i]] = m;
135  }
136  }
137 }
