13 {
14  TFile *file = M3::Open(File, "READ", __FILE__, __LINE__);
15  TDirectoryFile *PredicitveDir = file->Get<TDirectoryFile>("Predictive");
16 
17  std::vector<std::string> SampleNames;
18  //Get all entries in input file
19  TIter next(PredicitveDir->GetListOfKeys());
20  TKey *key = nullptr;
21 
22  // Loop through all entries
23  while ((key = static_cast<TKey*>(next()))) {
24  // get directory names, ignore flux
25  auto classname = std::string(key->GetClassName());
26  auto dirname = std::string(key->GetName());
27 
28  if (classname != "TDirectoryFile") continue;
29  dirname = std::string(key->GetName());
30 
31  if(dirname == "Total") continue;
32 
33  SampleNames.push_back(dirname);
34  MACH3LOG_DEBUG("Entering Sample {}", dirname);
35  }
36 
37  file->Close();
38  delete file;
39  return SampleNames;
40 }
