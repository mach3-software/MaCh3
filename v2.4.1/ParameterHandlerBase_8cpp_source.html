553  if (IsParameterFixed(i)) continue;
554 
555  _fPropVal[i] = _fPreFitValue[i] + corr_throw[i];
556 
557  int throws = 0;
558  // Try again if we the initial parameter proposal falls outside of the range of the parameter
559  while (_fPropVal[i] > _fUpBound[i] || _fPropVal[i] < _fLowBound[i]) {
560  randParams[i] = random_number[M3::GetThreadIndex()]->Gaus(0, 1);
561  const double corr_throw_single = M3::MatrixVectorMultiSingle(throwMatrixCholDecomp, randParams, _fNumPar, i);
562  _fPropVal[i] = _fPreFitValue[i] + corr_throw_single;
563  if (throws > 10000)
564  {
565  //KS: Since we are multithreading there is danger that those messages
566  //will be all over the place, small price to pay for faster code
567  MACH3LOG_WARN("Tried {} times to throw parameter {} but failed", throws, i);
568  MACH3LOG_WARN("Matrix: {}", matrixName);
569  MACH3LOG_WARN("Param: {}", _fNames[i]);
570  MACH3LOG_WARN("Setting _fPropVal: {} to {}", _fPropVal[i], _fPreFitValue[i]);
571  MACH3LOG_WARN("I live at {}:{}", __FILE__, __LINE__);
572  _fPropVal[i] = _fPreFitValue[i];
573  //throw MaCh3Exception(__FILE__ , __LINE__ );
574  }
575  throws++;
576  }
577  _fCurrVal[i] = _fPropVal[i];
578  }
579  }
580  else
581  {
582  PCAObj->ThrowParameters(random_number, throwMatrixCholDecomp,
583  randParams, corr_throw,
584  _fPreFitValue, _fLowBound, _fUpBound, _fNumPar);
585  } // end if pca
586 
587  // KS: At the end once we are happy with proposal do special proposal
588  SpecialStepProposal();
589 }

591 // *************************************
592 // Throw each parameter within their 1 sigma range
593 // Used to start the chain in different states
594 void ParameterHandlerBase::RandomConfiguration() {
595 // *************************************
596  // Have the 1 sigma for each parameter in each covariance class, sweet!
597  // Don't want to change the prior array because that's what determines our likelihood
598  // Want to change the _fPropVal, _fCurrVal, _fPreFitValue
599  // _fPreFitValue and the others will already be set
600  for (int i = 0; i < _fNumPar; ++i) {
601  // Check if parameter is fixed first: if so don't randomly throw
602  if (IsParameterFixed(i)) continue;
603  // Check that the sigma range is larger than the parameter range
604  // If not, throw in the valid parameter range instead
605  const double paramrange = _fUpBound[i] - _fLowBound[i];
606  const double sigma = sqrt((*covMatrix)(i,i));
607  double throwrange = sigma;
608  if (paramrange < sigma) throwrange = paramrange;
609 
610  _fPropVal[i] = _fPreFitValue[i] + random_number[0]->Gaus(0, 1)*throwrange;
611  // Try again if we the initial parameter proposal falls outside of the range of the parameter
612  int throws = 0;
613  while (_fPropVal[i] > _fUpBound[i] || _fPropVal[i] < _fLowBound[i]) {
614  if (throws > 1000) {
615  MACH3LOG_WARN("Tried {} times to throw parameter {} but failed", throws, i);
616  MACH3LOG_WARN("Matrix: {}", matrixName);
617  MACH3LOG_WARN("Param: {}", _fNames[i]);
618  throw MaCh3Exception(__FILE__ , __LINE__ );
619  }
620  _fPropVal[i] = _fPreFitValue[i] + random_number[0]->Gaus(0, 1)*throwrange;
621  throws++;
622  }
623  MACH3LOG_INFO("Setting current step in {} param {} = {} from {}", matrixName, i, _fPropVal[i], _fCurrVal[i]);
624  _fCurrVal[i] = _fPropVal[i];
625  }
626  if (pca) PCAObj->TransferToPCA();
627 
628  // KS: At the end once we are happy with proposal do special proposal
629  SpecialStepProposal();
630 }

689 // Used the proposal kernel and the current parameter value to set proposed step

691 void ParameterHandlerBase::Randomize() _noexcept_ {
692 // ************************************************
693  if (!pca) {
694  //KS: By multithreading here we gain at least factor 2 with 8 threads with ND only fit
695  #ifdef MULTITHREAD
696  #pragma omp parallel for
697  #endif
698  for (int i = 0; i < _fNumPar; ++i) {
699  // If parameter isn't fixed
700  if (!IsParameterFixed(i) > 0.0) {
701  randParams[i] = random_number[M3::GetThreadIndex()]->Gaus(0, 1);
702  // If parameter IS fixed
703  } else {
704  randParams[i] = 0.0;
705  }
706  } // end for
707  // If we're in the PCA basis we instead throw parameters there (only _fNumParPCA parameter)
708  } else {
709  // Scale the random parameters by the sqrt of eigen values for the throw
710  #ifdef MULTITHREAD
711  #pragma omp parallel for
712  #endif
713  for (int i = 0; i < PCAObj->GetNumberPCAedParameters(); ++i)
714  {
715  // If parameter IS fixed or out of bounds
716  if (PCAObj->IsParameterFixedPCA(i)) {
717  randParams[i] = 0.0;
718  } else {
719  randParams[i] = random_number[M3::GetThreadIndex()]->Gaus(0,1);
720  }
721  }
722  }

812 // ********************************************
813 // Get the likelihood in the case where we want to include priors on the parameters
814 // _fFlatPrior stores if we want to evaluate the likelihood for the given parameter
815 // true = don't evaluate likelihood (so run without a prior)
816 // false = evaluate likelihood (so run with a prior)
817 double ParameterHandlerBase::CalcLikelihood() const _noexcept_ {
818 // ********************************************
819  double logL = 0.0;
820  #ifdef MULTITHREAD
821  #pragma omp parallel for reduction(+:logL)
822  #endif
823  for(int i = 0; i < _fNumPar; ++i) {
824  if(_fFlatPrior[i]){
825  //HW: Flat prior, no need to calculate anything
826  continue;
827  }
828  // KS: Precalculate Diff once per "i" without doing this for every "j"
829  const double Diff = _fPropVal[i] - _fPreFitValue[i];
830  #ifdef MULTITHREAD
831  #pragma omp simd
832  #endif
833  for (int j = 0; j <= i; ++j) {
834  if (!_fFlatPrior[j]) {
835  //KS: Since matrix is symmetric we can calculate non diagonal elements only once and multiply by 2, can bring up to factor speed decrease.
836  double scale = (i != j) ? 1. : 0.5;
837  logL += scale * Diff * (_fPropVal[j] - _fPreFitValue[j])*InvertCovMatrix[i][j];
838  }
839  }
840  }
841  return logL;
842 }
