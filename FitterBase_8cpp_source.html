272 MACH3LOG_INFO("Getting starting position from {}", FitName);
273

275 TTree *posts = infile->Get<TTree>("posteriors");

289 // Config which was in MCMC from which we are starting

293

298 }

300 delete ConfigCov;
301 }
302

304 delete CovarianceFolder;
305
306 std::vector<double> branch_vals(systematics[s]->GetNumParams(), _BAD_DOUBLE_);

308 posts->SetBranchAddress(systematics[s]->GetParName(i).c_str(), &branch_vals[i]);

334 infile->Close();

391// *************************
392// Run Drag Race
393void FitterBase::DragRace(const int NLaps) {
394// *************************
395 MACH3LOG_INFO("Let the Race Begin!");
396 MACH3LOG_INFO("All tests will be performed with {} threads", M3::GetNThreads());
397
398 // Reweight the MC
399 for(unsigned int ivs = 0; ivs < samples.size(); ivs++ )
400 {
401 TStopwatch clockRace;
402 clockRace.Start();
403 for(int Lap = 0; Lap < NLaps; Lap++) {
404 samples[ivs]->reweight();
405 }
406 clockRace.Stop();
407 MACH3LOG_INFO("It took {:.4f} s to reweights {} times sample: {}", clockRace.RealTime(), NLaps, samples[ivs]->GetName());
408 MACH3LOG_INFO("On average {:.6f}", clockRace.RealTime()/NLaps);
409 }
410
411 for(unsigned int ivs = 0; ivs < samples.size(); ivs++ )
412 {
413 TStopwatch clockRace;
414 clockRace.Start();
415 for(int Lap = 0; Lap < NLaps; Lap++) {
416 samples[ivs]->GetLikelihood();
417 }
418 clockRace.Stop();
419 MACH3LOG_INFO("It took {:.4f} s to calculate GetLikelihood {} times sample: {}", clockRace.RealTime(), NLaps, samples[ivs]->GetName());
420 MACH3LOG_INFO("On average {:.6f}", clockRace.RealTime()/NLaps);
421 }
422 // Get vector of proposed steps. If we want to run LLH scan or something else after we need to revert changes after proposing steps multiple times
423 std::vector<std::vector<double>> StepsValuesBefore(systematics.size());
424 for (size_t s = 0; s < systematics.size(); ++s) {
425 StepsValuesBefore[s] = systematics[s]->getProposed();
426 }
427 for (size_t s = 0; s < systematics.size(); ++s) {
428 TStopwatch clockRace;
429 clockRace.Start();
430 for(int Lap = 0; Lap < NLaps; Lap++) {
431 systematics[s]->proposeStep();
432 }
433 clockRace.Stop();
434 MACH3LOG_INFO("It took {:.4f} s to propose step {} times cov: {}", clockRace.RealTime(), NLaps, systematics[s]->getName());
435 MACH3LOG_INFO("On average {:.6f}", clockRace.RealTime()/NLaps);
436 }
437 for (size_t s = 0; s < systematics.size(); ++s) {
438 systematics[s]->setParameters(StepsValuesBefore[s]);
439 }
440
441 for (size_t s = 0; s < systematics.size(); ++s) {
442 TStopwatch clockRace;
443 clockRace.Start();
444 for(int Lap = 0; Lap < NLaps; Lap++) {
445 systematics[s]->GetLikelihood();
446 }
447 clockRace.Stop();
448 MACH3LOG_INFO("It took {:.4f} s to calculate get likelihood {} times cov: {}", clockRace.RealTime(), NLaps, systematics[s]->getName());
449 MACH3LOG_INFO("On average {:.6f}", clockRace.RealTime()/NLaps);
450 }
451 MACH3LOG_INFO("End of race");
452}

775// *************************
776 TDirectory *Sample_LLH = outputFile->Get<TDirectory>("Sample_LLH");
777 MACH3LOG_INFO("Starting Get Step Scale Based On LLHScan");
778
779 if(Sample_LLH->IsZombie())

781 MACH3LOG_WARN("Couldn't find Sample_LLH, it looks like LLH scan wasn't run, will do this now");
782 RunLLHScan();

786 {
787 bool isxsec = (cov->getName() == "xsec_cov");
788
789 const int npars = cov->GetNumParams();
790 std::vector<double> StepScale(npars);
791 for (int i = 0; i < npars; ++i)
792 {
793 std::string name = cov->GetParName(i);
794 // For xsec we can get the actual name, hurray for being informative
795 if (isxsec) name = cov->GetParFancyName(i);
796
797
798 StepScale[i] = cov->GetIndivStepScale(i);
799 TH1D* LLHScan = Sample_LLH->Get<TH1D>((name+"_sam").c_str());
800 if(LLHScan == nullptr)
801 {
802 MACH3LOG_WARN("Couldn't find LLH scan, for {}, skipping", name);
803 continue;
804 }
805 const double LLH_val = std::max(LLHScan->GetBinContent(1), LLHScan->GetBinContent(LLHScan->GetNbinsX()));
806 //If there is no sensitivity leave it
807 if(LLH_val < 0.001) continue;
808
809 // EM: assuming that the likelihood is gaussian, approximate sigma value is given by variation/sqrt(-2LLH)
810 // can evaluate this at any point, simple to evaluate it in the first bin of the LLH scan
811 // KS: We assume variation is 1 sigma, each dial has different scale so it becomes faff...

813 const double approxSigma = TMath::Abs(Var)/std::sqrt(LLH_val);
814
815 // Based on Ewan comment I just took the 1sigma width from the LLH, assuming it was Gaussian, but then had to also scale by 2.38/sqrt(N_params)
816 const double NewStepScale = approxSigma * 2.38/std::sqrt(npars);
817 StepScale[i] = NewStepScale;
818 MACH3LOG_DEBUG("Sigma: {}", approxSigma);
819 MACH3LOG_DEBUG("optimal Step Size: {}", NewStepScale);
820 }
821 cov->setIndivStepScale(StepScale);
822 cov->SaveUpdatedMatrixConfig();
823 }
