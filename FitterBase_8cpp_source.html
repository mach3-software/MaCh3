391// Run Drag Race
392void FitterBase::DragRace(const int NLaps) {
393// *************************
394
395 MACH3LOG_INFO("Let the Race Begin!");
396 // Reweight the MC
397 for(unsigned int ivs = 0; ivs < samples.size(); ivs++ )
398 {
399 TStopwatch clockRace;
400 clockRace.Start();
401 for(int Lap = 0; Lap < NLaps; Lap++) {
402 samples[ivs]->reweight();
403 }
404 clockRace.Stop();
405 MACH3LOG_INFO("It took {:.4f} s to reweights {} times sample: {}", clockRace.RealTime(), NLaps, samples[ivs]->GetName());
406 MACH3LOG_INFO("On average {:.6f}", clockRace.RealTime()/NLaps);
407 }
408
409 for(unsigned int ivs = 0; ivs < samples.size(); ivs++ )
410 {
411 TStopwatch clockRace;
412 clockRace.Start();
413 for(int Lap = 0; Lap < NLaps; Lap++) {
414 samples[ivs]->GetLikelihood();
415 }
416 clockRace.Stop();
417 MACH3LOG_INFO("It took {:.4f} s to calculate GetLikelihood {} times sample: {}", clockRace.RealTime(), NLaps, samples[ivs]->GetName());
418 MACH3LOG_INFO("On average {:.6f}", clockRace.RealTime()/NLaps);
419 }
420 // Get vector of proposed steps. If we want to run LLH scan or something else after we need to revert changes after proposing steps multiple times
421 std::vector<std::vector<double>> StepsValuesBefore(systematics.size());
422 for (size_t s = 0; s < systematics.size(); ++s) {
423 StepsValuesBefore[s] = systematics[s]->getProposed();
424 }
425 for (size_t s = 0; s < systematics.size(); ++s) {
426 TStopwatch clockRace;
427 clockRace.Start();
428 for(int Lap = 0; Lap < NLaps; Lap++) {
429 systematics[s]->proposeStep();
430 }
431 clockRace.Stop();
432 MACH3LOG_INFO("It took {:.4f} s to propose step {} times cov: {}", clockRace.RealTime(), NLaps, systematics[s]->getName());
433 MACH3LOG_INFO("On average {:.6f}", clockRace.RealTime()/NLaps);
434 }
435 for (size_t s = 0; s < systematics.size(); ++s) {
436 systematics[s]->setParameters(StepsValuesBefore[s]);
437 }
438
439 for (size_t s = 0; s < systematics.size(); ++s) {
440 TStopwatch clockRace;
441 clockRace.Start();
442 for(int Lap = 0; Lap < NLaps; Lap++) {
443 systematics[s]->GetLikelihood();
444 }
445 clockRace.Stop();
446 MACH3LOG_INFO("It took {:.4f} s to calculate get likelihood {} times cov: {}", clockRace.RealTime(), NLaps, systematics[s]->getName());
447 MACH3LOG_INFO("On average {:.6f}", clockRace.RealTime()/NLaps);
448 }
449 MACH3LOG_INFO("End of race");
450}

815 for (int i = 0; i < npars; ++i)

825 {
826 MACH3LOG_WARN("Couldn't find LLH scan, for {}, skipping", name);

840 const double NewStepScale = approxSigma * 2.38/std::sqrt(npars);
841 StepScale[i] = NewStepScale;
842 MACH3LOG_DEBUG("Sigma: {}", approxSigma);
