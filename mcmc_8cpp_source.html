31// **********************
32// Do we accept the proposed step for all the parameters?
33void mcmc::CheckStep() {
34// **********************
35 bool accept = false;
36
37 // Set the acceptance probability to zero
38 accProb = 0.0;
39
40 // Calculate acceptance probability
41 if (anneal) accProb = TMath::Min(1.,TMath::Exp( -(logLProp-logLCurr) / (TMath::Exp(-step/AnnealTemp))));
42 else accProb = TMath::Min(1., TMath::Exp(logLCurr-logLProp));
43
44 // Get the random number
45 double fRandom = random->Rndm();
46
47 // Do the accept/reject
48 if (fRandom <= accProb) {
49 accept = true;
50 ++accCount;
51 } else {
52 accept = false;
53 }
54
55 #ifdef DEBUG
56 if (debug) debugFile << " logLProp: " << logLProp << " logLCurr: " << logLCurr << " accProb: " << accProb << " fRandom: " << fRandom << std::endl;
57 #endif
58
59 // Update all the handlers to accept the step
60 if (accept && !reject) {
61 logLCurr = logLProp;
62
63 // Loop over systematics and accept
64 for (size_t s = 0; s < systematics.size(); ++s) {
65 systematics[s]->acceptStep();
66 }
67 }
68
69 stepClock->Stop();
70 stepTime = stepClock->RealTime();
71
72 // Write step to output tree
73 outTree->Fill();
74}

78void mcmc::runMCMC() {
79 // *******************
80
81 // Save the settings into the output file
82 SaveSettings();
83

85 PrepareOutput();

100

124// *******************
125// Do the initial reconfigure of the MCMC
126void mcmc::ProposeStep() {
127// *******************
128
129 // Initial likelihood
130 double llh = 0.0;
131
132 // Initiate to false
133 reject = false;
134
135 // Loop over the systematics and propose the initial step
136 for (size_t s = 0; s < systematics.size(); ++s) {
137 // Could throw the initial value here to do MCMC stability studies
138 // Propose the steps for the systematics
139 systematics[s]->proposeStep();
140
141 // Get the likelihood from the systematics
142 syst_llh[s] = systematics[s]->GetLikelihood();
143 llh += syst_llh[s];
144
145 #ifdef DEBUG
146 if (debug) debugFile << "LLH after " << systematics[s]->getName() << " " << llh << std::endl;
147 #endif
148 }
149
150 // Check if we've hit a boundary in the systematics
151 // In this case we can save time by not having to reconfigure the simulation
152 if (llh >= _LARGE_LOGL_) {
153 reject = true;
154 #ifdef DEBUG
155 if (debug) debugFile << "Rejecting based on boundary" << std::endl;
156 #endif
157 }
158
159 // Only reweight when we have a good parameter configuration
160 // This speeds things up considerably because for every bad parameter configuration we don't have to reweight the MC
161 if (!reject)
162 {
163 // Could multi-thread this
164 // But since sample reweight is multi-threaded it's probably better to do that
165 for (size_t i = 0; i < samples.size(); ++i)
166 {
167 samples[i]->reweight();
168 }
169
170 //DB for atmospheric event by event sample migration, need to fully reweight all samples to allow event passing prior to likelihood evaluation
171 for (size_t i = 0; i < samples.size(); ++i) {
172 // Get the sample likelihoods and add them
173 sample_llh[i] = samples[i]->GetLikelihood();
174 llh += sample_llh[i];
175 #ifdef DEBUG
176 if (debug) debugFile << "LLH after sample " << i << " " << llh << std::endl;
177 #endif
178 }
179
180 // For when we don't have to reweight, set sample to madness
181 } else {
182 for (size_t i = 0; i < samples.size(); ++i) {
183 // Set the sample_llh[i] to be madly high also to signify a step out of bounds
184 sample_llh[i] = _LARGE_LOGL_;
185 #ifdef DEBUG
186 if (debug) debugFile << "LLH after REJECT sample " << i << " " << llh << std::endl;
187 #endif
188 }
189 }
190
191 // Save the proposed likelihood (class member)
192 logLProp = llh;
193}
