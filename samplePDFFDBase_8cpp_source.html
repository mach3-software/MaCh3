575 if (totalweight <= 0.){
576 MCSamples[iSample].xsec_w[iEvent] = 0.;
577 continue;
578 }
579
580 //DB Switch on BinningOpt to allow different binning options to be implemented
581 //The alternative would be to have inheritance based on BinningOpt
582 double XVar = (*(MCSamples[iSample].x_var[iEvent]));
583
584 //DB Commented out by default but if we ever want to consider shifts in theta this will be needed
585 //double YVar = MCSamples[iSample].rw_theta[iEvent];
586 //ETA - this would actually be with (*(MCSamples[iSample].y_var[iEvent])) and done extremely
587 //similarly to XVar now
588
589 //DB Find the relevant bin in the PDF for each event
590 int XBinToFill = -1;
591 int YBinToFill = MCSamples[iSample].NomYBin[iEvent];
592
593 //DB Check to see if momentum shift has moved bins
594 //DB - First, check to see if the event is still in the nominal bin
595 if (XVar < MCSamples[iSample].rw_upper_xbinedge[iEvent] && XVar >= MCSamples[iSample].rw_lower_xbinedge[iEvent]) {
596 XBinToFill = MCSamples[iSample].NomXBin[iEvent];
597 }
598 //DB - Second, check to see if the event is outside of the binning range and skip event if it is
599 else if (XVar < XBinEdges[0] || XVar >= XBinEdges[nXBins]) {
600 continue;
601 }
602 //DB - Thirdly, check the adjacent bins first as Eb+CC+EScale shifts aren't likely to move an Erec more than 1bin width
603 //Shifted down one bin from the event bin at nominal
604 else if (XVar < MCSamples[iSample].rw_lower_xbinedge[iEvent] && XVar >= MCSamples[iSample].rw_lower_lower_xbinedge[iEvent]) {
605 XBinToFill = MCSamples[iSample].NomXBin[iEvent]-1;
606 }
607 //Shifted up one bin from the event bin at nominal
608 else if (XVar < MCSamples[iSample].rw_upper_upper_xbinedge[iEvent] && XVar >= MCSamples[iSample].rw_upper_xbinedge[iEvent]) {
609 XBinToFill = MCSamples[iSample].NomXBin[iEvent]+1;
610 }
611 //DB - If we end up in this loop, the event has been shifted outside of its nominal bin, but is still within the allowed binning range
612 else {
613 for(unsigned int iBin=0;iBin<(XBinEdges.size()-1);iBin++) {
614 if (XVar >= XBinEdges[iBin] && XVar < XBinEdges[iBin+1]) {
615 XBinToFill = iBin;
616 }
617 }
618 }
619
620 //ETA - we can probably remove this final if check on the -1?
621 //Maybe we can add an overflow bin to the array and assign any events to this bin?
622 //Might save us an extra if call?
623 //DB Fill relevant part of thread array
624 if (XBinToFill != -1 && YBinToFill != -1) {
625 samplePDFFD_array_private[YBinToFill][XBinToFill] += totalweight;
626 samplePDFFD_array_private_w2[YBinToFill][XBinToFill] += totalweight*totalweight;
627 }
628 }
629 }
630 //End of Calc Weights and fill Array
631 //==================================================
632 // DB Copy contents of 'samplePDFFD_array_private' into 'samplePDFFD_array' which can then be used in GetLikelihood
633 for (size_t yBin = 0; yBin < nYBins; ++yBin) {
634 for (size_t xBin = 0; xBin < nXBins; ++xBin) {
635 #pragma omp atomic
636 samplePDFFD_array[yBin][xBin] += samplePDFFD_array_private[yBin][xBin];
637 #pragma omp atomic
638 samplePDFFD_array_w2[yBin][xBin] += samplePDFFD_array_private_w2[yBin][xBin];
639 }
640 }
641
642 for (size_t yBin = 0; yBin < nYBins; ++yBin) {
643 delete[] samplePDFFD_array_private[yBin];
644 delete[] samplePDFFD_array_private_w2[yBin];
645 }
646 delete[] samplePDFFD_array_private;
647 delete[] samplePDFFD_array_private_w2;
648 } //end of parallel region
649}

745// ************************************************
746//A way to check whether a normalisation parameter applies to an event or not
747void samplePDFFDBase::CalcXsecNormsBins(int iSample) {
748// ************************************************
749 FarDetectorCoreInfo *fdobj = &MCSamples[iSample];
750 #ifdef DEBUG
751 std::vector<int> VerboseCounter(xsec_norms.size(), 0);
752 #endif
753 for(int iEvent = 0; iEvent < fdobj->nEvents; ++iEvent){
754 std::vector< int > XsecBins = {};
755 if (XsecCov) {
756 // Skip oscillated NC events
757 // Not strictly needed, but these events don't get included in oscillated predictions, so
758 // no need to waste our time calculating and storing information about xsec parameters
759 // that will never be used.
760 if (fdobj->isNC[iEvent] && (*fdobj->nupdg[iEvent] != *fdobj->nupdgUnosc[iEvent]) ) {
761 MACH3LOG_TRACE("Event {}, missed NC/signal check", iEvent);
762 continue;
763 } //DB Abstract check on MaCh3Modes to determine which apply to neutral current
764 for (std::vector<XsecNorms4>::iterator it = xsec_norms.begin(); it != xsec_norms.end(); ++it) {
765 //Now check that the target of an interaction matches with the normalisation parameters
766 bool TargetMatch = MatchCondition((*it).targets, *(fdobj->Target[iEvent]));
767 if (!TargetMatch) {
768 MACH3LOG_TRACE("Event {}, missed target check ({}) for dial {}", iEvent, *(fdobj->Target[iEvent]), (*it).name);
769 continue;
770 }
771
772 //Now check that the neutrino flavour in an interaction matches with the normalisation parameters
773 bool FlavourMatch = MatchCondition((*it).pdgs, *(fdobj->nupdg[iEvent]));
774 if (!FlavourMatch) {
775 MACH3LOG_TRACE("Event {}, missed PDG check ({}) for dial {}", iEvent,(*fdobj->nupdg[iEvent]), (*it).name);
776 continue;
777 }
778
779 //Now check that the unoscillated neutrino flavour in an interaction matches with the normalisation parameters
780 bool FlavourUnoscMatch = MatchCondition((*it).preoscpdgs, *(fdobj->nupdgUnosc[iEvent]));
781 if (!FlavourUnoscMatch){
782 MACH3LOG_TRACE("Event {}, missed FlavourUnosc check ({}) for dial {}", iEvent,(*fdobj->nupdgUnosc[iEvent]), (*it).name);
783 continue;
784 }
785
786 //Now check that the mode of an interaction matches with the normalisation parameters
787 bool ModeMatch = MatchCondition((*it).modes, static_cast<int>(std::round(*(fdobj->mode[iEvent]))));
788 if (!ModeMatch) {
789 MACH3LOG_TRACE("Event {}, missed Mode check ({}) for dial {}", iEvent, *(fdobj->mode[iEvent]), (*it).name);
790 continue;
791 }
792
793 //Now check whether the norm has kinematic bounds
794 //i.e. does it only apply to events in a particular kinematic region?
795 bool IsSelected = true;
796 if ((*it).hasKinBounds) {
797 for (unsigned int iKinematicParameter = 0 ; iKinematicParameter < (*it).KinematicVarStr.size() ; ++iKinematicParameter ) {
798 if (ReturnKinematicParameter((*it).KinematicVarStr[iKinematicParameter], iSample, iEvent) <= (*it).Selection[iKinematicParameter][0]) {
799 IsSelected = false;
800 MACH3LOG_TRACE("Event {}, missed Kinematic var check ({}) for dial {}", iEvent, (*it).KinematicVarStr[iKinematicParameter], (*it).name);
801 continue;
802 }
803 else if (ReturnKinematicParameter((*it).KinematicVarStr[iKinematicParameter], iSample, iEvent) > (*it).Selection[iKinematicParameter][1]) {
804 MACH3LOG_TRACE("Event {}, missed Kinematic var check ({}) for dial {}", iEvent, (*it).KinematicVarStr[iKinematicParameter], (*it).name);
805 IsSelected = false;
806 continue;
807 }
808 }
809 }
810 //Need to then break the event loop
811 if(!IsSelected){
812 MACH3LOG_TRACE("Event {}, missed Kinematic var check for dial {}", iEvent, (*it).name);
813 continue;
814 }
815 // Now set 'index bin' for each normalisation parameter
816 // All normalisations are just 1 bin for 2015, so bin = index (where index is just the bin for that normalisation)
817 int bin = (*it).index;
818
819 XsecBins.push_back(bin);
820 MACH3LOG_TRACE("Event {}, will be affected by dial {}", iEvent, (*it).name);
821 #ifdef DEBUG
822 VerboseCounter[std::distance(xsec_norms.begin(), it)]++;
823 #endif
824 //}
825 } // end iteration over xsec_norms
826 } // end if (xsecCov)
827 fdobj->xsec_norms_bins[iEvent] = XsecBins;
828 }//end loop over events
829 #ifdef DEBUG
830 MACH3LOG_DEBUG("Channel {}", iSample);
831 MACH3LOG_DEBUG("┌──────────────────────────────────────────────────────────┐");
832 for (std::size_t i = 0; i < xsec_norms.size(); ++i) {
833 const auto& norm = xsec_norms[i];
834 double eventRatio = static_cast<double>(VerboseCounter[i]) / static_cast<double>(fdobj->nEvents);
835
836 MACH3LOG_DEBUG("│ Param {:<15}, affects {:<8} events ({:>6.2f}%) │",
837 XsecCov->GetParFancyName(norm.index), VerboseCounter[i], eventRatio);
838 }
839 MACH3LOG_DEBUG("└──────────────────────────────────────────────────────────┘");
840 #endif
841}
