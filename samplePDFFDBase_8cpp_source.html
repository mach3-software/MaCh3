594 MCSamples[iSample].xsec_w[iEvent] = 0.;
595 continue;
596 }
597
598 funcweight = CalcXsecWeightFunc(iSample,iEvent);
599 //DB Catch negative func weights and skip any event with a negative event. Previously we would set weight to zero and continue but that is inefficient
600 if (funcweight <= 0.){
601 MCSamples[iSample].xsec_w[iEvent] = 0.;
602 continue;
603 }
604
605 MCSamples[iSample].xsec_w[iEvent] = splineweight*normweight*funcweight;
606
607 totalweight = GetEventWeight(iSample, iEvent);
608
609 //DB Catch negative weights and skip any event with a negative event
610 if (totalweight <= 0.){
611 MCSamples[iSample].xsec_w[iEvent] = 0.;
612 continue;
613 }
614
615 //DB Switch on BinningOpt to allow different binning options to be implemented
616 //The alternative would be to have inheritance based on BinningOpt
617 double XVar = (*(MCSamples[iSample].x_var[iEvent]));
618
619 //DB Commented out by default but if we ever want to consider shifts in theta this will be needed
620 //double YVar = MCSamples[iSample].rw_theta[iEvent];
621 //ETA - this would actually be with (*(MCSamples[iSample].y_var[iEvent])) and done extremely
622 //similarly to XVar now
623
624 //DB Find the relevant bin in the PDF for each event
625 int XBinToFill = -1;
626 int YBinToFill = MCSamples[iSample].NomYBin[iEvent];
627
628 //DB Check to see if momentum shift has moved bins
629 //DB - First, check to see if the event is still in the nominal bin
630 if (XVar < MCSamples[iSample].rw_upper_xbinedge[iEvent] && XVar >= MCSamples[iSample].rw_lower_xbinedge[iEvent]) {
631 XBinToFill = MCSamples[iSample].NomXBin[iEvent];
632 }
633 //DB - Second, check to see if the event is outside of the binning range and skip event if it is
634 else if (XVar < XBinEdges[0] || XVar >= XBinEdges[nXBins]) {
635 continue;
636 }
637 //DB - Thirdly, check the adjacent bins first as Eb+CC+EScale shifts aren't likely to move an Erec more than 1bin width
638 //Shifted down one bin from the event bin at nominal
639 else if (XVar < MCSamples[iSample].rw_lower_xbinedge[iEvent] && XVar >= MCSamples[iSample].rw_lower_lower_xbinedge[iEvent]) {
640 XBinToFill = MCSamples[iSample].NomXBin[iEvent]-1;
641 }
642 //Shifted up one bin from the event bin at nominal
643 else if (XVar < MCSamples[iSample].rw_upper_upper_xbinedge[iEvent] && XVar >= MCSamples[iSample].rw_upper_xbinedge[iEvent]) {
644 XBinToFill = MCSamples[iSample].NomXBin[iEvent]+1;
645 }
646 //DB - If we end up in this loop, the event has been shifted outside of its nominal bin, but is still within the allowed binning range
647 else {
648 for(unsigned int iBin=0;iBin<(XBinEdges.size()-1);iBin++) {
649 if (XVar >= XBinEdges[iBin] && XVar < XBinEdges[iBin+1]) {
650 XBinToFill = iBin;
651 }
652 }
653 }
654
655 //ETA - we can probably remove this final if check on the -1?
656 //Maybe we can add an overflow bin to the array and assign any events to this bin?
657 //Might save us an extra if call?
658 //DB Fill relevant part of thread array
659 if (XBinToFill != -1 && YBinToFill != -1) {
660 samplePDFFD_array_private[YBinToFill][XBinToFill] += totalweight;
661 samplePDFFD_array_private_w2[YBinToFill][XBinToFill] += totalweight*totalweight;
662 }
663 }
664 }
665
666 //End of Calc Weights and fill Array
667 //==================================================
668 // DB Copy contents of 'samplePDFFD_array_private' into 'samplePDFFD_array' which can then be used in GetLikelihood
669 for (int yBin=0;yBin<nYBins;yBin++) {
670 for (int xBin=0;xBin<nXBins;xBin++) {
671#pragma omp atomic
672 samplePDFFD_array[yBin][xBin] += samplePDFFD_array_private[yBin][xBin];
673#pragma omp atomic
674 samplePDFFD_array_w2[yBin][xBin] += samplePDFFD_array_private_w2[yBin][xBin];
675 }
676 }
677
678 for (int yBin=0;yBin<nYBins;yBin++) {
679 delete[] samplePDFFD_array_private[yBin];
680 delete[] samplePDFFD_array_private_w2[yBin];
681 }
682 delete[] samplePDFFD_array_private;
683 delete[] samplePDFFD_array_private_w2;
684 } //end of parallel region
685}

785//A way to check whether a normalisation parameter applies to an event or not
786void samplePDFFDBase::CalcXsecNormsBins(int iSample){
787
788 fdmc_base *fdobj = &MCSamples[iSample];
789
790 for(int iEvent=0; iEvent < fdobj->nEvents; ++iEvent){
791 std::list< int > XsecBins = {};
792 if (XsecCov) {
793 for (std::vector<XsecNorms4>::iterator it = xsec_norms.begin(); it != xsec_norms.end(); ++it) {
794 // Skip oscillated NC events
795 // Not strictly needed, but these events don't get included in oscillated predictions, so
796 // no need to waste our time calculating and storing information about xsec parameters
797 // that will never be used.
798 if (fdobj->isNC[iEvent] && fdobj->signal) {continue;} //DB Abstract check on MaCh3Modes to determine which apply to neutral current
799
800 //Now check that the target of an interaction matches with the normalisation parameters
801 bool TargetMatch=false;
802 //If no target specified then apply to all modes
803 if ((*it).targets.size()==0) {
804 TargetMatch=true;
805 } else {
806 for (unsigned iTarget=0;iTarget<(*it).targets.size();iTarget++) {
807 if ((*it).targets.at(iTarget)== *(fdobj->Target[iEvent])) {
808 TargetMatch=true;
809 }
810 }
811 }
812 if (!TargetMatch) {continue;}
813
814 //Now check that the neutrino flavour in an interaction matches with the normalisation parameters
815 bool FlavourMatch=false;
816 //If no mode specified then apply to all modes
817 if ((*it).pdgs.size()==0) {
818 FlavourMatch=true;
819 } else {
820 for (unsigned iPDG=0;iPDG<(*it).pdgs.size();iPDG++) {
821 if ((*it).pdgs.at(iPDG)== fdobj->nupdg) {
822 FlavourMatch=true;
823 }
824 }
825 }
826 if (!FlavourMatch){continue;}
827
828 //Now check that the unoscillated neutrino flavour in an interaction matches with the normalisation parameters
829 bool FlavourUnoscMatch=false;
830 //If no mode specified then apply to all modes
831 if ((*it).preoscpdgs.size()==0) {
832 FlavourUnoscMatch=true;
833 } else {
834 for (unsigned iPDG=0;iPDG<(*it).preoscpdgs.size();iPDG++) {
835 if ((*it).preoscpdgs.at(iPDG) == fdobj->nupdgUnosc) {
836 FlavourUnoscMatch=true;
837 }
838 }
839 }
840 if (!FlavourUnoscMatch){continue;}
841
842 //Now check that the mode of an interaction matches with the normalisation parameters
843 bool ModeMatch=false;
844 //If no mode specified then apply to all modes
845 if ((*it).modes.size()==0) {
846 ModeMatch=true;
847 } else {
848 for (unsigned imode=0;imode<(*it).modes.size();imode++) {
849 if ((*it).modes.at(imode)== *(fdobj->mode[iEvent])) {
850 ModeMatch=true;
851 }
852 }
853 }
854 if (!ModeMatch) {continue;}
855
856 //Now check whether the norm has kinematic bounds
857 //i.e. does it only apply to events in a particular kinematic region?
858 bool IsSelected = true;
859 if ((*it).hasKinBounds) {
860 for (unsigned int iKinematicParameter = 0 ; iKinematicParameter < (*it).KinematicVarStr.size() ; ++iKinematicParameter ) {
861 if (ReturnKinematicParameter((*it).KinematicVarStr[iKinematicParameter], iSample, iEvent) <= (*it).Selection[iKinematicParameter][0]) {
862 IsSelected = false;
863 continue;
864 }
865 else if (ReturnKinematicParameter((*it).KinematicVarStr[iKinematicParameter], iSample, iEvent) > (*it).Selection[iKinematicParameter][1]) {
866 IsSelected = false;
867 continue;
868 }
869 }
870 }
871 //Need to then break the event loop
872 if(!IsSelected){
873 continue;
874 }
875
876 // Now set 'index bin' for each normalisation parameter
877 // All normalisations are just 1 bin for 2015, so bin = index (where index is just the bin for that normalisation)
878 int bin = (*it).index;
879
880 //If syst on applies to a particular detector
881 if ((XsecCov->GetParDetID(bin) & SampleDetID)==SampleDetID) {
882 XsecBins.push_back(bin);
883 }
884 } // end iteration over xsec_norms
885 } // end if (xsecCov)
886 fdobj->xsec_norms_bins[iEvent]=XsecBins;
887 }//end loop over events
888 return;
889}
