29 void FindKnot(std::vector<double>& SigmaValues,
30  const std::string& dirname,
31  const std::string& subdirname,
32  const std::string& ProjName,
33  std::string histname) {
34  auto StripPrefix = [](std::string& str, const std::string& prefix) {
35  if (str.find(prefix + "_") == 0) {
36  str.erase(0, prefix.length() + 1);
37  if (str.find(prefix) == 0) {
38  MACH3LOG_ERROR("Failed to strip prefix '{}' from string '{}'", prefix, str);
39  throw MaCh3Exception(__FILE__, __LINE__);
40  }
41  } else {
42  MACH3LOG_ERROR("String '{}' does not start with expected prefix '{}'", str, prefix);
43  throw MaCh3Exception(__FILE__, __LINE__);
44  }
45  };
46 
47  // Remove sample and dial name to avoid potential issues
48  StripPrefix(histname, subdirname);
49  StripPrefix(histname, ProjName);
50  StripPrefix(histname, dirname);
51 
52  MACH3LOG_DEBUG("Name afters striping {}", histname);
53 
54  double sigma = 0.0;
55  // Find the "_sig_" part in the name
56  size_t sig_pos = histname.find("_sig_");
57  // Extract the part after "_sig_"
58  std::string sigma_part = histname.substr(sig_pos + 5);
59 
60  // Check if it starts with 'n' (negative) or 'p' (positive) or 'nom' (0 or prior)
61  if (histname.find("nom_") != std::string::npos) {
62  sigma = 0.0;
63  PriorKnot = static_cast<int>(SigmaValues.size());
64  MACH3LOG_DEBUG("Found prior knot {}", PriorKnot);
65  } else if (sigma_part.size() > 0 && sigma_part[0] == 'n') {
66  sigma = -std::stod(sigma_part.substr(1));
67  } else if (sigma_part.size() > 0 && sigma_part[0] == 'p') {
68  sigma = std::stod(sigma_part.substr(1));
69  } else {
70  MACH3LOG_ERROR("Weirdly formatted string {}", sigma_part);
71  throw MaCh3Exception(__FILE__ , __LINE__ );
72  }
73  MACH3LOG_DEBUG("Adding sigma {}", sigma);
74 
75  SigmaValues.push_back(sigma);
76 }
