96// Poisson likelihood calc for data and MC event rates
97double samplePDFBase::getTestStatLLH(const double data, const double mc) const {
98// ***************************************************************************
99 // Need some MC
100 if(mc == 0) return 0.;
101
102 double negLogL = 0;
103 if(mc > 0 && data > 0)
104 {
105 //http://hyperphysics.phy-astr.gsu.edu/hbase/math/stirling.html
106 negLogL += (mc - data + data * std::log(data/mc));
107 }
108 else if(mc > 0 && data == 0) negLogL += mc;
109
110 return negLogL;
111}

113// *************************
114// data is data, mc is mc, w2 is Sum(w_{i}^2) (sum of weights squared), which is sigma^2_{MC stats}
115double samplePDFBase::getTestStatLLH(const double data, const double mc, const double w2) const {
116// *************************
117 // Need some MC
118 if (mc == 0) return 0.0;
119
120 // The MC used in the likelihood calculation
121 // Is allowed to be changed by Barlow Beeston beta parameters
122 double newmc = mc;
123
124 switch (fTestStatistic)
125 {
126 //CW: Not full Barlow-Beeston or what is referred to as "light": we're not introducing any more parameters
127 // Assume the MC has a Gaussian distribution around generated
128 // As in https://arxiv.org/abs/1103.0354 eq 10, 11
129 //CW: Calculate the Barlow-Beeston likelihood contribution from MC statistics
130 // Assumes the beta scaling parameters are Gaussian distributed
131 // Follows arXiv:1103.0354 section 5 and equation 8, 9, 10, 11 on page 4/5
132 // Essentially solves equation 11
133 case (kBarlowBeeston):
134 {
135 // The penalty from MC statistics using Conways approach (https://cds.cern.ch/record/1333496?)
136 double penalty = 0;
137 // Barlow-Beeston uses fractional uncertainty on MC, so sqrt(sum[w^2])/mc
138 const double fractional = std::sqrt(w2)/mc;
139 // b in quadratic equation
140 const double temp = mc*fractional*fractional-1;
141 // b^2 - 4ac in quadratic equation
142 const double temp2 = temp*temp + 4*data*fractional*fractional;
143 if (temp2 < 0) {
144 MACH3LOG_ERROR("Negative square root in Barlow Beeston coefficient calculation!");
145 throw MaCh3Exception(__FILE__ , __LINE__ );
146 }
147 // Solve for the positive beta
148 const double beta = (-1*temp+sqrt(temp2))/2.;
149 newmc = mc*beta;
150 // And penalise the movement in beta relative the mc uncertainty
151 if (fractional > 0) penalty = (beta-1)*(beta-1)/(2*fractional*fractional);
152 else penalty = 0;
153
154 // Calculate the new Poisson likelihood
155 // For Barlow-Beeston newmc is modified, so can only calculate Poisson likelihood after Barlow-Beeston
156 // For the Poisson likelihood, this is just the usual calculation
157 // For IceCube likelihood, we calculate it later
158 double stat = 0;
159 // All likelihood calculations may use the bare Poisson likelihood, so calculate here
160 if (data == 0) stat = newmc;
161 else if (newmc > 0) stat = newmc-data+data*std::log(data/newmc);
162
163 // Return the statistical contribution and penalty
164 return stat+penalty;
165 }
166 break;
167 //KS: Alternative calculation of Barlow-Beeston following Hans Dembinski and Ahmed Abdelmottele arXiv:2206.12346v2
168 case (kDembinskiAbdelmottele):
169 {
170 //KS: code follows authors implementation from:
171 //https://github.com/scikit-hep/iminuit/blob/059d06b00cae097ebf340b218b4eb57357111df8/src/iminuit/cost.py#L274-L300
172
173 //the so-called effective count
174 const double k = mc*mc / w2;
175 //Calculate beta which is scaling factor between true and generated MC
176 const double beta = (data + k) / (mc + k);
177
178 newmc = mc*beta;
179 // And penalise the movement in beta relative the mc uncertainty
180 const double penalty = k*beta-k+k*std::log(k/(k*beta));
181
182 // Calculate the new Poisson likelihood
183 // For Barlow-Beeston newmc is modified, so can only calculate Poisson likelihood after Barlow-Beeston
184 // For the Poisson likelihood, this is just the usual calculation
185 // For IceCube likelihood, we calculate it later
186 double stat = 0;
187 // All likelihood calculations may use the bare Poisson likelihood, so calculate here
188 if (data == 0) stat = newmc;
189 else if (newmc > 0) stat = newmc-data+data*std::log(data/newmc);
190
191 // Return the statistical contribution and penalty
192 return stat+penalty;
193 }
194 break;
195 //CW: Also try the IceCube likelihood
196 // It does not modify the MC content
197 // https://arxiv.org/abs/1901.04645
198 // Argüelles, C.A., Schneider, A. & Yuan, T. J. High Energ. Phys. (2019) 2019: 30. https://doi.org/10.1007/JHEP06(2019)030
199 // We essentially construct eq 3.16 and take the logarithm
200 // in eq 3.16, mu is MC, sigma2 is w2, k is data
201 case (kIceCube):
202 {
203 double stat = 0.0;
204
205 // If there for some reason is 0 mc uncertainty, return the Poisson LLH
206 if (w2 == 0)
207 {
208 // Calculate the new Poisson likelihood
209 if (data == 0) stat = newmc;
210 else if (newmc > 0) stat = newmc-data+data*std::log(data/newmc);
211
212 return stat;
213 }
214 // Auxiliary variables
215 const long double b = mc/w2;
216 const long double a = mc*b+1;
217 const long double k = data;
218 // Use C99's implementation of log of gamma function to not be C++11 dependent
219 stat = double(-1*(a * logl(b) + lgammal(k+a) - lgammal(k+1) - ((k+a)*log1pl(b)) - lgammal(a)));
220
221 // Return the statistical contribution and penalty
222 return stat;
223 }
224 break;
225 //KS: Pearson works on assumption that event distribution in each bin is described by a Gaussian which in our case is not fulfilled for all bins, hence use it at your own risk
226 case (kPearson):
227 {
228 //KS: 2 is because this function returns -LLH not -2LLH
229 const double stat = (data-mc)*(data-mc)/(2*mc);
230
231 // Return the statistical
232 return stat;
233 }
234 break;
235 case (kPoisson):
236 {
237 //Just call getTestStatLLH which doesn't take in weights
238 //and is a Poisson likelihood comparison.
239 return getTestStatLLH(data, mc);//stat;
240 }
241 break;
242 case TestStatistic::kNTestStatistics:
243 MACH3LOG_ERROR("kNTestStatistics is not a valid TestStatistic!");
244 throw MaCh3Exception(__FILE__, __LINE__);
245 default:
246 MACH3LOG_ERROR("Couldn't find TestStatistic {} exiting!", static_cast<int>(fTestStatistic));
247 throw MaCh3Exception(__FILE__ , __LINE__ );
248 } // end switch
249}
