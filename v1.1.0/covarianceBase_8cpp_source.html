756 // ************************************************
757 // Correlate the steps by setting the proposed step of a parameter to its current value + some correlated throw
758 void covarianceBase::CorrelateSteps() {
759 // ************************************************
760  //KS: Using custom function compared to ROOT one with 8 threads we have almost factor 2 performance increase, by replacing TMatrix with just double we increase it even more
761  MatrixVectorMulti(corr_throw, throwMatrixCholDecomp, randParams, _fNumPar);
762 
763  // If not doing PCA
764  if (!pca) {
765  #ifdef MULTITHREAD
766  #pragma omp parallel for
767  #endif
768  for (int i = 0; i < _fNumPar; ++i) {
769  if (_fError[i] > 0.) {
770  _fPropVal[i] = _fCurrVal[i] + corr_throw[i]*_fGlobalStepScale*_fIndivStepScale[i];
771  }
772  }
773  // If doing PCA throw uncorrelated in PCA basis (orthogonal basis by definition)
774  } else {
775  // Throw around the current step
776  #ifdef MULTITHREAD
777  #pragma omp parallel for
778  #endif
779  for (int i = 0; i < _fNumParPCA; ++i)
780  {
781  if (fParSigma_PCA[i] > 0.)
782  {
783  double IndStepScale = 1.;
784  //KS: If undecomposed parameter apply individual step scale and Cholesky for better acceptance rate
785  if(isDecomposed_PCA[i] >= 0)
786  {
787  IndStepScale *= _fIndivStepScale[isDecomposed_PCA[i]];
788  IndStepScale *= corr_throw[isDecomposed_PCA[i]];
789  }
790  //If decomposed apply only random number
791  else
792  {
793  IndStepScale *= randParams[i];
794  //KS: All PCA-ed parameters have the same step scale
795  IndStepScale *= _fIndivStepScale[FirstPCAdpar];
796  }
797  fParProp_PCA(i) = fParCurr_PCA(i)+_fGlobalStepScale*IndStepScale*PCAObj.eigen_values_master[i];
798  }
799  }
800  // Then update the parameter basis
801  TransferToParam();
802  }
803 }
