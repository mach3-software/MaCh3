198 inline double GetBetaParameter(const double data, const double mc, const double w2, TestStatistic TestStat) {
199 // ****************
200  double Beta = 0.0;
201 
202  if (TestStat == kDembinskiAbdelmottele) {
203  //the so-called effective count
204  const double k = mc*mc / w2;
205  //Calculate beta which is scaling factor between true and generated MC
206  Beta = (data + k) / (mc + k);
207  }
208  //KS: Below is technically only true for Cowan's BB, which will not be true for Poisson or IceCube, because why not...
209  else {
210  // CW: Barlow-Beeston uses fractional uncertainty on MC, so sqrt(sum[w^2])/mc
211  const double fractional = std::sqrt(w2)/mc;
212  // CW: -b/2a in quadratic equation
213  const double temp = mc*fractional*fractional-1;
214  // CW: b^2 - 4ac in quadratic equation
215  const double temp2 = temp*temp + 4*data*fractional*fractional;
216  if (temp2 < 0) {
217  MACH3LOG_ERROR("Negative square root in Barlow Beeston coefficient calculation!");
218  throw MaCh3Exception(__FILE__ , __LINE__ );
219  }
220  // CW: Solve for the positive beta
221  Beta = (-1*temp+std::sqrt(temp2))/2.;
222  }
223  return Beta;
224 }

230 inline double GetSubOptimality(const std::vector<double>& EigenValues, const int TotalTarameters) {
231 // *********************
232  double sum_eigenvalues_squared_inv = 0.0;
233  double sum_eigenvalues_inv = 0.0;
234  for (unsigned int j = 0; j < EigenValues.size(); j++)
235  {
236  //KS: IF Eigen values are super small skip them
237  //if(EigenValues[j] < 0.0000001) continue;
238  sum_eigenvalues_squared_inv += std::pow(EigenValues[j], -2);
239  sum_eigenvalues_inv += 1.0 / EigenValues[j];
240  }
241  const double SubOptimality = TotalTarameters * sum_eigenvalues_squared_inv / std::pow(sum_eigenvalues_inv, 2);
242  return SubOptimality;

290 inline void GetHPD(TH1D* const hist, double& Mean, double& Error, double& Error_p, double& Error_m, const double coverage = 0.6827) {
291 // ****************
292  // Get the bin which has the largest posterior density
293  const int MaxBin = hist->GetMaximumBin();
294  // And it's value
295  const double peakval = hist->GetBinCenter(MaxBin);
296 
297  // The total integral of the posterior
298  const long double Integral = hist->Integral();
299  //KS: and integral of left handed and right handed parts
300  const long double LowIntegral = hist->Integral(1, MaxBin-1) + hist->GetBinContent(MaxBin)/2.0;
301  const long double HighIntegral = hist->Integral(MaxBin+1, hist->GetNbinsX()) + hist->GetBinContent(MaxBin)/2.0;
302 
303  // Keep count of how much area we're covering
304  //KS: Take only half content of HPD bin as one half goes for right handed error and the other for left handed error
305  long double sum = hist->GetBinContent(MaxBin)/2.0;
306 
307  // Counter for current bin
308  int CurrBin = MaxBin;
309  while (sum/HighIntegral < coverage && CurrBin < hist->GetNbinsX()) {
310  CurrBin++;
311  sum += hist->GetBinContent(CurrBin);
312  }
313  const double sigma_p = std::fabs(hist->GetBinCenter(MaxBin)-hist->GetXaxis()->GetBinUpEdge(CurrBin));
314  // Reset the sum
315  //KS: Take only half content of HPD bin as one half goes for right handed error and the other for left handed error
316  sum = hist->GetBinContent(MaxBin)/2.0;
317 
318  // Reset the bin counter
319  CurrBin = MaxBin;
320  // Counter for current bin
321  while (sum/LowIntegral < coverage && CurrBin > 1) {
322  CurrBin--;
323  sum += hist->GetBinContent(CurrBin);
324  }
325  const double sigma_m = std::fabs(hist->GetBinCenter(CurrBin)-hist->GetBinLowEdge(MaxBin));
326 
327  // Now do the double sided HPD
328  //KS: Start sum from the HPD
329  sum = hist->GetBinContent(MaxBin);
330  int LowBin = MaxBin;
331  int HighBin = MaxBin;
332  long double LowCon = 0.0;
333  long double HighCon = 0.0;
334 
335  while (sum/Integral < coverage && (LowBin > 0 || HighBin < hist->GetNbinsX()+1))
336  {
337  LowCon = 0.0;
338  HighCon = 0.0;
339  //KS:: Move further only if you haven't reached histogram end
340  if(LowBin > 1)
341  {
342  LowBin--;
343  LowCon = hist->GetBinContent(LowBin);
344  }
345  if(HighBin < hist->GetNbinsX())
346  {
347  HighBin++;
348  HighCon = hist->GetBinContent(HighBin);
349  }
350 
351  // If we're on the last slice and the lower contour is larger than the upper
352  if ((sum+LowCon+HighCon)/Integral > coverage && LowCon > HighCon) {
353  sum += LowCon;
354  break;
355  // If we're on the last slice and the upper contour is larger than the lower
356  } else if ((sum+LowCon+HighCon)/Integral > coverage && HighCon >= LowCon) {
357  sum += HighCon;
358  break;
359  } else {
360  sum += LowCon + HighCon;
361  }
362  }
363 
364  double sigma_hpd = 0.0;
365  if (LowCon > HighCon) {
366  sigma_hpd = std::fabs(hist->GetBinLowEdge(LowBin)-hist->GetBinCenter(MaxBin));
367  } else {
368  sigma_hpd = std::fabs(hist->GetXaxis()->GetBinUpEdge(HighBin)-hist->GetBinCenter(MaxBin));
369  }
370 
371  Mean = peakval;
372  Error = sigma_hpd;
373  Error_p = sigma_p;
374  Error_m = sigma_m;
375 }

383 inline void GetCredibleInterval(TH1D* const hist, TH1D* hpost_copy, const double coverage = 0.6827) {
384 // ***************
385  if(coverage > 1)
386  {
387  MACH3LOG_ERROR("Specified Credible Interval is greater that 1 and equal to {} Should be between 0 and 1", coverage);
388  throw MaCh3Exception(__FILE__ , __LINE__ );
389  }
390  //KS: Reset first copy of histogram
391  hpost_copy->Reset("");
392  hpost_copy->Fill(0.0, 0.0);
393 
394  //KS: Temporary structure to be thread save
395  double *hist_copy = new double[hist->GetXaxis()->GetNbins()+1];
396  bool *hist_copy_fill = new bool[hist->GetXaxis()->GetNbins()+1];
397  for (int i = 0; i <= hist->GetXaxis()->GetNbins(); ++i)
398  {
399  hist_copy[i] = hist->GetBinContent(i);
400  hist_copy_fill[i] = false;
401  }
402 
404  const long double Integral = hist->Integral();
405  long double sum = 0;
406 
407  while ((sum / Integral) < coverage)
408  {
410  int max_entry_bin = 0;
411  double max_entries = 0.;
412  for (int i = 0; i <= hist->GetXaxis()->GetNbins(); ++i)
413  {
414  if (hist_copy[i] > max_entries)
415  {
416  max_entries = hist_copy[i];
417  max_entry_bin = i;
418  }
419  }
421  hist_copy[max_entry_bin] = -1.;
422  hist_copy_fill[max_entry_bin] = true;
423 
424  sum += max_entries;
425  }
426  //KS: Now fill our copy only for bins which got included in coverage region
427  for(int i = 0; i <= hist->GetXaxis()->GetNbins(); ++i)
428  {
429  if(hist_copy_fill[i]) hpost_copy->SetBinContent(i, hist->GetBinContent(i));
430  }
431 
432  delete[] hist_copy;
433  delete[] hist_copy_fill;
434 
435  return;
436 }
