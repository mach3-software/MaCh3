236void MCMCProcessor::MakePostfit() {
237// ****************************
238 // Check if we've already made post-fit
239 if (MadePostfit == true) return;
240 MadePostfit = true;
241
242 // Check if the output file is ready
243 if (OutputFile == nullptr) MakeOutputFile();
244
245 MACH3LOG_INFO("MCMCProcessor is making post-fit plots...");
246
247 // Directory for posteriors
248 TDirectory *PostDir = OutputFile->mkdir("Post");
249 TDirectory *PostHistDir = OutputFile->mkdir("Post_1d_hists");
250
251 // We fit with this Gaussian
252 Gauss = new TF1("Gauss","[0]/sqrt(2.0*3.14159)/[2]*TMath::Exp(-0.5*pow(x-[1],2)/[2]/[2])",-5,5);
253 Gauss->SetLineWidth(2);
254 Gauss->SetLineColor(kOrange-5);
255
256 // nDraw is number of draws we want to do
257 for (int i = 0; i < nDraw; ++i)
258 {
259 if (i % (nDraw/5) == 0) {
260 MaCh3Utils::PrintProgressBar(i, nDraw);
261 }
262 OutputFile->cd();
263 TString Title = "";
264 double Prior = 1.0, PriorError = 1.0;
265 GetNthParameter(i, Prior, PriorError, Title);
266
267 // This holds the posterior density
268 const double maxi = Chain->GetMaximum(BranchNames[i]);
269 const double mini = Chain->GetMinimum(BranchNames[i]);
270 // This holds the posterior density
271 hpost[i] = new TH1D(BranchNames[i], BranchNames[i], nBins, mini, maxi);
272 hpost[i]->SetMinimum(0);
273 hpost[i]->GetYaxis()->SetTitle("Steps");
274 hpost[i]->GetYaxis()->SetNoExponent(false);
275
276 //KS: Apply additional Cuts, like mass ordering
277 std::string CutPosterior1D = "";
278 if(Posterior1DCut != "")
279 {
280 CutPosterior1D = StepCut +" && " + Posterior1DCut;
281 }
282 else CutPosterior1D = StepCut;
283
284 // Project BranchNames[i] onto hpost, applying stepcut
285 Chain->Project(BranchNames[i], BranchNames[i], CutPosterior1D.c_str());
286
287 if(ApplySmoothing) hpost[i]->Smooth();
288
289 (*Central_Value)(i) = Prior;
290
291 double Mean, Err, Err_p, Err_m;
292 GetArithmetic(hpost[i], Mean, Err);
293 (*Means)(i) = Mean;
294 (*Errors)(i) = Err;
295
296 GetGaussian(hpost[i], Gauss, Mean, Err);
297 (*Means_Gauss)(i) = Mean;
298 (*Errors_Gauss)(i) = Err;
299
300 GetHPD(hpost[i], Mean, Err, Err_p, Err_m);
301 (*Means_HPD)(i) = Mean;
302 (*Errors_HPD)(i) = Err;
303 (*Errors_HPD_Positive)(i) = Err_p;
304 (*Errors_HPD_Negative)(i) = Err_m;
305
306 // Write the results from the projection into the TVectors and TMatrices
307 (*Covariance)(i,i) = (*Errors)(i)*(*Errors)(i);
308 (*Correlation)(i,i) = 1.0;
309
310 //KS: This need to be before SetMaximum(), this way plot is nicer as line end at the maximum
311 auto hpd = std::make_unique<TLine>((*Means_HPD)(i), hpost[i]->GetMinimum(), (*Means_HPD)(i), hpost[i]->GetMaximum());
312 SetTLineStyle(hpd.get(), kBlack, 2, kSolid);
313
314 hpost[i]->SetLineWidth(2);
315 hpost[i]->SetLineColor(kBlue-1);
316 hpost[i]->SetMaximum(hpost[i]->GetMaximum()*DrawRange);
317 hpost[i]->SetTitle(Title);
318 hpost[i]->GetXaxis()->SetTitle(hpost[i]->GetTitle());
319
320 // Now make the TLine for the Asimov
321 auto Asimov = std::make_unique<TLine>(Prior, hpost[i]->GetMinimum(), Prior, hpost[i]->GetMaximum());
322 SetTLineStyle(Asimov.get(), kRed-3, 2, kDashed);
323
324 auto leg = std::make_unique<TLegend>(0.12, 0.6, 0.6, 0.97);
325 SetLegendStyle(leg.get(), 0.04);
326 leg->AddEntry(hpost[i], Form("#splitline{PDF}{#mu = %.2f, #sigma = %.2f}", hpost[i]->GetMean(), hpost[i]->GetRMS()), "l");
327 leg->AddEntry(Gauss, Form("#splitline{Gauss}{#mu = %.2f, #sigma = %.2f}", Gauss->GetParameter(1), Gauss->GetParameter(2)), "l");
328 leg->AddEntry(hpd.get(), Form("#splitline{HPD}{#mu = %.2f, #sigma = %.2f (+%.2f-%.2f)}", (*Means_HPD)(i), (*Errors_HPD)(i), (*Errors_HPD_Positive)(i), (*Errors_HPD_Negative)(i)), "l");
329 leg->AddEntry(Asimov.get(), Form("#splitline{Prior}{x = %.2f , #sigma = %.2f}", Prior, PriorError), "l");
330
331 //CW: Don't plot if this is a fixed histogram (i.e. the peak is the whole integral)
332 if (hpost[i]->GetMaximum() == hpost[i]->Integral()*DrawRange)
333 {
334 MACH3LOG_WARN("Found fixed parameter, moving on");
335 IamVaried[i] = false;
336 //KS:Set mean and error to prior for fixed parameters, it looks much better when fixed parameter has mean on prior rather than on 0 with 0 error.
337 (*Means_HPD)(i) = Prior;
338 (*Errors_HPD)(i) = PriorError;
339 continue;
340 }
341
342 // Store that this parameter is indeed being varied
343 IamVaried[i] = true;
344
345 // Write to file
346 Posterior->SetName(Title);
347 Posterior->SetTitle(Title);
348
349 // Draw onto the TCanvas
350 hpost[i]->Draw();
351 hpd->Draw("same");
352 Asimov->Draw("same");
353 leg->Draw("same");
354
355 if(printToPDF) Posterior->Print(CanvasName);
356
357 // cd into params directory in root file
358 PostDir->cd();
359 Posterior->Write();
360
361 hpost[i]->SetName(Title);
362 hpost[i]->SetTitle(Title);
363 PostHistDir->cd();
364 hpost[i]->Write();
365 } // end the for loop over nDraw
366
367 OutputFile->cd();
368 TTree *SettingsBranch = new TTree("Settings", "Settings");
369 int CrossSectionParameters = nParam[kXSecPar];
370 SettingsBranch->Branch("CrossSectionParameters", &CrossSectionParameters);
371 int CrossSectionParametersStartingPos = ParamTypeStartPos[kXSecPar];
372 SettingsBranch->Branch("CrossSectionParametersStartingPos", &CrossSectionParametersStartingPos);
373 int FluxParameters = GetGroup("Flux");
374 SettingsBranch->Branch("FluxParameters", &FluxParameters);
375
376 int NDParameters = nParam[kNDPar];
377 SettingsBranch->Branch("NDParameters", &NDParameters);
378 int NDParametersStartingPos = ParamTypeStartPos[kNDPar];
379 SettingsBranch->Branch("NDParametersStartingPos", &NDParametersStartingPos);
380
381 int FDParameters = nParam[kFDDetPar];
382 SettingsBranch->Branch("FDParameters", &FDParameters);
383 int FDParametersStartingPos = ParamTypeStartPos[kFDDetPar];
384 SettingsBranch->Branch("FDParametersStartingPos", &FDParametersStartingPos);
385
386 int OscParameters = nParam[kOSCPar];
387 SettingsBranch->Branch("OscParameters", &OscParameters);
388 int OscParametersStartingPos = ParamTypeStartPos[kOSCPar];
389 SettingsBranch->Branch("OscParametersStartingPos", &OscParametersStartingPos);
390
391 SettingsBranch->Branch("NDSamplesBins", &NDSamplesBins);
392 SettingsBranch->Branch("NDSamplesNames", &NDSamplesNames);
393
394 SettingsBranch->Fill();
395 SettingsBranch->Write();
396 delete SettingsBranch;
397
398 TDirectory *Names = OutputFile->mkdir("Names");
399 Names->cd();
400 for (std::vector<TString>::iterator it = BranchNames.begin(); it != BranchNames.end(); ++it) {
401 TObjString((*it)).Write();
402 }
403 Names->Close();
404 delete Names;
405
406 OutputFile->cd();
407 Central_Value->Write("Central_Value");
408 Means->Write("PDF_Means");
409 Errors->Write("PDF_Error");
410 Means_Gauss->Write("Gauss_Means");
411 Errors_Gauss->Write("Gauss_Errors");
412 Means_HPD->Write("Means_HPD");
413 Errors_HPD->Write("Errors_HPD");
414 Errors_HPD_Positive->Write("Errors_HPD_Positive");
415 Errors_HPD_Negative->Write("Errors_HPD_Negative");
416
417 PostDir->Close();
418 delete PostDir;
419 PostHistDir->Close();
420 delete PostHistDir;
421} // Have now written the postfit projections

953
954 if (HaveMadeDiagonal == false) {
955 MakePostfit();

1032 //OutputFile->cd();

1034 } // End j loop

1041// ***************
1042//KS: Cache all steps to allow multithreading, hit RAM quite a bit
1043void MCMCProcessor::CacheSteps() {
1044// ***************
1045 if(CacheMCMC == true) return;
1046
1047 CacheMCMC = true;
1048
1049 if(ParStep != nullptr)
1050 {
1051 MACH3LOG_ERROR("It look like ParStep was already filled ");

1055 }
1056
1057 MACH3LOG_INFO("Caching input tree...");
1058 MACH3LOG_INFO("Allocating {:.2f} MB", double(sizeof(double)*nDraw*nEntries)/1.E6);
1059 TStopwatch clock;
1060 clock.Start();
1061
1062 ParStep = new double*[nDraw];
1063 StepNumber = new int[nEntries];

1067 {

1070 for (int j = 0; j < nEntries; ++j)

1100 Chain->GetEntry(j);

1103 // Set the branch addresses for params
1104 for (int i = 0; i < nDraw; ++i)
1105 {
1106 ParStep[i][j] = ParValBranch[i];
1107 }
1108 }
1109 delete[] ParValBranch;
1110
1111 // Set all the branches to on
1112 Chain->SetBranchStatus("*", true);
1113
1114 // Cache max and min in chain for covariance matrix
1115 for (int i = 0; i < nDraw; ++i)
1116 {
1117 const double Min_Chain_i = Chain->GetMinimum(BranchNames[i]);
1118 const double Max_Chain_i = Chain->GetMaximum(BranchNames[i]);

1124 for (int j = 0; j <= i; ++j)
1125 {
1126 const double Min_Chain_j = Chain->GetMinimum(BranchNames[j]);
1127 const double Max_Chain_j = Chain->GetMaximum(BranchNames[j]);
1128
1129 // TH2D to hold the Correlation
1130 hpost2D[i][j] = new TH2D(Form("hpost2D_%i_%i",i,j), Form("hpost2D_%i_%i",i,j), nBins, Min_Chain_i, Max_Chain_i, nBins, Min_Chain_j, Max_Chain_j);
1131 TString Title_j = "";
1132 double Prior_j, PriorError_j;
1133 GetNthParameter(j, Prior_j, PriorError_j, Title_j);
1134
1135 hpost2D[i][j]->SetMinimum(0);
1136 hpost2D[i][j]->GetXaxis()->SetTitle(Title_i);
1137 hpost2D[i][j]->GetYaxis()->SetTitle(Title_j);
1138 hpost2D[i][j]->GetZaxis()->SetTitle("Steps");
1139 }
1140 }
1141 clock.Stop();
1142 MACH3LOG_INFO("Caching steps took {:.2f}s to finish for {} steps", clock.RealTime(), nEntries );
1143}

1940// Scan the input trees
1941void MCMCProcessor::ScanInput() {
1942// **************************
1943 // KS: This can reduce time necessary for caching even by half
1944 #ifdef MULTITHREAD
1945 //ROOT::EnableImplicitMT();
1946 #endif
1947
1948 // Open the Chain
1949 Chain = new TChain("posteriors","posteriors");
1950 Chain->Add(MCMCFile.c_str());
1951
1952 nEntries = int(Chain->GetEntries());
1953
1954 //Only is suboptimality we might want to change it, therefore set it high enough so it doesn't affect other functionality
1955 UpperCut = nEntries+1;
1956
1957 // Get the list of branches
1958
1959 TObjArray* brlis = Chain->GetListOfBranches();
1960
1961 // Get the number of branches
1962 nBranches = brlis->GetEntries();
1963
1964 BranchNames.reserve(nBranches);
1965 IamVaried.reserve(nBranches);
1966 ParamType.reserve(nBranches);
1967
1968 // Set all the branches to off
1969 Chain->SetBranchStatus("*", false);
1970
1971 // Loop over the number of branches
1972 // Find the name and how many of each systematic we have
1973 for (int i = 0; i < nBranches; i++)
1974 {
1975 // Get the TBranch and its name
1976 TBranch* br = static_cast<TBranch*>(brlis->At(i));
1977 if(!br){
1978 MACH3LOG_ERROR("Invalid branch at position {}", i);
1979 throw MaCh3Exception(__FILE__,__LINE__);
1980 }
1981 TString bname = br->GetName();
1982
1983 //KS: Exclude parameter types
1984 bool rejected = false;
1985 for(unsigned int ik = 0; ik < ExcludedTypes.size(); ++ik )
1986 {
1987 if(bname.BeginsWith(ExcludedTypes[ik]))
1988 {
1989 rejected = true;
1990 break;
1991 }
1992 }
1993 if(rejected) continue;
1994
1995 // Turn on the branches which we want for parameters
1996 Chain->SetBranchStatus(bname.Data(), true);
1997
1998 // If we're on beam systematics
1999 if(bname.BeginsWith("xsec_"))
2000 {
2001 BranchNames.push_back(bname);
2002 ParamType.push_back(kXSecPar);
2003 nParam[kXSecPar]++;
2004 }
2005 else if (bname.BeginsWith("ndd_"))
2006 {
2007 BranchNames.push_back(bname);
2008 ParamType.push_back(kNDPar);
2009 nParam[kNDPar]++;
2010 }
2011 else if (bname.BeginsWith("skd_joint_"))
2012 {
2013 BranchNames.push_back(bname);
2014 ParamType.push_back(kFDDetPar);
2015 nParam[kFDDetPar]++;
2016 }
2017 else if (bname.BeginsWith("sin2th_") ||
2018 bname.BeginsWith("delm2_") ||
2019 bname.BeginsWith("delta_") ||
2020 bname.BeginsWith("baseline") ||
2021 bname.BeginsWith("density") )
2022 {
2023 BranchNames.push_back(bname);
2024 ParamType.push_back(kOSCPar);
2025 nParam[kOSCPar]++;
2026 }
2027
2028 //KS: as a bonus get LogL systematic
2029 if (bname.BeginsWith("LogL_sample_")) {
2030 SampleName_v.push_back(bname);
2031 nSamples++;
2032 }
2033 else if (bname.BeginsWith("LogL_systematic_")) {
2034 SystName_v.push_back(bname);
2035 nSysts++;
2036 }

2039 // Read the input Covariances
2040 ReadInputCov();
2041
2042 // Check order of parameter types
2043 ScanParameterOrder();
2044
2045 // Print useful Info
2046 PrintInfo();

2048 nSteps = Chain->GetMaximum("step");

2050 int cut = nSteps/5;
2051 SetStepCut(cut);
2052}

2786// **************************
2787// KS: Reweight prior of MCMC chain to another
2788void MCMCProcessor::ReweightPrior(const std::vector<std::string>& Names,
2789 const std::vector<double>& NewCentral,
2790 const std::vector<double>& NewError) {
2791// **************************
2792 MACH3LOG_INFO("Reweighting Prior");
2793
2794 if( (Names.size() != NewCentral.size()) || (NewCentral.size() != NewError.size()))
2795 {
2796 MACH3LOG_ERROR("Size of passed vectors doesn't match in ReweightPrior");
2797 throw MaCh3Exception(__FILE__ , __LINE__ );
2798 }
2799 std::vector<int> Param;
2800 std::vector<double> OldCentral;
2801 std::vector<double> OldError;
2802 std::vector<bool> FlatPrior;
2803
2804 //KS: First we need to find parameter number based on name
2805 for(unsigned int k = 0; k < Names.size(); ++k)
2806 {
2807 //KS: First we need to find parameter number based on name
2808 int ParamNo = GetParamIndexFromName(Names[k]);
2809 if(ParamNo == _UNDEF_)
2810 {
2811 MACH3LOG_WARN("Couldn't find param {}. Can't reweight Prior", Names[k]);
2812 continue;
2813 }
2814
2815 TString Title = "";
2816 double Prior = 1.0, PriorError = 1.0;
2817 GetNthParameter(ParamNo, Prior, PriorError, Title);
2818
2819 Param.push_back(ParamNo);
2820 OldCentral.push_back(Prior);
2821 OldError.push_back(PriorError);
2822
2823 ParameterEnum ParType = ParamType[ParamNo];
2824 int ParamTemp = ParamNo - ParamTypeStartPos[ParType];
2825
2826 FlatPrior.push_back(ParamFlat[ParType][ParamTemp]);
2827 }
2828 std::vector<double> ParameterPos(Names.size());
2829
2830 std::string InputFile = MCMCFile+".root";
2831 std::string OutputFilename = MCMCFile + "_reweighted.root";
2832
2833 //KS: Simply create copy of file and add there new branch
2834 int ret = system(("cp " + InputFile + " " + OutputFilename).c_str());
2835 if (ret != 0)
2836 MACH3LOG_WARN("Error: system call to copy file failed with code {}", ret);
2837
2838 TFile *OutputChain = new TFile(OutputFilename.c_str(), "UPDATE");
2839 OutputChain->cd();
2840 TTree *post = OutputChain->Get<TTree>("posteriors");
2841
2842 double Weight = 1.;
2843
2844 post->SetBranchStatus("*",false);
2845 // Set the branch addresses for params
2846 for (unsigned int j = 0; j < Names.size(); ++j) {
2847 post->SetBranchStatus(BranchNames[Param[j]].Data(), true);
2848 post->SetBranchAddress(BranchNames[Param[j]].Data(), &ParameterPos[j]);
2849 }
2850 TBranch *bpt = post->Branch("Weight", &Weight, "Weight/D");
2851 post->SetBranchStatus("Weight", true);
2852
2853 for (int i = 0; i < nEntries; ++i)
2854 {
2855 post->GetEntry(i);
2856 Weight = 1.;
2857
2858 //KS: Calculate reweight weight. Weights are multiplicative so we can do several reweights at once. FIXME Big limitation is that code only works for uncorrelated parameters :(
2859 for (unsigned int j = 0; j < Names.size(); ++j)
2860 {
2861 double new_chi = (ParameterPos[j] - NewCentral[j])/NewError[j];
2862 double new_prior = std::exp(-0.5 * new_chi * new_chi);
2863
2864 double old_chi = -1;
2865 double old_prior = -1;
2866 if(FlatPrior[j]) {
2867 old_prior = 1.0;
2868 } else {
2869 old_chi = (ParameterPos[j] - OldCentral[j])/OldError[j];
2870 old_prior = std::exp(-0.5 * old_chi * old_chi);
2871 }
2872 Weight *= new_prior/old_prior;
2873 }
2874 bpt->Fill();
2875 }
2876 post->SetBranchStatus("*",true);
2877 OutputChain->cd();
2878 post->Write("posteriors", TObject::kOverwrite);
2879 OutputChain->Close();
2880 delete OutputChain;
2881
2882 OutputFile->cd();
2883}

3600 (*SamplingEfficiency)(j) = _UNDEF_;

3711// **************************
3712// Get the batched means variance estimation and variable indicating if number of batches is sensible
3713void MCMCProcessor::BatchedAnalysis() {
3714// **************************
3715 if(BatchedAverages == nullptr)
3716 {
3717 MACH3LOG_ERROR("BatchedAverages haven't been initialises or have been deleted something is wrong");
3718 MACH3LOG_ERROR("I need it and refuse to go further");
3719 throw MaCh3Exception(__FILE__ , __LINE__ );
3720 }
3721
3722 // Calculate variance estimator using batched means following @cite chakraborty2019estimating see Eq. 1.2
3723 TVectorD* BatchedVariance = new TVectorD(nDraw);
3724 //KS: The hypothesis is rejected if C > z α for a given confidence level α. If the batch means do not pass the test, Correlated is reported for the half-width on the statistical reports following @cite rossetti2024batch alternatively for more old-school see Alexopoulos and Seila 1998 section 3.4.3
3725 TVectorD* C_Test_Statistics = new TVectorD(nDraw);
3726
3727 std::vector<double> OverallBatchMean(nDraw);
3728 std::vector<double> C_Rho_Nominator(nDraw);
3729 std::vector<double> C_Rho_Denominator(nDraw);
3730 std::vector<double> C_Nominator(nDraw);
3731 std::vector<double> C_Denominator(nDraw);
3732 const int BatchLength = nEntries/nBatches+1;
3733//KS: Start parallel region
3734#ifdef MULTITHREAD
3735#pragma omp parallel
3736{
3737#endif
3738 #ifdef MULTITHREAD
3739 #pragma omp for
3740 #endif
3741 //KS: First calculate mean of batched means for each param and Initialise everything to 0
3742 for (int j = 0; j < nDraw; ++j)
3743 {
3744 OverallBatchMean[j] = 0.0;
3745 C_Rho_Nominator[j] = 0.0;
3746 C_Rho_Denominator[j] = 0.0;
3747 C_Nominator[j] = 0.0;
3748 C_Denominator[j] = 0.0;
3749
3750 (*BatchedVariance)(j) = 0.0;
3751 (*C_Test_Statistics)(j) = 0.0;
3752 for (int i = 0; i < nBatches; ++i)
3753 {
3754 OverallBatchMean[j] += BatchedAverages[i][j];
3755 }
3756 OverallBatchMean[j] /= nBatches;
3757 }
3758
3759 #ifdef MULTITHREAD
3760 #pragma omp for nowait
3761 #endif
3762 //KS: next loop is completely independent thus nowait clause
3763 for (int j = 0; j < nDraw; ++j)
3764 {
3765 for (int i = 0; i < nBatches; ++i)
3766 {
3767 (*BatchedVariance)(j) += (OverallBatchMean[j] - BatchedAverages[i][j])*(OverallBatchMean[j] - BatchedAverages[i][j]);
3768 }
3769 (*BatchedVariance)(j) = (BatchLength/(nBatches-1))* (*BatchedVariance)(j);
3770 }
3771
3772 //KS: Now we focus on C test statistic, again use nowait as next is calculation is independent
3773 #ifdef MULTITHREAD
3774 #pragma omp for nowait
3775 #endif
3776 for (int j = 0; j < nDraw; ++j)
3777 {
3778 C_Nominator[j] = (OverallBatchMean[j] - BatchedAverages[0][j])*(OverallBatchMean[j] - BatchedAverages[0][j]) +
3779 (OverallBatchMean[j] - BatchedAverages[nBatches-1][j])*(OverallBatchMean[j] - BatchedAverages[nBatches-1][j]);
3780 for (int i = 0; i < nBatches; ++i)
3781 {
3782 C_Denominator[j] += (OverallBatchMean[j] - BatchedAverages[i][j])*(OverallBatchMean[j] - BatchedAverages[i][j]);
3783 }
3784 C_Denominator[j] = 2*C_Denominator[j];
3785 }
3786
3787 //KS: We still calculate C and for this we need rho wee need autocorrelations between batches
3788 #ifdef MULTITHREAD
3789 #pragma omp for
3790 #endif
3791 for (int j = 0; j < nDraw; ++j)
3792 {
3793 for (int i = 0; i < nBatches-1; ++i)
3794 {
3795 C_Rho_Nominator[j] += (OverallBatchMean[j] - BatchedAverages[i][j])*(OverallBatchMean[j] - BatchedAverages[i+1][j]);
3796 }
3797
3798 for (int i = 0; i < nBatches; ++i)
3799 {
3800 C_Rho_Denominator[j] += (OverallBatchMean[j] - BatchedAverages[i][j])*(OverallBatchMean[j] - BatchedAverages[i][j]);
3801 }
3802 }
3803
3804 //KS: Final calculations of C
3805 #ifdef MULTITHREAD
3806 #pragma omp for
3807 #endif
3808 for (int j = 0; j < nDraw; ++j)
3809 {
3810 (*C_Test_Statistics)(j) = std::sqrt((nBatches*nBatches - 1)/(nBatches-2)) * ( C_Rho_Nominator[j]/C_Rho_Denominator[j] + C_Nominator[j]/ C_Denominator[j]);
3811 }
3812#ifdef MULTITHREAD
3813} //End parallel region
3814#endif
3815
3816 //Save to file
3817 OutputFile->cd();
3818 BatchedVariance->Write("BatchedMeansVariance");
3819 C_Test_Statistics->Write("C_Test_Statistics");
3820
3821 //Delete all variables
3822 delete BatchedVariance;
3823 delete C_Test_Statistics;
3824}

3961 constexpr double LowerThreshold = 0;
3962 constexpr double UpperThreshold = 1.0;
3963 // Tells how many intervals between thresholds we want to check
3964 constexpr int NChecks = 100;
3965 constexpr double Division = (UpperThreshold - LowerThreshold)/NChecks;
3966

3975 GewekePlots[j]->GetXaxis()->SetTitle("Burn-In (%)");

3977 }
3978
3979//KS: Start parallel region
3980#ifdef MULTITHREAD
3981#pragma omp parallel
3982{
3983#endif
3984 //KS: First we calculate mean and spectral variance for the upper limit, this doesn't change and in literature is most often 50%
3985 #ifdef MULTITHREAD
3986 #pragma omp for
3987 #endif
3988 for (int j = 0; j < nDraw; ++j)
3989 {
3990 for(int i = 0; i < nEntries; ++i)
3991 {
3992 if(StepNumber[i] > Threshold)
3993 {
3994 MeanUp[j] += ParStep[j][i];
3995 DenomCounterUp[j]++;
3996 }
3997 }
3998 MeanUp[j] = MeanUp[j]/DenomCounterUp[j];
3999 }
4000
4001 //KS: now Spectral variance which in this case is sample variance
4002 #ifdef MULTITHREAD
4003 #pragma omp for collapse(2)
4004 #endif
4005 for (int j = 0; j < nDraw; ++j)
4006 {
4007 for(int i = 0; i < nEntries; ++i)
4008 {
4009 if(StepNumber[i] > Threshold)

4012 }
4013 }
4014 }
4015

4017 #ifdef MULTITHREAD
4018 #pragma omp for
4019 #endif
4020 for (int k = 1; k < NChecks+1; ++k)

4025 std::vector<int> DenomCounterDown(nDraw, 0);
4026

4028 //KS: First mean
4029 for (int j = 0; j < nDraw; ++j)
4030 {
4031 for(int i = 0; i < nEntries; ++i)

4034 {

4036 DenomCounterDown[j]++;

4043 {
4044 for(int i = 0; i < nEntries; ++i)
4045 {
4046 if(StepNumber[i] < ThresholsCheck)
4047 {
4048 SpectralVarianceDown[j] += (ParStep[j][i] - MeanDown[j])*(ParStep[j][i] - MeanDown[j]);
4049 }
4050 }
4051 }
4052 //Lastly calc T score and fill histogram entry
4053 for (int j = 0; j < nDraw; ++j)

4056 GewekePlots[j]->SetBinContent(k, T_score);
4057 }
4058 } //end loop over intervals
4059#ifdef MULTITHREAD
4060} //End parallel region
4061#endif
4062
4063 //Finally save it to TFile
4064 OutputFile->cd();

4066 for (int j = 0; j < nDraw; ++j)
4067 {
4068 GewekeDir->cd();

4070 }
4071 for (int i = 0; i < nDraw; ++i) {
4072 delete[] ParStep[i];
4073 }
4074 delete[] ParStep;

4077 delete GewekeDir;
4078 OutputFile->cd();
4079}
