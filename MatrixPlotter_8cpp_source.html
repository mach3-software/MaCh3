9TH2D* GetSubMatrix(TH2D *MatrixFull, const std::string& Title, const std::vector<std::string>& Params)
10{
11 std::vector<int> ParamIndex(Params.size(), -999);
12
13 for(size_t i = 0; i < Params.size(); i++)
14 {
15 for(int j = 0; j < MatrixFull->GetNbinsX(); j++)
16 {
17 if(MatrixFull->GetXaxis()->GetBinLabel(j+1) == Params[i])
18 {
19 ParamIndex[i] = j;
20 break;
21 }
22 }
23 }
24
25 for(size_t i = 0; i < Params.size(); i++)
26 {
27 if(ParamIndex[i] == -999 )
28 MACH3LOG_ERROR("Didn't find param {} in matrix within {} sub-block", Params[i], Title);
29 }
30
31 auto new_end = std::remove(ParamIndex.begin(), ParamIndex.end(), -999);
32 ParamIndex.erase(new_end, ParamIndex.end());
33
34 TH2D* Hist = new TH2D(Title.c_str(), Title.c_str(), ParamIndex.size(), 0, ParamIndex.size(), ParamIndex.size(), 0, ParamIndex.size());
35 Hist->GetZaxis()->SetTitle("Correlation");
36 Hist->SetMinimum(-1);
37 Hist->SetMaximum(1);
38 Hist->GetXaxis()->SetLabelSize(0.015);
39 Hist->GetYaxis()->SetLabelSize(0.015);
40
41 for(size_t x = 0; x < ParamIndex.size(); x++)
42 {
43 for(size_t y = 0; y < ParamIndex.size(); y++)
44 {
45 Hist->SetBinContent(x+1, y+1, MatrixFull->GetBinContent(ParamIndex[x]+1, ParamIndex[y]+1));
46 }
47 Hist->GetXaxis()->SetBinLabel(x+1, MatrixFull->GetXaxis()->GetBinLabel(ParamIndex[x]+1));
48 Hist->GetYaxis()->SetBinLabel(x+1, MatrixFull->GetXaxis()->GetBinLabel(ParamIndex[x]+1));
49 }
50 return Hist;
51}

53void SetupInfo(const std::string& Config, std::vector<std::string>& Title, std::vector<std::vector<std::string>>& Params)
54{
55 // Load the YAML file
56 YAML::Node config = M3OpenConfig(Config);
57
58 // Access the "MatrixPlotter" section
59 YAML::Node settings = config["MatrixPlotter"];
60
61 // Retrieve the list of Titles
62 Title = settings["Titles"].as<std::vector<std::string>>();
63 Params.resize(Title.size());
64
65 // Retrieve parameters for each title (using the titles as keys)
66 for(size_t it = 0; it < Title.size(); it++)
67 {
68 if (settings[Title[it]]) {
69 Params[it] = settings[Title[it]].as<std::vector<std::string>>();
70 } else {
71 MACH3LOG_ERROR("Missing key in YAML for: {}", Title[it]);
72 throw MaCh3Exception(__FILE__ , __LINE__ );
73 }
74 }
75
76 // Check if sizes match
77 if(Title.size() != Params.size())
78 {
79 MACH3LOG_ERROR("Size doesn't match");
80 throw MaCh3Exception(__FILE__ , __LINE__ );
81 }
82}
