233 inline int PDGToNuOscillatorFlavour(int NuPdg){
234 int NuOscillatorFlavour = M3::_BAD_INT_;
235 switch(std::abs(NuPdg)){
236 case NuPDG::kNue:
237 NuOscillatorFlavour = NuOscillator::kElectron;
238 break;
239 case NuPDG::kNumu:
240 NuOscillatorFlavour = NuOscillator::kMuon;
241 break;
242 case NuPDG::kNutau:
243 NuOscillatorFlavour = NuOscillator::kTau;
244 break;
245 default:
246 MACH3LOG_ERROR("Unknown Neutrino PDG {}, cannot convert to NuOscillator type", NuPdg);
247 break;
248 }
249
250 //This is very cheeky but if the PDG is negative then multiply the PDG by -1
251 // This is consistent with the treatment that NuOscillator expects as enums only
252 // exist for the generic matter flavour and not the anti-matter version
253 if(NuPdg < 0){NuOscillatorFlavour *= -1;}
254
255 return NuOscillatorFlavour;
256 }
