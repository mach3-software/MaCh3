321  void CheckBinsHaveNoGaps(const std::vector<BinInfo>& TestedBins,
322  const std::vector<double>& MinVal,
323  const std::vector<double>& MaxVal,
324  size_t ValidationBinsPerDim = 100) const {
325  bool gap_found = false;
326  if (TestedBins.empty()) return;
327  const size_t Dim = TestedBins[0].Extent.size();
328  if (MinVal.size() != Dim || MaxVal.size() != Dim) {
329  MACH3LOG_ERROR("MinVal/MaxVal size does not match dimension of bins");
330  throw MaCh3Exception(__FILE__, __LINE__);
331  }
332 
333  // Build a fine validation grid from the provided min/max
334  std::vector<std::vector<double>> TestGridEdges(Dim);
335  for (size_t d = 0; d < Dim; ++d) {
336  TestGridEdges[d].resize(ValidationBinsPerDim + 1);
337  const double width = (MaxVal[d] - MinVal[d]) / static_cast<double>(ValidationBinsPerDim);
338  for (size_t i = 0; i <= ValidationBinsPerDim; ++i)
339  TestGridEdges[d][i] = MinVal[d] + static_cast<double>(i) * width;
340  }
341  // Precompute midpoints of each test cell
342  std::vector<size_t> indices(Dim, 0);
343 
344  std::function<void(size_t)> scan = [&](size_t d) {
345  if (gap_found) return; // stop recursion
346 
347  // Base case: we have selected a cell in every dimension
348  if (d == Dim) {
349  std::vector<double> point(Dim);
350 
351  // Compute the midpoint of the current N-D grid cell
352  for (size_t i = 0; i < Dim; ++i) {
353  const double lo = TestGridEdges[i][indices[i]];
354  const double hi = TestGridEdges[i][indices[i] + 1];
355  point[i] = 0.5 * (lo + hi);
356  }
357 
358  // Check coverage
359  for (const auto& bin : TestedBins) {
360  if (bin.IsEventInside(point)) {
361  return; // covered
362  }
363  }
364 
365  // Not covered by any bin gap
366  MACH3LOG_WARN("Gap detected in non-uniform binning at point [{:.2f}]", fmt::join(point, ", "));
367  gap_found = true;
368  return;
369  }
370 
371  for (size_t i = 0; i + 1 < TestGridEdges[d].size(); ++i) {
372  indices[d] = i;
373  scan(d + 1);
374  }
375  };
376  // Start recursion at dimension 0
377  scan(0);
378  }

381  void InitialiseGridMapping() {
382  const size_t Dim = BinEdges.size();
383 
384  // Compute total number of "mega bins"
385  int NGridBins = 1;
386  for (size_t d = 0; d < Dim; ++d) {
387  NGridBins *= AxisNBins[d];
388  }
389  BinGridMapping.resize(NGridBins);
390 
391  // Helper: convert linear index to multi-dimensional index
392  std::vector<int> MultiIndex(Dim, 0);
393 
394  std::function<void(size_t, int)> scan;
395  scan = [&](size_t d, int LinearIndex) {
396  if (d == Dim) {
397  // Compute the edges of the current mega-bin
398  std::vector<std::array<double, 2>> CellEdges(Dim);
399  for (size_t i = 0; i < Dim; ++i) {
400  CellEdges[i][0] = BinEdges[i][MultiIndex[i]];
401  CellEdges[i][1] = BinEdges[i][MultiIndex[i] + 1];
402  }
403 
404  // Check overlap with all non-uniform bins
405  for (size_t iBin = 0; iBin < Bins.size(); ++iBin) {
406  auto& bin = Bins[iBin];
407  bool overlap = true;
408  for (size_t i = 0; i < Dim; ++i) {
409  const double a_lo = bin.Extent[i][0];
410  const double a_hi = bin.Extent[i][1];
411  const double b_lo = CellEdges[i][0];
412  const double b_hi = CellEdges[i][1];
413  if (!(a_hi > b_lo && a_lo < b_hi)) { // overlap condition
414  overlap = false;
415  break;
416  }
417  }
418  if (overlap) {
419  BinGridMapping[LinearIndex].push_back(static_cast<int>(iBin));
420 
421  // Debug statement: show both small bin extent AND mega-bin edges
422  std::vector<std::string> bin_extent_str(Dim);
423  std::vector<std::string> mega_edges_str(Dim);
424 
425  for (size_t i = 0; i < Dim; ++i) {
426  bin_extent_str[i] = fmt::format("[{:.3f}, {:.3f}]", bin.Extent[i][0], bin.Extent[i][1]);
427  mega_edges_str[i] = fmt::format("[{:.3f}, {:.3f}]", CellEdges[i][0], CellEdges[i][1]);
428  }
429 
430  MACH3LOG_DEBUG("MegaBin {} (multi-index [{}], edges {}) assigned Bin {} with extents {}",
431  LinearIndex, fmt::join(MultiIndex, ","), fmt::join(mega_edges_str, ", "),
432  iBin, fmt::join(bin_extent_str, ", "));
433  }
434  }
435  return;
436  }
437 
438  // Loop over all bins along this dimension
439  for (int i = 0; i < AxisNBins[d]; ++i) {
440  MultiIndex[d] = i;
441  int NewLinearIndex = LinearIndex;
442  if (d > 0) {
443  int stride = 1;
444  for (size_t s = 0; s < d; ++s) stride *= AxisNBins[s];
445  NewLinearIndex += i * stride;
446  } else {
447  NewLinearIndex = i;
448  }
449  scan(d + 1, NewLinearIndex);
450  }
451  };
452  // Start the recursive scan over all dimensions, beginning at dimension 0 with linear index 0
453  scan(0, 0);
454  }

506  BinEdges[iDim].resize(BinsPerDimension + 1);

508  for (size_t iEdge = 0; iEdge <= BinsPerDimension; iEdge++) {

510  }

515  GlobalOffset = 0;
517  InitialiseBinMigrationLookUp(static_cast<int>(BinEdges.size()));

520  }

564  int FindBin(const double KinVar,
565  const int NomBin,
566  const int N_Bins,
567  const std::vector<double>& Bin_Edges,
568  const std::vector<BinShiftLookup>& Bin_Lookup) const {
569  //DB Check to see if momentum shift has moved bins
570  //DB - First , check to see if the event is outside of the binning range and skip event if it is
571  if (KinVar < Bin_Edges[0] || KinVar >= Bin_Edges[N_Bins]) {
572  return M3::UnderOverFlowBin;
573  }
574  // KS: If NomBin is UnderOverFlowBin we must do binary search :(
575  if(NomBin > M3::UnderOverFlowBin) {
576  // KS: Get reference to avoid repeated indexing and help with performance
577  const BinShiftLookup& _restrict_ Bin = Bin_Lookup[NomBin];
578  const double lower = Bin.lower_binedge;
579  const double upper = Bin.upper_binedge;
580  const double lower_lower = Bin.lower_lower_binedge;
581  const double upper_upper = Bin.upper_upper_binedge;
582 
583  //DB - Second, check to see if the event is still in the nominal bin
584  if (KinVar < upper && KinVar >= lower) {
585  return NomBin;
586  }
587  //DB - Thirdly, check the adjacent bins first as Eb+CC+EScale shifts aren't likely to move an Erec more than 1bin width
588  //Shifted down one bin from the event bin at nominal
589  if (KinVar < lower && KinVar >= lower_lower) {
590  return NomBin-1;
591  }
592  //Shifted up one bin from the event bin at nominal
593  if (KinVar < upper_upper && KinVar >= upper) {
594  return NomBin+1;
595  }
596  }
597  //DB - If we end up in this loop, the event has been shifted outside of its nominal bin, but is still within the allowed binning range
598  // KS: Perform binary search to find correct bin. We already checked if isn't outside of bounds
599  return static_cast<int>(std::distance(Bin_Edges.begin(), std::upper_bound(Bin_Edges.begin(), Bin_Edges.end(), KinVar)) - 1);
600  }
