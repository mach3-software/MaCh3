168struct SampleBinningInfo {
169// ***************************
171 std::vector<double> XBinEdges;
173 std::vector<double> YBinEdges;
174
176 size_t nXBins = M3::_BAD_INT_;
178 size_t nYBins = M3::_BAD_INT_;
180 size_t nBins = M3::_BAD_INT_;
182 size_t GlobalOffset = M3::_BAD_INT_;
183
185 std::vector<double> rw_lower_xbinedge;
187 std::vector<double> rw_lower_lower_xbinedge;
189 std::vector<double> rw_upper_xbinedge;
191 std::vector<double> rw_upper_upper_xbinedge;

197 int GetBin(const int xBin, const int yBin) const {
198 return static_cast<int>(yBin * nXBins + xBin);
199 }
200
206 int GetBinSafe(const int xBin, const int yBin) const {
207 if (xBin < 0 || yBin < 0 || static_cast<size_t>(xBin) >= nXBins || static_cast<size_t>(yBin) >= nYBins) {
208 MACH3LOG_ERROR("GetBinSafe: Bin indices out of range: xBin={}, yBin={}, max xBin={}, max yBin={}",
209 xBin, yBin, nXBins - 1, nYBins - 1);
210 throw MaCh3Exception(__FILE__, __LINE__);
211 }
212 return GetBin(xBin, yBin);
213 }

218 int GetBinGlobal(const int xBin, const int yBin) const {
219 return static_cast<int>(GlobalOffset + GetBin(xBin, yBin));
220 }

223 int FindXBin(const double XVar, const int NomXBin) const {
224 //DB Check to see if momentum shift has moved bins
225 //DB - First , check to see if the event is outside of the binning range and skip event if it is
226 if (XVar < XBinEdges[0] || XVar >= XBinEdges[nXBins]) {
227 return -1;
228 }
229 //DB - Second, check to see if the event is still in the nominal bin
230 else if (XVar < rw_upper_xbinedge[NomXBin] && XVar >= rw_lower_xbinedge[NomXBin]) {
231 return NomXBin;
232 }
233 //DB - Thirdly, check the adjacent bins first as Eb+CC+EScale shifts aren't likely to move an Erec more than 1bin width
234 //Shifted down one bin from the event bin at nominal
235 else if (XVar < rw_lower_xbinedge[NomXBin] && XVar >= rw_lower_lower_xbinedge[NomXBin]) {
236 return NomXBin-1;
237 }
238 //Shifted up one bin from the event bin at nominal
239 else if (XVar < rw_upper_upper_xbinedge[NomXBin] && XVar >= rw_upper_xbinedge[NomXBin]) {
240 return NomXBin+1;
241 }
242 //DB - If we end up in this loop, the event has been shifted outside of its nominal bin, but is still within the allowed binning range
243 else {
244 // KS: Perform binary search to find correct bin. We already checked if isn't outside of bounds
245 return static_cast<int>(std::distance(XBinEdges.begin(), std::upper_bound(XBinEdges.begin(), XBinEdges.end(), XVar)) - 1);
246 }
247 }
248};

254inline int GetSampleFromGlobalBin(const std::vector<SampleBinningInfo>& BinningInfo, const size_t GlobalBin) {
255 for (size_t iSample = 0; iSample < BinningInfo.size(); ++iSample) {
256 const SampleBinningInfo& info = BinningInfo[iSample];
257 if (GlobalBin >= info.GlobalOffset && GlobalBin < info.GlobalOffset + info.nBins) {
258 return static_cast<int>(iSample);
259 }
260 }
261
262 MACH3LOG_ERROR("Couldn't find sample corresponding to bin {}", GlobalBin);
263 throw MaCh3Exception(__FILE__, __LINE__);
264
265 // GlobalBin is out of range for all samples
266 return M3::_BAD_INT_;
267}

272 if (BinningInfo.empty()) {
273 MACH3LOG_ERROR("No binning samples provided.");

350 inline int PDGToNuOscillatorFlavour(const int NuPdg){
351 int NuOscillatorFlavour = M3::_BAD_INT_;
352 switch(std::abs(NuPdg)){
353 case NuPDG::kNue:
354 NuOscillatorFlavour = NuOscillator::kElectron;
355 break;
356 case NuPDG::kNumu:
357 NuOscillatorFlavour = NuOscillator::kMuon;
358 break;
359 case NuPDG::kNutau:
360 NuOscillatorFlavour = NuOscillator::kTau;
361 break;
362 default:
363 MACH3LOG_ERROR("Unknown Neutrino PDG {}, cannot convert to NuOscillator type", NuPdg);
364 break;
365 }
366
367 //This is very cheeky but if the PDG is negative then multiply the PDG by -1
368 // This is consistent with the treatment that NuOscillator expects as enums only
369 // exist for the generic matter flavour and not the anti-matter version
370 if(NuPdg < 0){NuOscillatorFlavour *= -1;}
371
372 return NuOscillatorFlavour;
373 }
