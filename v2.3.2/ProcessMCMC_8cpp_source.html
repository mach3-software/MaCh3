304

695 // KS: Perform KS test to check if two posteriors for the same parameter came from the same distribution
696 void KolmogorovSmirnovTest(const std::vector<std::unique_ptr<MCMCProcessor>>& Processor,
697  const std::unique_ptr<TCanvas>& Posterior,
698  const TString& canvasname)
699 {
700  constexpr Color_t CumulativeColor[] = {kBlue-1, kRed, kGreen+2};
701  constexpr Style_t CumulativeStyle[] = {kSolid, kDashed, kDotted};
702 
703  for(int i = 0; i < Processor[0]->GetNParams(); ++i)
704  {
705  // This holds the posterior density
706  std::vector<std::unique_ptr<TH1D>> hpost(nFiles);
707  std::vector<std::unique_ptr<TH1D>> CumulativeDistribution(nFiles);
708 
709  TString Title;
710  double Prior = 1.0;
711  double PriorError = 1.0;
712 
713  Processor[0]->GetNthParameter(i, Prior, PriorError, Title);
714  bool Skip = false;
715  for (int ik = 0 ; ik < nFiles; ik++)
716  {
717  int Index = 0;
718  if(ik == 0 ) Index = i;
719  else
720  {
721  // KS: If somehow this chain doesn't given params we skip it
722  Index = Processor[ik]->GetParamIndexFromName(hpost[0]->GetTitle());
723  if(Index == M3::_BAD_INT_)
724  {
725  Skip = true;
726  break;
727  }
728  }
729  hpost[ik] = M3::Clone(Processor[ik]->GetHpost(Index));
730  CumulativeDistribution[ik] = M3::Clone(Processor[ik]->GetHpost(Index));
731  CumulativeDistribution[ik]->Fill(0., 0.);
732  CumulativeDistribution[ik]->Reset();
733  CumulativeDistribution[ik]->SetMaximum(1.);
734  TString TempTitle = Title+" Kolmogorov Smirnov";
735  CumulativeDistribution[ik]->SetTitle(TempTitle);
736 
737  TempTitle = Title+" Value";
738  CumulativeDistribution[ik]->GetXaxis()->SetTitle(TempTitle);
739  CumulativeDistribution[ik]->GetYaxis()->SetTitle("Cumulative Probability");
740 
741  CumulativeDistribution[ik]->SetLineWidth(2);
742  CumulativeDistribution[ik]->SetLineColor(CumulativeColor[ik]);
743  CumulativeDistribution[ik]->SetLineStyle(CumulativeStyle[ik]);
744  }
745 
746  // Don't plot if this is a fixed histogram (i.e. the peak is the whole integral)
747  if(hpost[0]->GetMaximum() == hpost[0]->Integral()*1.5 || Skip) {
748  continue;
749  }
750 
751  for (int ik = 0 ; ik < nFiles; ik++)
752  {
753  const int NumberOfBins = hpost[ik]->GetXaxis()->GetNbins();
754  double Cumulative = 0;
755  const double Integral = hpost[ik]->Integral();
756  for (int j = 1; j < NumberOfBins+1; ++j)
757  {
758  Cumulative += hpost[ik]->GetBinContent(j)/Integral;
759  CumulativeDistribution[ik]->SetBinContent(j, Cumulative);
760  }
761  //KS: Set overflow to 1 just in case
762  CumulativeDistribution[ik]->SetBinContent(NumberOfBins+1, 1.);
763  }
764 
765  std::vector<int> TestStatBin(nFiles, 0);
766  std::vector<double> TestStatD(nFiles, -999);
767  std::vector<std::unique_ptr<TLine>> LineD(nFiles);
768  //Find KS statistic
769  for (int ik = 1 ; ik < nFiles; ik++)
770  {
771  const int NumberOfBins = CumulativeDistribution[0]->GetXaxis()->GetNbins();
772  for (int j = 1; j < NumberOfBins+1; ++j)
773  {
774  const double BinValue = CumulativeDistribution[0]->GetBinCenter(j);
775  const int BinNumber = CumulativeDistribution[ik]->FindBin(BinValue);
776  //KS: Calculate D statistic for this bin, only save it if it's bigger than previously found value
777  double TempDstat = std::fabs(CumulativeDistribution[0]->GetBinContent(j) - CumulativeDistribution[ik]->GetBinContent(BinNumber));
778  if(TempDstat > TestStatD[ik])
779  {
780  TestStatD[ik] = TempDstat;
781  TestStatBin[ik] = j;
782  }
783  }
784  }
785 
786  for (int ik = 0 ; ik < nFiles; ik++)
787  {
788  LineD[ik] = std::make_unique<TLine>(CumulativeDistribution[0]->GetBinCenter(TestStatBin[ik]), 0, CumulativeDistribution[0]->GetBinCenter(TestStatBin[ik]), CumulativeDistribution[0]->GetBinContent(TestStatBin[ik]));
789  LineD[ik]->SetLineColor(CumulativeColor[ik]);
790  LineD[ik]->SetLineWidth(2.0);
791  }
792  CumulativeDistribution[0]->Draw();
793  for (int ik = 0 ; ik < nFiles; ik++)
794  CumulativeDistribution[ik]->Draw("SAME");
795 
796  auto leg = std::make_unique<TLegend>(0.15, 0.7, 0.5, 0.90);
797  leg->SetTextSize(0.04f);
798  for (int ik = 0; ik < nFiles; ik++)
799  leg->AddEntry(CumulativeDistribution[ik].get(), TitleNames[ik].c_str(), "l");
800  for (int ik = 1; ik < nFiles; ik++)
801  leg->AddEntry(LineD[ik].get(), Form("#Delta D = %.4f", TestStatD[ik]), "l");
802 
803  leg->SetLineColor(0);
804  leg->SetLineStyle(0);
805  leg->SetFillColor(0);
806  leg->SetFillStyle(0);
807  leg->Draw("SAME");
808 
809  for (int ik = 1; ik < nFiles; ik++)
810  LineD[ik]->Draw("sam");
811 
812  Posterior->cd();
813  Posterior->Print(canvasname);
814  } //End loop over parameter
815 }
