153 // ********************************************
154  /*
155  * In order to adapt efficient we want to setup our throw matrix to be a serious of block-diagonal (ish) matrices
156  *
157  * To do this we set sub-block in the config by parameter index. For example having
158  * [[0,4],[4, 6]] in your config will set up two blocks one with all indices 0<=i<4 and the other with 4<=i<6
159  */
160  // Set up block regions
161  adapt_block_matrix_indices = std::vector<int>(GetNumParams(), 0);
162 
163  // Should also make a matrix of block sizes
164  adapt_block_sizes = std::vector<int>(block_indices.size()+1, 0);
165  adapt_block_sizes[0] = GetNumParams();
166 
167  if(block_indices.size()==0 || block_indices[0].size()==0) return;
168 
169  int block_size = static_cast<int>(block_indices.size());
170  // Now we loop over our blocks
171  for(int iblock=0; iblock < block_size; iblock++){
172  // Loop over blocks in the block
173  int sub_block_size = static_cast<int>(block_indices.size()-1);
174  for(int isubblock=0; isubblock < sub_block_size ; isubblock+=2){
175  int block_lb = block_indices[iblock][isubblock];
176  int block_ub = block_indices[iblock][isubblock+1];
177 
178  if(block_lb > GetNumParams() || block_ub > GetNumParams()){
179  MACH3LOG_ERROR("Cannot set matrix block with edges {}, {} for matrix of size {}",
180  block_lb, block_ub, GetNumParams());

182  }
183  for(int ipar = block_lb; ipar < block_ub; ipar++){
184  adapt_block_matrix_indices[ipar] = iblock+1;

333 void AdaptiveMCMCHandler::UpdateAdaptiveCovariance() {
334 // ********************************************
335  std::vector<double> par_means_prev = par_means;
336  int steps_post_burn = total_steps - start_adaptive_update;
337 
338  prev_step_accepted = false;
339 
340  // Step 1: Update means and compute deviations
341  for (int i = 0; i < GetNumParams(); ++i) {
342  if (IsFixed(i)) continue;
343 
344  par_means[i] = (CurrVal(i) + par_means_prev[i]*steps_post_burn)/(steps_post_burn+1);
345  // Left over from cyclic means
346  }
347 
348  // Step 2: Update covariance
349  #ifdef MULTITHREAD
350  #pragma omp parallel for
351  #endif
352  for (int i = 0; i < GetNumParams(); ++i) {
353  if (IsFixed(i)) {
354  (*adaptive_covariance)(i, i) = 1.0;
355  continue;
356  }
357 
358  int block_i = adapt_block_matrix_indices[i];
359 
360  for (int j = 0; j <= i; ++j) {
361  if (IsFixed(j) || adapt_block_matrix_indices[j] != block_i) {
362  (*adaptive_covariance)(i, j) = 0.0;
363  (*adaptive_covariance)(j, i) = 0.0;
364  continue;
365  }
366 
367  double cov_prev = (*adaptive_covariance)(i, j);
368  double cov_updated = 0.0;
369 
370  // HH: Check if either parameter is skipped
371  // If parameter i and j are both skipped
372  if ((*_param_skip_adapt_flags)[i] && (*_param_skip_adapt_flags)[j]) {
373  // Set covariance to initial throw matrix value
374  (*adaptive_covariance)(i, j) = (*initial_throw_matrix)(i, j);
375  (*adaptive_covariance)(j, i) = (*initial_throw_matrix)(j, i);
376  continue;
377  }
378  // If only one parameter is skipped we need to make sure correlation is zero
379  // because we don't want to change one parameter while keeping the other fixed
380  // as this would lead to penalty terms in the prior
381  else if ((*_param_skip_adapt_flags)[i] || (*_param_skip_adapt_flags)[j]) {
382  continue;
383  } // otherwise adapt as normal
384 
385  if (steps_post_burn > 0) {
386  // Haario-style update
387  double cov_t = cov_prev * (steps_post_burn - 1) / steps_post_burn;
388  double prev_means_t = steps_post_burn * par_means_prev[i] * par_means_prev[j];
389  double curr_means_t = (steps_post_burn + 1) * par_means[i] * par_means[j];
390  double curr_step_t = CurrVal(i) * CurrVal(j);
391 
392  cov_updated = cov_t + (prev_means_t - curr_means_t + curr_step_t) / steps_post_burn;
393  }
394 
395  (*adaptive_covariance)(i, j) = cov_updated;
396  (*adaptive_covariance)(j, i) = cov_updated;
397  }
398  }

451 void AdaptiveMCMCHandler::CheckMatrixValidityForAdaption(const TMatrixDSym *covMatrix) const {
452 // ********************************************
453  int n = covMatrix->GetNrows();
454  std::vector<std::vector<double>> corrMatrix(n, std::vector<double>(n));
455 
456  // Extract standard deviations from the diagonal
457  std::vector<double> stdDevs(n);
458  for (int i = 0; i < n; ++i) {
459  stdDevs[i] = std::sqrt((*covMatrix)(i, i));
460  }
461 
462  // Compute correlation for each element
463  for (int i = 0; i < n; ++i) {
464  for (int j = 0; j < n; ++j) {
465  double cov = (*covMatrix)(i, j);
466  double corr = cov / (stdDevs[i] * stdDevs[j]);
467  corrMatrix[i][j] = corr;
468  }
469  }
470 
471  // KS: These numbers are completely arbitrary
472  constexpr double NumberOfParametersThreshold = 5;
473  constexpr double CorrelationThreshold = 0.5;
474 
475  bool FlagMessage = false;
476  // For each parameter, count how many others it is correlated with
477  for (int i = 0; i < n; ++i) {
478  if (IsFixed(i)) {
479  continue;
480  }
481  int block_i = adapt_block_matrix_indices[i];
482  int correlatedCount = 0;
483  std::vector<int> correlatedWith;
484 
485  for (int j = 0; j < n; ++j) {
486  if (i == j || IsFixed(j) || adapt_block_matrix_indices[j] != block_i) {
487  continue;
488  }
489  if (corrMatrix[i][j] > CorrelationThreshold) {
490  correlatedCount++;
491  correlatedWith.push_back(j);
492  }
493  }
494 
495  if (correlatedCount > NumberOfParametersThreshold) {
496  MACH3LOG_WARN("Parameter {} ({}) is highly correlated with {} other parameters (above threshold {}).", _ParamNames->at(i), i, correlatedCount, CorrelationThreshold);
497  MACH3LOG_WARN("Correlated with parameters: {}.", fmt::join(correlatedWith, ", "));
498  FlagMessage = true;
499  }
500  }
501  if(FlagMessage){
502  MACH3LOG_WARN("Found highly correlated block, adaptive may not work as intended, proceed with caution");
503  MACH3LOG_WARN("You can consider defying so called Adaption Block to not update this part of matrix");
504  }
505 }

517 void AdaptiveMCMCHandler::CalculateRobbinsMonroStepLength() {
518 // ********************************************
519  /*
520  Obtains the constant "step scale" for Robbins-Monro adaption
521  for simplicity and because it has the degrees of freedom "baked in"
522  it will be same across ALL blocks.
523  */
524 
526  double alpha = -ROOT::Math::normal_quantile(target_acceptance/2, 1.0);
527 
528  int non_fixed_pars = GetNumParams()-GetNFixed();
529 
530  if(non_fixed_pars == 0){
531  MACH3LOG_ERROR("Number of non_fixed_pars is 0, have you fixed all parameters");
532  MACH3LOG_ERROR("This will cause division by 0 and crash ()");
533  throw MaCh3Exception(__FILE__, __LINE__);
534  }
536  c_robbins_monro = (1- 1/non_fixed_pars)*std::sqrt(TMath::Pi()*2 * std::exp(alpha*alpha));
537  c_robbins_monro += 1/(non_fixed_pars*target_acceptance*(1-target_acceptance));
538 
539  MACH3LOG_INFO("Robbins-Monro scale factor set to {}", c_robbins_monro);
540 }

542 // ********************************************
543 void AdaptiveMCMCHandler::UpdateRobbinsMonroScale(){
544 // ********************************************
545  /*
546  Update the scale factor using Robbins-Monro.
547  TLDR: If acceptance rate is too high, scale factor goes up, if too low goes down
548  will pretty rapidly converge to the right value.
549  */
550  if ((total_steps>0) && (total_steps % acceptance_rate_batch_size == 0)) {
551  n_rm_restarts++;
552  }
553 
555  int batch_step;
556  if (n_rm_restarts > total_rm_restarts) {
557  batch_step = total_steps;
558  }
559  else{
560  batch_step = total_steps % acceptance_rate_batch_size;
561  }
562 
563  int non_fixed_pars = GetNumParams()-GetNFixed();
564 
565  // double root_scale = std::sqrt(adaption_scale);
566 
568  double scale_factor = adaption_scale * c_robbins_monro / std::max(200.0, static_cast<double>(batch_step) / non_fixed_pars);
569 
571  if(prev_step_accepted){
572  adaption_scale += (1 - target_acceptance) * scale_factor;
