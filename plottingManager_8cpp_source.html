196void PlottingManager::setOutFileName(std::string saveName) {
197 if (saveName.find(".") == std::string::npos)
198 {
199 _outputName = saveName + ".pdf";
200 return;
201 }
202
203 std::string ext = saveName;
204 // if there are .'s then remove everything before them until we only have the file extension left
205 while (ext.find(".") != std::string::npos)
206 ext.erase(0, ext.find(".") + 1);
207 if (ext == "pdf" || ext == "ps" || ext == "eps")
208 {
209 _outputName = saveName;
210 return;
211 }
212
213 MACH3LOG_WARN("file extension '{}' that you provided doesnt support multiple plots in one file", ext);
214 MACH3LOG_WARN("should be one of .pdf, .eps .ps, will use pdf");
215 _outputName = saveName + ".pdf";
216}

246void PlottingManager::parseFileLabels(std::string labelString, std::vector<std::string> &labelVec) {
247 // take in a string defining labels of the form "label1;label2;...;labelN" and parse it into a
248 // vector containing the labels
249
250 int end = labelString.find(";");
251 while (end != -1)
252 { // Loop until no delimiter is left in the string.
253 labelVec.push_back(labelString.substr(0, end));
254 labelString.erase(labelString.begin(), labelString.begin() + end + 1);
255 end = labelString.find(";");
256 }
257 labelVec.push_back(labelString.substr(0, end));
258}
