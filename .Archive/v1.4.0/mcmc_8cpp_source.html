125 void mcmc::ProposeStep() {
126 // *******************
127  // Initial likelihood
128  double llh = 0.0;
129 
130  // Initiate to false
131  reject = false;
132 
133  // Loop over the systematics and propose the initial step
134  for (size_t s = 0; s < systematics.size(); ++s) {
135  // Could throw the initial value here to do MCMC stability studies
136  // Propose the steps for the systematics
137  systematics[s]->proposeStep();
138 
139  // Get the likelihood from the systematics
140  syst_llh[s] = systematics[s]->GetLikelihood();
141  llh += syst_llh[s];
142 
143  #ifdef DEBUG
144  if (debug) debugFile << "LLH after " << systematics[s]->getName() << " " << llh << std::endl;
145  #endif
146  }
147 
148  // Check if we've hit a boundary in the systematics
149  // In this case we can save time by not having to reconfigure the simulation
150  if (llh >= M3::_LARGE_LOGL_) {
151  reject = true;
152  #ifdef DEBUG
153  if (debug) debugFile << "Rejecting based on boundary" << std::endl;
154  #endif
155  }
156 
157  // Only reweight when we have a good parameter configuration
158  // This speeds things up considerably because for every bad parameter configuration we don't have to reweight the MC
159  if (!reject)
160  {
161  // Could multi-thread this
162  // But since sample reweight is multi-threaded it's probably better to do that
163  for (size_t i = 0; i < samples.size(); ++i)
164  {
165  samples[i]->reweight();
166  }
167 
168  //DB for atmospheric event by event sample migration, need to fully reweight all samples to allow event passing prior to likelihood evaluation
169  for (size_t i = 0; i < samples.size(); ++i) {
170  // Get the sample likelihoods and add them
171  sample_llh[i] = samples[i]->GetLikelihood();
172  llh += sample_llh[i];
173  #ifdef DEBUG
174  if (debug) debugFile << "LLH after sample " << i << " " << llh << std::endl;
175  #endif
176  }
177 
178  // For when we don't have to reweight, set sample to madness
179  } else {
180  for (size_t i = 0; i < samples.size(); ++i) {
181  // Set the sample_llh[i] to be madly high also to signify a step out of bounds
182  sample_llh[i] = M3::_LARGE_LOGL_;
183  #ifdef DEBUG

185  #endif

187  }
