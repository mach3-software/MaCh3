30  * HW: Idea is that adaption can simply read the YAML config

32  * External Info:

34  * ExternalMatrixFileName [str] : Name of file containing external info

44  * Adaption blocks [vector<vector<int>>] : Splits the throw matrix into several block matrices
45  */

83 void AdaptiveMCMCHandler::SetAdaptiveBlocks(std::vector<std::vector<int>> block_indices, const int Npars) {
84 // ********************************************
85  /*
86  * In order to adapt efficient we want to setup our throw matrix to be a serious of block-diagonal (ish) matrices
87  *
88  * To do this we set sub-block in the config by parameter index. For example having
89  * [[0,4],[4, 6]] in your config will set up two blocks one with all indices 0<=i<4 and the other with 4<=i<6
90  */
91  // Set up block regions
92  adapt_block_matrix_indices = std::vector<int>(Npars, 0);
93 
94  // Should also make a matrix of block sizes
95  adapt_block_sizes = std::vector<int>(block_indices.size()+1, 0);
96  adapt_block_sizes[0] = Npars;
97 
98  if(block_indices.size()==0 || block_indices[0].size()==0) return;
99 
100  // Now we loop over our blocks
101  for(int iblock=0; iblock<int(block_indices.size()); iblock++){
102  // Loop over blocks in the block
103  for(int isubblock=0; isubblock<int(block_indices[iblock].size())-1; isubblock+=2){
104  int block_lb = block_indices[iblock][isubblock];
105  int block_ub = block_indices[iblock][isubblock+1];
106 
107  if(block_lb > Npars || block_ub > Npars){
108  MACH3LOG_ERROR("Cannot set matrix block with edges {}, {} for matrix of size {}",
109  block_lb, block_ub, Npars);
110  throw MaCh3Exception(__FILE__, __LINE__);;
111  }
112  for(int ipar = block_lb; ipar < block_ub; ipar++){
113  adapt_block_matrix_indices[ipar] = iblock+1;
114  adapt_block_sizes[iblock+1] += 1;
115  adapt_block_sizes[0] -= 1;
116  }
117  }
118  }
119 }

198 void AdaptiveMCMCHandler::UpdateAdaptiveCovariance(const std::vector<double>& _fCurrVal, const int Npars) {
199 // ********************************************
200  std::vector<double> par_means_prev = par_means;
201 
202  int steps_post_burn = total_steps - start_adaptive_update;
203 
204  #ifdef MULTITHREAD
205  #pragma omp parallel for
206  #endif
207  for(int iRow = 0; iRow < Npars; iRow++) {
208  par_means[iRow] = (_fCurrVal[iRow]+par_means[iRow]*steps_post_burn)/(steps_post_burn+1);
209  }
210 
211  //Now we update the covariances using cov(x,y)=E(xy)-E(x)E(y)
212  #ifdef MULTITHREAD
213  #pragma omp parallel for
214  #endif
215  for(int irow = 0; irow < Npars; irow++){
216  int block = adapt_block_matrix_indices[irow];
217  // int scale_factor = 5.76/double(adapt_block_sizes[block]);
218  for(int icol = 0; icol <= irow; icol++){
219  double cov_val=0;
220  // Not in the same blocks
221  if(adapt_block_matrix_indices[icol] == block){
222  // Calculate Covariance for block
223  // https://projecteuclid.org/journals/bernoulli/volume-7/issue-2/An-adaptive-Metropolis-algorithm/bj/1080222083.full
224  cov_val = (*adaptive_covariance)(irow, icol)*Npars/5.6644;
225  cov_val += par_means_prev[irow]*par_means_prev[icol]; //First we remove the current means
226  cov_val = (cov_val*steps_post_burn+_fCurrVal[irow]*_fCurrVal[icol])/(steps_post_burn+1); //Now get mean(iRow*iCol)
227  cov_val -= par_means[icol]*par_means[irow];
228  cov_val*=5.6644/Npars;
229  }
230  (*adaptive_covariance)(icol, irow) = cov_val;
231  (*adaptive_covariance)(irow, icol) = cov_val;
232  }
233  }
234 }
