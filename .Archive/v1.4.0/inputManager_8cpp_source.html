232 std::vector<std::string> InputManager::parseLocation(const std::string &rawLocationString, std::string &fitter,
233  fileTypeEnum fileType, const std::string &parameter,
234  const std::string &sample, const std::string &parameter2) const {
235 
236  std::string locationString(rawLocationString);
237  std::vector<std::string> tokens;
238 
239  std::size_t pos = 0;
240  std::string toReplace = "";
241 
242  // loop through the raw location string and replace any instance of {PARAMETER} with the name of
243  // the parameter
244  toReplace = "{PARAMETER}";
245  while ((pos = locationString.find(toReplace)) != std::string::npos)
246  {
247  locationString.replace(pos, toReplace.length(),
248  getFitterSpecificParamName(fitter, fileType, parameter));
249  }
250 
251  // same again with {SAMPLE}
252  toReplace = "{SAMPLE}";
253  while ((pos = locationString.find(toReplace)) != std::string::npos)
254  {
255  locationString.replace(pos, toReplace.length(),
256  getFitterSpecificSampleName(fitter, fileType, sample));
257  }
258 
259  // loop through the raw location string and replace any instance of {PARAMETER2} with the name of
260  // the second specified parameter
261  toReplace = "{PARAMETER2}";
262  while ((pos = locationString.find(toReplace)) != std::string::npos)
263  {
264  locationString.replace(pos, toReplace.length(),
265  getFitterSpecificParamName(fitter, fileType, parameter2));
266  }
267 
268  // Now we go through and look for ":" and split the location into two parts
269  // first part should be the directory to look for the parameter
270  // lats part should be the end of the name of the object to look for
271  std::string token;
272  std::string delimiter = ":";
273  while ((pos = locationString.find(delimiter)) != std::string::npos)
274  {
275  token = locationString.substr(0, pos);
276  tokens.push_back(token);
277  locationString.erase(0, pos + delimiter.length());
278  }
279  tokens.push_back(locationString);
280 
281  // should only have found 2 parts, anything more means there was too many
282  // ":" in the location string
283  if (tokens.size() > 2)
284  {
285  throw MaCh3Exception(__FILE__ , __LINE__, "Too many : tokens in location string: " + rawLocationString);
286  }
287 
288  return tokens;
289 }

291 std::shared_ptr<TObject> InputManager::findRootObject(const InputFile &fileDef,
292  const std::vector<std::string> &locationVec) const {
293 
294  std::shared_ptr<TObject> object = nullptr;
295 
296  // if vector only has one element, just interpret it as the absolute path to the object
297  if (locationVec.size() == 1)
298  {
299  object = std::shared_ptr<TObject>(fileDef.file->Get(locationVec[0].c_str()));
300  }
301 
302  // if vector has two elements, interpret the first as the directory and the second as a string to
303  // match with the end of the objects in the file
304  else if (locationVec.size() == 2)
305  {
306  TDirectoryFile *directory = fileDef.file->Get<TDirectoryFile>(locationVec[0].c_str());
307  size_t nMatchingObjects = 0;
308 
309  // let's make sure that the directory itself exists
310  if (directory == nullptr)
311  {
312  object = nullptr;
313  }
314  else
315  {
316  // loop through the keys in the directory and find objects whose name matches the specified
317  // pattern
318  TIter next(directory->GetListOfKeys());
319  while (TKey *key = static_cast<TKey*>(next()))
320  {
321  if (strEndsWith(std::string(key->GetName()), locationVec[1]))
322  {
323  object = std::shared_ptr<TObject>(directory->Get(key->GetName()));
324  nMatchingObjects++;
325  }
326  }
327  }
328  // check that only one object matched the pattern
329  if (nMatchingObjects > 1)
330  {
331  MACH3LOG_CRITICAL("Too many objects match the pattern specified by {} {}", locationVec[0], locationVec.size()==2 ? locationVec[1] : "");
332  MACH3LOG_CRITICAL("Found {} matching objects, should just be one", nMatchingObjects);
333 
334  throw MaCh3Exception(__FILE__ , __LINE__ );
335  }
336  }
337  else // Vector too big!!
338  {
339  MACH3LOG_CRITICAL("Invalid object location vector");
340  MACH3LOG_CRITICAL("Should have two elements: [ (directory to look in), (end of the name of the object) ]");
341  MACH3LOG_CRITICAL("Or one element that is just the absolute path to the object");
342 
343  throw MaCh3Exception(__FILE__ , __LINE__ );
344  }
345  return object;
346 }

653  inputFileDef.availableParams_map_LLHBySample[sample][parameter] = true;
654  numLLHBySampleParams++;
655  }
656  }
657 
658  if (numLLHBySampleParams != 0)
659  {
660  inputFileDef.availableSamples_LLH.push_back(sample);
661 
662  if (printThoughts)
663  MACH3LOG_INFO("........ Found {} LLH scans for sample {}", numLLHBySampleParams, sample);
664 
665  if ((numLLHParams != numLLHBySampleParams))
666  {
667  MACH3LOG_ERROR("hmmmmm something weird is happening here");
668  MACH3LOG_ERROR(" I have {} LLH scans for sample {}", numLLHBySampleParams, sample);
669  MACH3LOG_ERROR(" But {} parameters with Total_LLH scals", numLLHParams);
670  }
671  }
672  }
673 
674  // ######### Now for the main event: Look for MCMC chains and processed posteriors ###########
675 
676  // EM: if "MCMCsteps" was defined for this fitter, we assume that it is a MaCh3 raw MCMC file
677  // thus it needs an MCMCProcessor to read from it. This isn't super general and it would probably
678  // be good to have some additional "isMaCh3" option that can decide whether or not to use MCMCProcessor
679  // but hey ho it's good enough for now
680  if ( thisFitterSpec_config["MCMCsteps"] )
681  {
682  MACH3LOG_DEBUG("Initialising MCMCProcessor for the input file");
683  std::vector<std::string> posteriorTreeRawLocations = thisFitterSpec_config["MCMCsteps"]["location"].as<std::vector<std::string>>();
684 
685  TTree *postTree = nullptr;
686  for ( const std::string &rawLoc: posteriorTreeRawLocations )
687  {
688  MACH3LOG_DEBUG(" - Looking for MCMC chain parameter values at: {}", rawLoc);
689 
690  postTree = inputFileDef.file->Get<TTree>(rawLoc.c_str());
691 
692  if ( postTree != nullptr )
693  {
694  inputFileDef.mcmcProc = new MCMCProcessor(inputFileDef.fileName);
695  inputFileDef.mcmcProc->Initialise();
696 
697  MACH3LOG_DEBUG(" - FOUND!");
698  break;
699  }
700  }
701 
702  if ( postTree != nullptr )
703  {
704  inputFileDef.posteriorTree = postTree;
705  inputFileDef.nMCMCentries = int(postTree->GetEntries());
706  }
707  }
708 
709  if (printThoughts)
710  {
711  MACH3LOG_INFO("....Searching for MCMC related things");
712  }
713 
714  size_t num1dPosteriors = 0;
715  std::vector<std::string> enabled1dPosteriorParams;
716 
717  size_t numMCMCchainParams = 0;
718  std::vector<std::string> enabledMCMCchainParams;
719 
720  for ( const std::string &parameter : _knownParameters )
721  {
722  MACH3LOG_DEBUG(" - for {}", parameter);
723  // check for 1d post processing posterior.q
724  inputFileDef.availableParams_map_1dPosteriors[parameter] = false;
725  if ( thisFitterSpec_config["1dPosteriors"] && find1dPosterior(inputFileDef, parameter, fitter) )
726  {
727  MACH3LOG_DEBUG(" Found 1d processed posterior!");
728  enabled1dPosteriorParams.push_back(parameter);
729  num1dPosteriors++;
730  inputFileDef.availableParams_map_1dPosteriors[parameter] = true;
731  }
732  // now check for parameters in chain
733  inputFileDef.availableParams_map_MCMCchain[parameter] = false;
734  if ( thisFitterSpec_config["MCMCsteps"] && findRawChainSteps(inputFileDef, parameter, fitter) )
735  {
736  MACH3LOG_DEBUG(" Found raw MCMC steps!");
737  enabledMCMCchainParams.push_back(parameter);
738  numMCMCchainParams++;
739  inputFileDef.availableParams_map_MCMCchain[parameter] = true;
740  }
741  }
742 
743  if (num1dPosteriors > 0 )
744  {
745  foundFitter = true;
746  inputFileDef.has1dPosteriors = true;
747  inputFileDef.availableParams_1dPosteriors = enabled1dPosteriorParams;
748 
749  if (printThoughts)
750  MACH3LOG_INFO("........ Found {} 1d processed posteriors", num1dPosteriors);
751  }
752  if ( numMCMCchainParams > 0 )
753  {
754  foundFitter = true;
755  inputFileDef.hasMCMCchain = true;
756  inputFileDef.availableParams_MCMCchain = enabledMCMCchainParams;
757 
758  if (printThoughts)
759  MACH3LOG_INFO("........ Found {} parameters in MCMC chain", numMCMCchainParams);
760  }
761 
762  /*
763  switch (i) {
764  // any other weird fitter specific conditions/ edge cases should go in here
765  }
766  */
767 
768  if (foundFitter)
769  {
770  if (printThoughts)
771  {
772  MACH3LOG_INFO("This is a {} file!\n", fitter);
773  }
774  inputFileDef.fitter = fitter;
775  return;
776  }
777  }
778 
779  // if we didn't return above then the fitter type wasn't found
780  if (printThoughts)
781  MACH3LOG_WARN("I don't know what kinda fitter this came from, will proceed with caution");
782  inputFileDef.fitter = "UNKNOWN_FITTER";
783 }

785 void InputManager::fillFileData(InputFile &inputFileDef, bool printThoughts) {
786  // load in the data in the file using the info gotten above
787  // EM: a lot of this is copy paste from above, could be better organised
788  if (printThoughts)
789  MACH3LOG_INFO("....getting data from file {}", inputFileDef.fileName);
790 
791  YAML::Node thisFitterSpec_config = _fitterSpecConfig[inputFileDef.fitter];
792 
793  // set the default post fit error type so we can read it as default later
794  inputFileDef.defaultErrorType =
795  thisFitterSpec_config["defaultPostFitErrorType"].as<std::string>();
796 
797  // ########### First fill up the LLH vectors ############
798  for (const std::string LLHType : {"sample", "penalty", "total"})
799  {
800  if (!inputFileDef.hasLLHScans_map.at(LLHType))
801  continue;
802 
803  // vector of all the possible locations that we might find LLH scans for this type of LLH
804  YAML::Node testLLHConfig = thisFitterSpec_config[LLHType + "_LLH"];
805  const std::vector<std::string> testLLHRawLocations =
806  testLLHConfig["location"].as<std::vector<std::string>>();
807 
808  // get the expected root object type of the llh scans
809  std::string LLHObjType = thisFitterSpec_config["LLHObjectType"].as<std::string>();
810 
811  // EM: now get the objects from the file
812  for (const std::string &parameter : inputFileDef.availableParams_LLH)
813  {
814  std::shared_ptr<TObject> LLHObj = nullptr;
815 
816  // check the locations for the object
817  for (const std::string &rawLocation : testLLHRawLocations)
818  {
819  LLHObj = findRootObject(inputFileDef,
820  parseLocation(rawLocation, inputFileDef.fitter, kLLH, parameter));
821  // if we found it then break out of the loop
822  if (LLHObj != nullptr)
823  break;
824  }
825 
826  // double check that we actually found it.. just in case
827  if (LLHObj == nullptr)
828  {
829  MACH3LOG_ERROR("Hmmm, something seems to have gone wrong and I couldnt find the {} LLH scan for {} when attempting to read the data for it", LLHType, parameter);
830  MACH3LOG_ERROR("This will very likely cause segfaults and sadness");
831  }
832 
833  // now convert it to a TGraph
834  std::shared_ptr<TGraph> LLHGraph = std::make_shared<TGraph>();
835 
836  // EM: maybe have the type of the LLH object specified in the config to know what to downcast
837  // it to??
838  if (LLHObjType == "TH1D")
839  {
840  LLHGraph = std::make_shared<TGraph>(static_cast<TH1D*>(LLHObj.get()));
841  }
842 
843  else if (LLHObjType == "TGraph")
844  {
845  LLHGraph = std::shared_ptr<TGraph>(static_cast<TGraph*>(LLHObj->Clone()));
846  }
847 
848  else
849  {
850  MACH3LOG_CRITICAL("ERROR: uknown type of LLH object specified: {}", LLHObjType);
851  throw MaCh3Exception(__FILE__ , __LINE__ );
852  }
853 
854  inputFileDef.LLHScans_map[LLHType][parameter] = LLHGraph;
855  }
856  }
857 
858  // ####### Get the by sample LLH scans #######
859  for (const std::string &parameter : inputFileDef.availableParams_LLH)
860  {
861  for (const std::string &sample : _knownSamples)
862  {
863  findBySampleLLH(inputFileDef, parameter, inputFileDef.fitter, sample, true);
864  }
865  }
866 
867  // ####### Get the processed post fit errors #######
868  for (const std::string &parameter : inputFileDef.availableParams_postFitErrors)
869  {
870  std::vector<std::string> availableErrorTypes =
871  thisFitterSpec_config["AvailablePostFitErrorTypes"].as<std::vector<std::string>>();
872  for (const std::string &errorType : availableErrorTypes)
873  {
874  findPostFitParamError(inputFileDef, parameter, inputFileDef.fitter, errorType, true);
875  }
876  }
877 
878  // ########## Get the MCMC related posteriors ###########
879  for (const std::string &parameter : inputFileDef.availableParams_MCMCchain)
880  {
881  findRawChainSteps(inputFileDef, parameter, inputFileDef.fitter, true);
882  }
883  for (const std::string &parameter : inputFileDef.availableParams_1dPosteriors)
884  {
885  find1dPosterior(inputFileDef, parameter, inputFileDef.fitter, true);
886  }
887 }
