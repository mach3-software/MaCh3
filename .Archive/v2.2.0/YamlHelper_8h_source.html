179 inline bool compareYAMLNodes(const YAML::Node& node1, const YAML::Node& node2) {
180 // **********************
181  // Check if the types of the nodes match
182  if (node1.Type() != node2.Type()) {
183  return false;
184  }
185 
186  // Compare scalar types (like strings, numbers)
187  if (node1.IsScalar() && node2.IsScalar()) {
188  return node1.as<std::string>() == node2.as<std::string>();
189  }
190 
191  // Compare sequences (like YAML lists)
192  if (node1.IsSequence() && node2.IsSequence()) {
193  if (node1.size() != node2.size()) {
194  return false;
195  }
196  for (std::size_t i = 0; i < node1.size(); ++i) {
197  if (!compareYAMLNodes(node1[i], node2[i])) {
198  return false;
199  }
200  }
201  return true;
202  }
203 
204  // Compare maps (like YAML dictionaries)
205  if (node1.IsMap() && node2.IsMap()) {
206  if (node1.size() != node2.size()) {
207  return false;
208  }
209  for (auto it1 = node1.begin(); it1 != node1.end(); ++it1) {
210  auto key = it1->first.as<std::string>();
211  if (!node2[key] || !compareYAMLNodes(it1->second, node2[key])) {
212  return false;
213  }
214  }
215  return true;
216  }
217 
218  // Default case: if it's neither scalar, sequence, nor map, consider it unequal
219  return false;
220 }

314 inline YAML::Node LoadYamlConfig(const std::string& filename, const std::string& File, const int Line) {
315 // **********************
316  // KS: YAML can be dumb and not throw error if you pass toml for example...
317  if (!(filename.length() >= 5 && filename.compare(filename.length() - 5, 5, ".yaml") == 0) &&
318  !(filename.length() >= 4 && filename.compare(filename.length() - 4, 4, ".yml") == 0)) {
319  MACH3LOG_ERROR("Invalid file extension: {}\n", filename);
320  throw MaCh3Exception(File, Line);
321  }
322 
323  try {
324  YAML::Node yamlNode = YAML::LoadFile(filename);
325 
326  // Convert the YAML file to string
327  std::ifstream fileStream(filename);
328  std::stringstream buffer;
329  buffer << fileStream.rdbuf();
330  std::string fileContent = buffer.str();
331 
332  // Use regex to find any line starting with `-` but not followed by space, digit, or dot
333  // This excludes valid numeric negative values like -1760.0
334  std::regex linePattern(R"(^\s*-(?![\d\s\.]).*)");
335 
336  std::istringstream contentStream(fileContent);
337  std::string lineconfig;
338  int lineNumber = 1;
339 
340  // KS: Instead of boolean, track flow-style list nesting depth
341  int flowListDepth = 0;
342 
343  while (std::getline(contentStream, lineconfig)) {
344  // Ignore YAML document separator
345  if (lineconfig.find("---") != std::string::npos) {
346  lineNumber++;
347  continue;
348  }
349 
350  // Update flow list depth (increment for each '[' and decrement for each ']')
351  for (char c : lineconfig) {
352  if (c == '[') flowListDepth++;
353  else if (c == ']') flowListDepth = std::max(0, flowListDepth - 1);
354  }
355 
356  // Check lines only outside flow-style lists
357  if (flowListDepth == 0 && std::regex_match(lineconfig, linePattern)) {
358  MACH3LOG_ERROR("Warning: Missing space or tab after '-' at line {}: {}\n", lineNumber, lineconfig);
359  throw MaCh3Exception(File, Line);
360  }
361 
362  lineNumber++;
363  }
364 
365  return yamlNode;
366 
367  } catch (const std::exception& e) {
368  MACH3LOG_ERROR("{}\n", e.what());
369  MACH3LOG_ERROR("Can't open file {}\n", filename);
370  throw MaCh3Exception(File, Line);
371  }
372 }

394 inline YAML::Node MergeNodes(YAML::Node a, YAML::Node b) {
395 // **********************
396  if (!b.IsMap()) {
397  // If b is not a map, merge result is b, unless b is null
398  return b.IsNull() ? a : b;
399  }
400  if (!a.IsMap()) {
401  // If a is not a map, merge result is b
402  return b;
403  }
404  if (!b.size()) {
405  // If a is a map, and b is an empty map, return a
406  return a;
407  }
408  // Create a new map 'c' with the same mappings as a, merged with b
409  auto c = YAML::Node(YAML::NodeType::Map);
410  for (auto n : a) {
411  if (n.first.IsScalar()) {
412  const std::string & key = n.first.Scalar();
413  auto t = (Cnode(b)[key]);
414  if (t) {
415  c[n.first] = MergeNodes(n.second, t);
416  continue;
417  }
418  }
419  c[n.first] = n.second;
420  }
421  // Add the mappings from 'b' not already in 'c'
422  for (auto n : b) {
423  if (!n.first.IsScalar() || !Cnode(c)[n.first.Scalar()]) {
424  c[n.first] = n.second;
425  }
426  }
427  return c;
428 }

443 inline std::vector<double> ParseBounds(const YAML::Node& node, const std::string& File, const int Line) {
444 // **********************
445  std::vector<double> bounds;
446 
447  if (!node || !node.IsSequence() || (node.size() != 1 && node.size() != 2)) {
448  MACH3LOG_ERROR("Bounds must be a sequence of 1 or 2 elements, got size {}", static_cast<int>(node.size()));
449  throw MaCh3Exception(File, Line);
450  }
451 
452  if (node.size() == 1) {
453  const YAML::Node& val = node[0];
454 
455  if (!val || val.IsNull() || (val.IsScalar() && val.Scalar().empty())) {
456  MACH3LOG_ERROR("Single-element bounds must contain a valid numeric value.");
457  throw MaCh3Exception(File, Line);
458  } else if (val.IsScalar()) {
459  try {
460  const double center = val.as<double>();
461  bounds = {center - 0.5, center + 0.5};
462  } catch (const YAML::BadConversion& e) {
463  MACH3LOG_ERROR("Invalid value in Bounds[0]: '{}'. Expected a number.", static_cast<std::string>(val.Scalar()));
464  throw MaCh3Exception(File, Line);
465  }
466  } else {
467  MACH3LOG_ERROR("Invalid type in Bounds[0]");
468  throw MaCh3Exception(File, Line);
469  }
470  } else {
471  for (std::size_t i = 0; i < 2; ++i) {
472  const YAML::Node& val = node[i];
473 
474  if (!val || val.IsNull() || (val.IsScalar() && val.Scalar().empty())) {
475  bounds.push_back(i == 0 ? M3::KinematicLowBound : M3::KinematicUpBound);
476  } else if (val.IsScalar()) {
477  try {
478  bounds.push_back(val.as<double>());
479  } catch (const YAML::BadConversion& e) {
480  MACH3LOG_ERROR("Invalid value in Bounds[{}]: '{}'. Expected a number or empty string.",
481  static_cast<int>(i), static_cast<std::string>(val.Scalar()));
482  throw MaCh3Exception(File, Line);
483  }
484  } else {
485  MACH3LOG_ERROR("Invalid type in Bounds[{}]", static_cast<int>(i));
486  throw MaCh3Exception(File, Line);
487  }
488  }
489  }
490  return bounds;
491 }
