191  int FindXBin(const double XVar, const int NomXBin) const {
192  //DB Check to see if momentum shift has moved bins
193  //DB - First , check to see if the event is outside of the binning range and skip event if it is
194  if (XVar < XBinEdges[0] || XVar >= XBinEdges[nXBins]) {
195  return -1;
196  }
197  //DB - Second, check to see if the event is still in the nominal bin
198  else if (XVar < rw_upper_xbinedge[NomXBin] && XVar >= rw_lower_xbinedge[NomXBin]) {
199  return NomXBin;
200  }
201  //DB - Thirdly, check the adjacent bins first as Eb+CC+EScale shifts aren't likely to move an Erec more than 1bin width
202  //Shifted down one bin from the event bin at nominal
203  else if (XVar < rw_lower_xbinedge[NomXBin] && XVar >= rw_lower_lower_xbinedge[NomXBin]) {
204  return NomXBin-1;
205  }
206  //Shifted up one bin from the event bin at nominal
207  else if (XVar < rw_upper_upper_xbinedge[NomXBin] && XVar >= rw_upper_xbinedge[NomXBin]) {
208  return NomXBin+1;
209  }
210  //DB - If we end up in this loop, the event has been shifted outside of its nominal bin, but is still within the allowed binning range
211  else {
212  // KS: Perform binary search to find correct bin. We already checked if isn't outside of bounds
213  return static_cast<int>(std::distance(XBinEdges.begin(), std::upper_bound(XBinEdges.begin(), XBinEdges.end(), XVar)) - 1);
214  }
215  }
216 };
