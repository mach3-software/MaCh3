162 void PrepareChains() {
163 // *******************
164  auto rnd = std::make_unique<TRandom3>(0);
165 
166  MACH3LOG_INFO("Generating {}", Ntoys);
167 
168  TStopwatch clock;
169  clock.Start();
170 
171  std::vector<unsigned int> BurnIn(Nchains);
172  std::vector<unsigned int> nEntries(Nchains);
173  std::vector<int> nBranches(Nchains);
174  std::vector<unsigned int> step(Nchains);
175 
176  Draws = new double**[Nchains]();
177  DrawsFolded = new double**[Nchains]();
178 
179  // KS: This can reduce time necessary for caching even by half
180  #ifdef MULTITHREAD
181  //ROOT::EnableImplicitMT();
182  #endif
183 
184  // Open the Chain
185  //It is tempting to multithread here but unfortunately, ROOT files are not thread safe :(
186  for (int m = 0; m < Nchains; m++)
187  {
188  TChain* Chain = new TChain("posteriors");
189  Chain->Add(MCMCFile[m].c_str());
190  MACH3LOG_INFO("On file: {}", MCMCFile[m].c_str());
191  nEntries[m] = static_cast<unsigned int>(Chain->GetEntries());
192 
193  // Set the step cut to be 20%
194  BurnIn[m] = nEntries[m]/5;
195 
196  // Get the list of branches
197  TObjArray* brlis = Chain->GetListOfBranches();
198 
199  // Get the number of branches
200  nBranches[m] = brlis->GetEntries();
201 
202  if(m == 0) BranchNames.reserve(nBranches[m]);
203 
204  // Set all the branches to off
205  Chain->SetBranchStatus("*", false);
206 
207  // Loop over the number of branches
208  // Find the name and how many of each systematic we have
209  for (int i = 0; i < nBranches[m]; i++)
210  {
211  // Get the TBranch and its name
212  TBranch* br = static_cast<TBranch *>(brlis->At(i));
213  if(!br){
214  MACH3LOG_ERROR("Invalid branch at position {}", i);
215  throw MaCh3Exception(__FILE__,__LINE__);
216  }
217  TString bname = br->GetName();
218 
219  // Read in the step
220  if (bname == "step") {
221  Chain->SetBranchStatus(bname, true);
222  Chain->SetBranchAddress(bname, &step[m]);
223  }
224  //Count all branches
225  else if (bname.BeginsWith("PCA_") || bname.BeginsWith("accProb") || bname.BeginsWith("stepTime") )
226  {
227  continue;
228  }
229  else
230  {
231  //KS: Save branch name only for one chain, we assume all chains have the same branches, otherwise this doesn't make sense either way
232  if(m == 0)
233  {
234  BranchNames.push_back(bname);
235  //KS: We calculate R Hat also for LogL, just in case, however we plot them separately
236  if(bname.BeginsWith("LogL"))
237  {
238  ValidPar.push_back(false);
239  }
240  else
241  {
242  ValidPar.push_back(true);
243  }
244  }
245  Chain->SetBranchStatus(bname, true);
246  MACH3LOG_DEBUG("{}", bname);
247  }
248  }
249 
250  if(m == 0) nDraw = int(BranchNames.size());
251 
252  //TN: Qualitatively faster sanity check, with the very same outcome (all chains have the same #branches)
253  if(m > 0)
254  {
255  if(nBranches[m] != nBranches[0])
256  {
257  MACH3LOG_ERROR("Ups, something went wrong, chain {} called {} has {} branches, while 0 called {} has {} branches", m, MCMCFile[m], nBranches[m], MCMCFile[0], nBranches[0]);
258  MACH3LOG_ERROR("All chains should have the same number of branches");
259  throw MaCh3Exception(__FILE__ , __LINE__ );
260  }
261  }
262 
263  //TN: move the Draws here, so we need to iterate over every chain only once
264  Draws[m] = new double*[Ntoys]();
265  DrawsFolded[m] = new double*[Ntoys]();
266  for(int i = 0; i < Ntoys; i++)
267  {
268  Draws[m][i] = new double[nDraw]();
269  DrawsFolded[m][i] = new double[nDraw]();
270  for(int j = 0; j < nDraw; j++)
271  {
272  Draws[m][i][j] = 0.;
273  DrawsFolded[m][i][j] = 0.;
274  }
275  }
276 
277  // MJR: array to hold branch values; SetBranchAddress in every step is very
278  // expensive, so doing it once only here saves time
279  double* branch_values = new double[nDraw]();
280  for (int j = 0; j < nDraw; ++j)
281  {
282  Chain->SetBranchAddress(BranchNames[j].Data(), &branch_values[j]);
283  }
284 
285  //TN: move looping over toys here, so we don't need to loop over chains more than once
286  if(BurnIn[m] >= nEntries[m])
287  {
288  MACH3LOG_ERROR("You are running on a chain shorter than BurnIn cut");
289  MACH3LOG_ERROR("Number of entries {} BurnIn cut {}", nEntries[m], BurnIn[m]);
290  MACH3LOG_ERROR("You will run into the infinite loop");
291  MACH3LOG_ERROR("You can make a new chain or modify BurnIn cut");
292  throw MaCh3Exception(__FILE__ , __LINE__ );
293  }
294 
295  for (int i = 0; i < Ntoys; i++)
296  {
297  // Get a random entry after burn in
298  int entry = int(nEntries[m]*rnd->Rndm());
299 
300  Chain->GetEntry(entry);
301 
302  // If we have combined chains by hadd need to check the step in the chain
303  // Note, entry is not necessarily the same as the step due to merged ROOT files, so can't choose an entry in the range BurnIn - nEntries :(
304  if (step[m] < BurnIn[m])
305  {
306  i--;
307  continue;
308  }
309 
310  // Output some info for the user
311  if (Ntoys > 10 && i % (Ntoys/10) == 0) {
312  MaCh3Utils::PrintProgressBar(i+m*Ntoys, static_cast<Long64_t>(Ntoys)*Nchains);
313  MACH3LOG_DEBUG("Getting random entry {}", entry);
314  }
315 
316  // Set the branch addresses for params
317  for (int j = 0; j < nDraw; ++j)
318  {
319  Draws[m][i][j] = branch_values[j];
320  }
321 
322  }//end loop over toys
323 
324  //TN: There, we now don't need to keep the chain in memory anymore
325  delete Chain;
326  delete[] branch_values;
327  }
328 
329  //KS: Now prepare folded draws, quoting Gelman
330  //"We propose to report the maximum of rank normalized split-Rb and rank normalized folded-split-Rb for each parameter"
331  MedianArr = new double[nDraw]();
332  #ifdef MULTITHREAD
333  #pragma omp parallel for
334  #endif
335  for(int j = 0; j < nDraw; j++)
336  {
337  MedianArr[j] = 0.;
338  std::vector<double> TempDraws(static_cast<size_t>(Ntoys) * Nchains);
339  for(int m = 0; m < Nchains; m++)
340  {
341  for(int i = 0; i < Ntoys; i++)
342  {
343  const int im = i+m;
344  TempDraws[im] = Draws[m][i][j];
345  }
346  }
347  MedianArr[j] = CalcMedian(TempDraws.data(), Ntoys*Nchains);
348  }
349 
350  #ifdef MULTITHREAD
351  #pragma omp parallel for collapse(3)
352  #endif
353  for(int m = 0; m < Nchains; m++)
354  {
355  for(int i = 0; i < Ntoys; i++)
356  {
357  for(int j = 0; j < nDraw; j++)
358  {
359  DrawsFolded[m][i][j] = std::fabs(Draws[m][i][j] - MedianArr[j]);
360  }
361  }
362  }
363  clock.Stop();
364  MACH3LOG_INFO("Finished calculating Toys, it took {:.2f}s to finish", clock.RealTime());
365 }

437 //KS: Based on Gelman et. al. arXiv:1903.08008v5
438 // Probably most of it could be moved cleverly to MCMC Processor, keep it separate for now
439 void CalcRhat() {
440 // *******************
441 
442  TStopwatch clock;
443  clock.Start();
444 
445  //KS: Start parallel region
446  // If we would like to do this for thousands of chains we might consider using GPU for this
447  #ifdef MULTITHREAD
448  #pragma omp parallel
449  {
450  #endif
451 
452  #ifdef MULTITHREAD
453  #pragma omp for collapse(2)
454  #endif
455  //KS: loop over chains and draws are independent so might as well collapse for sweet cache hits
456  //Calculate the mean for each parameter within each considered chain
457  for (int m = 0; m < Nchains; ++m)
458  {
459  for (int j = 0; j < nDraw; ++j)
460  {
461  for(int i = 0; i < Ntoys; i++)
462  {
463  Mean[m][j] += Draws[m][i][j];
464  MeanFolded[m][j] += DrawsFolded[m][i][j];
465  }
466  Mean[m][j] = Mean[m][j]/Ntoys;
467  MeanFolded[m][j] = MeanFolded[m][j]/Ntoys;
468  }
469  }
470 
471  #ifdef MULTITHREAD
472  #pragma omp for
473  #endif
474  //Calculate the mean for each parameter global means we include information from several chains
475  for (int j = 0; j < nDraw; ++j)
476  {
477  for (int m = 0; m < Nchains; ++m)
478  {
479  MeanGlobal[j] += Mean[m][j];
480  MeanGlobalFolded[j] += MeanFolded[m][j];
481  }
482  MeanGlobal[j] = MeanGlobal[j]/Nchains;
483  MeanGlobalFolded[j] = MeanGlobalFolded[j]/Nchains;
484  }
485 
486 
487  #ifdef MULTITHREAD
488  #pragma omp for collapse(2)
489  #endif
490  //Calculate the standard deviation for each parameter within each considered chain
491  for (int m = 0; m < Nchains; ++m)
492  {
493  for (int j = 0; j < nDraw; ++j)
494  {
495  for(int i = 0; i < Ntoys; i++)
496  {
497  StandardDeviation[m][j] += (Draws[m][i][j] - Mean[m][j])*(Draws[m][i][j] - Mean[m][j]);
498  StandardDeviationFolded[m][j] += (DrawsFolded[m][i][j] - MeanFolded[m][j])*(DrawsFolded[m][i][j] - MeanFolded[m][j]);
499  }
500  StandardDeviation[m][j] = StandardDeviation[m][j]/(Ntoys-1);
501  StandardDeviationFolded[m][j] = StandardDeviationFolded[m][j]/(Ntoys-1);
502  }
503  }
504 
505  #ifdef MULTITHREAD
506  #pragma omp for
507  #endif
508  //Calculate the standard deviation for each parameter combining information from all chains
509  for (int j = 0; j < nDraw; ++j)
510  {
511  for (int m = 0; m < Nchains; ++m)
512  {
513  StandardDeviationGlobal[j] += StandardDeviation[m][j];
514  StandardDeviationGlobalFolded[j] += StandardDeviationFolded[m][j];
515  }
516  StandardDeviationGlobal[j] = StandardDeviationGlobal[j]/Nchains;
517  StandardDeviationGlobalFolded[j] = StandardDeviationGlobalFolded[j]/Nchains;
518  }
519 
520  #ifdef MULTITHREAD
521  #pragma omp for
522  #endif
523  for (int j = 0; j < nDraw; ++j)
524  {
525  //KS: This term only makes sense if we have at least 2 chains
526  if(Nchains == 1)
527  {
528  BetweenChainVariance[j] = 0.;
529  BetweenChainVarianceFolded[j] = 0.;
530  }
531  else
532  {
533  for (int m = 0; m < Nchains; ++m)
534  {
535  BetweenChainVariance[j] += ( Mean[m][j] - MeanGlobal[j])*( Mean[m][j] - MeanGlobal[j]);
536  BetweenChainVarianceFolded[j] += ( MeanFolded[m][j] - MeanGlobalFolded[j])*( MeanFolded[m][j] - MeanGlobalFolded[j]);
537  }
538  BetweenChainVariance[j] = BetweenChainVariance[j]*Ntoys/(Nchains-1);
539  BetweenChainVarianceFolded[j] = BetweenChainVarianceFolded[j]*Ntoys/(Nchains-1);
540  }
541  }
542 
543  #ifdef MULTITHREAD
544  #pragma omp for
545  #endif
546  for (int j = 0; j < nDraw; ++j)
547  {
548  MarginalPosteriorVariance[j] = (Ntoys-1) * StandardDeviationGlobal[j] /(Ntoys) + BetweenChainVariance[j]/Ntoys;
549  MarginalPosteriorVarianceFolded[j] = (Ntoys-1) * StandardDeviationGlobalFolded[j] /(Ntoys) + BetweenChainVarianceFolded[j]/Ntoys;
550  }
551 
552  #ifdef MULTITHREAD
553  #pragma omp for
554  #endif
555  //Finally calculate our estimator
556  for (int j = 0; j < nDraw; ++j)
557  {
558  RHat[j] = sqrt(MarginalPosteriorVariance[j]/StandardDeviationGlobal[j]);
559  RHatFolded[j] = sqrt(MarginalPosteriorVarianceFolded[j]/StandardDeviationGlobalFolded[j]);
560 
561  //KS: For flat params values can be crazy so cap at 0
562  CapVariable(RHat[j], 0);
563  CapVariable(RHatFolded[j], 0);
564  }
565 
566  #ifdef MULTITHREAD
567  #pragma omp for
568  #endif
569  //KS: Additionally calculates effective step size which is an estimate of the sample size required to achieve the same level of precision if that sample was a simple random sample.
570  for (int j = 0; j < nDraw; ++j)
571  {
572  if(Nchains > 1) EffectiveSampleSize[j] = Nchains * Ntoys * MarginalPosteriorVariance[j] / BetweenChainVariance[j];
573  if(Nchains > 1) EffectiveSampleSizeFolded[j] = Nchains * Ntoys * MarginalPosteriorVarianceFolded[j] / BetweenChainVarianceFolded[j];
574 
575  //KS: For flat params values can be crazy so cap at 0
576  CapVariable(EffectiveSampleSize[j], 0);
577  CapVariable(EffectiveSampleSizeFolded[j], 0);
578  }
579  #ifdef MULTITHREAD
580  } //End parallel region
581  #endif
582 
583  clock.Stop();
584  MACH3LOG_INFO("Finished calculating RHat, it took {:.2f}s to finish", clock.RealTime());
585 }

630

649  RhatFoldedPlot->Fill(RHatFolded[j]);
650  EffectiveSampleSizeFoldedPlot->Fill(EffectiveSampleSizeFolded[j]);
