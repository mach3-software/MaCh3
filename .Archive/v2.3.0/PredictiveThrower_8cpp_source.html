444 std::vector<std::unique_ptr<TH2D>> PredictiveThrower::ProduceSpectra(const std::vector<std::vector<std::unique_ptr<TH1D>>>& Toys,
445  const std::string suffix) {
446 // *************************
447  std::vector<double> MaxValue(TotalNumberOfSamples);
448  #ifdef MULTITHREAD
449  #pragma omp parallel for collapse(2)
450  #endif
451  for (int sample = 0; sample < TotalNumberOfSamples; ++sample) {
452  for (int toy = 0; toy < Ntoys; ++toy) {
453  double max_val = Toys[sample][toy]->GetMaximum();
454  MaxValue[sample] = std::max(MaxValue[sample], max_val);
455  }
456  }
457 
458  std::vector<std::unique_ptr<TH2D>> Spectra(TotalNumberOfSamples);
459  for (int sample = 0; sample < TotalNumberOfSamples; ++sample) {
460  // Get MC histogram x-axis binning
461  TH1D* refHist = Toys[sample][0].get();
462 
463  const int n_bins_x = refHist->GetNbinsX();
464  std::vector<double> x_bin_edges(n_bins_x + 1);
465  for (int b = 0; b <= n_bins_x; ++b) {
466  x_bin_edges[b] = refHist->GetXaxis()->GetBinLowEdge(b + 1); // ROOT bins start at 1
467  }
468  // Last edge is upper edge of last bin:
469  x_bin_edges[n_bins_x] = refHist->GetXaxis()->GetBinUpEdge(n_bins_x);
470 
471  constexpr int n_bins_y = 400;
472  constexpr double y_min = 0.0;
473  const double y_max = MaxValue[sample] * 1.05;
474 
475  // Create TH2D with variable binning on x axis
476  Spectra[sample] = std::make_unique<TH2D>(
477  (SampleNames[sample] + "_" + suffix).c_str(), // name
478  (SampleNames[sample] + "_" + suffix).c_str(), // title
479  n_bins_x, x_bin_edges.data(), // x axis bins
480  n_bins_y, y_min, y_max // y axis bins
481  );
482 
483  Spectra[sample]->SetDirectory(nullptr);
484  Spectra[sample]->Sumw2(true);
485  }
486 
487  #ifdef MULTITHREAD
488  #pragma omp parallel for
489  #endif
490  for (int sample = 0; sample < TotalNumberOfSamples; ++sample) {
491  for (int toy = 0; toy < Ntoys; ++toy) {
492  FastViolinFill(Spectra[sample].get(), Toys[sample][toy].get());
493  }
494  }
495 
496  return Spectra;
497 }

607 void PredictiveThrower::PosteriorPredictivepValue(const std::vector<std::unique_ptr<TH1D>>& PostPred_mc,

641  chi2_pred_vec[iToy].back() += chi2_pred_vec[iToy][iSample];
642  }
