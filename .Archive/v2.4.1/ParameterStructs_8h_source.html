249 inline std::string GetTF1(const SplineInterpolation i) {
250  // **************************************************
251  std::string Func = "";
252  switch(i) {
253  case SplineInterpolation::kLinearFunc:
254  Func = "([1]+[0]*x)";
255  break;
256  case SplineInterpolation::kTSpline3:
257  case SplineInterpolation::kLinear:
258  case SplineInterpolation::kMonotonic:
259  case SplineInterpolation::kAkima:
260  case SplineInterpolation::kKochanekBartels:
261  case SplineInterpolation::kSplineInterpolations:
262  MACH3LOG_ERROR("Interpolation type {} not supported for TF1!", static_cast<int>(i));
263  throw MaCh3Exception(__FILE__, __LINE__);
264  default:
265  MACH3LOG_ERROR("UNKNOWN SPLINE INTERPOLATION SPECIFIED!");
266  MACH3LOG_ERROR("You gave {}", static_cast<int>(i));
267  throw MaCh3Exception(__FILE__ , __LINE__ );
268  }
269  return Func;
270 }
