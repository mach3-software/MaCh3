49 bool checkSoftwareVersions(TFile *file, TFile *prevFile, const std::string& ConfigName)
50 {
51  bool weirdFile = false;
52 
53  TMacro *versionHeader = file->Get<TMacro>(ConfigName.c_str());
54  TMacro *prevVersionHeader = prevFile->Get<TMacro>(ConfigName.c_str());
55 
56  // EM: compare the digest of the version header file in this file, with the previous one
57  if((versionHeader == NULL) && (prevVersionHeader == NULL)){
58  MACH3LOG_WARN("files don't contain an embedded version header, indicating they were made before this feature was added");
59  MACH3LOG_WARN(" I can still combine them but I can't guarantee they were made with matching software versions");
60  MACH3LOG_WARN(" This is ok but you'll just have to be extra careful and make sure you check this yourself!");
61  }
62  else if((versionHeader != NULL) && (prevVersionHeader == NULL)){
63  MACH3LOG_ERROR("looks like file {} has a version header file but previous ones do not", file->GetName());
64  MACH3LOG_ERROR("This is odd and suggests this file was made with an MaCh3 version");
65  MACH3LOG_ERROR("from after this feature was added, whereas other files were made with an older MaCh3");
66  weirdFile = true;
67  }
68  else if((versionHeader == NULL) && (prevVersionHeader != NULL)){
69  MACH3LOG_ERROR("looks like file {} doesn't have a version header file but previous ones do", file->GetName());
70  MACH3LOG_ERROR("This is odd and suggests this file was made with an MaCh3 version");
71  MACH3LOG_ERROR("from before this feature was added, whereas other files were made with newer MaCh3");
72  weirdFile = true;
73  }
74  else{
75  MACH3LOG_DEBUG(" Prev header digest: {} :: current: {}", (prevVersionHeader->Checksum())->AsString(), (versionHeader->Checksum())->AsString());
76  if(std::strcmp((versionHeader->Checksum())->AsString(), (prevVersionHeader->Checksum())->AsString()) != 0){
77  MACH3LOG_ERROR("Looks like the {} embedded config for file {} is different to the previous ones", ConfigName, file->GetName());
78  MACH3LOG_ERROR("This strongly suggests that this file was made with different software versions than the previous ones");
79  CompareTwoConfigs(TMacroToString(*versionHeader), TMacroToString(*prevVersionHeader));
80  weirdFile = true;
81  }
82  }
83  return weirdFile;
84 }
