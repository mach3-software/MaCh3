597 // ************************************************
598 // "Randomize" the parameters in the covariance class for the proposed step
599 // Used the proposal kernel and the current parameter value to set proposed step
600 // Also get a new random number for the randParams
601 void covarianceBase::randomize() _noexcept_ {
602 // ************************************************
603  if (!pca) {
604 //KS: By multithreading here we gain at least factor 2 with 8 threads with ND only fit
605  #ifdef MULTITHREAD
606  #pragma omp parallel for
607  #endif
608  for (int i = 0; i < _fNumPar; ++i) {
609  // If parameter isn't fixed
610  if (_fError[i] > 0.0) {
611 #ifdef MULTITHREAD
612  randParams[i] = random_number[omp_get_thread_num()]->Gaus(0, 1);
613 #else
614  randParams[i] = random_number[0]->Gaus(0, 1);
615 #endif
616  // If parameter IS fixed
617  } else {
618  randParams[i] = 0.0;
619  }
620  } // end for
621  // If we're in the PCA basis we instead throw parameters there (only _fNumParPCA parameter)
622  } else {
623  // Scale the random parameters by the sqrt of eigen values for the throw
624  #ifdef MULTITHREAD
625  #pragma omp parallel for
626  #endif
627  for (int i = 0; i < _fNumPar; ++i)
628  {
629  if (fParSigma_PCA[i] > 0. && i < _fNumParPCA)
630  {
631 #ifdef MULTITHREAD
632  randParams[i] = random_number[omp_get_thread_num()]->Gaus(0,1);
633 #else
634  randParams[i] = random_number[0]->Gaus(0,1);
635 #endif
636  } else { // If parameter IS fixed or out od bounds
637  randParams[i] = 0.0;
638  }
639  }
640  }

643 // ************************************************
644 // Correlate the steps by setting the proposed step of a parameter to its current value + some correlated throw
645 void covarianceBase::CorrelateSteps() _noexcept_ {
646 // ************************************************
647  //KS: Using custom function compared to ROOT one with 8 threads we have almost factor 2 performance increase, by replacing TMatrix with just double we increase it even more
648  MatrixVectorMulti(corr_throw, throwMatrixCholDecomp, randParams, _fNumPar);
649 
650  // If not doing PCA
651  if (!pca) {
652  #ifdef MULTITHREAD
653  #pragma omp parallel for
654  #endif
655  for (int i = 0; i < _fNumPar; ++i) {
656  if (_fError[i] > 0.) {
657  _fPropVal[i] = _fCurrVal[i] + corr_throw[i]*_fGlobalStepScale*_fIndivStepScale[i];
658  }
659  }
660  // If doing PCA throw uncorrelated in PCA basis (orthogonal basis by definition)
661  } else {
662  // Throw around the current step
663  #ifdef MULTITHREAD
664  #pragma omp parallel for
665  #endif
666  for (int i = 0; i < _fNumParPCA; ++i)
667  {
668  if (fParSigma_PCA[i] > 0.)
669  {
670  double IndStepScale = 1.;
671  //KS: If undecomposed parameter apply individual step scale and Cholesky for better acceptance rate
672  if(isDecomposed_PCA[i] >= 0)
673  {
674  IndStepScale *= _fIndivStepScale[isDecomposed_PCA[i]];
675  IndStepScale *= corr_throw[isDecomposed_PCA[i]];
676  }
677  //If decomposed apply only random number
678  else
679  {
680  IndStepScale *= randParams[i];
681  //KS: All PCA-ed parameters have the same step scale
682  IndStepScale *= _fIndivStepScale[FirstPCAdpar];
683  }
684  fParProp_PCA(i) = fParCurr_PCA(i)+_fGlobalStepScale*IndStepScale*PCAObj.eigen_values_master[i];
685  }
686  }
687  // Then update the parameter basis
688  TransferToParam();
689  }
690 }

806 double covarianceBase::CalcLikelihood() _noexcept_ {
807 // ********************************************

809  #ifdef MULTITHREAD
810  #pragma omp parallel for reduction(+:logL)
811  #endif
812  for(int i = 0; i < _fNumPar; ++i){
813  #ifdef MULTITHREAD

815  #endif
816  for (int j = 0; j <= i; ++j) {

821  }
822  }
823  }
824  return logL;
825 }
