616  XPos = YResp = NULL;
617  }

621  inline int FindX(double x) {
622  // The segment we're interested in (klow in ROOT code)
623  int segment = 0;
624  int kHigh = nPoints-1;
625  // If the variation is below the lowest saved spline point
626  if (x <= XPos[0]){
627  segment = 0;
628  // If the variation is above the highest saved spline point
629  } else if (x >= XPos[nPoints-1]) {
630  // Yes, the -2 is indeed correct, see TSpline.cxx:814 and //see: https://savannah.cern.ch/bugs/?71651
631  segment = kHigh;
632  // If the variation is between the maximum and minimum, perform a binary search
633  } else {
634  // The top point we've got
635  int kHalf = 0;
636  // While there is still a difference in the points (we haven't yet found the segment)
637  // This is a binary search, incrementing segment and decrementing kHalf until we've found the segment
638  while (kHigh - segment > 1) {
639  // Increment the half-step
640  kHalf = (segment + kHigh)/2;
641  // If our variation is above the kHalf, set the segment to kHalf
642  if (x > XPos[kHalf]) {
643  segment = kHalf;
644  // Else move kHigh down
645  } else {
646  kHigh = kHalf;
647  }
648  } // End the while: we've now done our binary search
649  } // End the else: we've now found our point
650  if (segment >= nPoints-1 && nPoints > 1) segment = nPoints-2;
651  return segment;
652  }

655  inline double Eval(double var) override {
656  // Get the segment for this variation
657  int segment = FindX(var);
658  // The get the coefficients for this variation
659  _float_ x = -999.99, y = -999.99, b = -999.99, c = -999.99, d = -999.99;
660  GetCoeff(segment, x, y, b, c, d);
661  double dx = var - x;
662  // Evaluate the third order polynomial
663  double weight = y+dx*(b+dx*(c+d*dx));
664  return weight;
665  }
