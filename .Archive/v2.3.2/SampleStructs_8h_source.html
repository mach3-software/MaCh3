256  int FindBin(const double KinVar,
257  const int NomBin,
258  const size_t N_Bins,
259  const std::vector<double>& Bin_Edges,
260  const std::vector<BinShiftLookup>& Bin_Lookup) const {
261  //DB Check to see if momentum shift has moved bins
262  //DB - First , check to see if the event is outside of the binning range and skip event if it is
263  if (KinVar < Bin_Edges[0] || KinVar >= Bin_Edges[N_Bins]) {
264  return M3::UnderOverFlowBin;
265  }
266 
267  // KS: Get reference to avoid repeated indexing and help with performance
268  const BinShiftLookup& _restrict_ Bin = Bin_Lookup[NomBin];
269  const double lower = Bin.lower_binedge;
270  const double upper = Bin.upper_binedge;
271  const double lower_lower = Bin.lower_lower_binedge;
272  const double upper_upper = Bin.upper_upper_binedge;
273 
274  //DB - Second, check to see if the event is still in the nominal bin
275  if (KinVar < upper && KinVar >= lower) {
276  return NomBin;
277  }
278  //DB - Thirdly, check the adjacent bins first as Eb+CC+EScale shifts aren't likely to move an Erec more than 1bin width
279  //Shifted down one bin from the event bin at nominal
280  if (KinVar < lower && KinVar >= lower_lower) {
281  return NomBin-1;
282  }
283  //Shifted up one bin from the event bin at nominal
284  if (KinVar < upper_upper && KinVar >= upper) {
285  return NomBin+1;
286  }
287  //DB - If we end up in this loop, the event has been shifted outside of its nominal bin, but is still within the allowed binning range
288  // KS: Perform binary search to find correct bin. We already checked if isn't outside of bounds
289  return static_cast<int>(std::distance(Bin_Edges.begin(), std::upper_bound(Bin_Edges.begin(), Bin_Edges.end(), KinVar)) - 1);
290  }

296  void InitialiseLookUpSingleDimension(std::vector<BinShiftLookup>& Bin_Lookup, const std::vector<double>& Bin_Edges, const size_t TotBins) {
297  Bin_Lookup.resize(TotBins);
298  //Set rw_pdf_bin and upper_binedge and lower_binedge for each skmc_base
299  for(size_t bin_i = 0; bin_i < TotBins; bin_i++){
300  double low_lower_edge = M3::_DEFAULT_RETURN_VAL_;
301  double low_edge = Bin_Edges[bin_i];
302  double upper_edge = Bin_Edges[bin_i+1];
303  double upper_upper_edge = M3::_DEFAULT_RETURN_VAL_;
304 
305  if (bin_i == 0) {
306  low_lower_edge = Bin_Edges[0];
307  } else {
308  low_lower_edge = Bin_Edges[bin_i-1];
309  }
310 
311  if (bin_i + 2 < TotBins) {
312  upper_upper_edge = Bin_Edges[bin_i + 2];
313  } else if (bin_i + 1 < TotBins) {
314  upper_upper_edge = Bin_Edges[bin_i + 1];
315  }
316 
317  Bin_Lookup[bin_i].lower_binedge = low_edge;
318  Bin_Lookup[bin_i].upper_binedge = upper_edge;
319  Bin_Lookup[bin_i].lower_lower_binedge = low_lower_edge;
320  Bin_Lookup[bin_i].upper_upper_binedge = upper_upper_edge;
321  }
322  }
