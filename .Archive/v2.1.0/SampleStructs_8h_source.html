241  inline int PDGToNuOscillatorFlavour(int NuPdg){
242  int NuOscillatorFlavour = M3::_BAD_INT_;
243  switch(std::abs(NuPdg)){
244  case NuPDG::kNue:
245  NuOscillatorFlavour = NuOscillator::kElectron;
246  break;
247  case NuPDG::kNumu:
248  NuOscillatorFlavour = NuOscillator::kMuon;
249  break;
250  case NuPDG::kNutau:
251  NuOscillatorFlavour = NuOscillator::kTau;
252  break;
253  default:
254  MACH3LOG_ERROR("Unknown Neutrino PDG {}, cannot convert to NuOscillator type", NuPdg);
255  break;
256  }
257 
258  //This is very cheeky but if the PDG is negative then multiply the PDG by -1
259  // This is consistent with the treatment that NuOscillator expects as enums only
260  // exist for the generic matter flavour and not the anti-matter version
261  if(NuPdg < 0){NuOscillatorFlavour *= -1;}
262 
263  return NuOscillatorFlavour;
264  }
