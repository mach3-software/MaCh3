29 inline double* MatrixMult(double *A, double *B, int n) {
30  //CW: First transpose to increse cache hits
31  double *BT = new double[n*n];
32  #ifdef MULTITHREAD
33  #pragma omp parallel for
34  #endif
35  for (int i = 0; i < n; i++) {
36  for (int j = 0; j < n; j++) {
37  BT[j*n+i] = B[i*n+j];
38  }
39  }
40 
41  // Now multiply
42  double *C = new double[n*n];
43  #ifdef MULTITHREAD
44  #pragma omp parallel for
45  #endif
46  for (int i = 0; i < n; i++) {
47  for (int j = 0; j < n; j++) {
48  double sum = 0;
49  for (int k = 0; k < n; k++) {
50  sum += A[i*n+k]*BT[j*n+k];
51  }
52  C[i*n+j] = sum;
53  }
54  }
55  delete BT;
56 
57  return C;
58 }

173  }

181  const std::vector<double>& Values,

184 // *************************************

186  YAML::Node systematics = NodeCopy["Systematics"];
187 
188  if (!systematics || !systematics.IsSequence()) {
189  MACH3LOG_ERROR("'Systematics' node is missing or not a sequence in the YAML copy");
190  throw MaCh3Exception(__FILE__, __LINE__);
191  }
192 
193  if (!FancyNames.empty() && FancyNames.size() != Values.size()) {
194  MACH3LOG_ERROR("Mismatch in sizes: FancyNames has {}, but Values has {}", FancyNames.size(), Values.size());
195  throw MaCh3Exception(__FILE__, __LINE__);
196  }
197 
198  if (FancyNames.empty() && systematics.size() != Values.size()) {
199  MACH3LOG_ERROR("Mismatch in sizes: Values has {}, but YAML 'Systematics' has {} entries",
200  Values.size(), systematics.size());
201  throw MaCh3Exception(__FILE__, __LINE__);
202  }
203 
204  if (!FancyNames.empty()) {
205  for (std::size_t i = 0; i < FancyNames.size(); ++i) {
206  bool matched = false;
207  for (std::size_t j = 0; j < systematics.size(); ++j) {
208  YAML::Node systematicNode = systematics[j]["Systematic"];
209  if (!systematicNode) continue;
210  auto nameNode = systematicNode["Names"];
211  if (!nameNode || !nameNode["FancyName"]) continue;
212  if (nameNode["FancyName"].as<std::string>() == FancyNames[i]) {
213  if (!systematicNode["ParameterValues"]) {
214  MACH3LOG_ERROR("Missing 'ParameterValues' for matched FancyName '{}'", FancyNames[i]);
215  throw MaCh3Exception(__FILE__, __LINE__);
216  }
217  systematicNode["ParameterValues"][Tune] = MaCh3Utils::FormatDouble(Values[i], 4);
218  matched = true;
219  break;
220  }
221  }
222  if (!matched) {
223  MACH3LOG_ERROR("Could not find a matching FancyName '{}' in the systematics", FancyNames[i]);
224  throw MaCh3Exception(__FILE__, __LINE__);
225  }
226  }
227  } else {
228  for (std::size_t i = 0; i < systematics.size(); ++i) {
229  YAML::Node systematicNode = systematics[i]["Systematic"];
230  if (!systematicNode || !systematicNode["ParameterValues"]) {
231  MACH3LOG_ERROR("Missing 'Systematic' or 'ParameterValues' entry at index {}", i);
232  throw MaCh3Exception(__FILE__, __LINE__);
233  }
234  systematicNode["ParameterValues"][Tune] = MaCh3Utils::FormatDouble(Values[i], 4);
235  }
236  }
237

239  std::string YAMLString = YAMLtoSTRING(NodeCopy);
240  FixSampleNamesQuotes(YAMLString);
241  // Write to output file

243  std::ofstream outFile(OutName);
244  if (!outFile) {
245  MACH3LOG_ERROR("Failed to open file for writing: {}", OutName);
246  throw MaCh3Exception(__FILE__, __LINE__);

248

251 }
