248 inline std::string GetTF1(const SplineInterpolation i) {
249  // **************************************************
250  std::string Func = "";
251  switch(i) {
252  case SplineInterpolation::kLinearFunc:

254  break;
255  case SplineInterpolation::kTSpline3:

258  case SplineInterpolation::kAkima:
259  case SplineInterpolation::kSplineInterpolations:
260  MACH3LOG_ERROR("Interpolation type {} not supported for TF1!", static_cast<int>(i));
261  throw MaCh3Exception(__FILE__, __LINE__);
262  default:
263  MACH3LOG_ERROR("UNKNOWN SPLINE INTERPOLATION SPECIFIED!");
264  MACH3LOG_ERROR("You gave {}", static_cast<int>(i));
265  throw MaCh3Exception(__FILE__ , __LINE__ );
266  }
