6std::unique_ptr<FitterBase> MaCh3FitterFactory(manager *fitMan, std::vector<samplePDFBase*>& Samples, std::vector<covarianceBase*>& Covariances) {
7// ********************************************
8 std::unique_ptr<FitterBase> MaCh3Fitter = nullptr;
9
10 auto Algorithm = GetFromManager<std::string>(fitMan->raw()["General"]["FittingAlgorithm"], "MCMC");
11
12 if(Algorithm == "MCMC") {
13 MaCh3Fitter = std::make_unique<mcmc>(fitMan);
14 } else if (Algorithm == "PSO") {
15 MaCh3Fitter = std::make_unique<PSO>(fitMan);
16 } else if (Algorithm == "Minuit2") {
17 #ifdef MaCh3_MINUIT2
18 MaCh3Fitter = std::make_unique<MinuitFit>(fitMan);
19 #else
20 MACH3LOG_ERROR("Trying to use Minuit2 however MaCh3 was compiled without Minuit2 support");
21 throw MaCh3Exception(__FILE__ , __LINE__ );
22 #endif
23 } else {
24 MACH3LOG_ERROR("You want to use algorithm {}, I don't recognize it, sry", Algorithm);
25 throw MaCh3Exception(__FILE__ , __LINE__ );
26 }
27
28 //KS: Adding samples and covariances to the Fitter class could be in the factory
29 for(unsigned int i = 0; Samples.size(); i++)
30 MaCh3Fitter->addSamplePDF(Samples[i]);
31 for(unsigned int i = 0; Covariances.size(); i++)
32 MaCh3Fitter->addSystObj(Covariances[i]);
33
34 return MaCh3Fitter;
35}

47

52
54 covarianceXsec* xsec = new covarianceXsec(CovMatrixFile, "xsec_cov", PCAThreshold, PCAParamRegion[0], PCAParamRegion[1]);
55
56 // Fill the parameter values with their nominal values
57 // should _ALWAYS_ be done before overriding with fix or flat
58 xsec->setParameters();
59
60 auto FixParams = GetFromManager<std::vector<std::string>>(Settings[std::string(PreFix) + "FixParams"], {});
61
62 // Fixed xsec parameters loop
63 if (FixParams.size() == 1 && FixParams.at(0) == "All") {
64 for (int j = 0; j < xsec->GetNumParams(); j++) {
65 xsec->toggleFixParameter(j);
66 }
67 } else {
68 for (unsigned int j = 0; j < FixParams.size(); j++) {
69 xsec->toggleFixParameter(FixParams.at(j));
70 }
71 }
72 //Global step scale for matrix
73 auto StepScale = Settings[std::string(PreFix) + "StepScale"].as<double>();

75 xsec->setStepScale(StepScale);

77 // Adaptive MCMC stuff

79 xsec->initialiseAdaption(FitManager->raw());
80
81 MACH3LOG_INFO("Factory successful");
