5std::unique_ptr<FitterBase> MaCh3FitterFactory(manager *fitMan) {
6// ********************************************
7 std::unique_ptr<FitterBase> MaCh3Fitter = nullptr;
8
9 auto Algorithm = GetFromManager<std::string>(fitMan->raw()["General"]["FittingAlgorithm"], "MCMC");
10
11 if(Algorithm == "MCMC") {
12 MaCh3Fitter = std::make_unique<mcmc>(fitMan);
13 } else if (Algorithm == "PSO") {
14 MaCh3Fitter = std::make_unique<PSO>(fitMan);
15 } else if (Algorithm == "Minuit2") {
16 #ifdef MaCh3_MINUIT2
17 MaCh3Fitter = std::make_unique<MinuitFit>(fitMan);
18 #else
19 MACH3LOG_ERROR("Trying to use Minuit2 however MaCh3 was compiled without Minuit2 support");
20 throw MaCh3Exception(__FILE__ , __LINE__ );
21 #endif
22 } else {
23 MACH3LOG_ERROR("You want to use algorithm {}, I don't recognize it, sry", Algorithm);
24 throw MaCh3Exception(__FILE__ , __LINE__ );
25 }
26 return MaCh3Fitter;
27}

32 if (argc < 2) {
33 MACH3LOG_ERROR("Wrong usage of MaCh3 executable!");

39 // Initialise manger responsible for config handling

42 //KS: Lambda to make sure we are not overwriting setting which should be committed
43 auto SanityOverwrite = [](const std::string& Name) {
44 if (Name.find("Systematics") != std::string::npos ||
45 Name.find("Samples") != std::string::npos)
46 {
47 MACH3LOG_CRITICAL("You are overwriting settings ({}) that are highly likely intended to be committed.", Name);
48 throw MaCh3Exception(__FILE__ , __LINE__ );

50 };
51
52 for (int i = 2; i < argc; ++i)
53 {
54 const std::string arg = argv[i];
55 const size_t colonCount = std::count(arg.begin(), arg.end(), ':');
56
58 if (colonCount == 1) {
59 const size_t firstColon = arg.find(':');
60 const std::string section = arg.substr(0, firstColon);
61 const std::string value = arg.substr(firstColon + 1);
62
63 MACH3LOG_INFO("Overriding setting: Section={}, Value={}", section, value);
64 SanityOverwrite(section);
65 FitManager->OverrideSettings(section, value);
66 } else if (colonCount == 2) {
67 const size_t firstColon = arg.find(':');
68 const size_t secondColon = arg.find(':', firstColon + 1);
69
70 const std::string section = arg.substr(0, firstColon);
71 const std::string key = arg.substr(firstColon + 1, secondColon - firstColon - 1);
72 const std::string value = arg.substr(secondColon + 1);
73
74 MACH3LOG_INFO("Overriding setting: Section={}, Key={}, Value={}", section, key, value);
75 SanityOverwrite(section);
76 SanityOverwrite(key);

78 } else if (colonCount == 3) {
79 const size_t firstColon = arg.find(':');

84 const std::string key = arg.substr(firstColon + 1, secondColon - firstColon - 1);

87
88 MACH3LOG_INFO("Overriding setting: Section={}, Key={}, Key={}, Value={}", section, key, key2, value);
89 SanityOverwrite(section);

92 FitManager->OverrideSettings(section, key, key2, value);
93 } else {
94 MACH3LOG_ERROR("Invalid override argument format: {}", arg);
95 MACH3LOG_ERROR("Expected format:Section:Key:Key:Valu, Section:Key:Value or Section:Value");
96 throw MaCh3Exception(__FILE__, __LINE__);
