257 inline void SetEntries(const int NewEntries) {
258 if (NewEntries > nEntries) {
259 MACH3LOG_ERROR("Cannot increase entries from {} to {}. Only decreasing is allowed.", nEntries, NewEntries);
260 throw MaCh3Exception(__FILE__, __LINE__);
261 }
262 if (NewEntries <= 0) {
263 MACH3LOG_ERROR("Entries cannot be below 0, but {} was passed.", NewEntries);
264 throw MaCh3Exception(__FILE__, __LINE__);
265 }
266
267 if (static_cast<int>(BurnInCut) > NewEntries) {
268 MACH3LOG_ERROR("BurnInCut ({}) is larger than NewEntries ({})", BurnInCut, NewEntries);
269 throw MaCh3Exception(__FILE__, __LINE__);
270 }
271
272 MACH3LOG_INFO("Setting entries to {} from {}.", NewEntries, nEntries);
273 MACH3LOG_WARN("This may behave not as expected when using merged multiple chains");
274 nEntries = NewEntries;
