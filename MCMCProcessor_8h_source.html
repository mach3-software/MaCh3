260 inline void SetEntries(const int NewEntries) {
261 if (NewEntries > nEntries) {
262 MACH3LOG_ERROR("Cannot increase entries from {} to {}. Only decreasing is allowed.", nEntries, NewEntries);
263 throw MaCh3Exception(__FILE__, __LINE__);
264 }
265 if (NewEntries <= 0) {
266 MACH3LOG_ERROR("Entries cannot be below 0, but {} was passed.", NewEntries);
267 throw MaCh3Exception(__FILE__, __LINE__);
268 }
269

272 throw MaCh3Exception(__FILE__, __LINE__);
273 }
274
275 MACH3LOG_INFO("Setting entries to {} from {}.", NewEntries, nEntries);
276 MACH3LOG_WARN("This may behave not as expected when using merged multiple chains");
277 nEntries = NewEntries;
