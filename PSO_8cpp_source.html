332double PSO::swarmIterate(){
333// *************************
334
335 std::vector<double> total_pos(fDim,0.0);
336
337 for (int i = 0; i < fParticles; ++i) {
338 std::vector<double> part1 = vector_multiply(system[i]->get_velocity(), fInertia);
339 std::vector<double> part2 = vector_multiply(vector_subtract(system[i]->get_personal_best_position(), system[i]->get_position()), (fOne * random->Rndm()));
340 std::vector<double> part3 = vector_multiply(vector_subtract(get_best_particle()->get_personal_best_position(), system[i]->get_position()),(fTwo * random->Rndm()));
341 std::vector<double> new_velocity = three_vector_addition(part1, part2, part3);
342 std::vector<double> new_pos = vector_add(system[i]->get_position(), new_velocity);
343 transform(total_pos.begin(), total_pos.end(), new_pos.begin(), total_pos.begin(),[](double x, double y) {return x+y;});
344
345 for (int j = 0; j < fDim; ++j) {
346 // Check if out of bounds and reflect if so
347 if(ranges_min[j] > new_pos[j]){
348 new_pos[j] = ranges_min[j];
349 }
350 else if(new_pos[j] > ranges_max[j]) {
351 new_pos[j] = ranges_max[j];
352 }
353 // If parameter fixed don't update it
354 if(fixed[j]) new_pos[j] = system[i]->get_position()[j];
355 }
356
357 if(fTestLikelihood){
358 double velo = 0.0;
359 for (int j = 0; j < fDim; ++j) {
360 paramlist[j][i] = new_pos[j];
361 velo += new_velocity[j]*new_velocity[j];
362 }
363 vel[i] = sqrt(velo);
364 }
365
366 system[i]->set_velocity(new_velocity);
367 system[i]->set_position(new_pos);
368 double new_value = CalcChi(new_pos);
369 if(new_value <= system[i]->get_personal_best_value()) {
370 system[i]->set_personal_best_value(new_value);
371 system[i]->set_personal_best_position(new_pos);
372 if(new_value < fBestValue){
373 fBestValue = new_value;
374 set_best_particle(system[i]);
375 }
376 }
377 }
378
379 std::vector<double> best_pos = get_best_particle()->get_personal_best_position();
380 std::vector<double> result(best_pos.size(), 0.0);
381 transform(total_pos.begin(), total_pos.end(), total_pos.begin(), [=](double val){return val/fParticles;});
382 transform(total_pos.begin(),total_pos.end(),best_pos.begin(),result.begin(),[](double x, double y) {return x-y;});
383
384 double mean_dist_sq = 0;
385 for (int i = 0; i<fDim; i++){
386 mean_dist_sq += result[i]*result[i];
387 }
388
389 return mean_dist_sq;
390}
