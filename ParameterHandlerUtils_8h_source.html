29 inline double* MatrixMult(double *A, double *B, int n) {
30 //CW: First transpose to increse cache hits
31 double *BT = new double[n*n];
32 #ifdef MULTITHREAD
33 #pragma omp parallel for
34 #endif
35 for (int i = 0; i < n; i++) {
36 for (int j = 0; j < n; j++) {
37 BT[j*n+i] = B[i*n+j];
38 }
39 }
40
41 // Now multiply
42 double *C = new double[n*n];
43 #ifdef MULTITHREAD
44 #pragma omp parallel for
45 #endif
46 for (int i = 0; i < n; i++) {
47 for (int j = 0; j < n; j++) {
48 double sum = 0;
49 for (int k = 0; k < n; k++) {
50 sum += A[i*n+k]*BT[j*n+k];
51 }
52 C[i*n+j] = sum;
53 }
54 }
55 delete BT;
56
57 return C;
58 }

61 inline double** MatrixMult(double **A, double **B, int n) {
62 // First make into monolithic array
63 double *A_mon = new double[n*n];
64 double *B_mon = new double[n*n];
65
66 #ifdef MULTITHREAD
67 #pragma omp parallel for
68 #endif
69 for (int i = 0; i < n; ++i) {
70 for (int j = 0; j < n; ++j) {
71 A_mon[i*n+j] = A[i][j];
72 B_mon[i*n+j] = B[i][j];
73 }
74 }
75 //CW: Now call the monolithic calculator
76 double *C_mon = MatrixMult(A_mon, B_mon, n);
77 delete A_mon;
78 delete B_mon;
79
80 // Return the double pointer
81 double **C = new double*[n];
82 #ifdef MULTITHREAD
83 #pragma omp parallel for
84 #endif
85 for (int i = 0; i < n; ++i) {
86 C[i] = new double[n];
87 for (int j = 0; j < n; ++j) {
88 C[i][j] = C_mon[i*n+j];
89 }
90 }
91 delete C_mon;
92
93 return C;
94 }

107inline void FixSampleNamesQuotes(std::string& yamlStr) {
108// *************************************
109 std::stringstream input(yamlStr);
110 std::string line;
111 std::string fixedYaml;
112 std::regex sampleNamesRegex(R"(SampleNames:\s*\[([^\]]+)\])");
113
114 while (std::getline(input, line)) {
115 std::smatch match;
116 if (std::regex_search(line, match, sampleNamesRegex)) {
117 std::string contents = match[1]; // inside the brackets
118 std::stringstream ss(contents);
119 std::string item;
120 std::vector<std::string> quotedItems;
121
122 while (std::getline(ss, item, ',')) {
123 item = std::regex_replace(item, std::regex(R"(^\s+|\s+$)"), ""); // trim
124 quotedItems.push_back("\"" + item + "\"");
125 }
126
127 std::string replacement = "SampleNames: [" + fmt::format("{}", fmt::join(quotedItems, ", ")) + "]";
128 line = std::regex_replace(line, sampleNamesRegex, replacement);
129 }
130 fixedYaml += line + "\n";
131 }
132
133 yamlStr = fixedYaml;
134}

138inline void AddTuneValues(YAML::Node& root,
139 const std::vector<double>& Values,
140 const std::string& Tune,
141 const std::vector<std::string>& FancyNames = {}) {
142// *************************************
143 YAML::Node NodeCopy = YAML::Clone(root);
144 YAML::Node systematics = NodeCopy["Systematics"];
145
146 if (!systematics || !systematics.IsSequence()) {
147 MACH3LOG_ERROR("'Systematics' node is missing or not a sequence in the YAML copy");
148 throw MaCh3Exception(__FILE__, __LINE__);
149 }
150
151 if (!FancyNames.empty() && FancyNames.size() != Values.size()) {
152 MACH3LOG_ERROR("Mismatch in sizes: FancyNames has {}, but Values has {}", FancyNames.size(), Values.size());
153 throw MaCh3Exception(__FILE__, __LINE__);
154 }
155
156 if (FancyNames.empty() && systematics.size() != Values.size()) {
157 MACH3LOG_ERROR("Mismatch in sizes: Values has {}, but YAML 'Systematics' has {} entries",
158 Values.size(), systematics.size());
159 throw MaCh3Exception(__FILE__, __LINE__);
160 }
161
162 if (!FancyNames.empty()) {
163 for (std::size_t i = 0; i < FancyNames.size(); ++i) {
164 bool matched = false;
165 for (std::size_t j = 0; j < systematics.size(); ++j) {
166 YAML::Node systematicNode = systematics[j]["Systematic"];
167 if (!systematicNode) continue;
168 auto nameNode = systematicNode["Names"];
169 if (!nameNode || !nameNode["FancyName"]) continue;
170 if (nameNode["FancyName"].as<std::string>() == FancyNames[i]) {
171 if (!systematicNode["ParameterValues"]) {
172 MACH3LOG_ERROR("Missing 'ParameterValues' for matched FancyName '{}'", FancyNames[i]);
173 throw MaCh3Exception(__FILE__, __LINE__);
174 }
175 systematicNode["ParameterValues"][Tune] = MaCh3Utils::FormatDouble(Values[i], 4);
176 matched = true;
177 break;
178 }
179 }
180 if (!matched) {
181 MACH3LOG_ERROR("Could not find a matching FancyName '{}' in the systematics", FancyNames[i]);
182 throw MaCh3Exception(__FILE__, __LINE__);
183 }
184 }
185 } else {
186 for (std::size_t i = 0; i < systematics.size(); ++i) {
187 YAML::Node systematicNode = systematics[i]["Systematic"];
188 if (!systematicNode || !systematicNode["ParameterValues"]) {
189 MACH3LOG_ERROR("Missing 'Systematic' or 'ParameterValues' entry at index {}", i);
190 throw MaCh3Exception(__FILE__, __LINE__);
191 }
192 systematicNode["ParameterValues"][Tune] = MaCh3Utils::FormatDouble(Values[i], 4);
193 }
194 }
195
196 // Convert updated copy to string
197 std::string YAMLString = YAMLtoSTRING(NodeCopy);
198 FixSampleNamesQuotes(YAMLString);
199 // Write to output file
200 std::string OutName = "UpdatedMatrixWithTune" + Tune + ".yaml";
201 std::ofstream outFile(OutName);
202 if (!outFile) {
203 MACH3LOG_ERROR("Failed to open file for writing: {}", OutName);
204 throw MaCh3Exception(__FILE__, __LINE__);
205 }
206
207 outFile << YAMLString;
208 outFile.close();
209}
