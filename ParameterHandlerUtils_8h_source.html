30inline double* MatrixMult(double *A, double *B, int n) {
31 //CW: First transpose to increse cache hits
32 double *BT = new double[n*n];
33 #ifdef MULTITHREAD
34 #pragma omp parallel for
35 #endif
36 for (int i = 0; i < n; i++) {
37 for (int j = 0; j < n; j++) {
38 BT[j*n+i] = B[i*n+j];
39 }
40 }
41
42 // Now multiply
43 double *C = new double[n*n];
44 #ifdef MULTITHREAD
45 #pragma omp parallel for
46 #endif
47 for (int i = 0; i < n; i++) {
48 for (int j = 0; j < n; j++) {
49 double sum = 0;
50 for (int k = 0; k < n; k++) {
51 sum += A[i*n+k]*BT[j*n+k];
52 }
53 C[i*n+j] = sum;
54 }
55 }
56 delete BT;
57
58 return C;
59}

62inline double** MatrixMult(double **A, double **B, int n) {
63 // First make into monolithic array
64 double *A_mon = new double[n*n];
65 double *B_mon = new double[n*n];
66
67 #ifdef MULTITHREAD
68 #pragma omp parallel for
69 #endif
70 for (int i = 0; i < n; ++i) {
71 for (int j = 0; j < n; ++j) {
72 A_mon[i*n+j] = A[i][j];
73 B_mon[i*n+j] = B[i][j];
74 }
75 }
76 //CW: Now call the monolithic calculator
77 double *C_mon = MatrixMult(A_mon, B_mon, n);
78 delete A_mon;
79 delete B_mon;
80
81 // Return the double pointer
82 double **C = new double*[n];
83 #ifdef MULTITHREAD
84 #pragma omp parallel for
85 #endif
86 for (int i = 0; i < n; ++i) {
87 C[i] = new double[n];
88 for (int j = 0; j < n; ++j) {
89 C[i][j] = C_mon[i*n+j];
90 }
91 }
92 delete C_mon;
93
94 return C;
95}

151inline void FixSampleNamesQuotes(std::string& yamlStr) {
152// *************************************
153 std::stringstream input(yamlStr);
154 std::string line;
155 std::string fixedYaml;
156 std::regex sampleNamesRegex(R"(SampleNames:\s*\[([^\]]+)\])");
157
158 while (std::getline(input, line)) {
159 std::smatch match;
160 if (std::regex_search(line, match, sampleNamesRegex)) {
161 std::string contents = match[1]; // inside the brackets
162 std::stringstream ss(contents);
163 std::string item;
164 std::vector<std::string> quotedItems;
165
166 while (std::getline(ss, item, ',')) {
167 item = std::regex_replace(item, std::regex(R"(^\s+|\s+$)"), ""); // trim
168 quotedItems.push_back("\"" + item + "\"");
169 }
170
171 std::string replacement = "SampleNames: [" + fmt::format("{}", fmt::join(quotedItems, ", ")) + "]";
172 line = std::regex_replace(line, sampleNamesRegex, replacement);
173 }
174 fixedYaml += line + "\n";
175 }
176
177 yamlStr = fixedYaml;
178}

182inline void AddTuneValues(YAML::Node& root,
183 const std::vector<double>& Values,
184 const std::string& Tune,
185 const std::vector<std::string>& FancyNames = {}) {
186// *************************************
187 YAML::Node NodeCopy = YAML::Clone(root);
188 YAML::Node systematics = NodeCopy["Systematics"];
189
190 if (!systematics || !systematics.IsSequence()) {
191 MACH3LOG_ERROR("'Systematics' node is missing or not a sequence in the YAML copy");
192 throw MaCh3Exception(__FILE__, __LINE__);
193 }
194
195 if (!FancyNames.empty() && FancyNames.size() != Values.size()) {
196 MACH3LOG_ERROR("Mismatch in sizes: FancyNames has {}, but Values has {}", FancyNames.size(), Values.size());
197 throw MaCh3Exception(__FILE__, __LINE__);
198 }
199
200 if (FancyNames.empty() && systematics.size() != Values.size()) {
201 MACH3LOG_ERROR("Mismatch in sizes: Values has {}, but YAML 'Systematics' has {} entries",
202 Values.size(), systematics.size());
203 throw MaCh3Exception(__FILE__, __LINE__);
204 }
205
206 if (!FancyNames.empty()) {
207 for (std::size_t i = 0; i < FancyNames.size(); ++i) {
208 bool matched = false;
209 for (std::size_t j = 0; j < systematics.size(); ++j) {
210 YAML::Node systematicNode = systematics[j]["Systematic"];
211 if (!systematicNode) continue;
212 auto nameNode = systematicNode["Names"];
213 if (!nameNode || !nameNode["FancyName"]) continue;
214 if (nameNode["FancyName"].as<std::string>() == FancyNames[i]) {
215 if (!systematicNode["ParameterValues"]) {
216 MACH3LOG_ERROR("Missing 'ParameterValues' for matched FancyName '{}'", FancyNames[i]);
217 throw MaCh3Exception(__FILE__, __LINE__);
218 }
219 systematicNode["ParameterValues"][Tune] = MaCh3Utils::FormatDouble(Values[i], 4);
220 matched = true;
221 break;
222 }
223 }
224 if (!matched) {
225 MACH3LOG_ERROR("Could not find a matching FancyName '{}' in the systematics", FancyNames[i]);
226 throw MaCh3Exception(__FILE__, __LINE__);
227 }
228 }
229 } else {
230 for (std::size_t i = 0; i < systematics.size(); ++i) {
231 YAML::Node systematicNode = systematics[i]["Systematic"];
232 if (!systematicNode || !systematicNode["ParameterValues"]) {
233 MACH3LOG_ERROR("Missing 'Systematic' or 'ParameterValues' entry at index {}", i);
234 throw MaCh3Exception(__FILE__, __LINE__);
235 }
236 systematicNode["ParameterValues"][Tune] = MaCh3Utils::FormatDouble(Values[i], 4);
237 }
238 }
239
240 // Convert updated copy to string
241 std::string YAMLString = YAMLtoSTRING(NodeCopy);
242 FixSampleNamesQuotes(YAMLString);
243 // Write to output file
244 std::string OutName = "UpdatedMatrixWithTune" + Tune + ".yaml";
245 std::ofstream outFile(OutName);
246 if (!outFile) {
247 MACH3LOG_ERROR("Failed to open file for writing: {}", OutName);
248 throw MaCh3Exception(__FILE__, __LINE__);
249 }
250
251 outFile << YAMLString;
252 outFile.close();
253}

257inline void MakeCorrelationMatrix(YAML::Node& root,
258 const std::vector<double>& Values,
259 const std::vector<double>& Errors,
260 const std::vector<std::vector<double>>& Correlation,
261 const std::string& OutYAMLName,
262 const std::vector<std::string>& FancyNames = {}) {
263// *************************************
264 if (Values.size() != Errors.size() || Values.size() != Correlation.size()) {
265 MACH3LOG_ERROR("Size mismatch between Values, Errors, and Correlation matrix");
266 throw MaCh3Exception(__FILE__, __LINE__);
267 }
268
269 for (const auto& row : Correlation) {
270 if (row.size() != Correlation.size()) {
271 MACH3LOG_ERROR("Correlation matrix is not square");
272 throw MaCh3Exception(__FILE__, __LINE__);
273 }
274 }
275
276 YAML::Node NodeCopy = YAML::Clone(root);
277 YAML::Node systematics = NodeCopy["Systematics"];
278
279 if (!systematics || !systematics.IsSequence()) {
280 MACH3LOG_ERROR("'Systematics' node is missing or not a sequence");
281 throw MaCh3Exception(__FILE__, __LINE__);
282 }
283
284 if (!FancyNames.empty() && FancyNames.size() != Values.size()) {
285 MACH3LOG_ERROR("FancyNames size ({}) does not match Values size ({})", FancyNames.size(), Values.size());
286 throw MaCh3Exception(__FILE__, __LINE__);
287 }
288
289 // Map from FancyName to Systematic node
290 std::unordered_map<std::string, YAML::Node> nameToNode;
291 for (std::size_t i = 0; i < systematics.size(); ++i) {
292 YAML::Node syst = systematics[i]["Systematic"];
293 if (!syst || !syst["Names"] || !syst["Names"]["FancyName"]) continue;
294 std::string name = syst["Names"]["FancyName"].as<std::string>();
295 nameToNode[name] = syst;
296 }
297
298 if (!FancyNames.empty()) {
299 for (std::size_t i = 0; i < FancyNames.size(); ++i) {
300 const std::string& name_i = FancyNames[i];
301 auto it_i = nameToNode.find(name_i);
302 if (it_i == nameToNode.end()) {
303 MACH3LOG_ERROR("Could not find FancyName '{}' in YAML", name_i);
304 throw MaCh3Exception(__FILE__, __LINE__);
305 }
306 YAML::Node& syst_i = it_i->second;
307
308 syst_i["ParameterValues"]["PreFitValue"] = MaCh3Utils::FormatDouble(Values[i], 4);
309 syst_i["Error"] = std::round(Errors[i] * 100.0) / 100.0;
310
311 YAML::Node correlationsNode = YAML::Node(YAML::NodeType::Sequence);
312 for (std::size_t j = 0; j < FancyNames.size(); ++j) {
313 if (i == j) continue;
314 YAML::Node singleEntry;
315 singleEntry[FancyNames[j]] = MaCh3Utils::FormatDouble(Correlation[i][j], 4);
316 correlationsNode.push_back(singleEntry);
317 }
318 syst_i["Correlations"] = correlationsNode;
319 }
320 } else {
321 if (systematics.size() != Values.size()) {
322 MACH3LOG_ERROR("Mismatch in sizes: Values has {}, but YAML 'Systematics' has {} entries",
323 Values.size(), systematics.size());
324 throw MaCh3Exception(__FILE__, __LINE__);
325 }
326
327 for (std::size_t i = 0; i < systematics.size(); ++i) {
328 YAML::Node syst = systematics[i]["Systematic"];
329 if (!syst) {
330 MACH3LOG_ERROR("Missing 'Systematic' node at index {}", i);
331 throw MaCh3Exception(__FILE__, __LINE__);
332 }
333
334 syst["ParameterValues"]["PreFitValue"] = MaCh3Utils::FormatDouble(Values[i], 4);
335 syst["Error"] = std::round(Errors[i] * 100.0) / 100.0;
336
337 YAML::Node correlationsNode = YAML::Node(YAML::NodeType::Sequence);
338 for (std::size_t j = 0; j < Correlation[i].size(); ++j) {
339 if (i == j) continue;
340 YAML::Node singleEntry;
341 const std::string& otherName = systematics[j]["Systematic"]["Names"]["FancyName"].as<std::string>();
342 singleEntry[otherName] = MaCh3Utils::FormatDouble(Correlation[i][j], 4);
343 correlationsNode.push_back(singleEntry);
344 }
345 syst["Correlations"] = correlationsNode;
346 }
347 }
348
349 // Convert and write
350 std::string YAMLString = YAMLtoSTRING(NodeCopy);
351 FixSampleNamesQuotes(YAMLString);
352 std::ofstream outFile(OutYAMLName);
353 if (!outFile) {
354 MACH3LOG_ERROR("Failed to open file for writing: {}", OutYAMLName);
355 throw MaCh3Exception(__FILE__, __LINE__);
356 }
357
358 outFile << YAMLString;
359 outFile.close();
360}

367inline TMacro* GetConfigMacroFromChain(TDirectory* CovarianceFolder) {
368// *************************************
369 if (!CovarianceFolder) {
370 MACH3LOG_ERROR("Null TDirectory passed to {}", __func__);
371 throw MaCh3Exception(__FILE__, __LINE__);
372 }
373
374 TMacro* foundMacro = nullptr;
375 int macroCount = 0;
376
377 TIter next(CovarianceFolder->GetListOfKeys());
378 TKey* key;
379 while ((key = dynamic_cast<TKey*>(next()))) {
380 if (std::string(key->GetClassName()) == "TMacro") {
381 ++macroCount;
382 if (macroCount == 1) {
383 foundMacro = dynamic_cast<TMacro*>(key->ReadObj());
384 }
385 }
386 }
387
388 if (macroCount == 1 && foundMacro) {
389 MACH3LOG_INFO("Found single TMacro in directory: using it.");
390 return foundMacro;
391 } else {
392 MACH3LOG_WARN("Found {} TMacro objects. Using hardcoded macro name: Config_xsec_cov.", macroCount);
393 TMacro* fallback = CovarianceFolder->Get<TMacro>("Config_xsec_cov");
394 if (!fallback) {
395 MACH3LOG_WARN("Fallback macro 'Config_xsec_cov' not found in directory.");
396 }
397 return fallback;
398 }
399}
