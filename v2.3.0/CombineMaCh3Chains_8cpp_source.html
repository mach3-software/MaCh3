157 bool CheckFolder(TFile* file, TFile* prevFile, const std::string& FolderName, const std::vector<std::string>& SkipVector = {})
158 {
159  bool mismatch = false;
160  TDirectory* dir = file->GetDirectory(FolderName.c_str());
161  TDirectory* prevDir = prevFile->GetDirectory(FolderName.c_str());
162 
163  if (!dir || !prevDir) {
164  MACH3LOG_ERROR("Could not find folder '{}' in one or both files", FolderName);
165  return true;
166  }
167 
168  TIter nextKey(dir->GetListOfKeys());
169  TKey* key;
170 
171  while ((key = static_cast<TKey*>(nextKey()))) {
172  const std::string objName = key->GetName();
173  TObject* obj = key->ReadObj();
174  if (!obj) continue;
175 
176  // Handle TH1 comparison
177  if (obj->InheritsFrom("TH1")) {
178  TH1* hist = static_cast<TH1*>(obj);
179  TH1* prevHist = dynamic_cast<TH1*>(prevDir->Get(objName.c_str()));
180  if (!prevHist) {
181  MACH3LOG_ERROR("Missing histogram '{}' in previous file (folder '{}')", objName, FolderName);
182  mismatch = true;
183  continue;
184  }
185  if (!CompareHistograms(hist, prevHist, objName, FolderName)) {
186  mismatch = true;
187  }
188  }
189  // Handle TMacro comparison
190  else if (obj->InheritsFrom("TMacro")) {
191  TMacro* macro = static_cast<TMacro*>(obj);
192  TMacro* prevMacro = dynamic_cast<TMacro*>(prevDir->Get(objName.c_str()));
193  if (!prevMacro) {
194  MACH3LOG_ERROR("Missing TMacro '{}' in previous file (folder '{}')", objName, FolderName);
195  mismatch = true;
196  continue;
197  }
198  if (!CompareTwoConfigs(TMacroToString(*macro), TMacroToString(*prevMacro), SkipVector)) {
199  mismatch = true;
200  }
201  }
202  }
203  return mismatch;
