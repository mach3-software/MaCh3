18double GetMinimumInRange(TH1D *hist, double minRange, double maxRange)
19{
20 double MinVale = 1234567890.;
21 TAxis* xaxis = hist->GetXaxis();
22 for(int x = 1; x <= hist->GetNbinsX(); x++)
23 {
24 if(xaxis->GetBinLowEdge(x)>minRange && xaxis->GetBinUpEdge(x)<maxRange){
25 if(MinVale > hist->GetBinContent(x))MinVale = hist->GetBinContent(x);
26 }
27 }
28 return MinVale;
29}

33bool IsHistogramAllOnes(TH1D *hist, double tolerance = 0.001, int max_failures = 100)
34{
35 int failure_count = 0;
36
37 for (int bin = 2; bin <= hist->GetNbinsX(); ++bin)
38 {
39 if (fabs(hist->GetBinContent(bin) - 1.0) > tolerance)
40 {
41 if (++failure_count > max_failures)
42 {
43 return false;
44 }
45 }
46 }
47 return true;
48}

217std::pair<TGraph *, TGraph *> CreateMinMaxBand(TH1D *hist, Color_t color)
218{
219 int nBins = hist->GetNbinsX();
220 std::vector<double> x(nBins), ymin(nBins), ymax(nBins);
221
222 for (int i = 0; i < nBins; ++i)
223 {
224 x[i] = hist->GetBinCenter(i + 1);
225 ymin[i] = hist->GetBinContent(i + 1);
226 ymax[i] = hist->GetBinContent(i + 1);
227 }
228
229 TGraph *minGraph = new TGraph(nBins, x.data(), ymin.data());
230 TGraph *maxGraph = new TGraph(nBins, x.data(), ymax.data());
231
232 minGraph->SetLineColor(color);
233 maxGraph->SetLineColor(color);
234 minGraph->SetFillColorAlpha(color, float(0.3));
235 maxGraph->SetFillColorAlpha(color, float(0.3));
236
237 return {minGraph, maxGraph};
238}

240TGraph* CalculateMinMaxBand(const std::vector<TH1D *> &histograms, Color_t color)
241{
242 if (histograms.empty())
243 {
244 throw std::invalid_argument("Empty histogram vector provided");
245 }
246
247 int nBins = histograms[0]->GetNbinsX();
248 std::vector<double> x(nBins), ymin(nBins, 1e10), ymax(nBins, -1e10);
249
250 for (int bin = 0; bin < nBins; bin++)
251 {
252 x[bin] = histograms[0]->GetBinCenter(bin + 1);
253
254 for (const auto &hist : histograms)
255 {
256 double content = hist->GetBinContent(bin + 1);
257 ymin[bin] = std::min(ymin[bin], content);
258 ymax[bin] = std::max(ymax[bin], content);
259 }
260 }
261
262
263 // Create a band using TGraphAsymmErrors for the shaded region
264 TGraphAsymmErrors *band = new TGraphAsymmErrors(nBins);
265 for (int i = 0; i < nBins; ++i)
266 {
267 band->SetPoint(i, x[i], (ymax[i] + ymin[i]) / 2.0);
268 band->SetPointError(i, 0, 0, (ymax[i] - ymin[i]) / 2.0, (ymax[i] - ymin[i]) / 2.0);
269 }
270
271 band->SetFillColorAlpha(color, float(0.2));
272 band->SetLineWidth(1);
273
274 return band; // Return band and min graph (min graph can be used for legend)
275}

278std::pair<TH1D *, TH1D *> CalculateMinMaxHistograms(const std::vector<TH1D *> &histograms)
279{
280 if (histograms.empty())
281 {
282 throw std::invalid_argument("Empty histogram vector provided");
283 }
284
285 TH1D *min_hist = static_cast<TH1D *>(histograms[0]->Clone());
286 TH1D *max_hist = static_cast<TH1D *>(histograms[0]->Clone());
287
288 for (const auto &hist : histograms)
289 {
290 for (int bin = 1; bin <= min_hist->GetNbinsX(); ++bin)
291 {
292 double current_min = min_hist->GetBinContent(bin);
293 double current_max = max_hist->GetBinContent(bin);
294 double bin_content = hist->GetBinContent(bin);
295
296 min_hist->SetBinContent(bin, std::min(current_min, bin_content));
297 max_hist->SetBinContent(bin, std::max(current_max, bin_content));
298 }
299 }
300
301 return {min_hist, max_hist};
302}

305void ProcessAutoCorrelationDirectory(TDirectoryFile *autocor_dir,
306 TH1D *&average_hist,
307 int &parameter_count,
308 std::vector<TH1D *> &histograms)
309{
310 TIter next(autocor_dir->GetListOfKeys());
311 TKey *key;
312
313 while ((key = dynamic_cast<TKey *>(next())))
314 {
315 TH1D *current_hist = nullptr;
316 autocor_dir->GetObject(key->GetName(), current_hist);
317
318 if (!current_hist ||
319 current_hist->GetMaximum() <= 0 ||
320 IsHistogramAllOnes(current_hist))
321 {
322 continue;
323 }
324
325 current_hist->SetDirectory(nullptr); // Detach from file
326 histograms.push_back(current_hist);
327
328 if (!average_hist)
329 {
330 average_hist = static_cast<TH1D *>(current_hist->Clone());
331 average_hist->SetDirectory(nullptr);
332 }
333 else
334 {
335 average_hist->Add(current_hist);
336 }
337 parameter_count++;
338 }
339}
