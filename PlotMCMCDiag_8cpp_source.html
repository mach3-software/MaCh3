13double GetMinimumInRange(TH1D *hist, double minRange, double maxRange)
14{
15 double MinVale = 1234567890.;
16 TAxis* xaxis = hist->GetXaxis();
17 for(int x = 1; x <= hist->GetNbinsX(); x++)
18 {
19 if(xaxis->GetBinLowEdge(x)>minRange && xaxis->GetBinUpEdge(x)<maxRange){
20 if(MinVale > hist->GetBinContent(x))MinVale = hist->GetBinContent(x);
