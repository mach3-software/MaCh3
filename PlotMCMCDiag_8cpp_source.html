265 std::unique_ptr<TGraphAsymmErrors> CalculateMinMaxBand(const std::vector<std::unique_ptr<TH1D>> &histograms, Color_t color)
266 {
267  if (histograms.empty()) {
268  throw MaCh3Exception(__FILE__, __LINE__, "Empty histogram vector provided");
269  }
270 
271  int nBins = histograms[0]->GetNbinsX();
272  std::vector<double> x(nBins), ymin(nBins, 1e10), ymax(nBins, -1e10);
273 
274  for (int bin = 0; bin < nBins; bin++)
275  {
276  x[bin] = histograms[0]->GetBinCenter(bin + 1);
277 
278  for (const auto &hist : histograms)
279  {
280  double content = hist->GetBinContent(bin + 1);
281  ymin[bin] = std::min(ymin[bin], content);
282  ymax[bin] = std::max(ymax[bin], content);

284  }
285 
286  // Create a band using TGraphAsymmErrors for the shaded region
287  auto band = std::make_unique<TGraphAsymmErrors>(nBins);
288  for (int i = 0; i < nBins; ++i)
289  {
290  band->SetPoint(i, x[i], (ymax[i] + ymin[i]) / 2.0);
291  band->SetPointError(i, 0, 0, (ymax[i] - ymin[i]) / 2.0, (ymax[i] - ymin[i]) / 2.0);
292  }
293 
294  band->SetFillColorAlpha(color, 0.2f);
295  band->SetLineWidth(1);
296
