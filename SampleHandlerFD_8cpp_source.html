406 if (normweight <= 0.){

408 continue;
409 }

411 CalcWeightFunc(iEvent);
412
413 MCSamples.xsec_w[iEvent] = splineweight*normweight;
414
415 //DB Total weight
416 totalweight = GetEventWeight(iEvent);
417 //DB Catch negative weights and skip any event with a negative event
418 if (totalweight <= 0.){
419 MCSamples.xsec_w[iEvent] = 0.;
420 continue;
421 }
422 //DB Switch on BinningOpt to allow different binning options to be implemented
423 //The alternative would be to have inheritance based on BinningOpt
424 double XVar = *(MCSamples.x_var[iEvent]);
425
426 //DB Find the relevant bin in the PDF for each event
427 int XBinToFill = -1;

429
430 //DB - First, check to see if the event is still in the nominal bin
431 if (XVar < MCSamples.rw_upper_xbinedge[iEvent] && XVar >= MCSamples.rw_lower_xbinedge[iEvent]) {
432 XBinToFill = MCSamples.NomXBin[iEvent];
433 }
434 //DB - Second, check to see if the event is outside of the binning range and skip event if it is
435 //ETA- note that nXBins is XBinEdges.size() - 1
436 else if (XVar < XBinEdges[0] || XVar >= XBinEdges[nXBins]) {

438 }
439 //DB - Thirdly, check the adjacent bins first as Eb+CC+EScale shifts aren't likely to move an Erec more than 1bin width
440 //Shifted down one bin from the event bin at nominal
441 else if (XVar < MCSamples.rw_lower_xbinedge[iEvent] && XVar >= MCSamples.rw_lower_lower_xbinedge[iEvent]) {
442 XBinToFill = MCSamples.NomXBin[iEvent]-1;
443 }

446 XBinToFill = MCSamples.NomXBin[iEvent]+1;
447 }

450 for (unsigned int iBin=0;iBin<(XBinEdges.size()-1);iBin++)
451 {
452 if (XVar >= XBinEdges[iBin] && XVar < XBinEdges[iBin+1]) {
453 XBinToFill = iBin;
454 }
455 }
456 }
457
458 //DB Fill relevant part of thread array
459 if (XBinToFill != -1 && YBinToFill != -1) {
460 SampleHandlerFD_array[YBinToFill][XBinToFill] += totalweight;
461 if (FirstTimeW2) SampleHandlerFD_array_w2[YBinToFill][XBinToFill] += totalweight*totalweight;
462 }
463 }

673 funcParsGrid.resize(GetNEvents());
674
675 // For every functional parameter in XsecCov that matches the name in funcParsNames, add it to the map
676 for (FunctionalParameter & fp : funcParsVec) {
677 for (std::string name : funcParsNamesVec) {
678 if (fp.name == name) {

680 fp.funcPtr = &funcParsFuncMap[funcParsNamesMap[fp.name]];

724 }

867 }

899 #ifdef DEBUG

901 #endif
902 //}
