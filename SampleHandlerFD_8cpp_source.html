601void SampleHandlerFD::SetupFunctionalParameters() {
602 funcParsVec = ParHandler->GetFunctionalParametersFromSampleName(SampleName);
603 // RegisterFunctionalParameters is implemented in experiment-specific code,
604 // which calls RegisterIndividualFuncPar to populate funcParsNamesMap, funcParsNamesVec, and funcParsFuncMap
605 RegisterFunctionalParameters();
606 funcParsMap.resize(funcParsNamesMap.size());
607 funcParsGrid.resize(GetNEvents());
608
609 // For every functional parameter in XsecCov that matches the name in funcParsNames, add it to the map

611 for (std::string name : funcParsNamesVec) {
612 if (fp.name == name) {

621 MACH3LOG_ERROR("Functional parameter {} not found, did you define it in RegisterFunctionalParameters()?", fp.name);

627 // For each event, make a vector of pointers to the functional parameters
628 for (std::size_t iEvent = 0; iEvent < static_cast<std::size_t>(GetNEvents()); ++iEvent) {
629 // Now loop over the functional parameters and get a vector of enums corresponding to the functional parameters
630 for (std::vector<FunctionalParameter>::iterator it = funcParsVec.begin(); it != funcParsVec.end(); ++it) {
631 // Check whether the interaction modes match
632 bool ModeMatch = MatchCondition((*it).modes, static_cast<int>(std::round(*(MCSamples[iEvent].mode))));
633 if (!ModeMatch) {
634 MACH3LOG_TRACE("Event {}, missed Mode check ({}) for dial {}", iEvent, *(MCSamples[iEvent].mode), (*it).name);
635 continue;
636 }
637 // Now check whether within kinematic bounds
638 bool IsSelected = true;
639 if ((*it).hasKinBounds) {
640 const auto& kinVars = (*it).KinematicVarStr;
641 const auto& selection = (*it).Selection;
642
643 for (std::size_t iKinPar = 0; iKinPar < kinVars.size(); ++iKinPar) {
644 const double kinVal = ReturnKinematicParameter(kinVars[iKinPar], static_cast<int>(iEvent));
645
646 bool passedAnyBound = false;
647 const auto& boundsList = selection[iKinPar];
648
649 for (const auto& bounds : boundsList) {
650 if (kinVal > bounds[0] && kinVal <= bounds[1]) {
651 passedAnyBound = true;
652 break;
653 }
654 }
655
656 if (!passedAnyBound) {
657 MACH3LOG_TRACE("Event {}, missed kinematic check ({}) for dial {}",
658 iEvent, kinVars[iKinPar], (*it).name);
659 IsSelected = false;
660 break;
661 }
662 }
663 }
664 // Need to then break the event loop
665 if(!IsSelected){
666 MACH3LOG_TRACE("Event {}, missed Kinematic var check for dial {}", iEvent, (*it).name);
667 continue;
668 }
669 auto funcparenum = funcParsNamesMap[(*it).name];
670 funcParsGrid.at(iEvent).push_back(funcparenum);
671 }
672 }
673 MACH3LOG_INFO("Finished setting up functional parameters");
674}

767//A way to check whether a normalisation parameter applies to an event or not
768void SampleHandlerFD::CalcNormsBins(std::vector<NormParameter>& norm_parameters, std::vector< std::vector< int > >& xsec_norms_bins) {
769// ************************************************
770 #ifdef DEBUG
771 std::vector<int> VerboseCounter(norm_parameters.size(), 0);
772 #endif
773 for(unsigned int iEvent = 0; iEvent < GetNEvents(); ++iEvent){
774 std::vector< int > NormBins = {};
775 if (ParHandler) {
776 // Skip oscillated NC events
777 // Not strictly needed, but these events don't get included in oscillated predictions, so
778 // no need to waste our time calculating and storing information about xsec parameters
779 // that will never be used.
780 if (MCSamples[iEvent].isNC && (*MCSamples[iEvent].nupdg != *MCSamples[iEvent].nupdgUnosc) ) {
781 MACH3LOG_TRACE("Event {}, missed NC/signal check", iEvent);
782 continue;
783 } //DB Abstract check on MaCh3Modes to determine which apply to neutral current
784 for (std::vector<NormParameter>::iterator it = norm_parameters.begin(); it != norm_parameters.end(); ++it) {
785 //Now check that the target of an interaction matches with the normalisation parameters
786 bool TargetMatch = MatchCondition((*it).targets, *(MCSamples[iEvent].Target));
787 if (!TargetMatch) {
788 MACH3LOG_TRACE("Event {}, missed target check ({}) for dial {}", iEvent, *(MCSamples[iEvent].Target), (*it).name);
789 continue;
790 }
791
792 //Now check that the neutrino flavour in an interaction matches with the normalisation parameters
793 bool FlavourMatch = MatchCondition((*it).pdgs, *(MCSamples[iEvent].nupdg));
794 if (!FlavourMatch) {
795 MACH3LOG_TRACE("Event {}, missed PDG check ({}) for dial {}", iEvent,(*MCSamples[iEvent].nupdg), (*it).name);
796 continue;
797 }
798
799 //Now check that the unoscillated neutrino flavour in an interaction matches with the normalisation parameters
800 bool FlavourUnoscMatch = MatchCondition((*it).preoscpdgs, *(MCSamples[iEvent].nupdgUnosc));
801 if (!FlavourUnoscMatch){
802 MACH3LOG_TRACE("Event {}, missed FlavourUnosc check ({}) for dial {}", iEvent,(*MCSamples[iEvent].nupdgUnosc), (*it).name);
803 continue;
804 }
805
806 //Now check that the mode of an interaction matches with the normalisation parameters
807 bool ModeMatch = MatchCondition((*it).modes, static_cast<int>(std::round(*(MCSamples[iEvent].mode))));
808 if (!ModeMatch) {
809 MACH3LOG_TRACE("Event {}, missed Mode check ({}) for dial {}", iEvent, *(MCSamples[iEvent].mode), (*it).name);
810 continue;
811 }
812
813 //Now check whether the norm has kinematic bounds
814 //i.e. does it only apply to events in a particular kinematic region?
815 // Now check whether within kinematic bounds
816 bool IsSelected = true;
817 if ((*it).hasKinBounds) {
818 const auto& kinVars = (*it).KinematicVarStr;
819 const auto& selection = (*it).Selection;
820
821 for (std::size_t iKinPar = 0; iKinPar < kinVars.size(); ++iKinPar) {
822 const double kinVal = ReturnKinematicParameter(kinVars[iKinPar], static_cast<int>(iEvent));
823
824 bool passedAnyBound = false;
825 const auto& boundsList = selection[iKinPar];
826
827 for (const auto& bounds : boundsList) {
828 if (kinVal > bounds[0] && kinVal <= bounds[1]) {
829 passedAnyBound = true;
830 break;
831 }
832 }
833
834 if (!passedAnyBound) {
835 MACH3LOG_TRACE("Event {}, missed kinematic check ({}) for dial {}",
836 iEvent, kinVars[iKinPar], (*it).name);
837 IsSelected = false;
838 break;
839 }
840 }
841 }
842 // Need to then break the event loop
843 if(!IsSelected){
844 MACH3LOG_TRACE("Event {}, missed Kinematic var check for dial {}", iEvent, (*it).name);
845 continue;
846 }
847 // Now set 'index bin' for each normalisation parameter
848 // All normalisations are just 1 bin for 2015, so bin = index (where index is just the bin for that normalisation)
849 int bin = (*it).index;
850
851 NormBins.push_back(bin);
852 MACH3LOG_TRACE("Event {}, will be affected by dial {}", iEvent, (*it).name);
853 #ifdef DEBUG
854 VerboseCounter[std::distance(norm_parameters.begin(), it)]++;
855 #endif
856 //}
857 } // end iteration over norm_parameters
858 } // end if (ParHandler)
859 xsec_norms_bins[iEvent] = NormBins;
860 }//end loop over events
861 #ifdef DEBUG
862 MACH3LOG_DEBUG("┌──────────────────────────────────────────────────────────┐");
863 for (std::size_t i = 0; i < norm_parameters.size(); ++i) {
864 const auto& norm = norm_parameters[i];
865 double eventRatio = static_cast<double>(VerboseCounter[i]) / static_cast<double>(GetNEvents());
866
867 MACH3LOG_DEBUG("│ Param {:<15}, affects {:<8} events ({:>6.2f}%) │",
868 ParHandler->GetParFancyName(norm.index), VerboseCounter[i], eventRatio);
869 }
870 MACH3LOG_DEBUG("└──────────────────────────────────────────────────────────┘");
871 #endif
872}
