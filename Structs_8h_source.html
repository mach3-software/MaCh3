248inline std::string GetTF1(const SplineInterpolation i) {
249 // **************************************************
250 std::string Func = "";
251 switch(i) {
252 case SplineInterpolation::kLinearFunc:
253 Func = "([1]+[0]*x)";
254 break;
255 case SplineInterpolation::kTSpline3:
256 case SplineInterpolation::kLinear:
257 case SplineInterpolation::kMonotonic:
258 case SplineInterpolation::kAkima:
259 case SplineInterpolation::kSplineInterpolations:
260 MACH3LOG_ERROR("Interpolation type {} not supported for TF1!", static_cast<int>(i));
261 throw MaCh3Exception(__FILE__, __LINE__);
262 default:
263 MACH3LOG_ERROR("UNKNOWN SPLINE INTERPOLATION SPECIFIED!");
264 MACH3LOG_ERROR("You gave {}", static_cast<int>(i));
265 throw MaCh3Exception(__FILE__ , __LINE__ );
266 }
267 return Func;
268}

581 inline int PDGToNuOscillatorFlavour(int NuPdg){
582 int NuOscillatorFlavour = M3::_BAD_INT_;
583 switch(std::abs(NuPdg)){
584 case NuPDG::kNue:
585 NuOscillatorFlavour = NuOscillator::kElectron;
586 break;
587 case NuPDG::kNumu:
588 NuOscillatorFlavour = NuOscillator::kMuon;
589 break;
590 case NuPDG::kNutau:
591 NuOscillatorFlavour = NuOscillator::kTau;
592 break;
593 default:
594 MACH3LOG_ERROR("Unknown Neutrino PDG {}, cannot convert to NuOscillator type", NuPdg);
595 break;
596 }
597
598 //This is very cheeky but if the PDG is negative then multiply the PDG by -1
599 // This is consistent with the treatment that NuOscillator expects as enums only
600 // exist for the generic matter flavour and not the anti-matter version
601 if(NuPdg < 0){NuOscillatorFlavour *= -1;}
602
603 return NuOscillatorFlavour;
604 }
