214inline std::string GetTF1(const SplineInterpolation i) {
215 // **************************************************
216 std::string Func = "";
217 switch(i) {
218 case SplineInterpolation::kLinearFunc:
219 Func = "([1]+[0]*x)";
220 break;
221 case SplineInterpolation::kTSpline3:
222 case SplineInterpolation::kLinear:
223 case SplineInterpolation::kMonotonic:
224 case SplineInterpolation::kAkima:
225 case SplineInterpolation::kSplineInterpolations:
226 MACH3LOG_ERROR("Interpolation type {} not supported for TF1!", static_cast<int>(i));
227 throw MaCh3Exception(__FILE__, __LINE__);
228 default:
229 MACH3LOG_ERROR("UNKNOWN SPLINE INTERPOLATION SPECIFIED!");
230 MACH3LOG_ERROR("You gave {}", static_cast<int>(i));
231 throw MaCh3Exception(__FILE__ , __LINE__ );
232 }
233 return Func;
234}

612 inline int PDGToNuOscillatorFlavour(int NuPdg){
613 int NuOscillatorFlavour = M3::_BAD_INT_;
614 switch(std::abs(NuPdg)){
615 case NuPDG::kNue:
616 NuOscillatorFlavour = NuOscillator::kElectron;
617 break;
618 case NuPDG::kNumu:
619 NuOscillatorFlavour = NuOscillator::kMuon;
620 break;
621 case NuPDG::kNutau:
622 NuOscillatorFlavour = NuOscillator::kTau;
623 break;
624 default:
625 MACH3LOG_ERROR("Unknown Nuetrino PDG {}, cannot convert to NuOscillator type", NuPdg);
626 break;
627 }
628
629 //This is very cheeky but if the PDG is negative then multiply the PDG by -1
630 // This is consistent with the treatment that NuOscillator expects as enums only
631 // exist for the generic matter flavour and not the anti-matter version
632 if(NuPdg < 0){NuOscillatorFlavour *= -1;}
633
634 return NuOscillatorFlavour;
635 }
