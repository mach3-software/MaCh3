149// **************************************************
152inline RespFuncType SplineInterpolation_ToRespFuncType(const SplineInterpolation i) {
153// **************************************************
154 RespFuncType Type = kRespFuncTypes;
155 switch(i) {
156 // TSpline3 (third order spline in ROOT)
157 case SplineInterpolation::kTSpline3:
158 Type = RespFuncType::kTSpline3_red;
159 break;
160 case SplineInterpolation::kLinear:
161 Type = RespFuncType::kTSpline3_red;
162 break;
163 case SplineInterpolation::kMonotonic:
164 Type = RespFuncType::kTSpline3_red;
165 break;
166 // (Experimental) Akima_Spline (crd order spline which is allowed to be discontinuous in 2nd deriv)
167 case SplineInterpolation::kAkima:
168 Type = RespFuncType::kTSpline3_red;
169 break;
170 case SplineInterpolation::kLinearFunc:
171 Type = RespFuncType::kTF1_red;
172 break;
173 default:
174 std::cerr << "UNKNOWN SPLINE INTERPOLATION SPECIFIED!" << std::endl;
175 std::cerr << "You gave " << static_cast<int>(i) << std::endl;
176 std::cerr << __FILE__ << ":" << __LINE__ << std::endl;
177 throw;
178 }
179 return Type;
180}
