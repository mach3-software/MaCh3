158inline RespFuncType SplineInterpolation_ToRespFuncType(const SplineInterpolation i) {
159// **************************************************
160 RespFuncType Type = kRespFuncTypes;
161 switch(i) {
162 // TSpline3 (third order spline in ROOT)
163 case SplineInterpolation::kTSpline3:
164 Type = RespFuncType::kTSpline3_red;
165 break;
166 case SplineInterpolation::kLinear:
167 Type = RespFuncType::kTSpline3_red;
168 break;
169 case SplineInterpolation::kMonotonic:
170 Type = RespFuncType::kTSpline3_red;
171 break;
172 // (Experimental) Akima_Spline (crd order spline which is allowed to be discontinuous in 2nd deriv)
173 case SplineInterpolation::kAkima:
174 Type = RespFuncType::kTSpline3_red;
175 break;
176 case SplineInterpolation::kLinearFunc:
177 Type = RespFuncType::kTF1_red;
178 break;
179 default:
180 MACH3LOG_ERROR("UNKNOWN SPLINE INTERPOLATION SPECIFIED!");
181 MACH3LOG_ERROR("You gave {}", static_cast<int>(i));
182 throw MaCh3Exception(__FILE__ , __LINE__ );
183 }
184 return Type;
185}

523 inline int PDGToNuOscillatorFlavour(int NuPdg){
524 int NuOscillatorFlavour = _BAD_INT_;
525 switch(std::abs(NuPdg)){
526 case NuPDG::kNue:
527 NuOscillatorFlavour = NuOscillator::kElectron;
528 break;
529 case NuPDG::kNumu:
530 NuOscillatorFlavour = NuOscillator::kMuon;
531 break;
532 case NuPDG::kNutau:
533 NuOscillatorFlavour = NuOscillator::kTau;
534 break;
535 default:
536 MACH3LOG_ERROR("Unknown Nuetrino PDG {}, cannot convert to NuOscillator type", NuPdg);
537 break;
538 }
539
540 //This is very cheeky but if the PDG is negative then multiply the PDG by -1
541 // This is consistent with the treatment that NuOscillator expects as enums only
542 // exist for the generic matter flavour and not the anti-matter version
543 if(NuPdg < 0){NuOscillatorFlavour *= -1;}
544
545 return NuOscillatorFlavour;
546 }
