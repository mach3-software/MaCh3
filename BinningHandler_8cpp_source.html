8 auto BinRangeToBinEdges(YAML::Node const &bin_range) {
9  bool is_lin = true;
10  YAML::Node bin_range_specifier;
11  if (bin_range["linspace"]) {
12  bin_range_specifier = bin_range["linspace"];
13  } else if (bin_range["logspace"]) {
14  is_lin = false;
15  bin_range_specifier = bin_range["logspace"];
16  } else {
17  std::stringstream ss;
18  ss << bin_range;
19  MACH3LOG_ERROR("When parsing binning, expected bin range specifier with "
20  "key linspace or logspace, but found,\n{}",
21  ss.str());
22  throw MaCh3Exception(__FILE__, __LINE__);
23  }
24 
25  auto nb = Get<int>(bin_range_specifier["nb"], __FILE__, __LINE__);
26  auto low = Get<double>(bin_range_specifier["low"], __FILE__, __LINE__);
27  auto up = Get<double>(bin_range_specifier["up"], __FILE__, __LINE__);
28 
29  std::vector<double> edges(nb + 1, low);
30  // force the last bin to be exactly as parsed to avoid numerical instabilities
31  // not quite lining back up with the end of the range, which could
32  // cause spurious errors or infinitesimally small bins for specifications
33  // like: [ { logspace: { nb: 10, 1E-1, 10}, 10, 11} ]
34  edges.back() = up;
35 
36  if (is_lin) {
37  double bw = (up - low) / nb;
38  for (int i = 0; i < (nb - 1); ++i) {
39  edges[i + 1] = edges[i] + bw;
40  }
41  } else {
42  double llow = std::log10(low);
43  double lup = std::log10(up);
44  double lbw = (lup - llow) / nb;
45  for (int i = 0; i < (nb - 1); ++i) {
46  edges[i + 1] = std::pow(10, llow + (i + 1) * lbw);
47  }
48  }
49 
50  return edges;
51 }

60 auto BuildBinEdgesFromNode(YAML::Node const &bin_edges_node,
61  bool &found_range_specifier) {
62  if (bin_edges_node.IsMap()) {
63  found_range_specifier = true;
64  return BinRangeToBinEdges(bin_edges_node);
65  }
66  std::vector<double> edges_builder;
67  if (!bin_edges_node.IsSequence()) {
68  std::stringstream ss;
69  ss << bin_edges_node;
70  MACH3LOG_ERROR(
71  "When parsing binning, expected to find a YAML map or sequence, "
72  "but found:\n{}",
73  ss.str());
74  throw MaCh3Exception(__FILE__, __LINE__);
75  }
76  for (auto const &it : bin_edges_node) {
77  if (it.IsScalar()) {
78  edges_builder.push_back(it.as<double>());
79  } else if (it.IsMap()) {
80  found_range_specifier = true;
81  auto range_edges = BinRangeToBinEdges(it);
82  std::copy(range_edges.begin(), range_edges.end(),
83  std::back_inserter(edges_builder));
84  } else {
85  std::stringstream ss;
86  ss << bin_edges_node;
87  MACH3LOG_ERROR(
88  "When parsing binning, expected elements in outer sequence to all be "
89  "either scalars or maps, but found:\n{}",
90  ss.str());
91  throw MaCh3Exception(__FILE__, __LINE__);
92  }
93  }
94 
95  // Check for duplicates or out-of-order bins
96  std::vector<double> edges;
97  for (size_t eb_it = 0; eb_it < edges_builder.size(); ++eb_it) {
98  if (edges.size()) {
99  if (edges_builder[eb_it] == edges.back()) { // remove duplicate edges
100  continue;
101  } else if (edges_builder[eb_it] < edges.back()) {
102  std::stringstream ss;
103  ss << "[ ";
104  for(auto const & e : edges_builder){
105  ss << fmt::format("{:.3g} ", e);
106  }
107  ss << "]";
108  MACH3LOG_ERROR(
109  "When parsing binning, found edges that were not monotonically "
110  "increasing, problem bin at index: {}:\n{}",
111  eb_it, ss.str());
112  throw MaCh3Exception(__FILE__, __LINE__);
113  }
114  }
115  edges.push_back(edges_builder[eb_it]);
116  }
117 
118  return edges;
119 }

257 int BinningHandler::FindGlobalBin(const int NomSample,
258  const std::vector<const double*>& KinVar,
259  const std::vector<int>& NomBin) const {
260 // ************************************************
261  //DB Find the relevant bin in the PDF for each event
262  const int Dim = static_cast<int>(KinVar.size());
263  const SampleBinningInfo& _restrict_ Binning = SampleBinning[NomSample];
264  int GlobalBin = 0;
265 
266  for(int i = 0; i < Dim; ++i) {
267  const double Var = *KinVar[i];
268  const int Bin = Binning.FindBin(i, Var, NomBin[i]);
269  // KS: If we are outside of range in only one dimension this mean out of bounds, we can simply quickly finish
270  if(Bin < 0) return M3::UnderOverFlowBin;
271  // KS: inline GetBin computation to avoid any memory allocation, which in reweight loop is very costly
272  GlobalBin += Bin * Binning.Strides[i];
273  }
274 
275  if(Binning.Uniform) {
276  GlobalBin += static_cast<int>(Binning.GlobalOffset);
277  return GlobalBin;
278  } else {
279  const auto& _restrict_ BinMapping = Binning.BinGridMapping[GlobalBin];
280  const size_t nNonUniBins = BinMapping.size();
281  for(size_t iBin = 0; iBin < nNonUniBins; iBin++) {
282  const int BinNumber = BinMapping[iBin];
283  const auto& _restrict_ NonUniBin = Binning.Bins[BinNumber];
284  if(NonUniBin.IsEventInside(KinVar)){
285  return BinNumber + Binning.GlobalOffset;
286  }
287  }
288  MACH3LOG_DEBUG("Didn't find any bin so returning UnderOverFlowBin");
289  return M3::UnderOverFlowBin;
290  }

370 // Get fancy name for a given bin, to help match it with global properties
371 std::string BinningHandler::GetBinName(const int iSample, const int SampleBin) const {
372 // ************************************************
373  const auto& Binning = SampleBinning[iSample];
374 
375  // Safety checks
376  if (SampleBin < 0 || SampleBin >= static_cast<int>(Binning.nBins)) {
377  MACH3LOG_ERROR("Requested bin {} is out of range for sample {}", SampleBin, iSample);
378  throw MaCh3Exception(__FILE__, __LINE__);
379  }
380  std::string BinName;
381 
382  if(Binning.Uniform) {
383  int Dim = static_cast<int>(Binning.Strides.size());
384  std::vector<int> Bins(Dim, 0);
385  int Remaining = SampleBin;
386 
387  // Convert the flat/global bin index into per-dimension indices
388  // Dim0 is the fastest-changing axis, Dim1 the next, etc.
389  //
390  // For example (2D):
391  // x = bin % Nx
392  // y = bin / Nx
393  //
394  // For 3D:
395  // x = bin % Nx
396  // y = (bin / Nx) % Ny
397  // z = bin / (Nx * Ny)
398  for (int i = 0; i < Dim; ++i) {
399  const int nBinsDim = static_cast<int>(Binning.BinEdges[i].size()) - 1;
400  Bins[i] = Remaining % nBinsDim;
401  Remaining /= nBinsDim;
402  }
403 
404  for (int i = 0; i < Dim; ++i) {
405  if (i > 0) BinName += ", ";
406  const double min = Binning.BinEdges[i].at(Bins[i]);
407  const double max = Binning.BinEdges[i].at(Bins[i] + 1);
408  BinName += fmt::format("Dim{} ({:g}, {:g})", i, min, max);
409  }
410  } else{
411  const BinInfo& bin = Binning.Bins[SampleBin];
412  const int Dim = static_cast<int>(bin.Extent.size());
413 
414  for (int i = 0; i < Dim; ++i) {
415  if (i > 0) BinName += ", ";
416  const double min = bin.Extent[i][0];
417  const double max = bin.Extent[i][1];
418  BinName += fmt::format("Dim{} ({:g}, {:g})", i, min, max);
419  }
420  }
421  return BinName;
422 }
