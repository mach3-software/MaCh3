65template <typename CovType>
66std::unique_ptr<CovType> MaCh3CovarianceFactory(manager *FitManager, const std::string& PreFix){
67// ********************************************
68 // config for our matrix
69 YAML::Node Settings = FitManager->raw()["General"]["Systematics"];
70 auto CovMatrixName = Get<std::string>(Settings[std::string(PreFix) + "CovName"], __FILE__, __LINE__);
71 MACH3LOG_INFO("Initialising {} matrix", CovMatrixName);
72
73 // yaml files initialising out matrix
74 auto CovMatrixFile = Get<std::vector<std::string>>(Settings[std::string(PreFix) + "CovFile"], __FILE__, __LINE__);
75
76 // PCA threshold, -1 means no pca
77 auto PCAThreshold = GetFromManager<int>(Settings[std::string(PreFix) + "PCAThreshold"], -1);
78 // do we pca whole matrix or only submatrix
79 auto PCAParamRegion = GetFromManager<std::vector<int>>(Settings[std::string(PreFix) + "PCAParams"], {-999, -999});
80
81 auto CovObject = std::make_unique<CovType>(CovMatrixFile, CovMatrixName, PCAThreshold, PCAParamRegion[0], PCAParamRegion[1]);
82
83 // Fill the parameter values with their nominal values
84 // should _ALWAYS_ be done before overriding with fix or flat
85 CovObject->SetParameters();
86
87 auto FixParams = GetFromManager<std::vector<std::string>>(Settings[std::string(PreFix) + "Fix"], {});
88
89 // Fixed CovObject parameters loop
90 if (FixParams.size() == 1 && FixParams.at(0) == "All") {
91 for (int j = 0; j < CovObject->GetNumParams(); j++) {
92 CovObject->ToggleFixParameter(j);
93 }
94 } else {
95 for (unsigned int j = 0; j < FixParams.size(); j++) {
96 CovObject->ToggleFixParameter(FixParams.at(j));
97 }
98 }
99 //Global step scale for matrix
100 auto StepScale = Get<double>(Settings[std::string(PreFix) + "StepScale"], __FILE__, __LINE__);
101
102 CovObject->SetStepScale(StepScale);
103
104 // Adaptive MCMC stuff
105 if (CheckNodeExists(FitManager->raw(), "AdaptionOptions")) {
106 CovObject->InitialiseAdaption(FitManager->raw());
107 }
108
109 return CovObject;
110}
