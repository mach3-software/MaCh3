57template <typename CovType>
58CovType* MaCh3CovarianceFactory(manager *FitManager, const std::string& PreFix){
59// ********************************************
60 // config for our matrix
61 YAML::Node Settings = FitManager->raw()["General"]["Systematics"];
62 auto CovMatrixName = Settings[std::string(PreFix) + "CovName"].as<std::string>();
63 MACH3LOG_INFO("Initialising {} matrix", CovMatrixName);
64
65 // yaml files initialising out matrix
66 auto CovMatrixFile = Settings[std::string(PreFix) + "CovFile"].as<std::vector<std::string>>();
67
68 // PCA threshold, -1 means no pca
69 auto PCAThreshold = GetFromManager<int>(Settings[std::string(PreFix) + "PCAThreshold"], -1);
70 // do we pca whole matrix or only submatrix
71 auto PCAParamRegion = GetFromManager<std::vector<int>>(Settings[std::string(PreFix) + "PCAParams"], {-999, -999});
72
73 CovType* CovObject = new CovType(CovMatrixFile, CovMatrixName, PCAThreshold, PCAParamRegion[0], PCAParamRegion[1]);
74
75 // Fill the parameter values with their nominal values
76 // should _ALWAYS_ be done before overriding with fix or flat
77 CovObject->setParameters();
78
79 auto FixParams = GetFromManager<std::vector<std::string>>(Settings[std::string(PreFix) + "Fix"], {});
80
81 // Fixed CovObject parameters loop
82 if (FixParams.size() == 1 && FixParams.at(0) == "All") {
83 for (int j = 0; j < CovObject->GetNumParams(); j++) {
84 CovObject->toggleFixParameter(j);
85 }
86 } else {
87 for (unsigned int j = 0; j < FixParams.size(); j++) {
88 CovObject->toggleFixParameter(FixParams.at(j));
89 }
90 }
91 //Global step scale for matrix
92 auto StepScale = Settings[std::string(PreFix) + "StepScale"].as<double>();
93
94 CovObject->setStepScale(StepScale);
95
96 // Adaptive MCMC stuff
97 if(FitManager->raw()["AdaptionOptions"])
98 CovObject->initialiseAdaption(FitManager->raw());
99
100 MACH3LOG_INFO("Factory successful");
101
102 return CovObject;
103}
