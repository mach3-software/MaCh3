71template <typename CovType>
72CovType* MaCh3CovarianceFactory(manager *FitManager, const std::string& PreFix){
73// ********************************************
74 // config for our matrix
75 YAML::Node Settings = FitManager->raw()["General"]["Systematics"];
76 auto CovMatrixName = Settings[std::string(PreFix) + "CovName"].as<std::string>();
77 MACH3LOG_INFO("Initialising {} matrix", CovMatrixName);
78
79 // yaml files initialising out matrix
80 auto CovMatrixFile = Settings[std::string(PreFix) + "CovFile"].as<std::vector<std::string>>();
81
82 // PCA threshold, -1 means no pca
83 auto PCAThreshold = GetFromManager<int>(Settings[std::string(PreFix) + "PCAThreshold"], -1);
84 // do we pca whole matrix or only submatrix
85 auto PCAParamRegion = GetFromManager<std::vector<int>>(Settings[std::string(PreFix) + "PCAParams"], {-999, -999});
86
87 CovType* CovObject = new CovType(CovMatrixFile, CovMatrixName, PCAThreshold, PCAParamRegion[0], PCAParamRegion[1]);
88
89 // Fill the parameter values with their nominal values
90 // should _ALWAYS_ be done before overriding with fix or flat
91 CovObject->setParameters();
92
93 auto FixParams = GetFromManager<std::vector<std::string>>(Settings[std::string(PreFix) + "Fix"], {});
94
95 // Fixed CovObject parameters loop
96 if (FixParams.size() == 1 && FixParams.at(0) == "All") {
97 for (int j = 0; j < CovObject->GetNumParams(); j++) {
98 CovObject->toggleFixParameter(j);
99 }
100 } else {
101 for (unsigned int j = 0; j < FixParams.size(); j++) {
102 CovObject->toggleFixParameter(FixParams.at(j));
103 }
104 }
105 //Global step scale for matrix
106 auto StepScale = Settings[std::string(PreFix) + "StepScale"].as<double>();
107
108 CovObject->setStepScale(StepScale);
109
110 // Adaptive MCMC stuff
111 if(FitManager->raw()["AdaptionOptions"])
112 CovObject->initialiseAdaption(FitManager->raw());
113
114 MACH3LOG_INFO("Factory successful");
115
116 return CovObject;
117}
