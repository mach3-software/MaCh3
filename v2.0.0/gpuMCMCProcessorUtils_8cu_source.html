115  float* NumeratorSum_gpu,

117 //*********************************************************
118 
119  const unsigned int CurrentLagNum = (blockIdx.x * blockDim.x + threadIdx.x);
120 
121  //KS: Accessing shared memory is much much faster than global memory hence we use shared memory for calculation and then write to global memory
122  __shared__ float shared_NumeratorSum[_BlockSize_];
123  __shared__ float shared_DenomSum[_BlockSize_];
124 
125  // this is the stopping condition!
126  if (CurrentLagNum < d_nLag*d_nDraws)
127  {
128  shared_NumeratorSum[threadIdx.x] = 0;
129  shared_DenomSum[threadIdx.x] = 0;
130 
131  //KS: Might consider caching this information, which MIGHT be faster too lazy right now
132  const int Param = int(CurrentLagNum/d_nLag);
133  const int nLag = CurrentLagNum - Param*d_nLag;
134  // Loop over the number of entries
135  for (int i = 0; i < d_nEntries; ++i)
136  {
137  //KS: Use fmaf to have it tiny bit faster, for something easier to read: Param*d_nEntries + i
138  int CurrParStep = fmaf(Param, d_nEntries, i);
139  const float Diff = ParStep_gpu[CurrParStep]-ParamSums_gpu[Param];
140  // Only sum the numerator up to i = N-k
141  if (i < d_nEntries-nLag)
142  {
143  //KS: Use fmaf to have it tiny bit faster, for something easier to read: Param*d_nEntries + (i + nLag)
144  CurrParStep = fmaf(Param, d_nEntries, i + nLag);
145  const float LagTerm = ParStep_gpu[CurrParStep]-ParamSums_gpu[Param];
146  const float Product = Diff*LagTerm;
147  shared_NumeratorSum[threadIdx.x] += Product;
148  }
149  // Square the difference to form the denominator
150  const float Denom = Diff*Diff;
151  shared_DenomSum[threadIdx.x] += Denom;
152  }
153 
154  //KS: Make sure threads are synchronised before moving to global memory
155  __syncthreads();
156  NumeratorSum_gpu[CurrentLagNum] = shared_NumeratorSum[threadIdx.x];
157  DenomSum_gpu[CurrentLagNum] = shared_DenomSum[threadIdx.x];
158  }
159 }
