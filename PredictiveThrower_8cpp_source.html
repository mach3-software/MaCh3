105// *************************
106 int counter = 0;
107 for (size_t s = 0; s < systematics.size(); ++s) {

116
117 if(Is_PriorPredictive) {

126 YAML::Node ConfigInChain = Processor.GetCovConfig(kXSecPar);
127 if(ModelSystematic){
128 YAML::Node ConfigNow = ModelSystematic->GetConfig();

190bool PredictiveThrower::LoadToys() {
191// *************************
192 auto PosteriorFileName = Get<std::string>(fitMan->raw()["Predictive"]["PosteriorFile"], __FILE__, __LINE__);
193 // Open the ROOT file
194 TFile* file = TFile::Open(PosteriorFileName.c_str(), "READ");
195 TDirectory* ToyDir = nullptr;
196 if (!file || file->IsZombie()) {
197 return false;
198 } else {
199 // Check for the "toys" directory
200 if ((ToyDir = file->GetDirectory("Toys"))) {
201 MACH3LOG_INFO("Found toys in Posterior file will attempt toy reading");
202 } else {
203 file->Close();
204 delete file;
205 return false;
206 }
207 }
208
209 // Finally get the TTree branch with the penalty vectors for each of the toy throws
210 TTree* PenaltyTree = static_cast<TTree*>(file->Get("ToySummary"));
211 if (!PenaltyTree) {
212 MACH3LOG_WARN("ToySummary TTree not found in file.");
213 file->Close();
214 delete file;
215 return false;
216 }
217
218 Ntoys = static_cast<int>(PenaltyTree->GetEntries());
219 int ConfigNtoys = Get<int>(fitMan->raw()["Predictive"]["Ntoy"], __FILE__, __LINE__);;
220 if (Ntoys != ConfigNtoys) {
221 MACH3LOG_WARN("Found different number of toys in saved file than asked to run!");
222 MACH3LOG_INFO("Will read _ALL_ toys in the file");
223 MACH3LOG_INFO("Ntoys in file: {}", Ntoys);
224 MACH3LOG_INFO("Ntoys specified: {}", ConfigNtoys);
225 }
226
227 PenaltyTerm.resize(Ntoys);
228 ReweightWeight.resize(Ntoys);
229
230 double Penalty = 0, Weight = 1;
231 PenaltyTree->SetBranchAddress("Penalty", &Penalty);
232 PenaltyTree->SetBranchAddress("Weight", &Weight);
233 PenaltyTree->SetBranchAddress("NModelParams", &NModelParams);
234
235 for (int i = 0; i < Ntoys; ++i) {
236 PenaltyTree->GetEntry(i);
237 if (FullLLH) {
238 PenaltyTerm[i] = Penalty;
239 } else {
240 PenaltyTerm[i] = 0.0;
241 }
242
243 ReweightWeight[i] = Weight;
244 }
245 // Resize all vectors and get sample names
246 SetupSampleInformation();
247
248 for (int sample = 0; sample < TotalNumberOfSamples; ++sample) {
249 TH1D* DataHist1D = static_cast<TH1D*>(ToyDir->Get((SampleNames[sample] + "_data").c_str()));
250 Data_Hist[sample] = M3::Clone(DataHist1D);
251
252 TH1D* MCHist1D = static_cast<TH1D*>(ToyDir->Get((SampleNames[sample] + "_mc").c_str()));
253 MC_Nom_Hist[sample] = M3::Clone(MCHist1D);
254
255 TH1D* W2Hist1D = static_cast<TH1D*>(ToyDir->Get((SampleNames[sample] + "_w2").c_str()));
256 W2_Nom_Hist[sample] = M3::Clone(W2Hist1D);
257 }
258
259
260 for (int iToy = 0; iToy < Ntoys; ++iToy)
261 {
262 if (iToy % 100 == 0) MACH3LOG_INFO(" Loaded toy {}", iToy);
263
264 for (int sample = 0; sample < TotalNumberOfSamples; ++sample) {
265 TH1D* MCHist1D = static_cast<TH1D*>(ToyDir->Get((SampleNames[sample] + "_mc_" + std::to_string(iToy)).c_str()));
266 TH1D* W2Hist1D = static_cast<TH1D*>(ToyDir->Get((SampleNames[sample] + "_w2_" + std::to_string(iToy)).c_str()));
267
268 MC_Hist_Toy[sample][iToy] = M3::Clone(MCHist1D);
269 W2_Hist_Toy[sample][iToy] = M3::Clone(W2Hist1D);
270 }
271 }
272
273 file->Close();
274 delete file;
275 return true;
276}

278// *************************
279// Produce MaCh3 toys:
280void PredictiveThrower::ProduceToys() {
281// *************************
283 if(LoadToys()) return;
284
286 SetupToyGeneration();
287
288 auto PosteriorFileName = Get<std::string>(fitMan->raw()["Predictive"]["PosteriorFile"], __FILE__, __LINE__);
289
290 MACH3LOG_INFO("Starting {}", __func__);
291
292 outputFile->cd();
293 double Penalty = 0, Weight = 1;
294 int Draw = 0;
295
296 TTree *ToyTree = new TTree("ToySummary", "ToySummary");
297 ToyTree->Branch("Penalty", &Penalty, "Penalty/D");
298 ToyTree->Branch("Weight", &Weight, "Weight/D");
299 ToyTree->Branch("Draw", &Draw, "Draw/I");
300 ToyTree->Branch("NModelParams", &NModelParams, "NModelParams/I");
301
302 TDirectory* ToyDirectory = outputFile->mkdir("Toys");
303 ToyDirectory->cd();
304
305 for (size_t iPDF = 0; iPDF < samples.size(); iPDF++)
306 {
307 auto* MaCh3Sample = dynamic_cast<SampleHandlerFD*>(samples[iPDF]);
308 // Get nominal spectra and event rates
309 TH1D* DataHist1D = static_cast<TH1D*>(MaCh3Sample->GetDataHist(1));
310 Data_Hist[iPDF] = M3::Clone(DataHist1D, MaCh3Sample->GetTitle() + "_data");
311 Data_Hist[iPDF]->Write((MaCh3Sample->GetTitle() + "_data").c_str());
312
313 TH1D* MCHist1D = static_cast<TH1D*>(MaCh3Sample->GetMCHist(1));
314 MC_Nom_Hist[iPDF] = M3::Clone(MCHist1D, MaCh3Sample->GetTitle() + "_mc");
315 MCHist1D->Write((MaCh3Sample->GetTitle() + "_mc").c_str());
316
317 TH1D* W2Hist1D = static_cast<TH1D*>(MaCh3Sample->GetW2Hist(1));
318 W2_Nom_Hist[iPDF] = M3::Clone(W2Hist1D, MaCh3Sample->GetTitle() + "_w2");
319 W2Hist1D->Write((MaCh3Sample->GetTitle() + "_w2").c_str());
320 delete W2Hist1D;
321 }
322
324 std::vector<std::vector<double>> branch_vals(systematics.size());
325 std::vector<std::vector<std::string>> branch_name(systematics.size());
326
327 TChain* PosteriorFile = new TChain("posteriors");
328 PosteriorFile->Add(PosteriorFileName.c_str());
329 int Step = 0;
330 PosteriorFile->SetBranchAddress("step", &Step);
331
332 if (PosteriorFile->GetBranch("Weight")) {
333 PosteriorFile->SetBranchStatus("Weight", true);
334 PosteriorFile->SetBranchAddress("Weight", &Weight);
335 } else {
336 MACH3LOG_WARN("Not applying reweighting weight");
337 Weight = 1.0;
338 }
339
340 for (size_t s = 0; s < systematics.size(); ++s) {
341 systematics[s]->MatchMaCh3OutputBranches(PosteriorFile, branch_vals[s], branch_name[s]);
342 }
343 //Get the burn-in from the config
344 auto burn_in = Get<int>(fitMan->raw()["Predictive"]["BurnInSteps"], __FILE__, __LINE__);
345
346 //DL: Adding sanity check for chains shorter than burn in
347 const unsigned int maxNsteps = static_cast<unsigned int>(PosteriorFile->GetMaximum("step"));
348 if(static_cast<unsigned int>(burn_in) >= maxNsteps)
349 {
350 MACH3LOG_ERROR("You are running on a chain shorter than burn in cut");
351 MACH3LOG_ERROR("Maximal value of nSteps: {}, burn in cut {}", maxNsteps, burn_in);
352 MACH3LOG_ERROR("You will run into infinite loop");
353 MACH3LOG_ERROR("You can make new chain or modify burn in cut");
354 throw MaCh3Exception(__FILE__,__LINE__);
355 }
356
357 TStopwatch TempClock;
358 TempClock.Start();
359 for(int i = 0; i < Ntoys; i++)
360 {
361 if( i % (Ntoys/10) == 0) {
362 MaCh3Utils::PrintProgressBar(i, Ntoys);
363 }
364 int entry = 0;
365 Step = -999;
366
367 //YSP: Ensures you get an entry from the mcmc even when burn_in is set to zero (Although not advised :p ).
368 //Take 200k burn in steps, WP: Eb C in 1st peaky
369 // If we have combined chains by hadd need to check the step in the chain
370 // Note, entry is not necessarily same as step due to merged ROOT files, so can't choose entry in the range BurnIn - nEntries :(
371 while(Step < burn_in){
372 entry = random->Integer(static_cast<unsigned int>(PosteriorFile->GetEntries()));
373 PosteriorFile->GetEntry(entry);
374 }
375 if(!Is_PriorPredictive) Draw = entry;
376 for (size_t s = 0; s < systematics.size(); ++s)
377 {
378 systematics[s]->SetParameters(branch_vals[s]);
379
380 //KS: Below line can help you get prior predictive distributions which are helpful for getting pre and post ND fit spectra
381 //YSP: If not set in the config, the code runs SK Posterior Predictive distributions by default. If true, then the code runs SK prior predictive.
382 if(Is_PriorPredictive) systematics[s]->ThrowParameters();
383 }
384
385 // This set some params to prior value this way you can evaluate errors from subset of errors
386 SetParamters();
387
388 Penalty = 0;
389 if(FullLLH) {
390 for (size_t s = 0; s < systematics.size(); ++s) {
391 //KS: do times 2 because banff reports chi2
392 Penalty = 2.0 * systematics[s]->GetLikelihood();
393 }
394 }
395
396 PenaltyTerm[i] = Penalty;
397 ReweightWeight[i] = Weight;
398
399 for (size_t iPDF = 0; iPDF < samples.size(); iPDF++) {
400 samples[iPDF]->Reweight();
401 }
402
403 for (size_t iPDF = 0; iPDF < samples.size(); iPDF++)
404 {
405 auto* MaCh3Sample = dynamic_cast<SampleHandlerFD*>(samples[iPDF]);
406
407 TH1D* MCHist1D = static_cast<TH1D*>(MaCh3Sample->GetMCHist(1));
408 MC_Hist_Toy[iPDF][i] = M3::Clone(MCHist1D, MaCh3Sample->GetTitle() + "_mc_" + std::to_string(i));
409 MC_Hist_Toy[iPDF][i]->Write();
410
411 TH1D* W2Hist1D = static_cast<TH1D*>(MaCh3Sample->GetW2Hist(1));
412 W2_Hist_Toy[iPDF][i] = M3::Clone(W2Hist1D, MaCh3Sample->GetTitle() + "_w2_" + std::to_string(i));
413 W2_Hist_Toy[iPDF][i]->Write();
414 delete W2Hist1D;
415 }
416 ToyTree->Fill();
417 }//end of toys loop
418 TempClock.Stop();
419
420 delete PosteriorFile;
421 ToyDirectory->Close();
422 delete ToyDirectory;
423
424 outputFile->cd();
425 ToyTree->Write();
426 delete ToyTree;
427
428 MACH3LOG_INFO("{} took {:.2f}s to finish for {} toys", __func__, TempClock.RealTime(), Ntoys);
429}

432std::vector<std::unique_ptr<TH2D>> PredictiveThrower::ProduceSpectra(const std::vector<std::vector<std::unique_ptr<TH1D>>>& Toys,
433 const std::string suffix) {
434// *************************
435 std::vector<double> MaxValue(TotalNumberOfSamples);
436 #ifdef MULTITHREAD
437 #pragma omp parallel for
438 #endif
439 for (int sample = 0; sample < TotalNumberOfSamples; ++sample) {
440 for (int toy = 0; toy < Ntoys; ++toy) {
441 double max_val = Toys[sample][toy]->GetMaximum();
442 MaxValue[sample] = std::max(MaxValue[sample], max_val);
443 }
444 }
445
446 std::vector<std::unique_ptr<TH2D>> Spectra(TotalNumberOfSamples);
447 for (int sample = 0; sample < TotalNumberOfSamples; ++sample) {
448 // Get MC histogram x-axis binning
449 TH1D* refHist = Toys[sample][0].get();
450
451 const int n_bins_x = refHist->GetNbinsX();
452 std::vector<double> x_bin_edges(n_bins_x + 1);
453 for (int b = 0; b <= n_bins_x; ++b) {
454 x_bin_edges[b] = refHist->GetXaxis()->GetBinLowEdge(b + 1); // ROOT bins start at 1
455 }
456 // Last edge is upper edge of last bin:
457 x_bin_edges[n_bins_x] = refHist->GetXaxis()->GetBinUpEdge(n_bins_x);
458
459 constexpr int n_bins_y = 400;
460 constexpr double y_min = 0.0;
461 const double y_max = MaxValue[sample] * 1.05;
462
463 // Create TH2D with variable binning on x axis
464 Spectra[sample] = std::make_unique<TH2D>(
465 (SampleNames[sample] + "_" + suffix).c_str(), // name
466 (SampleNames[sample] + "_" + suffix).c_str(), // title
467 n_bins_x, x_bin_edges.data(), // x axis bins
468 n_bins_y, y_min, y_max // y axis bins
469 );
470
471 Spectra[sample]->SetDirectory(nullptr);
472 Spectra[sample]->Sumw2(true);
473 }
474
475 #ifdef MULTITHREAD
476 #pragma omp parallel for
477 #endif
478 for (int sample = 0; sample < TotalNumberOfSamples; ++sample) {
479 for (int toy = 0; toy < Ntoys; ++toy) {
480 FastViolinFill(Spectra[sample].get(), Toys[sample][toy].get());
481 }
482 }
483
484 return Spectra;
485}

488std::unique_ptr<TH1D> PredictiveThrower::MakePredictive(const std::vector<std::unique_ptr<TH1D>>& Toys,
489 const std::string& Sample_Name,
490 const std::string& suffix,
491 const bool DebugHistograms) {
492// *************************
493 constexpr int nXBins = 100;
494 int nbinsx = Toys[0]->GetNbinsX();
495
496 auto PredictiveHist = std::make_unique<TH1D>((Sample_Name + "_" + suffix + "_PostPred").c_str(),
497 (Sample_Name + "_" + suffix + "_PostPred").c_str(),
498 nbinsx, Toys[0]->GetXaxis()->GetXbins()->GetArray());
499 PredictiveHist->SetDirectory(nullptr);
500
501 for (int i = 1; i <= nbinsx; ++i) {
502 double x_low = Toys[0]->GetXaxis()->GetBinLowEdge(i);
503 double x_high = Toys[0]->GetXaxis()->GetBinUpEdge(i);
504 TString projName = TString::Format("%s %s X: [%.3f, %.3f]", Sample_Name.c_str(), suffix.c_str(), x_low, x_high);
505 auto PosteriorHist = std::make_unique<TH1D>(projName, projName, nXBins, 1, -1);
506 PosteriorHist->SetDirectory(nullptr);
507
508 for (size_t iToy = 0; iToy < Toys.size(); ++iToy) {
509 const double Content = Toys[iToy]->GetBinContent(i);
510 PosteriorHist->Fill(Content, ReweightWeight[iToy]);
511 }
512
513 if(DebugHistograms) PosteriorHist->Write();
514
515 const double nMean = PosteriorHist->GetMean();
516 const double nMeanError = PosteriorHist->GetRMS();
517
518 PredictiveHist->SetBinContent(i, nMean);
519 PredictiveHist->SetBinError(i, nMeanError);
520 }
521
522 PredictiveHist->Write();
523 return PredictiveHist;
524}

576// *************************
577double PredictiveThrower::GetLLH(const std::unique_ptr<TH1D>& DatHist,
578 const std::unique_ptr<TH1D>& MCHist,
579 const std::unique_ptr<TH1D>& W2Hist,
580 SampleHandlerBase* SampleHandler) {
581// *************************
582 double llh = 0.0;
583 for (int i = 1; i <= DatHist->GetXaxis()->GetNbins(); ++i)
584 {
585 const double data = DatHist->GetBinContent(i);
586 const double mc = MCHist->GetBinContent(i);
587 const double w2 = W2Hist->GetBinContent(i);
588 llh += SampleHandler->GetTestStatLLH(data, mc, w2);

595void PredictiveThrower::PosteriorPredictivepValue(const std::vector<std::unique_ptr<TH1D>>& PostPred_mc,
596 //const std::vector<std::unique_ptr<TH1D>>& PostPred_w2,
597 const std::vector<TDirectory*>& SampleDir) {

626

628 chi2_mc_vec[iToy].back() += chi2_mc_vec[iToy][iSample];
629 chi2_pred_vec[iToy].back() += chi2_pred_vec[iToy][iSample];
630 }
631 }
632
633 MakeChi2Plots(chi2_mc_vec, "-2LLH (Draw Fluc, Draw)", chi2_dat_vec, "-2LLH (Data, Draw)", SampleDir, "_drawfluc_draw");
634 MakeChi2Plots(chi2_pred_vec, "-2LLH (Pred Fluc, Draw)", chi2_dat_vec, "-2LLH (Data, Draw)", SampleDir, "_predfluc_draw");

670 chi2_hist->Fill(chi2_x_per_sample[iToy], chi2_y_sample[iToy]);
