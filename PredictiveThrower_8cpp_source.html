483  const std::vector<std::vector<std::unique_ptr<TH1>>>& Toys,
484  const std::vector<TDirectory*>& SampleDirectories,
485  const std::string suffix) {
486 // *************************

488  //Projections[sample][toy][dim]
489  std::vector<std::vector<std::vector<std::unique_ptr<TH1>>>> Projections(TotalNumberOfSamples);
490 
491  // 1. Create 1D projections, this is not thread safe, also we will use it a lot later on
492  for (int sample = 0; sample < TotalNumberOfSamples; ++sample) {
493  SampleDirectories[sample]->cd();
494 
495  const int nDims = SampleInfo[sample].Dimenstion;
496  MaxValue[sample].assign(nDims, 0);
497  Projections[sample].resize(Ntoys);
498  for (int toy = 0; toy < Ntoys; ++toy) {
499  if (nDims == 1) {
500  // For 1D histograms, no projection needed.
501  // Leave Projections[sample][toy][0] == nullptr
502  } else if (nDims == 2) {
503  Projections[sample][toy].resize(nDims);
504  TH2* h2 = dynamic_cast<TH2*>(Toys[sample][toy].get());
505  if (!h2) {
506  MACH3LOG_ERROR("Histogram is not TH2 for nDims==2");
507  throw MaCh3Exception(__FILE__, __LINE__);
508  }
509  auto px = h2->ProjectionX();
510  px->SetDirectory(nullptr);
511  Projections[sample][toy][0] = M3::Clone(px);
512 
513  auto py = h2->ProjectionY();
514  py->SetDirectory(nullptr);
515  Projections[sample][toy][1] = M3::Clone(py);
516 
517  delete px;
518  delete py;
519  } else {
520  MACH3LOG_ERROR("Asking for {} with N Dimension = {}. This is not implemented", __func__, nDims);
521  throw MaCh3Exception(__FILE__, __LINE__);
522  }
523  }
524  }
525 
526  // 2. Find maximum entries over all toys
527  #ifdef MULTITHREAD
528  #pragma omp parallel for collapse(2)
529  #endif
530  for (int sample = 0; sample < TotalNumberOfSamples; ++sample) {
531  for (int toy = 0; toy < Ntoys; ++toy) {
532  const int nDims = Toys[sample][0]->GetDimension();
533  for (int dim = 0; dim < nDims; dim++) {
534  double max_val = 0;
535  if (nDims == 1) {
536  max_val = Toys[sample][toy]->GetMaximum();
537  }
538  else if (nDims == 2) {
539  if (dim == 0) {
540  max_val = Projections[sample][toy][0]->GetMaximum();
541  } else {
542  max_val = Projections[sample][toy][1]->GetMaximum();
543  }
544  } else {
545  MACH3LOG_ERROR("Asking for {} with N Dimension = {}. This is not implemented", __func__, nDims);
546  throw MaCh3Exception(__FILE__, __LINE__);
547  }
548  MaxValue[sample][dim] = std::max(MaxValue[sample][dim], max_val);
549  }
550  }
551  }
552 
553  // 3. Make actual spectra histogram (this is because making ROOT histograms is not save)
554  // And we now have actual max values
555  std::vector<std::vector<std::unique_ptr<TH2D>>> Spectra(TotalNumberOfSamples);
556  for (int sample = 0; sample < TotalNumberOfSamples; ++sample) {
557  const int nDims = SampleInfo[sample].Dimenstion;
558  Spectra[sample].resize(nDims);
559  for (int dim = 0; dim < nDims; dim++) {
560  // Get MC histogram x-axis binning
561  TH1D* refHist = nullptr;
562  if (nDims == 1) {
563  refHist = static_cast<TH1D*>(Toys[sample][0].get());
564  }
