16// However if you use ROOT6 and have ROOT6 file exactly the same code will work. Something have changed with how TH2Poly bins are stored in TFile
17void CheckTH2PolyFileVersion(TFile *file) {
18// **************************************************
19 int FileROOTVersion = file->GetVersion();
20 int MainFileROOTVersion = FileROOTVersion;
21
22 // Remove last digit from number
23 // till only one digit is left
24 while (MainFileROOTVersion >= 10)
25 MainFileROOTVersion /= 10;
26
27 std::string SystemROOTVersion = std::string(ROOT_RELEASE);
28 int MainSystemROOTVersion = SystemROOTVersion.at(0) - '0';
29
30 if(MainFileROOTVersion != MainSystemROOTVersion)
31 {
32 MACH3LOG_ERROR("File was produced with: {} ROOT version", FileROOTVersion);
33 MACH3LOG_ERROR("Found: {} ROOT version in the system", SystemROOTVersion);
34 MACH3LOG_ERROR("For some documentation please visit me");
35 throw MaCh3Exception(__FILE__, __LINE__);
36 }
37}

41double OverflowIntegral(TH2Poly* poly) {
42// **************************************************
43 double overflow = 0.;
44 //TH2Polys have 9 overflow bins
45 for(int iOverflow = -1; iOverflow > -10; iOverflow--)

68//WP: Helper function for projecting TH2Poly onto the X axis
69TH1D* PolyProjectionX(TObject* poly, std::string TempName, const std::vector<double>& xbins, const bool computeErrors) {
70// **************************************************
71 TH1D* hProjX = new TH1D((TempName+"_x").c_str(),(TempName+"_x").c_str(), int(xbins.size()-1), &xbins[0]);
72
73 //KS: Temp Histogram to store error, use double as this is thread safe
74 std::vector<double> hProjX_Error(hProjX->GetXaxis()->GetNbins() + 1, 0.0);
75 double xlow, xup, frac = 0.;
76
77 //loop over bins in the poly
78 for (int i = 0; i < static_cast<TH2Poly*>(poly)->GetNumberOfBins(); i++)
79 {
80 //get bin and its edges
81 TH2PolyBin* bin = static_cast<TH2PolyBin*>(static_cast<TH2Poly*>(poly)->GetBins()->At(i));
82 xlow = bin->GetXMin();
83 xup = bin->GetXMax();
84
85 //Loop over projected bins, find fraction of poly bin in each
86 for(int dx = 0; dx < int(xbins.size()); dx++)
87 {
88 if(xbins[dx+1] <= xlow || xbins[dx] >= xup)
89 {
90 frac = 0;
91 }
92 else if(xbins[dx] <= xlow && xbins[dx+1] >= xup)
93 {
94 frac = 1;
95 }
96 else if(xbins[dx] <= xlow && xbins[dx+1] <= xup)
97 {
98 frac = (xbins[dx+1]-xlow)/(xup-xlow);
99 }
100 else if(xbins[dx] >= xlow && xbins[dx+1] >= xup)
101 {
102 frac = (xup-xbins[dx])/(xup-xlow);
103 }
104 else if(xbins[dx] >= xlow && xbins[dx+1] <= xup)
105 {
106 frac = (xbins[dx+1]-xbins[dx])/(xup-xlow);
107 }
108 else
109 {
110 frac = 0;
111 }
112 hProjX->SetBinContent(dx+1,hProjX->GetBinContent(dx+1)+frac*bin->GetContent());
113 //KS: Follow ROOT implementation and sum up the variance
114 if(computeErrors)
115 {
116 //KS: TH2PolyBin doesn't have GetError so we have to use TH2Poly,
117 //but numbering of GetBinError is different than GetBins...
118 const double Temp_Err = frac*static_cast<TH2Poly*>(poly)->GetBinError(i+1)*frac*static_cast<TH2Poly*>(poly)->GetBinError(i+1);
119 hProjX_Error[dx+1] += Temp_Err;
120 }
121 }
122 }
123 //KS: The error is sqrt(summed variance)) https://root.cern.ch/doc/master/TH2_8cxx_source.html#l02266
124 if(computeErrors)
125 {
126 for (int i = 1; i <= hProjX->GetXaxis()->GetNbins(); ++i)
127 {
128 const double Error = std::sqrt(hProjX_Error[i]);
129 hProjX->SetBinError(i, Error);
130 }
131 }
132 return hProjX;
133} // end project poly X function

251// ****************
252 // Make the x axis from the momentum of lepton
253 TAxis* xaxis = hist->GetXaxis();
254 // Make the y axis from the cos theta of lepton
255 TAxis* yaxis = hist->GetYaxis();
256
257 TString histname = hist->GetName();
258 // Convert TH2D binning to TH2Poly
259 TH2Poly* poly = new TH2Poly();
260 poly->SetName(histname);
261 poly->SetTitle(histname);
262
263 // Copy axis titles
264 poly->GetXaxis()->SetTitle(xaxis->GetTitle());
265 poly->GetYaxis()->SetTitle(yaxis->GetTitle());
266
267 double xmax, xmin, ymax, ymin;
268 for (int iy = 1; iy <= yaxis->GetNbins(); iy++) {
269 ymax = yaxis->GetBinUpEdge(iy);
270 ymin = yaxis->GetBinLowEdge(iy);
271 for (int ix = 1; ix <= xaxis->GetNbins(); ix++) {
272 xmax = xaxis->GetBinUpEdge(ix);
273 xmin = xaxis->GetBinLowEdge(ix);
274 double binofx[] = {xmin, xmax, xmax, xmin};
275 double binofy[] = {ymin, ymin, ymax, ymax};
276 poly->AddBin(4, binofx, binofy);
277 }
278 }
279
280 for (int iy = 1; iy <= yaxis->GetNbins(); iy++) {
281 ymax = yaxis->GetBinUpEdge(iy);
282 ymin = yaxis->GetBinLowEdge(iy);
283 for (int ix = 1; ix <= xaxis->GetNbins(); ix++) {
284 xmax = xaxis->GetBinUpEdge(ix);
285 xmin = xaxis->GetBinLowEdge(ix);
286
287 // Get the content of the corresponding bin in TH2D and set it in TH2Poly
288 int bin = hist->GetBin(ix, iy);

298//WP: Scale a TH2Poly and divide by bin width
299TH2Poly* PolyScaleWidth(TH2Poly *Histogram, double scale) {
300// ****************
301 TH2Poly* HistCopy = static_cast<TH2Poly*>(Histogram->Clone());
302 double xlow, xup, ylow, yup, area;
303
304 for(int i = 1; i < HistCopy->GetNumberOfBins()+1; i++)
305 {
306 TH2PolyBin* bin = static_cast<TH2PolyBin*>(Histogram->GetBins()->At(i - 1));
307 xlow = bin->GetXMin();
308 xup = bin->GetXMax();
309 ylow = bin->GetYMin();
310 yup = bin->GetYMax();
311 area = (xup-xlow)*(yup-ylow);
312 HistCopy->SetBinContent(i, Histogram->GetBinContent(i)/(area*scale));
313 }
314 return HistCopy;
315}

395

397 var->SetPointEYlow(m, var->GetY()[m] - xlow);

412// **************************************************************************
413// Recalculate Q^2 after Eb shift. Takes in shifted lepton momentum, lepton angle, and true neutrino energy
414double CalculateQ2(double PLep, double PUpd, double EnuTrue, double InitialQ2){
415// ***************************************************************************
416 constexpr double MLep = 0.10565837;
417
418 // Caluclate muon energy
419 double ELep = sqrt((MLep*MLep)+(PLep*PLep));
420
421 double CosTh = (2*EnuTrue*ELep - MLep*MLep - InitialQ2)/(2*EnuTrue*PLep);
422
423 ELep = sqrt((MLep*MLep)+(PUpd*PUpd));
424
425 // Calculate the new Q2
426 double Q2Upd = -(MLep*MLep) + 2.0*EnuTrue*(ELep - PUpd*CosTh);
427
428 return Q2Upd - InitialQ2;
429}

431// **************************************************************************
432// Recalculate Enu after Eb shift. Takes in shifted lepton momentum, lepton angle, and binding energy change, and if nu/anu
433double CalculateEnu(double PLep, double costh, double Eb, bool neutrino){
434// ***************************************************************************
435 double mNeff = 0.93956536 - Eb / 1000.;
436 double mNoth = 0.93827203;
437
438 if (!neutrino) {
439 mNeff = 0.93827203 - Eb / 1000.;
440 mNoth = 0.93956536;
441 }
442
443 double mLep = 0.10565837;
444 double eLep = sqrt(PLep * PLep + mLep * mLep);
445
446 double Enu = (2 * mNeff * eLep - mLep * mLep + mNoth * mNoth - mNeff * mNeff) /(2 * (mNeff - eLep + PLep * costh));
447
448 return Enu;
449}
