312void SampleSummary::AddData(std::vector<TH2Poly*> &Data) {

512 {

2552// ****************
2553// Get the mode error from a TH1D
2554double SampleSummary::GetModeError(TH1D* hpost){
2555// ****************
2556 // Get the bin which has the largest posterior density
2557 int MaxBin = hpost->GetMaximumBin();
2558
2559 // The total integral of the posterior
2560 const double Integral = hpost->Integral();
2561
2562 int LowBin = MaxBin;
2563 int HighBin = MaxBin;
2564 double sum = hpost->GetBinContent(MaxBin);;
2565 double LowCon = 0.0;
2566 double HighCon = 0.0;
2567 while (sum/Integral < 0.6827 && (LowBin > 0 || HighBin < hpost->GetNbinsX()+1) )
2568 {
2569 LowCon = 0.0;
2570 HighCon = 0.0;
2571 //KS:: Move further only if you haven't reached histogram end
2572 if(LowBin > 1)
2573 {
2574 LowBin--;
2575 LowCon = hpost->GetBinContent(LowBin);
2576 }
2577 if(HighBin < hpost->GetNbinsX())
2578 {
2579 HighBin++;
2580 HighCon = hpost->GetBinContent(HighBin);
2581 }
2582
2583 // If we're on the last slice and the lower contour is larger than the upper
2584 if ((sum+LowCon+HighCon)/Integral > 0.6827 && LowCon > HighCon) {
2585 sum += LowCon;
2586 break;
2587 // If we're on the last slice and the upper contour is larger than the lower
2588 } else if ((sum+LowCon+HighCon)/Integral > 0.6827 && HighCon >= LowCon) {
2589 sum += HighCon;
2590 break;
2591 } else {
2592 sum += LowCon + HighCon;
2593 }
2594 }
2595
2596 double Mode_Error = 0.0;
2597 if (LowCon > HighCon) {
2598 Mode_Error = std::fabs(hpost->GetBinCenter(LowBin)-hpost->GetBinCenter(MaxBin));
2599 } else {
2600 Mode_Error = std::fabs(hpost->GetBinCenter(HighBin)-hpost->GetBinCenter(MaxBin));
2601 }
2602
2603 return Mode_Error;
2604}
