81int manager::GetMCStatLLH() {
82// *************************
83 int mc_stat_llh = kNTestStatistics;
84 if (config["LikelihoodOptions"])
85 {
86 auto likelihood = GetFromManager<std::string>(config["LikelihoodOptions"]["TestStatistic"], "Barlow-Beeston", __FILE__ , __LINE__);
87
88 for(int i = 0; i < kNTestStatistics; i++) {
89 if(likelihood == TestStatistic_ToString(TestStatistic(i))) {
90 mc_stat_llh = TestStatistic(i);
91 break;
92 }
93 }
94 if(mc_stat_llh == kNTestStatistics)
95 {
96 MACH3LOG_ERROR("Wrong form of test-statistic specified!");
97 MACH3LOG_ERROR("You gave {} and I only support:", likelihood);
98 for(int i = 0; i < kNTestStatistics; i++)
99 {
100 MACH3LOG_ERROR("{}", TestStatistic_ToString(TestStatistic(i)));
101 }
102 throw MaCh3Exception(__FILE__ , __LINE__ );
103 }
104 } else {
105 mc_stat_llh = kPoisson;
106 }
107 return mc_stat_llh;
108}
