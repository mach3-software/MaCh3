54// ***************************************************************************
55 if(xsecgraph == NULL){

58 }
59
60 // EM: cap the weights of the knots if specified in the config
61 if(
62 (XsecCov->GetParSplineKnotUpperBound(splineParsIndex) != 9999)
63 || (XsecCov->GetParSplineKnotLowerBound(splineParsIndex) != -9999))
64 {
65 for(int knotId = 0; knotId < xsecgraph->GetN(); knotId++){
66 double x,y;
67
68 // EM: get the x and y of the point. Also double check that the requested point was valid just to be super safe
69 if( xsecgraph->GetPoint(knotId, x, y) == -1) {
70 MACH3LOG_ERROR("Invalid knot requested: {}", knotId);
71 throw MaCh3Exception(__FILE__ , __LINE__ );
72 }
73
74 y = std::min(y, XsecCov->GetParSplineKnotUpperBound(splineParsIndex));
75 y = std::max(y, XsecCov->GetParSplineKnotLowerBound(splineParsIndex));
76
77 xsecgraph->SetPoint(knotId, x, y);
78 }
79
80 // EM: check if our cap made the spline flat, if so we set to 1 knot to avoid problems later on
81 bool isFlat = true;
82
83 for(int knotId = 0; knotId < xsecgraph->GetN(); knotId++){
84 double x,y;
85
86 // EM: get the x and y of the point. Also double check that the requested point was valid just to be super safe
87 if( xsecgraph->GetPoint(knotId, x, y) == -1) {
88 MACH3LOG_ERROR("Invalid knot requested: {}", knotId);
89 throw MaCh3Exception(__FILE__ , __LINE__ );
90 }

538 } // end of if(nPoints !=2)

568 inline int FindX(double x) {
569 // The segment we're interested in (klow in ROOT code)
570 int segment = 0;
571 int kHigh = nPoints-1;
572 // If the variation is below the lowest saved spline point
573 if (x <= XPos[0]){
574 segment = 0;
575 // If the variation is above the highest saved spline point
576 } else if (x >= XPos[nPoints-1]) {
577 // Yes, the -2 is indeed correct, see TSpline.cxx:814 and //see: https://savannah.cern.ch/bugs/?71651
578 segment = kHigh;
579 // If the variation is between the maximum and minimum, perform a binary search
580 } else {
581 // The top point we've got
582 int kHalf = 0;
583 // While there is still a difference in the points (we haven't yet found the segment)
584 // This is a binary search, incrementing segment and decrementing kHalf until we've found the segment
585 while (kHigh - segment > 1) {
586 // Increment the half-step
587 kHalf = (segment + kHigh)/2;
588 // If our variation is above the kHalf, set the segment to kHalf
589 if (x > XPos[kHalf]) {
590 segment = kHalf;
591 // Else move kHigh down
592 } else {
593 kHigh = kHalf;
594 }
595 } // End the while: we've now done our binary search
596 } // End the else: we've now found our point
597 if (segment >= nPoints-1 && nPoints > 1) segment = nPoints-2;
598 return segment;
599 }

604 int segment = FindX(var);
605 // The get the coefficients for this variation
606 M3::float_t x = M3::float_t(-999.99), y = M3::float_t(-999.99), b = M3::float_t(-999.99), c = M3::float_t(-999.99), d = M3::float_t(-999.99);
607 GetCoeff(segment, x, y, b, c, d);
608 double dx = var - x;
609 // Evaluate the third order polynomial
610 double weight = y+dx*(b+dx*(c+d*dx));
611 return weight;
612 }

673// *****************************************
674 int Np = spl->GetNp();
675 M3::float_t x, y, b, c, d;
676 // Go through spline segment parameters,
677 // Get y values for each spline knot,
678 // Every knot must evaluate to 1.0 to create a flat spline
679 for(int i = 0; i < Np; i++) {
680 spl->GetCoeff(i, x, y, b, c, d);

682 return false;
683 }
684 }

703 // Make the temp vector

705 TempVector.reserve(OuterIt->size());
706 int InnerCounter = 0;
707 // Loop over each TSpline3 pointer
708 for (InnerIt = OuterIt->begin(); InnerIt != OuterIt->end(); ++InnerIt, ++InnerCounter) {

717 // Push back onto new vector
718 TempVector.push_back(red);
719 } // End inner for loop
720 ReducedVector.push_back(TempVector);
721 } // End outer for loop
