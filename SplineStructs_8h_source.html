118 for(int knotId = 0; knotId < xsecgraph->GetN(); knotId++){
119 double x,y;

122 if( xsecgraph->GetPoint(knotId, x, y) == -1) {

134 bool isFlat = true;

137 double x,y;

352 // Save the y response at each knot
353 YResp = new _float_[nPoints];
354
355 //KS: Default TSpline3 ROOT implementation

357 {
358 for (int i = 0; i < nPoints; ++i) {
359 // 3 is the size of the TSpline3 coefficients

363 XPos[i] = x;
364 YResp[i] = y;
365 Par[i][0] = b;
366 Par[i][1] = c;
367 Par[i][2] = d;
368 }
369 }
370 //CW: Reduce to use linear spline interpolation for certain parameters
371 // Not the most elegant way: use TSpline3 object but set coefficients to zero and recalculate spline points; the smart way (but more human intensive) would be to save memory here and simply not store the zeros at all
372 // Get which parameters should be linear from the fit manager
373 // Convert the spline number to global xsec parameter
374 // Loop over the splines points
375 // KS: kLinearFunc should be used with TF1, this is just as safety
376 else if(InterPolation == kLinear || InterPolation == kLinearFunc)
377 {
378 for (int k = 0; k < nPoints; ++k) {
379 // 3 is the size of the TSpline3 coefficients
380 Par[k] = new _float_[3];
381 Double_t x1, y1, b1, c1, d1, x2, y2, b2, c2, d2 = 0;
382 spline->GetCoeff(k, x1, y1, b1, c1, d1);
383 spline->GetCoeff(k+1, x2, y2, b2, c2, d2);
384 double tempb = (y2-y1)/(x2-x1);
385
386 XPos[k] = x1;
387 YResp[k] = y1;
388 Par[k][0] = tempb;
389 Par[k][1] = 0;
390 Par[k][2] = 0;
391 }
392 }
393 //EM: Akima spline is similar to regular cubic spline but is allowed to be discontinuous in 2nd derivative and coefficients in any segment
394 // only depend on th 2 nearest points on either side
395 else if(InterPolation == kAkima)
396 {
397 // get the knot values for the spline
398 for (int i = 0; i < nPoints; ++i) {
399 // 3 is the size of the TSpline3 coefficients
400 Par[i] = new _float_[3];
401
402 double x = -999.99, y = -999.99;
403 spline->GetKnot(i, x, y);
404
405 XPos[i] = x;
406 YResp[i] = y;
407 }
408
409 _float_* mvals = new _float_[nPoints + 3];
410 _float_* svals = new _float_[nPoints + 1];
411
412 for (int i = -2; i <= nPoints; ++i) {
413 // if segment is first or last or 2nd to first or last, needs to be dealt with slightly differently;
414 // need to estimate the values for additinal points which would lie outside of the spline
415 if(i ==-2){
416 mvals[i+2] = 3.0 * (YResp[1] - YResp[0]) / (XPos[1] - XPos[0]) - 2.0*(YResp[2] - YResp[1]) / (XPos[2] - XPos[1]);
417 }
418 else if(i==-1){
419 mvals[i+2] = 2.0 * (YResp[1] - YResp[0]) / (XPos[1] - XPos[0]) - (YResp[2] - YResp[1]) / (XPos[2] - XPos[1]);
420 }
421 else if(i==nPoints){
422 mvals[i+2] = 3.0 * (YResp[nPoints-1] - YResp[nPoints-2]) / (XPos[nPoints-1] - XPos[nPoints-2]) - 2.0*(YResp[nPoints-2] - YResp[nPoints-3]) / (XPos[nPoints-2] - XPos[nPoints-3]);
423 }
424 else if(i == nPoints - 1){
425 mvals[i+2] = 2.0 * (YResp[nPoints-1] - YResp[nPoints-2]) / (XPos[nPoints-1] - XPos[nPoints-2]) - (YResp[nPoints-2] - YResp[nPoints-3]) / (XPos[nPoints-2] - XPos[nPoints-3]);
426 }
427 //standard internal segment
428 else{
429 mvals[i+2] = (YResp[i+1] - YResp[i])/ (XPos[i+1] - XPos[i]);
430 }
431 }
432
433 for(int i =2; i<=nPoints+2; i++){
434 if (abs(mvals[i+1] - mvals[i]) + abs(mvals[i-1] - mvals[i-2]) != 0.0){
435 svals[i-2] = (abs(mvals[i+1] - mvals[i]) * mvals[i-1] + abs(mvals[i-1] - mvals[i-2]) *mvals[i]) / (abs(mvals[i+1] - mvals[i]) + abs(mvals[i-1] - mvals[i-2]));
436 }
437 else{svals[i-2] = mvals[i];}
438 }
439
440 // calculate the coefficients for the spline
441 for(int i = 0; i <nPoints; i++){
442 _float_ b, c, d = -999.999;
443
444 b = svals[i];
445 c = (3.0* (YResp[i+1] - YResp[i]) / (XPos[i+1] - XPos[i]) -2.0 *svals[i] - svals[i +1]) /(XPos[i+1] - XPos[i]);
446 d = ((svals[i + 1] +svals[i]) - 2.0*(YResp[i+1] - YResp[i]) / (XPos[i+1] - XPos[i])) / ((XPos[i+1] - XPos[i]) * (XPos[i+1] - XPos[i]));
447
448 Par[i][0] = b;
449 Par[i][1] = c;
450 Par[i][2] = d;
451 }
452
453 // check the input spline for linear segments, if there are any then overwrite the calculated coefficients
454 // this will pretty much only ever be the case if they are set to be linear in samplePDFND i.e. the user wants it to be linear
455 for(int i = 0; i <nPoints-1; i++){
456 double x = -999.99, y = -999.99, b = -999.99, c = -999.99, d = -999.99;
457 spline->GetCoeff(i, x, y, b, c, d);
458
459 if((c == 0.0 && d == 0.0)){
460 Par[i][0] = b;
461 Par[i][1] = 0.0;
462 Par[i][2] = 0.0;
463 }
464 }
465 delete[] mvals;
466 delete[] svals;
467 }

469 // must be between its two nearest knots, DOES NOT make the entire spline monotonic, only the segments
470 else if(InterPolation == kMonotonic)
471 {
472 // values of the secants at each point (for calculating monotone spline)
473 _float_ * Secants = new _float_[nPoints -1];
474 // values of the tangens at each point (for calculating monotone spline)

478 for (int i = 0; i < nPoints; ++i) {
479 // 3 is the size of the TSpline3 coefficients

489 }

533
534 if (alpha <0.0){
535 Tangents[i] = 0.0;
536 }
537 if (beta < 0.0){
538 Tangents[i+1] = 0.0;
539 }
540
541 if (alpha * alpha + beta * beta >9.0){
542 _float_ tau = 3.0 / sqrt(alpha * alpha + beta * beta);
543 Tangents[i] = tau * alpha * Secants[i];
544 Tangents[i+1] = tau * beta * Secants[i];
545 }
546 }
547 } // finished rescaling tangents
548 // fourth pass over knots to calculate the coefficients for the spline
549 _float_ dx;
550 for(int i = 0; i <nPoints-1; i++){
551 _float_ b, c, d = -999.999;
552 dx = XPos[i+1] - XPos[i];
553
554 b = Tangents[i] * dx;
555 c = 3.0* (YResp[i+1] - YResp[i]) -2.0 *dx * Tangents[i] - dx * Tangents[i +1];
556 d = 2.0* (YResp[i] - YResp[i+1]) + dx * (Tangents[i] + Tangents[i+1]);
557
558 Par[i][0] = b / dx;
559 Par[i][1] = c / (dx * dx);
560 Par[i][2] = d / (dx * dx * dx);
561
562 if((Par[i][0] == -999) | (Par[i][1] == -999) | (Par[i][2] ==-999) | (Par[i][0] == -999.999) | (Par[i][1] == -999.999) | (Par[i][2] ==-999.999)){
563 MACH3LOG_INFO("Bad spline parameters for segment {}: (b, c, d) = {}, {}, {}. This will cause problems with GPU.",
564 i, Par[i][0], Par[i][1], Par[i][2]);
565 }
566 MACH3LOG_TRACE("b: {}", b);
567 MACH3LOG_TRACE("dx: {}, x_0: {}, x_1: {}", dx, XPos[i], XPos[i+1]);
568 MACH3LOG_TRACE(" y_0: {}, y_1: {}", YResp[i], YResp[i+1]);
569 }
570
571 // include params for final "segment" outside of the spline so that par array fits with x and y arrays,
572 // should never actually get used but if not set then the GPU code gets very angry
573 Par[nPoints-1][0] = 0.0;
574 Par[nPoints-1][1] = 0.0;
575 Par[nPoints-1][2] = 0.0;
576
577 // check the input spline for linear segments, if there are any then overwrite the calculated coefficients
578 // this will pretty much only ever be the case if they are set to be linear in samplePDFND i.e. the user wants it to be linear
579 for(int i = 0; i <nPoints-1; i++){
580 double x = -999.99, y = -999.99, b = -999.99, c = -999.99, d = -999.99;
581 spline->GetCoeff(i, x, y, b, c, d);
582
583 if((c == 0.0 && d == 0.0)){
584 Par[i][0] = b;
585 Par[i][1] = 0.0;
586 Par[i][2] = 0.0;
587 }
588 }
589 delete[] Secants;
590 delete[] Tangents;
591 } // end of if(nPoints !=2)
592 }
593 else
594 {
595 MACH3LOG_ERROR("Unsupported interpolation type {}", static_cast<int>(InterPolation));
596 throw MaCh3Exception(__FILE__ , __LINE__ );
597 }

621 inline int FindX(double x) {
622 // The segment we're interested in (klow in ROOT code)
623 int segment = 0;
624 int kHigh = nPoints-1;
625 // If the variation is below the lowest saved spline point
626 if (x <= XPos[0]){
627 segment = 0;
628 // If the variation is above the highest saved spline point
629 } else if (x >= XPos[nPoints-1]) {
630 // Yes, the -2 is indeed correct, see TSpline.cxx:814 and //see: https://savannah.cern.ch/bugs/?71651
631 segment = kHigh;
632 // If the variation is between the maximum and minimum, perform a binary search
633 } else {
634 // The top point we've got
635 int kHalf = 0;
636 // While there is still a difference in the points (we haven't yet found the segment)
637 // This is a binary search, incrementing segment and decrementing kHalf until we've found the segment
638 while (kHigh - segment > 1) {
639 // Increment the half-step
640 kHalf = (segment + kHigh)/2;
641 // If our variation is above the kHalf, set the segment to kHalf
642 if (x > XPos[kHalf]) {
643 segment = kHalf;
644 // Else move kHigh down
645 } else {
646 kHigh = kHalf;
647 }
648 } // End the while: we've now done our binary search
649 } // End the else: we've now found our point
650 if (segment >= nPoints-1 && nPoints > 1) segment = nPoints-2;
651 return segment;
652 }

657 int segment = FindX(var);
658 // The get the coefficients for this variation
659 _float_ x = -999.99, y = -999.99, b = -999.99, c = -999.99, d = -999.99;
660 GetCoeff(segment, x, y, b, c, d);

662 // Evaluate the third order polynomial
663 double weight = y+dx*(b+dx*(c+d*dx));
664 return weight;
665 }

725inline bool isFlat(TSpline3_red* &spl) {
726// *****************************************
727 int Np = spl->GetNp();
728 _float_ x, y, b, c, d;
729 // Go through spline segment parameters,
730 // Get y values for each spline knot,
731 // Every knot must evaluate to 1.0 to create a flat spline
732 for(int i = 0; i < Np; i++) {
733 spl->GetCoeff(i, x, y, b, c, d);
734 if (y != 1) {
735 return false;
736 }
737 }
738 return true;
