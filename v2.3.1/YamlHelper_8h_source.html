186 inline bool compareYAMLNodes(const YAML::Node& node1, const YAML::Node& node2, bool Mute = false) {
187 // **********************
188  // Check if the types of the nodes match
189  if (node1.Type() != node2.Type()) {
190  return false;
191  }
192 
193  // Compare scalar types (like strings, numbers)
194  if (node1.IsScalar() && node2.IsScalar()) {
195  auto v1 = node1.as<std::string>();
196  auto v2 = node2.as<std::string>();
197  if (v1 != v2) {
198  if (!Mute) MACH3LOG_WARN("Scalar value mismatch: '{}' != '{}'", v1, v2);
199  return false;
200  }
201  return true;
202  }
203 
204  // Compare sequences (like YAML lists)
205  if (node1.IsSequence() && node2.IsSequence()) {
206  if (node1.size() != node2.size()) {
207  return false;
208  }
209  for (std::size_t i = 0; i < node1.size(); ++i) {
210  if (!compareYAMLNodes(node1[i], node2[i])) {
211  return false;
212  }
213  }
214  return true;
215  }
216 
217  // Compare maps (like YAML dictionaries)
218  if (node1.IsMap() && node2.IsMap()) {
219  if (node1.size() != node2.size()) {
220  MACH3LOG_WARN("Map size mismatch: {} != {}",
221  node1.size(), node2.size());
222  return false;
223  }
224  for (auto it1 = node1.begin(); it1 != node1.end(); ++it1) {
225  auto key = it1->first.as<std::string>();
226  if (!node2[key]) {
227  MACH3LOG_WARN("Key '{}' missing in second YAML", key);
228  return false;
229  }
230  if (!compareYAMLNodes(it1->second, node2[key])) {
231  MACH3LOG_WARN("Value mismatch at key '{}'", key);
232  return false;
233  }
234  }
235  return true;
236  }
237 
238  // Default case: if it's neither scalar, sequence, nor map, consider it unequal
239  if (!Mute) MACH3LOG_WARN("Unsupported YAML node type encountered");
240  return false;
241 }

353 inline YAML::Node LoadYamlConfig(const std::string& filename, const std::string& File, const int Line) {
354 // **********************
355  // KS: YAML can be dumb and not throw error if you pass toml for example...
356  if (!(filename.length() >= 5 && filename.compare(filename.length() - 5, 5, ".yaml") == 0) &&
357  !(filename.length() >= 4 && filename.compare(filename.length() - 4, 4, ".yml") == 0)) {
358  MACH3LOG_ERROR("Invalid file extension: {}\n", filename);
359  throw MaCh3Exception(File, Line);
360  }
361 
362  try {
363  YAML::Node yamlNode = YAML::LoadFile(filename);
364 
365  // Convert the YAML file to string
366  std::ifstream fileStream(filename);
367  std::stringstream buffer;
368  buffer << fileStream.rdbuf();
369  std::string fileContent = buffer.str();
370 
371  // Use regex to find any line starting with `-` but not followed by space, digit, or dot
372  // This excludes valid numeric negative values like -1760.0
373  std::regex linePattern(R"(^\s*-(?![\d\s\.]).*)");
374 
375  std::istringstream contentStream(fileContent);
376  std::string lineconfig;
377  int lineNumber = 1;
378 
379  // KS: Instead of boolean, track flow-style list nesting depth
380  int flowListDepth = 0;
381 
382  while (std::getline(contentStream, lineconfig)) {
383  // Ignore YAML document separator
384  if (lineconfig.find("---") != std::string::npos) {
385  lineNumber++;
386  continue;
387  }
388 
389  // Update flow list depth (increment for each '[' and decrement for each ']')
390  for (char c : lineconfig) {
391  if (c == '[') flowListDepth++;
392  else if (c == ']') flowListDepth = std::max(0, flowListDepth - 1);
393  }
394 
395  // Check lines only outside flow-style lists
396  if (flowListDepth == 0 && std::regex_match(lineconfig, linePattern)) {
397  MACH3LOG_ERROR("Warning: Missing space or tab after '-' at line {}: {}\n", lineNumber, lineconfig);
398  throw MaCh3Exception(File, Line);
399  }
400 
401  lineNumber++;
402  }
403 
404  return yamlNode;
405 
406  } catch (const std::exception& e) {
407  MACH3LOG_ERROR("{}\n", e.what());
408  MACH3LOG_ERROR("Can't open file {}\n", filename);
409  throw MaCh3Exception(File, Line);
410  }
411 }

433 inline YAML::Node MergeNodes(const YAML::Node& a, const YAML::Node& b) {
434 // **********************
435  if (!b.IsMap()) {
436  // If b is not a map, merge result is b, unless b is null
437  return b.IsNull() ? a : b;
438  }
439  if (!a.IsMap()) {
440  // If a is not a map, merge result is b
441  return b;
442  }
443  if (!b.size()) {
444  // If a is a map, and b is an empty map, return a
445  return a;
446  }
447  // Create a new map 'c' with the same mappings as a, merged with b
448  auto c = YAML::Node(YAML::NodeType::Map);
449  for (auto n : a) {
450  if (n.first.IsScalar()) {
451  const std::string & key = n.first.Scalar();
452  auto t = (Cnode(b)[key]);
453  if (t) {
454  c[n.first] = MergeNodes(n.second, t);
455  continue;
456  }
457  }
458  c[n.first] = n.second;
459  }
460  // Add the mappings from 'b' not already in 'c'
461  for (auto n : b) {
462  if (!n.first.IsScalar() || !Cnode(c)[n.first.Scalar()]) {
463  c[n.first] = n.second;
464  }
465  }
466  return c;
467 }

482 inline std::vector<double> ParseBounds(const YAML::Node& node, const std::string& File, const int Line) {
483 // **********************
484  std::vector<double> bounds;
485 
486  if (!node || !node.IsSequence() || (node.size() != 1 && node.size() != 2)) {
487  MACH3LOG_ERROR("Bounds must be a sequence of 1 or 2 elements, got size {}", static_cast<int>(node.size()));
488  throw MaCh3Exception(File, Line);
489  }
490 
491  if (node.size() == 1) {
492  const YAML::Node& val = node[0];
493 
494  if (!val || val.IsNull() || (val.IsScalar() && val.Scalar().empty())) {
495  MACH3LOG_ERROR("Single-element bounds must contain a valid numeric value.");
496  throw MaCh3Exception(File, Line);
497  } else if (val.IsScalar()) {
498  try {
499  const double center = val.as<double>();
500  if (std::floor(center) != center) {
501  MACH3LOG_ERROR("Invalid center value in Bounds[0]: '{}'. Expected an integer (e.g., 0 or 1).", static_cast<std::string>(val.Scalar()));
502  throw MaCh3Exception(File, Line);
503  }
504  bounds = {center - 0.5, center + 0.5};
505  } catch (const YAML::BadConversion& e) {
506  MACH3LOG_ERROR("Invalid value in Bounds[0]: '{}'. Expected a number.", static_cast<std::string>(val.Scalar()));
507  throw MaCh3Exception(File, Line);
508  }
509  } else {
510  MACH3LOG_ERROR("Invalid type in Bounds[0]");
511  throw MaCh3Exception(File, Line);
512  }
513  } else {
514  for (std::size_t i = 0; i < 2; ++i) {
515  const YAML::Node& val = node[i];
516 
517  if (!val || val.IsNull() || (val.IsScalar() && val.Scalar().empty())) {
518  bounds.push_back(i == 0 ? M3::KinematicLowBound : M3::KinematicUpBound);
519  } else if (val.IsScalar()) {
520  try {
521  bounds.push_back(val.as<double>());
522  } catch (const YAML::BadConversion& e) {
523  MACH3LOG_ERROR("Invalid value in Bounds[{}]: '{}'. Expected a number or empty string.",
524  static_cast<int>(i), static_cast<std::string>(val.Scalar()));
525  throw MaCh3Exception(File, Line);
526  }
527  } else {
528  MACH3LOG_ERROR("Invalid type in Bounds[{}]", static_cast<int>(i));
529  throw MaCh3Exception(File, Line);
530  }
531  }
532  }
533  return bounds;
534 }

566 inline std::vector<std::vector<double>> Parse2DBounds(const YAML::Node& node, const std::string& File, const int Line) {
567 // **********************
568  std::vector<std::vector<double>> bounds;

573

577  } else { // Case 2: List of pairs like [[0.25, 0.5], [0.75, 1.0]]

579  const YAML::Node& subnode = node[i];
580  bounds.push_back(ParseBounds(subnode, File, Line));
581  }
582  }
583 
584  return bounds;
