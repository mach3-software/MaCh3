26void PCAHandler::ConstructPCA(TMatrixDSym * covMatrix, const int firstPCAd, const int lastPCAd,
27 const double eigen_thresh, int& _fNumParPCA, const int _fNumPar) {
28// ********************************************
29 FirstPCAdpar = firstPCAd;
30 LastPCAdpar = lastPCAd;
31 eigen_threshold = eigen_thresh;
32
33 // Check that covariance matrix exists
34 if (covMatrix == NULL) {
35 MACH3LOG_ERROR("Covariance matrix for has not yet been set");
36 MACH3LOG_ERROR("Can not construct PCA until it is set");
37 throw MaCh3Exception(__FILE__ , __LINE__ );
38 }
39
40 if(FirstPCAdpar > covMatrix->GetNrows()-1 || LastPCAdpar>covMatrix->GetNrows()-1) {
41 MACH3LOG_ERROR("FirstPCAdpar and LastPCAdpar are higher than the number of parameters");
42 MACH3LOG_ERROR("first: {} last: {}, params: {}", FirstPCAdpar, LastPCAdpar, covMatrix->GetNrows()-1);
43 throw MaCh3Exception(__FILE__ , __LINE__ );
44 }
45 if(FirstPCAdpar < 0 || LastPCAdpar < 0){
46 MACH3LOG_ERROR("FirstPCAdpar and LastPCAdpar are less than 0 but not default -999");
47 MACH3LOG_ERROR("first: {} last: {}", FirstPCAdpar, LastPCAdpar);
48 throw MaCh3Exception(__FILE__ , __LINE__ );
49 }
50 MACH3LOG_INFO("PCAing parameters {} through {} inclusive", FirstPCAdpar, LastPCAdpar);
51 int numunpcadpars = covMatrix->GetNrows()-(LastPCAdpar-FirstPCAdpar+1);
52
53 TMatrixDSym submat(covMatrix->GetSub(FirstPCAdpar,LastPCAdpar,FirstPCAdpar,LastPCAdpar));
54
55 //CW: Calculate how many eigen values this threshold corresponds to
56 TMatrixDSymEigen eigen(submat);
57 eigen_values.ResizeTo(eigen.GetEigenValues());
58 eigen_vectors.ResizeTo(eigen.GetEigenVectors());
59 eigen_values = eigen.GetEigenValues();
60 eigen_vectors = eigen.GetEigenVectors();
61 double sum = 0;
62 // Loop over eigen values and sum them up
63 for (int i = 0; i < eigen_values.GetNrows(); ++i) {
64 sum += eigen_values(i);
65 }
66 nKeptPCApars = eigen_values.GetNrows();
67 //CW: Now go through again and see how many eigen values correspond to threshold
68 for (int i = 0; i < eigen_values.GetNrows(); ++i) {
69 // Get the relative size of the eigen value
70 double sig = eigen_values(i)/sum;
71 // Check against the threshold
72 if (sig < eigen_threshold) {
73 nKeptPCApars = i;
74 break;
75 }
76 }
77 _fNumParPCA = numunpcadpars+nKeptPCApars;
78 NumParPCA = _fNumParPCA;
79 MACH3LOG_INFO("Threshold of {} on eigen values relative sum of eigen value ({}) generates {} eigen vectors, plus we have {} unpcad pars, for a total of {}", eigen_threshold, sum, nKeptPCApars, numunpcadpars, _fNumParPCA);
80
81 //DB Create array of correct size so eigen_values can be used in CorrelateSteps
82 eigen_values_master = std::vector<double>(_fNumParPCA, 1.0);
83 for (int i = FirstPCAdpar; i < FirstPCAdpar+nKeptPCApars; ++i) {eigen_values_master[i] = eigen_values(i-FirstPCAdpar);}
84
85 // Now construct the transfer matrices
86 //These matrices will be as big as number of unPCAd pars plus number of eigenvalues kept
87 TransferMat.ResizeTo(covMatrix->GetNrows(), _fNumParPCA);
88 TransferMatT.ResizeTo(covMatrix->GetNrows(), _fNumParPCA);
89
90 // Get a subset of the eigen vector matrix
91 TMatrixD temp(eigen_vectors.GetSub(0, eigen_vectors.GetNrows()-1, 0, nKeptPCApars-1));
92
93 //Make transfer matrix which is two blocks of identity with a block of the PCA transfer matrix in between
94 TMatrixD temp2;
95 temp2.ResizeTo(covMatrix->GetNrows(), _fNumParPCA);
96
97 //First set the whole thing to 0
98 for(int iRow = 0; iRow < covMatrix->GetNrows(); iRow++){
99 for(int iCol = 0; iCol < _fNumParPCA; iCol++){
100 temp2[iRow][iCol] = 0;
101 }
102 }
103 //Set the first identity block
104 if(FirstPCAdpar != 0){
105 for(int iRow = 0; iRow < FirstPCAdpar; iRow++){
106 temp2[iRow][iRow] = 1;
107 }
108 }
109
110 //Set the transfer matrix block for the PCAd pars
111 temp2.SetSub(FirstPCAdpar,FirstPCAdpar,temp);
112
113 //Set the second identity block
114 if(LastPCAdpar != covMatrix->GetNrows()-1){
115 for(int iRow = 0;iRow < (covMatrix->GetNrows()-1)-LastPCAdpar; iRow++){
116 temp2[LastPCAdpar+1+iRow][FirstPCAdpar+nKeptPCApars+iRow] = 1;
117 }
118 }
119
120 TransferMat = temp2;
121 // Copy the contents
122 TransferMatT = TransferMat;
123 // And then transpose
124 TransferMatT.T();
125
126 #ifdef DEBUG_PCA
127 //KS: Let's dump all useful matrices to properly validate PCA
128 DebugPCA(sum, temp, submat, covMatrix->GetNrows());
129 #endif
130
131 // Make the PCA parameter arrays
132 fParCurr_PCA.ResizeTo(_fNumParPCA);
133 fParProp_PCA.ResizeTo(_fNumParPCA);
134 _fPreFitValue_PCA.resize(_fNumParPCA);
135
136 //KS: make easy map so we could easily find un-decomposed parameters
137 isDecomposed_PCA.resize(_fNumParPCA);
138 fParSigma_PCA.resize(_fNumParPCA);
139 for (int i = 0; i < _fNumParPCA; ++i)
140 {
141 fParSigma_PCA[i] = 1;
142 isDecomposed_PCA[i] = -1;
143 }
144 for (int i = 0; i < FirstPCAdpar; ++i) isDecomposed_PCA[i] = i;
145
146 for (int i = FirstPCAdpar+nKeptPCApars+1; i < _fNumParPCA; ++i) isDecomposed_PCA[i] = i+(_fNumPar-_fNumParPCA);
147}

151void PCAHandler::AcceptStep() _noexcept_ {
152// ********************************************

164// ************************************************
165// Correlate the steps by setting the proposed step of a parameter to its current value + some correlated throw
166void PCAHandler::CorrelateSteps(const std::vector<double>& IndivStepScale,
167 const double GlobalStepScale,
168 const double* _restrict_ randParams,
169 const double* _restrict_ corr_throw) _noexcept_ {
170// ************************************************
171 // Throw around the current step
172 #ifdef MULTITHREAD
173 #pragma omp parallel for
174 #endif
175 for (int i = 0; i < NumParPCA; ++i)
176 {
177 if (fParSigma_PCA[i] > 0.)
178 {
179 double IndStepScale = 1.;
180 //KS: If undecomposed parameter apply individual step scale and Cholesky for better acceptance rate
181 if(isDecomposed_PCA[i] >= 0)
182 {
183 IndStepScale *= IndivStepScale[isDecomposed_PCA[i]];
184 IndStepScale *= corr_throw[isDecomposed_PCA[i]];
185 }
186 //If decomposed apply only random number
187 else
188 {
189 IndStepScale *= randParams[i];
190 //KS: All PCA-ed parameters have the same step scale
191 IndStepScale *= IndivStepScale[FirstPCAdpar];
192 }
193 fParProp_PCA(i) = fParCurr_PCA(i)+GlobalStepScale*IndStepScale*eigen_values_master[i];
194 }
195 }
196 // Then update the parameter basis
197 TransferToParam();
198}

200// ********************************************
201// Transfer a parameter variation in the parameter basis to the eigen basis
202void PCAHandler::TransferToPCA() {
203// ********************************************
204 // Make the temporary vectors
205 TVectorD fParCurr_vec(static_cast<Int_t>(_pCurrVal->size()));
206 TVectorD fParProp_vec(static_cast<Int_t>(_pCurrVal->size()));
207 for(int i = 0; i < static_cast<int>(_pCurrVal->size()); ++i) {
208 fParCurr_vec(i) = (*_pCurrVal)[i];
209 fParProp_vec(i) = (*_pPropVal)[i];
210 }
211
212 fParCurr_PCA = TransferMatT*fParCurr_vec;
213 fParProp_PCA = TransferMatT*fParProp_vec;
214}

216// ********************************************
217// Transfer a parameter variation in the eigen basis to the parameter basis
218void PCAHandler::TransferToParam() {
219// ********************************************
220 // Make the temporary vectors
221 TVectorD fParProp_vec = TransferMat*fParProp_PCA;
222 TVectorD fParCurr_vec = TransferMat*fParCurr_PCA;
223 #ifdef MULTITHREAD
224 #pragma omp parallel for
225 #endif
226 for(int i = 0; i < static_cast<int>(_pCurrVal->size()); ++i) {
227 (*_pPropVal)[i] = fParProp_vec(i);
228 (*_pCurrVal)[i] = fParCurr_vec(i);
229 }
230}
