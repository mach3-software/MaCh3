18void PCAHandler::ConstructPCA(TMatrixDSym * covMatrix, const int firstPCAd, const int lastPCAd, const double eigen_thresh, int& _fNumParPCA) {
19// ********************************************
20 FirstPCAdpar = firstPCAd;
21 LastPCAdpar = lastPCAd;
22 eigen_threshold = eigen_thresh;
23
24 // Check that covariance matrix exists
25 if (covMatrix == NULL) {
26 MACH3LOG_ERROR("Covariance matrix for has not yet been set");
27 MACH3LOG_ERROR("Can not construct PCA until it is set");
28 throw MaCh3Exception(__FILE__ , __LINE__ );
29 }
30
31 if(FirstPCAdpar > covMatrix->GetNrows()-1 || LastPCAdpar>covMatrix->GetNrows()-1) {
32 MACH3LOG_ERROR("FirstPCAdpar and LastPCAdpar are higher than the number of parameters");
33 MACH3LOG_ERROR("first: {} last: {}, params: {}", FirstPCAdpar, LastPCAdpar, covMatrix->GetNrows()-1);
34 throw MaCh3Exception(__FILE__ , __LINE__ );
35 }
36 if(FirstPCAdpar < 0 || LastPCAdpar < 0){
37 MACH3LOG_ERROR("FirstPCAdpar and LastPCAdpar are less than 0 but not default -999");
38 MACH3LOG_ERROR("first: {} last: {}", FirstPCAdpar, LastPCAdpar);
39 throw MaCh3Exception(__FILE__ , __LINE__ );
40 }
41
42 MACH3LOG_INFO("PCAing parameters {} through {} inclusive", FirstPCAdpar, LastPCAdpar);
43 int numunpcadpars = covMatrix->GetNrows()-(LastPCAdpar-FirstPCAdpar+1);
44
45 TMatrixDSym submat(covMatrix->GetSub(FirstPCAdpar,LastPCAdpar,FirstPCAdpar,LastPCAdpar));
46
47 //CW: Calculate how many eigen values this threshold corresponds to
48 TMatrixDSymEigen eigen(submat);
49 eigen_values.ResizeTo(eigen.GetEigenValues());
50 eigen_vectors.ResizeTo(eigen.GetEigenVectors());
51 eigen_values = eigen.GetEigenValues();
52 eigen_vectors = eigen.GetEigenVectors();
53 double sum = 0;
54 // Loop over eigen values and sum them up
55 for (int i = 0; i < eigen_values.GetNrows(); ++i) {
56 sum += eigen_values(i);
57 }
58 nKeptPCApars = eigen_values.GetNrows();
59 //CW: Now go through again and see how many eigen values correspond to threshold
60 for (int i = 0; i < eigen_values.GetNrows(); ++i) {
61 // Get the relative size of the eigen value
62 double sig = eigen_values(i)/sum;
63 // Check against the threshold
64 if (sig < eigen_threshold) {
65 nKeptPCApars = i;
66 break;
67 }
68 }
69 _fNumParPCA = numunpcadpars+nKeptPCApars;
70 MACH3LOG_INFO("Threshold of {} on eigen values relative sum of eigen value ({}) generates {} eigen vectors, plus we have {} unpcad pars, for a total of {}", eigen_threshold, sum, nKeptPCApars, numunpcadpars, _fNumParPCA);
71
72 //DB Create array of correct size so eigen_values can be used in CorrelateSteps
73 eigen_values_master = std::vector<double>(_fNumParPCA, 1.0);
74 for (int i = FirstPCAdpar; i < FirstPCAdpar+nKeptPCApars; ++i) {eigen_values_master[i] = eigen_values(i-FirstPCAdpar);}
75
76 // Now construct the transfer matrices
77 //These matrices will be as big as number of unPCAd pars plus number of eigenvalues kept
78 TransferMat.ResizeTo(covMatrix->GetNrows(), _fNumParPCA);
79 TransferMatT.ResizeTo(covMatrix->GetNrows(), _fNumParPCA);
80
81 // Get a subset of the eigen vector matrix
82 TMatrixD temp(eigen_vectors.GetSub(0, eigen_vectors.GetNrows()-1, 0, nKeptPCApars-1));
83
84 //Make transfer matrix which is two blocks of identity with a block of the PCA transfer matrix in between
85 TMatrixD temp2;
86 temp2.ResizeTo(covMatrix->GetNrows(), _fNumParPCA);
87
88 //First set the whole thing to 0
89 for(int iRow = 0; iRow < covMatrix->GetNrows(); iRow++){
90 for(int iCol = 0; iCol < _fNumParPCA; iCol++){
91 temp2[iRow][iCol] = 0;
92 }
93 }
94 //Set the first identity block
95 if(FirstPCAdpar != 0){
96 for(int iRow = 0; iRow < FirstPCAdpar; iRow++){
97 temp2[iRow][iRow] = 1;
98 }
99 }
100
101 //Set the transfer matrix block for the PCAd pars
102 temp2.SetSub(FirstPCAdpar,FirstPCAdpar,temp);
103
104 //Set the second identity block
105 if(LastPCAdpar != covMatrix->GetNrows()-1){
106 for(int iRow = 0;iRow < (covMatrix->GetNrows()-1)-LastPCAdpar; iRow++){
107 temp2[LastPCAdpar+1+iRow][FirstPCAdpar+nKeptPCApars+iRow] = 1;
108 }
109 }
110
111 TransferMat = temp2;
112 // Copy the contents
113 TransferMatT = TransferMat;
114 // And then transpose
115 TransferMatT.T();
116
117 #ifdef DEBUG_PCA
118 //KS: Let's dump all useful matrices to properly validate PCA
119 DebugPCA(sum, temp, submat, covMatrix->GetNrows());
120 #endif
121}
