208inline TH1D* makeRatio(TH1D *PrefitCopy, TH1D *PostfitCopy, bool setAxes){
209 // set up the ratio hist
210 TH1D* Ratio = static_cast<TH1D*>(PrefitCopy->Clone());
211 Ratio->GetYaxis()->SetTitle("(x_{Post}-#mu_{Prior})/#sigma_{Prior}");
212 Ratio->SetMinimum(-3.7);
213 Ratio->SetMaximum(3.7);
214
215 for (int j = 0; j < Ratio->GetXaxis()->GetNbins(); ++j)
216 {
217 if ( PrefitCopy->GetBinError(j+1) > 1.e-5 )
218 {
219 Ratio->SetBinContent(j+1, (PostfitCopy->GetBinContent(j+1)-PrefitCopy->GetBinContent(j+1))/PrefitCopy->GetBinError(j+1));
220
221 double up = (PostfitCopy->GetBinContent(j+1)+PostfitCopy->GetBinError(j+1)-PrefitCopy->GetBinContent(j+1))/PrefitCopy->GetBinError(j+1);
222 double down = (PostfitCopy->GetBinContent(j+1)-PostfitCopy->GetBinError(j+1)-PrefitCopy->GetBinContent(j+1))/PrefitCopy->GetBinError(j+1);
223
224 double maximum = up-Ratio->GetBinContent(j+1);
225 double minimum = Ratio->GetBinContent(j+1)-down;
226
227 Ratio->SetBinError(j+1, std::max(maximum, minimum));
228 }
229 //KS: Most likely flat prior
230 else {
231 Ratio->SetBinContent(j+1, (PostfitCopy->GetBinContent(j+1)-PrefitCopy->GetBinContent(j+1)));
232
233 double up = (PostfitCopy->GetBinContent(j+1)+PostfitCopy->GetBinError(j+1)-PrefitCopy->GetBinContent(j+1));
234 double down = (PostfitCopy->GetBinContent(j+1)-PostfitCopy->GetBinError(j+1)-PrefitCopy->GetBinContent(j+1));
235
236 double maximum = up-Ratio->GetBinContent(j+1);
237 double minimum = Ratio->GetBinContent(j+1)-down;
238
239 Ratio->SetBinError(j+1, std::max(maximum, minimum));
240 }
241 } //end loop over parameters
242
243 if(setAxes){
244 Ratio->SetFillStyle(0);
245 Ratio->SetFillColor(0);
246
247 Ratio->SetLineColor(PostfitCopy->GetLineColor());
248 if (Ratio->GetLineColor() == 0) Ratio->SetLineColor(kBlack);
249 Ratio->SetMarkerColor(PostfitCopy->GetMarkerColor());
250
251 Ratio->SetLineWidth(man->getOption<int>("plotLineWidth"));
252 Ratio->SetTitle("");
253
254 Ratio->SetMarkerSize(2);
255 Ratio->SetMarkerStyle(20);
256
257 Ratio->GetYaxis()->SetTitleSize(25);
258 Ratio->GetYaxis()->SetTitleFont(43);
259 Ratio->GetYaxis()->SetTitleOffset(2.0);
260 Ratio->GetYaxis()->SetLabelFont(43);
261 Ratio->GetYaxis()->SetLabelSize(25);
262 Ratio->GetYaxis()->CenterTitle();
263 Ratio->GetYaxis()->SetNdivisions(5,2,0);
264
265 Ratio->GetXaxis()->SetTitleSize(25);
266 Ratio->GetXaxis()->SetTitleFont(43);
267 Ratio->GetXaxis()->SetTitleOffset(4.0);
268 Ratio->GetXaxis()->SetLabelOffset(0.025);
269 Ratio->GetXaxis()->SetLabelFont(43);
270 Ratio->GetXaxis()->SetLabelSize(25);
271 }
272
273 return Ratio;
274}

355void MakeXsecPlots()
356{
357 // get the names of the blocks of parameters to group together
358 std::vector<std::string> const blockNames = man->getOption<std::vector<std::string>>("paramGroups");
359 const int XsecPlots = static_cast<int>(blockNames.size());
360
361 for (int i = 0; i < XsecPlots; i++)
362 {
363 // get the configuration for this parameter
364 std::string blockName = blockNames[i];
365 YAML::Node paramBlock = man->getOption(blockName);
366 std::string blockTitle = paramBlock[0].as<std::string>();
367 std::vector<double> blockLimits = paramBlock[1].as<std::vector<double>>();
368 std::vector<std::string> blockContents = paramBlock[2].as<std::vector<std::string>>();
369
370 // get num of params in the block
371 const int nParams = static_cast<int>(blockContents.size());
372
373 // set some plot things
374 TH1D *blockHist_prefit = new TH1D(blockName.c_str(),
375 blockTitle.c_str(), nParams, 0.0, static_cast<double>(nParams));
376
377 man->style().setTH1Style(blockHist_prefit, man->getOption<std::string>("prefitHistStyle"));
378
379 // set the errors for the prefit block hist
380 for(int localBin=0; localBin < nParams; localBin ++){
381 // the "local" bin is the params index within the group of parameters
382 std::string paramName = blockContents[localBin];
383 copyParToBlockHist(localBin, paramName, blockHist_prefit, "Prior", 0);
384 }
385
386 // now set for the postfit blocks for all files
387 std::vector <TH1D *> blockHist_postfit_Vec;
388 for(unsigned int fileId = 0; fileId < man->getNFiles(); fileId++){
389
390 TH1D *blockHist_postfit = new TH1D((blockName + man->getFileName(fileId)).c_str(),
391 blockTitle.c_str(), nParams, 0.0, static_cast<double>(nParams));
392
393 // loop through all the parameters in this block and set the contents in the blocks TH1
394 for(int localBin=0; localBin < nParams; localBin ++){
395 // the "local" bin is the params index within the group of parameters
396 std::string paramName = blockContents[localBin];
397 copyParToBlockHist(localBin, paramName, blockHist_postfit, "", fileId);
398 }
399
400 blockHist_postfit_Vec.push_back(blockHist_postfit);
401 }
402
403 // set the y axis limits we got from config
404 blockHist_prefit->GetYaxis()->SetRangeUser(blockLimits[0], blockLimits[1]);
405
406 // Do some fancy replacements
407 PrettifyTitles(blockHist_prefit);
408
409 DrawPlots(canv, blockHist_prefit, blockHist_postfit_Vec, p3, p4);
410 delete blockHist_prefit;
411 for (auto hist : blockHist_postfit_Vec) {
412 delete hist;
413 }
414 blockHist_postfit_Vec.clear();
415 }
416}

418void MakeFluxPlots()
419{
420 // these for non named params where we dont need as much space
421 TPad* p1 = new TPad("p1", "p1", 0.0, 0.3, 1.0, 1.0);
422 TPad* p2 = new TPad("p2", "p2", 0.0, 0.0, 1.0, 0.3);

432 p2->SetGrid();
433
434 p1->SetLogx(true);
435 p2->SetLogx(true);
436
437 // get the names of the blocks of parameters to group together
438 std::vector<std::string> const fluxBlockNames = man->getOption<std::vector<std::string>>("fluxGroups");
439 auto const fluxBinningTable = man->getOption("FluxBinning");
440
441 const int FluxPlots = static_cast<int>(fluxBlockNames.size());
442
443 for (int i = 0; i < FluxPlots; i++)
444 {
445 // get the configuration for this block
446 std::string fluxBlockName = fluxBlockNames[i];
447 YAML::Node paramBlock = man->getOption(fluxBlockName);
448 std::string blockTitle = paramBlock[0].as<std::string>();
449 std::vector<double> blockLimits = paramBlock[1].as<std::vector<double>>();
450 std::string blockBinningName = paramBlock[2].as<std::string>();
451 std::vector<int> blockContents = paramBlock[3].as<std::vector<int>>();
452
453 // get the binning for this block of flux params
454 std::vector<double> binning = fluxBinningTable[blockBinningName].as<std::vector<double>>();
455
456 // get num of params in the block
457 int nParams = blockContents[1] - blockContents[0] +1;
458 // check for sanity
459 if(nParams <= 0 || blockContents.size() > 2){
460 MACH3LOG_CRITICAL("Invalid flux parameter block endpoints specified for {}", fluxBlockName);
461 MACH3LOG_CRITICAL(" Should have the form [<low index>, <up index>]");
462 throw MaCh3Exception(__FILE__ , __LINE__ );
463 }
464 if (nParams != static_cast<int>(binning.size()) - 1) {
465 MACH3LOG_CRITICAL("Binning provided for flux param block {} does not match the number of parameters specified for the block", fluxBlockName);
466 MACH3LOG_CRITICAL(" Provided {} parameters but {} bins", nParams, binning.size() -1);
467 throw MaCh3Exception(__FILE__ , __LINE__ );
468 }
469
470 TH1D *blockHist_prefit = new TH1D(fluxBlockName.c_str(), blockTitle.c_str(), nParams, binning.data());
471 blockHist_prefit->GetYaxis()->SetTitle("Parameter Variation");
472 blockHist_prefit->GetXaxis()->SetTitle("E_{#nu} (GeV)");
473 blockHist_prefit->GetXaxis()->SetTitleOffset(blockHist_prefit->GetXaxis()->GetTitleOffset()*1.2);
474 man->style().setTH1Style(blockHist_prefit, man->getOption<std::string>("prefitHistStyle"));
475 // set the errors for the prefit block hist
476 for(int fluxParId = blockContents[0]; fluxParId <= blockContents[1]; fluxParId++){
477 int localBin = fluxParId - blockContents[0];
478 std::string paramName = "b_" + std::to_string(fluxParId);
479 copyParToBlockHist(localBin, paramName, blockHist_prefit, "Prior", 0, false);
480 }
481
482 // now set for the postfit blocks for all files
483 std::vector <TH1D *> blockHist_postfit_Vec;
484 for(unsigned int fileId = 0; fileId < man->getNFiles(); fileId++){
485 TH1D *blockHist_postfit = new TH1D(fluxBlockName.c_str(), blockTitle.c_str(), nParams, binning.data());
486
487 for(int fluxParId = blockContents[0]; fluxParId <= blockContents[1]; fluxParId++){
488 int localBin = fluxParId - blockContents[0];
489 std::string paramName = "b_" + std::to_string(fluxParId);
490
491 copyParToBlockHist(localBin, paramName, blockHist_postfit, "", fileId, false);
492 }
493
494 blockHist_postfit_Vec.push_back(blockHist_postfit);
495 }
496
497 // set the y axis limits we got from config
498 blockHist_prefit->GetYaxis()->SetRangeUser(blockLimits[0], blockLimits[1]);
499
500 DrawPlots(canv, blockHist_prefit, blockHist_postfit_Vec, p1, p2);
501 delete blockHist_prefit;
502 for (auto hist : blockHist_postfit_Vec) {
503 delete hist;
504 }
505 blockHist_postfit_Vec.clear();
506 }
507
508 canv->cd();
509 canv->SetLogx(false);
510 canv->SetBottomMargin(canv->GetBottomMargin()*1.7);
511 delete p1;
512 delete p2;
513}

635{

737 posteriorDist->SetLineColor(kWhite);
738 posteriorDist->SetFillColorAlpha(TColor::GetColorPalette(floor(static_cast<float>(parId) *
739 TColor::GetNumberOfColors() / static_cast<float>(nParams))), 0.85);

745 gPad->Modified(); gPad->Update();

749 ridgeCanv->cd();
750 label->DrawLatexNDC(0.29, padBottom + 0.005, man->style().prettifyParamName(paramName).c_str());
751 line->DrawLine(0.1, padBottom, 0.9, padBottom);

755 ridgeCanv->SetGrid(1,1);
756 TPad *axisPad = new TPad("AxisPad", "", 0.3, 0.0, 0.9, 1.0, -1, 0, -1);

771 axisPlot->Draw("AXIG SAME");

773 axisPlot->SetFillStyle(4000);
774 axisPad->SetFillStyle(4000);
775
776 axisPad->SetGrid(1,1);
777 gPad->Modified(); gPad->Update();
778 gPad->SetFrameFillStyle(4000);
779

873inline TGraphAsymmErrors* MakeTGraphAsymmErrors(std::shared_ptr<TFile> File, std::vector<int> Index = {})
874{
875 int GraphBins = Index.size() == 0 ? nBins : Index.size();
876 std::vector<double> x(GraphBins);
877 std::vector<double> y(GraphBins);
878 std::vector<double> exl(GraphBins);
879 std::vector<double> eyl(GraphBins);
880 std::vector<double> exh(GraphBins);
881 std::vector<double> eyh(GraphBins);
882
883 TH1D* PostHist = static_cast<TH1D*>(File->Get( ("param_xsec_"+plotType).c_str() ));
884
885 TVectorD* Errors_HPD_Positive = static_cast<TVectorD*>(File->Get( "Errors_HPD_Positive" ));
886 TVectorD* Errors_HPD_Negative = static_cast<TVectorD*>(File->Get( "Errors_HPD_Negative" ));
887 //KS: I am tempted to multithread this...
888 for(int i = 0; i < GraphBins; ++i)
889 {
890 int Counter = Index.size() == 0 ? i : Index[i];
891 //KS: We are extracting value from three object each having different numbering scheme, I have checked carefully so this is correct please don't change all these +1 +0.5 etc. it just work...
892 x[i] = i + 0.5;
893 y[i] = PostHist->GetBinContent(Counter+1);
894
895 //KS: We don't want x axis errors as they are confusing in Violin plot
896 exh[i] = 0.00001;
897 exl[i] = 0.00001;
898 eyh[i] = (*Errors_HPD_Positive)(Counter);
899 eyl[i] = (*Errors_HPD_Negative)(Counter);
900 }
901 TGraphAsymmErrors* PostGraph = new TGraphAsymmErrors(GraphBins, x.data(), y.data(), exl.data(), exh.data(), eyl.data(), eyh.data());
902 PostGraph->SetTitle("");
903
904 return PostGraph;
905}
