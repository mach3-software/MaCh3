147inline TH1D* makeRatio(TH1D *PrefitCopy, TH1D *PostfitCopy, bool setAxes){
148 // set up the ratio hist
149 TH1D *Ratio = (TH1D*)PrefitCopy->Clone();
150 Ratio->GetYaxis()->SetTitle("(x_{Post}-#mu_{Prior})/#sigma_{Prior}");
151 Ratio->SetMinimum(-3.7);
152 Ratio->SetMaximum(3.7);
153
154 for (int j = 0; j < Ratio->GetXaxis()->GetNbins(); ++j)
155 {
156 if ( PrefitCopy->GetBinError(j+1) > 1.e-5 )
157 {
158 Ratio->SetBinContent(j+1, (PostfitCopy->GetBinContent(j+1)-PrefitCopy->GetBinContent(j+1))/PrefitCopy->GetBinError(j+1));
159
160 double up = (PostfitCopy->GetBinContent(j+1)+PostfitCopy->GetBinError(j+1)-PrefitCopy->GetBinContent(j+1))/PrefitCopy->GetBinError(j+1);
161 double down = (PostfitCopy->GetBinContent(j+1)-PostfitCopy->GetBinError(j+1)-PrefitCopy->GetBinContent(j+1))/PrefitCopy->GetBinError(j+1);
162
163 double maximum = up-Ratio->GetBinContent(j+1);
164 double minimum = Ratio->GetBinContent(j+1)-down;
165
166 Ratio->SetBinError(j+1, std::max(maximum, minimum));
167 }
168 //KS: Most likely flat prior
169 else {
170 Ratio->SetBinContent(j+1, (PostfitCopy->GetBinContent(j+1)-PrefitCopy->GetBinContent(j+1)));
171
172 double up = (PostfitCopy->GetBinContent(j+1)+PostfitCopy->GetBinError(j+1)-PrefitCopy->GetBinContent(j+1));
173 double down = (PostfitCopy->GetBinContent(j+1)-PostfitCopy->GetBinError(j+1)-PrefitCopy->GetBinContent(j+1));
174
175 double maximum = up-Ratio->GetBinContent(j+1);
176 double minimum = Ratio->GetBinContent(j+1)-down;
177
178 Ratio->SetBinError(j+1, std::max(maximum, minimum));
179 }
180 } //end loop over parameters
181
182 if(setAxes){
183 Ratio->SetFillStyle(0);
184 Ratio->SetFillColor(0);
185
186 Ratio->SetLineColor(PostfitCopy->GetLineColor());
187 if (Ratio->GetLineColor() == 0) Ratio->SetLineColor(kBlack);
188 Ratio->SetMarkerColor(PostfitCopy->GetMarkerColor());
189
190 Ratio->SetLineWidth(man->getOption<int>("plotLineWidth"));
191 Ratio->SetTitle("");
192
193 Ratio->SetMarkerSize(2);
194 Ratio->SetMarkerStyle(20);
195
196 Ratio->GetYaxis()->SetTitleSize(25);
197 Ratio->GetYaxis()->SetTitleFont(43);
198 Ratio->GetYaxis()->SetTitleOffset(2.0);
199 Ratio->GetYaxis()->SetLabelFont(43);
200 Ratio->GetYaxis()->SetLabelSize(25);
201 Ratio->GetYaxis()->CenterTitle();
202 Ratio->GetYaxis()->SetNdivisions(5,2,0);
203
204 Ratio->GetXaxis()->SetTitleSize(25);

207 Ratio->GetXaxis()->SetLabelOffset(0.025);

294void MakeXsecPlots()
295{
296 // get the names of the blocks of parameters to group together
297 std::vector<std::string> const blockNames = man->getOption<std::vector<std::string>>("paramGroups");
298 const int XsecPlots = (int)blockNames.size();
299
300 for (int i = 0; i < XsecPlots; i++)
301 {
302 // get the configuration for this parameter
303 std::string blockName = blockNames[i];
304 YAML::Node paramBlock = man->getOption(blockName);
305 std::string blockTitle = paramBlock[0].as<std::string>();
306 std::vector<double> blockLimits = paramBlock[1].as<std::vector<double>>();
307 std::vector<std::string> blockContents = paramBlock[2].as<std::vector<std::string>>();
308
309 // get num of params in the block
310 int nParams = (int)blockContents.size();
311
312 // set some plot things
313 TH1D *blockHist_prefit = new TH1D(blockName.c_str(), blockTitle.c_str(), nParams, 0.0, (double)nParams);
314
315 man->style().setTH1Style(blockHist_prefit, man->getOption<std::string>("prefitHistStyle"));
316
317 // set the errors for the prefit block hist
318 for(int localBin=0; localBin < nParams; localBin ++){
319 // the "local" bin is the params index within the group of parameters
320 std::string paramName = blockContents[localBin];
321 copyParToBlockHist(localBin, paramName, blockHist_prefit, "Prior", 0);
322 }
323
324 // now set for the postfit blocks for all files
325 std::vector <TH1D *> blockHist_postfit_Vec;
326 for(int fileId = 0; fileId < man->getNFiles(); fileId++){
327
328 TH1D *blockHist_postfit = new TH1D((blockName + man->getFileName(fileId)).c_str(), blockTitle.c_str(), nParams, 0.0, (double)nParams);
329
330 // loop throught all the parameters in this block and set the contents in the blocks TH1
331 for(int localBin=0; localBin < nParams; localBin ++){
332 // the "local" bin is the params index within the group of parameters
333 std::string paramName = blockContents[localBin];
334 copyParToBlockHist(localBin, paramName, blockHist_postfit, "", fileId);
335 }
336
337 blockHist_postfit_Vec.push_back(blockHist_postfit);
338 }
339
340 // set the y axis limits we got from config
341 blockHist_prefit->GetYaxis()->SetRangeUser(blockLimits[0], blockLimits[1]);
342
343 // Do some fancy replacements
344 PrettifyTitles(blockHist_prefit);
345
346 DrawPlots(canv, blockHist_prefit, blockHist_postfit_Vec, p3, p4);
347 blockHist_postfit_Vec.clear();

352{
353 p1->SetLogx(true);
354 p2->SetLogx(true);
355
356 // get the names of the blocks of parameters to group together
357 std::vector<std::string> const fluxBlockNames = man->getOption<std::vector<std::string>>("fluxGroups");
358 auto const fluxBinningTable = man->getOption("FluxBinning");
359
360 const int FluxPlots = (int)fluxBlockNames.size();
361
362 for (int i = 0; i < FluxPlots; i++)
363 {
364 // get the configuration for this block
365 std::string fluxBlockName = fluxBlockNames[i];
366 YAML::Node paramBlock = man->getOption(fluxBlockName);
367 std::string blockTitle = paramBlock[0].as<std::string>();
368 std::vector<double> blockLimits = paramBlock[1].as<std::vector<double>>();
369 std::string blockBinningName = paramBlock[2].as<std::string>();
370 std::vector<int> blockContents = paramBlock[3].as<std::vector<int>>();
371
372 // get the binning for this block of flux params
373 std::vector<double> binning = fluxBinningTable[blockBinningName].as<std::vector<double>>();
374
375 // get num of params in the block
376 int nParams = blockContents[1] - blockContents[0] +1;
377 // check for sanity
378 if(nParams <= 0 || blockContents.size() > 2){
379 MACH3LOG_CRITICAL("Invalid flux parameter block endpoints specified for {}", fluxBlockName);
380 MACH3LOG_CRITICAL(" Should have the form [<low index>, <up index>]");
381 throw MaCh3Exception(__FILE__ , __LINE__ );
382 }
383 if(nParams != (int)binning.size() -1){
384 MACH3LOG_CRITICAL("Binning provided for flux param block {} does not match the number of parameters specified for the block", fluxBlockName);
385 MACH3LOG_CRITICAL(" Provided {} parameters but {} bins", nParams, binning.size() -1);
386 throw MaCh3Exception(__FILE__ , __LINE__ );
387 }
388
389 TH1D *blockHist_prefit = new TH1D(fluxBlockName.c_str(), blockTitle.c_str(), nParams, binning.data());
390 blockHist_prefit->GetYaxis()->SetTitle("Parameter Variation");
391 blockHist_prefit->GetXaxis()->SetTitle("E_{#nu} (GeV)");
392 blockHist_prefit->GetXaxis()->SetTitleOffset(blockHist_prefit->GetXaxis()->GetTitleOffset()*1.2);
393 man->style().setTH1Style(blockHist_prefit, man->getOption<std::string>("prefitHistStyle"));
394 // set the errors for the prefit block hist
395 for(int fluxParId = blockContents[0]; fluxParId <= blockContents[1]; fluxParId++){
396 int localBin = fluxParId - blockContents[0];
397 std::string paramName = "b_" + std::to_string(fluxParId);
398 copyParToBlockHist(localBin, paramName, blockHist_prefit, "Prior", 0, false);
399 }
400
401 // now set for the postfit blocks for all files
402 std::vector <TH1D *> blockHist_postfit_Vec;
403 for(int fileId = 0; fileId < man->getNFiles(); fileId++){
404 TH1D *blockHist_postfit = new TH1D(fluxBlockName.c_str(), blockTitle.c_str(), nParams, binning.data());
405
406 for(int fluxParId = blockContents[0]; fluxParId <= blockContents[1]; fluxParId++){
407 int localBin = fluxParId - blockContents[0];
408 std::string paramName = "b_" + std::to_string(fluxParId);
409
410 copyParToBlockHist(localBin, paramName, blockHist_postfit, "", fileId, false);
411 }
412
413 blockHist_postfit_Vec.push_back(blockHist_postfit);
414 }
415
416 // set the y axis limits we got from config
417 blockHist_prefit->GetYaxis()->SetRangeUser(blockLimits[0], blockLimits[1]);
418
419 DrawPlots(canv, blockHist_prefit, blockHist_postfit_Vec, p1, p2);
420 blockHist_postfit_Vec.clear();
421 }
422
423 canv->cd();
424 canv->SetLogx(false);
425 canv->SetBottomMargin(canv->GetBottomMargin()*1.7);
