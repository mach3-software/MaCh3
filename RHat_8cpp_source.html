142// Load chain and prepare toys
143void PrepareChains() {
144// *******************
145
146 TStopwatch clock;
147 clock.Start();
148
149 Ntoys_requested = new int[Nchains]();
150 Ntoys_filled = new int[Nchains]();
151 TotToys = 0;
152 std::vector<int> BurnIn(Nchains);
153 std::vector<int> nEntries(Nchains);
154 std::vector<int> nBranches(Nchains);
155 std::vector<int> step(Nchains);
156
157 S1_chain = new double*[Nchains]();
158 S2_chain = new double*[Nchains]();
159
160 // KS: This can reduce time necessary for caching even by half
161 #ifdef MULTITHREAD
162 //ROOT::EnableImplicitMT();
163 #endif
164
165 // Open the Chain
166 //It is tempting to multithread here but unfortunately, ROOT files are not thread safe :(
167 for (int m = 0; m < Nchains; m++)
168 {
169 TChain* Chain = new TChain("posteriors");
170 Chain->Add(MCMCFile[m].c_str());
171
172 nEntries[m] = int(Chain->GetEntries());
173 Ntoys_requested[m] = nEntries[m]/NThin;
174 Ntoys_filled[m] = 0;
175
176 MACH3LOG_INFO("On file: {}", MCMCFile[m].c_str());
177 MACH3LOG_INFO("Generating {} Toys", Ntoys_requested[m]);
178
179 // Set the step cut to be 20%
180 BurnIn[m] = nEntries[m]/5;
181
182 // Get the list of branches
183 TObjArray* brlis = Chain->GetListOfBranches();
184
185 // Get the number of branches
186 nBranches[m] = brlis->GetEntries();
187
188 if(m == 0) BranchNames.reserve(nBranches[m]);
189
190 // Set all the branches to off
191 Chain->SetBranchStatus("*", false);
192
193 // Loop over the number of branches
194 // Find the name and how many of each systematic we have
195 for (int i = 0; i < nBranches[m]; i++)
196 {
197 // Get the TBranch and its name
198 TBranch* br = static_cast<TBranch *>(brlis->At(i));
199 if(!br){
200 MACH3LOG_ERROR("Invalid branch at position {}", i);
201 throw MaCh3Exception(__FILE__,__LINE__);
202 }
203 TString bname = br->GetName();
204
205 // Read in the step
206 if (bname == "step") {
207 Chain->SetBranchStatus(bname, true);
208 Chain->SetBranchAddress(bname, &step[m]);
209 }
210 //Count all branches
211 else if (bname.BeginsWith("PCA_") || bname.BeginsWith("accProb") || bname.BeginsWith("stepTime") )
212 {
213 continue;
214 }
215 else
216 {
217 //KS: Save branch name only for one chain, we assume all chains have the same branches, otherwise this doesn't make sense either way
218 if(m == 0)
219 {
220 BranchNames.push_back(bname);
221 //KS: We calculate R Hat also for LogL, just in case, however we plot them separately
222 if(bname.BeginsWith("LogL"))
223 {
224 ValidPar.push_back(false);
225 }
226 else
227 {
228 ValidPar.push_back(true);
229 }
230 }
231 Chain->SetBranchStatus(bname, true);
232 MACH3LOG_DEBUG("{}", bname);
233 }
234 }
235
236 if(m == 0) nDraw = int(BranchNames.size());
237
238 // MJR: Initialize quantities needed for calculating RHat
239 S1_chain[m] = new double[nDraw]();
240 S2_chain[m] = new double[nDraw]();
241 if (m == 0)
242 {
243 S1_global = new double[nDraw]();
244 S2_global = new double[nDraw]();
245 }
246 for (int id = 0; id < nDraw; ++id)
247 {
248 S1_chain[m][id] = 0.0;
249 S2_chain[m][id] = 0.0;
250 if (m == 0)
251 {
252 S1_global[id] = 0.0;
253 S2_global[id] = 0.0;
254 }
255 }
256
257 //TN: Qualitatively faster sanity check, with the very same outcome (all chains have the same #branches)
258 if(m > 0)
259 {
260 if(nBranches[m] != nBranches[0])
261 {
262 MACH3LOG_ERROR("Ups, something went wrong, chain {} called {} has {} branches, while 0 called {} has {} branches", m, MCMCFile[m], nBranches[m], MCMCFile[0], nBranches[0]);
263 MACH3LOG_ERROR("All chains should have the same number of branches");
264 throw MaCh3Exception(__FILE__ , __LINE__ );
265 }
266 }
267
268 // MJR: Create an array to hold branch values. Resetting branch addresses
269 // for every step is very expensive.
270 double* branch_values = new double[nDraw]();
271 for (int id = 0; id < nDraw; ++id)
272 {
273 Chain->SetBranchAddress(BranchNames[id].Data(), &branch_values[id]);
274 }
275
276 //TN: move looping over toys here, so we don't need to loop over chains more than once
277 if(BurnIn[m] >= nEntries[m])
278 {
279 MACH3LOG_ERROR("You are running on a chain shorter than BurnIn cut");
280 MACH3LOG_ERROR("Number of entries {} BurnIn cut {}", nEntries[m], BurnIn[m]);
281 MACH3LOG_ERROR("You will run into the infinite loop");
282 MACH3LOG_ERROR("You can make a new chain or modify BurnIn cut");
283 throw MaCh3Exception(__FILE__ , __LINE__ );
284 }
285
286 MACH3LOG_INFO("Loading chain {} / {}...", m, Nchains);
287 for (int i = 0; i < Ntoys_requested[m]; i++)
288 {
289 // This is here as a placeholder in case we want to do some thinning later
290 int entry = i*NThin;
291
292 Chain->GetEntry(entry);
293
294 // If we have combined chains by hadd need to check the step in the chain
295 // Note, entry is not necessarily the same as the step due to merged ROOT files, so can't choose an entry in the range BurnIn - nEntries :(
296 if (step[m] < BurnIn[m])
297 {
298 continue;
299 }
300
301 // Output some info for the user
302 if (Ntoys_requested[m] > 10 && i % (Ntoys_requested[m]/10) == 0) {
303 MaCh3Utils::PrintProgressBar(i+m*Ntoys_requested[m], static_cast<Long64_t>(Ntoys_requested[m])*Nchains);
304 MACH3LOG_DEBUG("Getting random entry {}", entry);
305 }
306
307 // MJR: Fill running quantities instead of loading everything into RAM.
308 // This is where we save big on both memory and time (resetting
309 // branch addresses and calling GetEntry() again here is very slow).
310 for (int j = 0; j < nDraw; ++j)
311 {
312 S1_global[j] += branch_values[j];
313 S2_global[j] += branch_values[j]*branch_values[j];
314 S1_chain[m][j] += branch_values[j];
315 S2_chain[m][j] += branch_values[j]*branch_values[j];
316 }
317
318 // Increment counters
319 Ntoys_filled[m]++;
320 TotToys++;
321
322 }//end loop over toys
323
324 //TN: There, we now don't need to keep the chain in memory anymore
325 delete Chain;
326 delete[] branch_values;
327 MACH3LOG_INFO("Finished loading chain {}!", m);
328 }
329
330 clock.Stop();
331 MACH3LOG_INFO("Finished calculating Toys, it took {:.2f}s to finish", clock.RealTime());
332}

382//KS: Based on Gelman et. al. arXiv:1903.08008v5
383// Probably most of it could be moved cleverly to MCMC Processor, keep it separate for now
384void CalcRhat() {
385// *******************
386
387 TStopwatch clock;
388 clock.Start();
389
390 //KS: Start parallel region
391 // If we would like to do this for thousands of chains we might consider using GPU for this
392 #ifdef MULTITHREAD
393 #pragma omp parallel
394 {
395 #endif
396
397 #ifdef MULTITHREAD
398 #pragma omp for
399 #endif
400 //KS: loop over chains and draws are independent so might as well collapse for sweet cache hits
401 //Calculate the mean for each parameter within each considered chain
402 // MJR: Calculate using running totals to massively save on time and memory
403 for (int m = 0; m < Nchains; ++m)
404 {
405 for (int j = 0; j < nDraw; ++j)
406 {
407 Mean[m][j] = S1_chain[m][j] / static_cast<double>(Ntoys_filled[m]);
408 StandardDeviation[m][j] = S2_chain[m][j]/static_cast<double>(Ntoys_filled[m]) - Mean[m][j]*Mean[m][j];
409 }
410 }
411
412 #ifdef MULTITHREAD
413 #pragma omp for
414 #endif
415 //Calculate the mean for each parameter global means we include information from several chains
416 for (int j = 0; j < nDraw; ++j)
417 {
418 for (int m = 0; m < Nchains; ++m)
419 {
420 StandardDeviationGlobal[j] += StandardDeviation[m][j];
421 }
422 MeanGlobal[j] = S1_global[j] / static_cast<double>(TotToys);
423 StandardDeviationGlobal[j] = StandardDeviationGlobal[j] / static_cast<double>(Nchains);
424 }
425
426 #ifdef MULTITHREAD
427 #pragma omp for
428 #endif
429 for (int j = 0; j < nDraw; ++j)
430 {
431 //KS: This term only makes sense if we have at least 2 chains
432 if(Nchains == 1)
433 {
434 BetweenChainVariance[j] = 0.;
435 }
436 else
437 {
438 for (int m = 0; m < Nchains; ++m)
439 {
440 BetweenChainVariance[j] += ( Mean[m][j] - MeanGlobal[j])*( Mean[m][j] - MeanGlobal[j]) * Ntoys_filled[m];
441 }
442 BetweenChainVariance[j] = BetweenChainVariance[j]/(Nchains-1);
443 }
444 }
445
446 int avgNtoys = TotToys/Nchains;
447 #ifdef MULTITHREAD
448 #pragma omp for
449 #endif
450 for (int j = 0; j < nDraw; ++j)
451 {
452 MarginalPosteriorVariance[j] = (avgNtoys-1) * StandardDeviationGlobal[j] / (avgNtoys) + BetweenChainVariance[j]/avgNtoys;
453 }
454
455 #ifdef MULTITHREAD
456 #pragma omp for
457 #endif
458 //Finally calculate our estimator
459 for (int j = 0; j < nDraw; ++j)
460 {
461 RHat[j] = sqrt(MarginalPosteriorVariance[j]/StandardDeviationGlobal[j]);
462
463 //KS: For flat params values can be crazy so cap at 0
464 CapVariable(RHat[j], 0);
465 }
466
467 #ifdef MULTITHREAD
468 #pragma omp for
469 #endif
470 //KS: Additionally calculates effective step size which is an estimate of the sample size required to achieve the same level of precision if that sample was a simple random sample.
471 for (int j = 0; j < nDraw; ++j)
472 {
473 if(Nchains > 1) EffectiveSampleSize[j] = TotToys * MarginalPosteriorVariance[j] / BetweenChainVariance[j];
474
475 //KS: For flat params values can be crazy so cap at 0
476 CapVariable(EffectiveSampleSize[j], 0);
477 }
478 #ifdef MULTITHREAD
479 } //End parallel region
480 #endif
481
482 clock.Stop();
483 MACH3LOG_INFO("Finished calculating RHat, it took {:.2f}s to finish", clock.RealTime());
484}
