24void getSplitSampleStack(int fileIdx, std::string parameterName, TH1D LLH_allSams,
25 std::vector<float> &cumSums, std::vector<bool> &drawLabel,
26 THStack *sampleStack, TLegend *splitSamplesLegend,
27 float baselineLLH_main = 0.00001)
28 {
29 std::vector<std::string> sampNames = man->input().getTaggedSamples(man->getOption<std::vector<std::string>>("sampleTags"));
30 size_t nSamples = sampNames.size();
31
32 cumSums.resize(nSamples);
33 drawLabel.resize(nSamples);
34
35 float LLH_main_integ = LLH_allSams.Integral();
36 float cumSum = 0.0;
37 int nBins = LLH_allSams.GetNbinsX();
38
39 MACH3LOG_DEBUG("getting split sample THStack for {} known samples", nSamples);
40 for (uint i = 0; i < nSamples; i++)
41 {
42 std::string sampName = sampNames[i];
43
44 MACH3LOG_DEBUG(" on sample {}/{}: {}", i, nSamples, sampName);
45
46 TH1D *LLH_indivSam =
47 new TH1D(man->input().getSampleSpecificLLHScan_TH1D(fileIdx, parameterName, sampName));
48 LLH_indivSam->SetName(Form("%i_%s_%s", fileIdx, parameterName.c_str(), sampName.c_str()));
49 LLH_indivSam->SetBit(kCanDelete);
50
51 // the hist for this sample didn't exist so move on
52 if (LLH_indivSam->GetNbinsX() == 1)
53 {
54 MACH3LOG_DEBUG(" sample hist had only 1 bin - assuming it doesn't exist");
55 delete LLH_indivSam;
56 drawLabel[i] = false;
57 cumSums[i] = -999.9;
58 continue;
59 }
60
61 LLH_indivSam->SetStats(0);
62 LLH_indivSam->SetLineColor(TColor::GetColorPalette(
63 floor(static_cast<float>(i) * TColor::GetNumberOfColors() / static_cast<float>(nSamples))));
64 LLH_indivSam->SetFillColor(TColor::GetColorPalette(
65 floor(static_cast<float>(i) * TColor::GetNumberOfColors() / static_cast<float>(nSamples))));
66 sampleStack->Add(LLH_indivSam);
67 splitSamplesLegend->AddEntry(LLH_indivSam, man->style().prettifySampleName(sampName).c_str(),
68 "lf");
69
70 float lastBinLLH = LLH_indivSam->GetBinContent(nBins);
71 cumSum += lastBinLLH;
72
73 MACH3LOG_DEBUG(" Last bin LLH = {} :: cumulative LLH = {}", lastBinLLH, cumSum);
74 MACH3LOG_DEBUG(" LLH fraction = {} / {} = {}", LLH_indivSam->Integral(), LLH_main_integ, LLH_indivSam->Integral() / LLH_main_integ);
75
76 cumSums[i] = cumSum;
77
78 // dont draw a label if the likelihood contribution is less than threshold%
79 if ((LLH_indivSam->Integral() / LLH_main_integ > sampleLabelThreshold) &&
80 (LLH_indivSam->Integral() / baselineLLH_main > sampleLabelThreshold))
81 {
82 drawLabel[i] = true;
83 } else
84 {
85 drawLabel[i] = false;
86 }
87 MACH3LOG_DEBUG(" drawLabel = {}", drawLabel.back());
88 MACH3LOG_DEBUG("");
89 }
90}

92// handy dandy helper function for drawing and nicely formatting a stack of ratio plots
93void drawRatioStack(THStack *ratioCompStack) {
94 // do this so 1.0 is in the middle of the plot vertically
95 double stackMax = ratioCompStack->GetMaximum("NOSTACK");
96 double stackMin = ratioCompStack->GetMinimum("NOSTACK");
97
98 double stackLim = std::max(std::abs(1.0 - stackMax), std::abs(1.0 - stackMin));
99
100 ratioCompStack->SetMinimum(1.0 - 1.05 * stackLim);
101 ratioCompStack->SetMaximum(1.0 + 1.05 * stackLim);
102
103 // draw it
104 ratioCompStack->Draw(Form("NOSTACK%s", man->getDrawOptions().c_str()));
105
106 // make it look a bit nicer
107 ratioCompStack->GetXaxis()->SetLabelSize(ratioLabelScaling *
108 ratioCompStack->GetXaxis()->GetLabelSize());
109 ratioCompStack->GetXaxis()->SetTitleSize(ratioLabelScaling *
110 ratioCompStack->GetXaxis()->GetLabelSize());
111 ratioCompStack->GetXaxis()->SetTitle("Parameter Variation");
112
113 ratioCompStack->GetYaxis()->SetLabelSize(ratioLabelScaling *
114 ratioCompStack->GetYaxis()->GetLabelSize());
115 ratioCompStack->GetYaxis()->SetTitleSize(ratioLabelScaling *
116 ratioCompStack->GetYaxis()->GetLabelSize());
117 ratioCompStack->GetYaxis()->SetTitleOffset(yTitleOffset);
118 ratioCompStack->GetYaxis()->SetNdivisions(5, 2, 0);
119}

121void makeLLHScanComparisons(std::string paramName, std::string LLHType, std::string outputFileName,
122 TCanvas *canv, TPad *LLHPad, TPad *ratioPad) {
123 // will use these to make comparisons of likelihoods
124 THStack *compStack = new THStack("LLH_Stack", "");
125 THStack *ratioCompStack = new THStack("LLH_Ratio_Stack", "");
126 auto legend = std::make_unique<TLegend>(0.3, 0.6, 0.7, 0.8);
127
128 // get the sample reweight hist from the main file
129 TH1D LLH_main = man->input().getLLHScan_TH1D(0, paramName, LLHType);
130 LLH_main.SetStats(0);
131
132 LLH_main.SetLineColor(kBlack);
133 compStack->Add(&LLH_main);
134 legend->AddEntry(&LLH_main, man->getFileLabel(0).c_str(), "l");
135
136 int nBins = LLH_main.GetNbinsX();
137
138 // go through the other files
139 for (unsigned int extraFileIdx = 1; extraFileIdx < man->input().getNInputFiles(); extraFileIdx++)
140 {
141 TH1D *compHist = new TH1D(man->input().getLLHScan_TH1D(extraFileIdx, paramName, LLHType));
142 compHist->SetBit(kCanDelete); // <- will allow this to be deleted by root once done plotting
143 if (compHist->GetNbinsX() == 0)
144 continue;
145
146 // make them look different to each other
147 compHist->SetLineColor(
148 TColor::GetColorPalette(floor(static_cast<float>(extraFileIdx) * TColor::GetNumberOfColors() /
149 static_cast<float>(man->input().getNInputFiles()))));
150 compHist->SetLineStyle(2 + extraFileIdx % 9);
151 compHist->SetLineWidth(lineWidth);
152
153 TH1D *divHist = static_cast<TH1D*>(compHist->Clone(Form("RatioHist_%i", extraFileIdx)));
154 divHist->SetBit(kCanDelete);
155
156 if (man->getPlotRatios())
157 {
158 // get the ratio hist
159 divHist->Divide(compHist, &LLH_main);
160 ratioCompStack->Add(divHist);
161 }
162
163 // add it to the comparisson hstack and legend
164 compStack->Add(compHist);
165 legend->AddEntry(compHist, man->getFileLabel(extraFileIdx).c_str(), "l");
166 }
167
168 // draw the log likelihoods
169 canv->cd();
170 canv->Draw();
171 LLHPad->Draw();
172 LLHPad->cd();
173 if (man->getDrawGrid())
174 LLHPad->SetGrid();
175 compStack->Draw(Form("NOSTACK%s", man->getDrawOptions().c_str()));
176 if (!man->getPlotRatios())
177 compStack->GetXaxis()->SetTitle("Parameter Variation");
178 compStack->GetYaxis()->SetTitle(Form("-2LLH_{%s}", LLHType.c_str()));
179 compStack->SetTitle(man->style().prettifyParamName(paramName).c_str());
180 compStack->GetYaxis()->SetTitleOffset(yTitleOffset);
181 legend->Draw();
182
183 // add the ratio plot if specified
184 if (man->getPlotRatios())
185 {
186 canv->cd();
187 ratioPad->Draw();
188 ratioPad->cd();
189 if (man->getDrawGrid())
190 ratioPad->SetGrid();
191
192 drawRatioStack(ratioCompStack);
193
194 // add horizontal line at 1 for reference
195 TLine line = TLine();
196 line.SetLineColor(kBlack);
197 line.SetLineWidth(lineWidth);
198 line.DrawLine(LLH_main.GetBinLowEdge(1), 1.0, LLH_main.GetBinLowEdge(nBins + 1), 1.0);
199 }
200
201 // save to the output file
202 canv->SaveAs(outputFileName.c_str());
203
204 // will use these to make comparisons of likelihoods
205 delete compStack;
206 delete ratioCompStack;
207}

358 }
359
360 canv->cd(2 + extraFileIdx);
361 if (man->getDrawGrid())
362 ratioPad->SetGrid();
363 ratioPad->Draw();
364 ratioPad->cd();
365
366 drawRatioStack(splitSamplesStackRatios);
367 }
368 }
369
370 canv->SaveAs(outputFileName.c_str());
371
372 delete label;
373 delete baseSplitSamplesStack;
374 delete baseSplitSamplesLegend;

377int PlotLLH() {
378
379 man->style().setPalette(man->getOption<std::string>("colorPalette"));
380
381 // make the canvas and other plotting stuff
382 TCanvas *canv = new TCanvas("canv", "", 1024, 1024);
383 gStyle->SetOptTitle(2);
384 // split up the canvas so can have side by side plots, one for each file
385 TCanvas *splitSamplesCanv = new TCanvas("splitSampCanv", "", 4096 * man->getNFiles(), 4096);
386
387 // split the canvas if plotting ratios
388 TPad *LLHPad, *ratioPad;
389 if (man->getPlotRatios())
390 {
391 LLHPad = new TPad("LLHPad", "LLHPad", 0.0, ratioPlotSplit, 1.0, 1.0);
392 LLHPad->SetBottomMargin(0.0);
393 ratioPad = new TPad("ratioPad", "ratioPad", 0.0, 0.0, 1.0, ratioPlotSplit);
394 ratioPad->SetTopMargin(0.0);
395 ratioPad->SetBottomMargin(0.3);
396 } else
397 {
398 LLHPad = new TPad("AllSampPad", "AllSampPad", 0.0, 0.0, 1.0, 1.0);
399 ratioPad = new TPad("AllSampRatioPad", "AllSampRatioPad", 0.0, 0.0, 0.0, 0.0);
400 }
401
402 canv->SaveAs((man->getOutputName("_Sample") + "[").c_str());
403 canv->SaveAs((man->getOutputName("_Penalty") + "[").c_str());
404 canv->SaveAs((man->getOutputName("_Total") + "[").c_str());
405
406 if (man->getSplitBySample())
407 canv->SaveAs((man->getOutputName("_bySample") + "[").c_str());
408
409 for( std::string par: man->getOption<std::vector<std::string>>("parameterTags")) std::cout << par << ", ";
410 // loop over the spline parameters
411 for (std::string paramName : man->input().getTaggedParameters(man->getOption<std::vector<std::string>>("parameterTags")))
412 {
413 MACH3LOG_DEBUG("working on parameter {}", paramName);
414 // ###############################################################
415 // First lets do just the straight up likelihoods from all samples
416 // ###############################################################
417
418 makeLLHScanComparisons(paramName, "sample", man->getOutputName("_Sample"), canv, LLHPad,
419 ratioPad);
420 makeLLHScanComparisons(paramName, "penalty", man->getOutputName("_Penalty"), canv, LLHPad,
421 ratioPad);
422 makeLLHScanComparisons(paramName, "total", man->getOutputName("_Total"), canv, LLHPad,
423 ratioPad);
424
425 // #########################################
426 // ## now lets make plots split by sample ##
427 // #########################################
428 if (man->getSplitBySample())
429 {
430 makeSplitSampleLLHScanComparisons(paramName, man->getOutputName("_bySample"),
431 splitSamplesCanv, LLHPad, ratioPad);
432 }
433 }
434
435 canv->SaveAs((man->getOutputName("_Sample") + "]").c_str());
436 canv->SaveAs((man->getOutputName("_Penalty") + "]").c_str());
437 canv->SaveAs((man->getOutputName("_Total") + "]").c_str());
438 if (man->getSplitBySample())
439 canv->SaveAs((man->getOutputName("_bySample") + "]").c_str());
440
441 delete LLHPad;
442 delete ratioPad;
443 delete canv;
444 if(man != nullptr) delete man;
445
446 return 0;
447}
