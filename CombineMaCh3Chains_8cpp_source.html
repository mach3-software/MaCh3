25bool ShouldSkipLine(const std::string& line, const std::vector<std::string>& SkipVector) {
26 // Otherwise, check if the line contains any word from SkipVector
27 for (const auto& word : SkipVector) {
28 MACH3LOG_TRACE("{} : {}",line, word);
29 if (line.find(word) != std::string::npos) {
30 MACH3LOG_TRACE("Found matching word, therefore Skipping");
31 return true;
32 }
33 }
34 return false;
35}

37bool CompareTwoConfigs(const std::string& File1, const std::string& File2, const std::vector<std::string>& SkipVector) {
38 std::istringstream file1(File1);
39 std::istringstream file2(File2);
40
41 std::string line1, line2;
42 int lineNumber = 1;
43 bool areEqual = true;
44
45 while (std::getline(file1, line1) && std::getline(file2, line2)) {
46 if (ShouldSkipLine(line1, SkipVector) || ShouldSkipLine(line2, SkipVector)) {
47 ++lineNumber;
48 continue;
49 }
50 if (line1 != line2) {
51 areEqual = false;
52 MACH3LOG_WARN("Difference found on line {}:", lineNumber);
53 MACH3LOG_WARN("Config1: {}", line1);
54 MACH3LOG_WARN("Config2: {}", line2);
55 }
56 ++lineNumber;
57 }
58 // Check if one file has extra lines
59 while (std::getline(file1, line1)) {
60 MACH3LOG_WARN("Extra line in {} on line {}: {}", File1, lineNumber, line1);
61 ++lineNumber;
62 }
63 while (std::getline(file2, line2)) {
64 MACH3LOG_WARN("Extra line in {} on line {}: {}", File2, lineNumber, line2);
65 ++lineNumber;
66 }
67 return areEqual;
68}

71bool checkSoftwareVersions(TFile *file, TFile *prevFile, const std::string& ConfigName, const std::vector<std::string>& SkipVector = {})
72{
73 bool weirdFile = false;
74
75 TMacro *versionHeader = file->Get<TMacro>(ConfigName.c_str());
76 TMacro *prevVersionHeader = prevFile->Get<TMacro>(ConfigName.c_str());
77
78 // EM: compare the digest of the version header file in this file, with the previous one
79 if(!CompareTwoConfigs(TMacroToString(*versionHeader), TMacroToString(*prevVersionHeader), SkipVector)){
80 MACH3LOG_ERROR("Looks like the {} embedded config for file {} is different to the previous ones", ConfigName, file->GetName());
81 MACH3LOG_ERROR("This strongly suggests that this file was made with different software versions than the previous ones");
82 weirdFile = true;
83 }
84

88void CopyDir(TDirectory *source) {
89 //copy all objects and subdirs of directory source as a subdir of the current directory
90 source->ls();
91 TDirectory *savdir = gDirectory;
92 TDirectory *adir = savdir->Get<TDirectory>(source->GetName());
93 adir->cd();
94 //loop on all entries of this directory
95 TKey *key;
96 TIter nextkey(source->GetListOfKeys());
97 while ((key = static_cast<TKey*>(nextkey()))) {
98 const char *classname = key->GetClassName();
99 TClass *cl = gROOT->GetClass(classname);
100 if (!cl) continue;
101 if (cl->InheritsFrom("TDirectory")) {
102 source->cd(key->GetName());
103 TDirectory *subdir = gDirectory;
104 adir->cd();
105 CopyDir(subdir);
106 adir->cd();
107 } else if (cl->InheritsFrom("TTree")) {
108 TTree *T = source->Get<TTree>(key->GetName());
109 adir->cd();
110 TTree *newT = T->CloneTree();
111 newT->Write();
112 } else {
113 source->cd();
114 TObject *obj = key->ReadObj();
115 adir->cd();
116 obj->Write();
117 delete obj;
118 }
119 }
120 adir->SaveSelf(kTRUE);
121 savdir->cd();
122}

147 // EM: loop through all the files in the provided list, compare the embedded version and config files
148 // If they match, we add the file to the list of files to be merged.
149 // If not, we throw an error and provide a (hopefully) helpful message telling the user why the files couldn't be merged.
150 for(uint fileId = 0; fileId < inpFileList.size(); fileId++)

153 TFile *file = new TFile(fileName.c_str());
154
155 if(file->Get<TTree>("posteriors")->GetEntries() == 0){
156 MACH3LOG_WARN("Hmmm, file {} Doesn't seem to have any entries", fileName.c_str());
157 MACH3LOG_WARN("That's weird but I guess there's no rule that says a file can't be empty");
158 MACH3LOG_WARN("I'll skip it but maybe double check that this doesn't indicate some deeper problem");
159 continue;
160 }
161
162 // EM: need to set this in the initial case
163 if(prevFile == nullptr) {
164 prevFile = file;
165 }
166
167 MACH3LOG_DEBUG("############ File {} #############", fileId);
168
169 bool weirdFile = false;
170 if(checkSoftwareVersions(file, prevFile, "MaCh3Engine/version_header")) weirdFile = true;
171 if(checkSoftwareVersions(file, prevFile, "MaCh3_Config", {"OutputFile:", "NSteps:"})) weirdFile = true;
172
173 if(weirdFile && !forceMerge){
174 MACH3LOG_ERROR("");
175 MACH3LOG_ERROR("=====================================================================================");
176 MACH3LOG_ERROR("This is not a great idea and could lead to weird outputs and cause some big headaches");

187 // EM: set these for the next iteration
188 prevFile = file;
189 }
190
191 TFile *outputFile = fileMerger->GetOutputFile();
192 outputFile->cd();
193
194 // EM: write out the version and config files to the combined file
195 TMacro *MaCh3_Config = prevFile->Get<TMacro>("MaCh3_Config");
196
197 if(MaCh3_Config != NULL) MaCh3_Config->Write();
198 delete MaCh3_Config;
199
200 // EM: now let's combine all the trees and write to the output file
201 bool mergeSuccess = fileMerger->PartialMerge(TFileMerger::kRegular | TFileMerger::kAll | TFileMerger::kOnlyListed);
202 if(mergeSuccess){
203 MACH3LOG_INFO("Files merged successfully");
204 } else{
205 MACH3LOG_ERROR("Failed to merge files");
206 }
207 delete fileMerger;
208
209 //KS: Sadly we need to open file to save TDirectories to not have weird copy of several obejcts there...
210 outputFile = new TFile(OutFileName.c_str(), "UPDATE");
211
212 // Get the source directory
213 TDirectory *MaCh3EngineDir = prevFile->Get<TDirectory>("MaCh3Engine");
214 TDirectory *CovarianceFolderDir = prevFile->Get<TDirectory>("CovarianceFolder");
215
216 outputFile->cd();

222}
