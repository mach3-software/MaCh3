471// Throw the parameters according to the covariance matrix
472// This shouldn't be used in MCMC code ase it can break Detailed Balance;
473void covarianceBase::throwParameters() {
474// *************************************
475 // First draw new randParams
476 randomize();
477
478 // KS: We use PCA very rarely on top PCA functionality isn't implemented for this function.
479 // Use __builtin_expect to give compiler a hint which option is more likely, which should help
480 // with better optimisation. This isn't critical but more to have example
481 if (__builtin_expect(!pca, 1)) {
482 MatrixVectorMulti(corr_throw, throwMatrixCholDecomp, randParams, _fNumPar);
483
484 #ifdef MULTITHREAD
485 #pragma omp parallel for
486 #endif
487 for (int i = 0; i < _fNumPar; ++i) {
488 // Check if parameter is fixed first: if so don't randomly throw
489 if (isParameterFixed(i)) continue;
490
491 _fPropVal[i] = _fPreFitValue[i] + corr_throw[i];
492 int throws = 0;
493 // Try again if we the initial parameter proposal falls outside of the range of the parameter
494 while (_fPropVal[i] > _fUpBound[i] || _fPropVal[i] < _fLowBound[i]) {
495#ifdef MULTITHREAD
496 randParams[i] = random_number[omp_get_thread_num()]->Gaus(0, 1);
497#else
498 randParams[i] = random_number[0]->Gaus(0,1);
499#endif
500 const double corr_throw_single = MatrixVectorMultiSingle(throwMatrixCholDecomp, randParams, _fNumPar, i);
501 _fPropVal[i] = _fPreFitValue[i] + corr_throw_single;
502 if (throws > 10000)
503 {
504 //KS: Since we are multithreading there is danger that those messages
505 //will be all over the place, small price to pay for faster code
506 MACH3LOG_WARN("Tried {} times to throw parameter {} but failed", throws, i);
507 MACH3LOG_WARN("Matrix: {}", matrixName);
508 MACH3LOG_WARN("Param: {}", _fNames[i]);
509 MACH3LOG_WARN("Setting _fPropVal: {} to {}", _fPropVal[i], _fPreFitValue[i]);
510 MACH3LOG_WARN("I live at {}:{}", __FILE__, __LINE__);
511 _fPropVal[i] = _fPreFitValue[i];
512 //throw MaCh3Exception(__FILE__ , __LINE__ );
513 }
514 throws++;
515 }
516 _fCurrVal[i] = _fPropVal[i];
517 }
518 }
519 else
520 {
521 MACH3LOG_CRITICAL("Hold on, you are trying to run Prior Predictive Code with PCA, which is wrong");
522 MACH3LOG_CRITICAL("Sorry I have to kill you, I mean your job");

527// *************************************
528// Throw each parameter within their 1 sigma range
529// Used to start the chain in different states
530void covarianceBase::RandomConfiguration() {
531// *************************************
532 // Have the 1 sigma for each parameter in each covariance class, sweet!
533 // Don't want to change the nominal array because that's what determines our likelihood
534 // Want to change the fParProp, fParCurr, fParInit
535 // fParInit and the others will already be set
536 for (int i = 0; i < _fNumPar; ++i) {
537 // Check if parameter is fixed first: if so don't randomly throw
538 if (isParameterFixed(i)) continue;
539 // Check that the sigma range is larger than the parameter range
540 // If not, throw in the valid parameter range instead
541 const double paramrange = _fUpBound[i] - _fLowBound[i];
542 const double sigma = sqrt((*covMatrix)(i,i));
543 double throwrange = sigma;
544 if (paramrange < sigma) throwrange = paramrange;
545
546 _fPropVal[i] = _fPreFitValue[i] + random_number[0]->Gaus(0, 1)*throwrange;
547 // Try again if we the initial parameter proposal falls outside of the range of the parameter
548 // Really only relevant for the xsec parameters; the flux and ND280 have -999 and 999 set to the limits!
549 int throws = 0;
550 while (_fPropVal[i] > _fUpBound[i] || _fPropVal[i] < _fLowBound[i]) {
551 if (throws > 1000) {
552 MACH3LOG_WARN("Tried {} times to throw parameter {} but failed", throws, i);
553 MACH3LOG_WARN("Matrix: {}", matrixName);
554 MACH3LOG_WARN("Param: {}", _fNames[i]);
555 throw MaCh3Exception(__FILE__ , __LINE__ );
556 }
557 _fPropVal[i] = _fPreFitValue[i] + random_number[0]->Gaus(0, 1)*throwrange;
558 throws++;
559 }
560 MACH3LOG_INFO("Setting current step in {} param {} = {} from {}", matrixName, i, _fPropVal[i], _fCurrVal[i]);
561 _fCurrVal[i] = _fPropVal[i];
562 }
563 if (pca) TransferToPCA();
564}

609 randParams[i] = random_number[omp_get_thread_num()]->Gaus(0, 1);
610#else
611 randParams[i] = random_number[0]->Gaus(0, 1);
612#endif
613 // If parameter IS fixed
614 } else {
615 randParams[i] = 0.0;
616 }
617 } // end for
618 // If we're in the PCA basis we instead throw parameters there (only _fNumParPCA parameter)
619 } else {
620 // Scale the random parameters by the sqrt of eigen values for the throw
621 #ifdef MULTITHREAD
622 #pragma omp parallel for
623 #endif
624 for (int i = 0; i < _fNumPar; ++i)
625 {
626 if (fParSigma_PCA[i] > 0. && i < _fNumParPCA)
627 {
628#ifdef MULTITHREAD
629 randParams[i] = random_number[omp_get_thread_num()]->Gaus(0,1);
630#else
631 randParams[i] = random_number[0]->Gaus(0,1);
632#endif
633 } else { // If parameter IS fixed or out od bounds
634 randParams[i] = 0.0;
635 }
636 }
637 }

640// ************************************************
641// Correlate the steps by setting the proposed step of a parameter to its current value + some correlated throw
642void covarianceBase::CorrelateSteps() {
643// ************************************************
644 //KS: Using custom function compared to ROOT one with 8 threads we have almost factor 2 performance increase, by replacing TMatrix with just double we increase it even more
645 MatrixVectorMulti(corr_throw, throwMatrixCholDecomp, randParams, _fNumPar);
646
647 // If not doing PCA
648 if (!pca) {
649 #ifdef MULTITHREAD
650 #pragma omp parallel for
651 #endif
652 for (int i = 0; i < _fNumPar; ++i) {
653 if (_fError[i] > 0.) {
654 _fPropVal[i] = _fCurrVal[i] + corr_throw[i]*_fGlobalStepScale*_fIndivStepScale[i];
655 }
656 }
657 // If doing PCA throw uncorrelated in PCA basis (orthogonal basis by definition)
658 } else {
659 // Throw around the current step
660 #ifdef MULTITHREAD
661 #pragma omp parallel for
662 #endif
663 for (int i = 0; i < _fNumParPCA; ++i)
664 {
665 if (fParSigma_PCA[i] > 0.)
666 {
667 double IndStepScale = 1.;
668 //KS: If undecomposed parameter apply individual step scale and Cholesky for better acceptance rate
669 if(isDecomposed_PCA[i] >= 0)
670 {
671 IndStepScale *= _fIndivStepScale[isDecomposed_PCA[i]];
672 IndStepScale *= corr_throw[isDecomposed_PCA[i]];
673 }
674 //If decomposed apply only random number
675 else
676 {
677 IndStepScale *= randParams[i];
678 //KS: All PCA-ed parameters have the same step scale
679 IndStepScale *= _fIndivStepScale[FirstPCAdpar];
680 }
681 fParProp_PCA(i) = fParCurr_PCA(i)+_fGlobalStepScale*IndStepScale*PCAObj.eigen_values_master[i];
682 }
683 }
684 // Then update the parameter basis
685 TransferToParam();
686 }

715void covarianceBase::throwParProp(const double mag) {
716// ********************************************
717 randomize();
718 if (!pca) {
719 // Make the correlated throw
720 MatrixVectorMulti(corr_throw, throwMatrixCholDecomp, randParams, _fNumPar);
721 // Number of sigmas we throw

733 for (int i = 0; i < _fNumPar; ++i) {
734 if (fParSigma_PCA[i] > 0.) {
735 _fPropVal[i] = proposed(i);
736 }
737 }

798// ********************************************
799// Get the likelihood in the case where we want to include priors on the parameters
800// fParEvalLikelihood stores if we want to evaluate the likelihood for the given parameter
801// true = evaluate likelihood (so run with a prior)
802// false = don't evaluate likelihood (so run without a prior)
803double covarianceBase::CalcLikelihood() {
804// ********************************************
805 double logL = 0.0;
806 #ifdef MULTITHREAD
807 #pragma omp parallel for reduction(+:logL)
808 #endif
809 for(int i = 0; i < _fNumPar; ++i){
810 #ifdef MULTITHREAD
811 #pragma omp simd
812 #endif
813 for (int j = 0; j <= i; ++j) {
814 if (!_fFlatPrior[i] && !_fFlatPrior[j]) {
815 //KS: Since matrix is symmetric we can calculate non diagonal elements only once and multiply by 2, can bring up to factor speed decrease.
816 short int scale = (i != j) ? 2 : 1;
817 logL += scale * 0.5*(_fPropVal[i] - _fPreFitValue[i])*(_fPropVal[j] - _fPreFitValue[j])*InvertCovMatrix[i][j];
818 }
819 }
820 }
821 return logL;
822}

1090//Makes sure that matrix is positive-definite by adding a small number to on-diagonal elements
1091void covarianceBase::MakePosDef(TMatrixDSym *cov) {
1092// ********************************************
1093 //DB Save original warning state and then increase it in this function to suppress 'matrix not positive definite' messages
1094 //Means we no longer need to overload
1095 if(cov == nullptr){
1096 cov = &*covMatrix;
1097 }
1098
1099 int originalErrorWarning = gErrorIgnoreLevel;
1100 gErrorIgnoreLevel = kFatal;
1101
1102 //DB Loop 1000 times adding 1e-9 which tops out at 1e-6 shift on the diagonal before throwing error
1103 int MaxAttempts = 1e5;
1104 int iAttempt = 0;
1105 bool CanDecomp = false;
1106 TDecompChol chdcmp;
1107
1108 for (iAttempt = 0; iAttempt < MaxAttempts; iAttempt++) {
1109 chdcmp = TDecompChol(*cov);
1110 if (chdcmp.Decompose()) {
1111 CanDecomp = true;
1112 break;
1113 } else {
1114 #ifdef MULTITHREAD
1115 #pragma omp parallel for
1116 #endif
1117 for (int iVar = 0 ; iVar < _fNumPar; iVar++) {
1118 (*cov)(iVar,iVar) += pow(10,-9);
1119 }
1120 }
1121 }
1122
1123 if (!CanDecomp) {
1124 MACH3LOG_ERROR("Tried {} times to shift diagonal but still can not decompose the matrix", MaxAttempts);
1125 MACH3LOG_ERROR("This indicates that something is wrong with the input matrix");
1126 throw MaCh3Exception(__FILE__ , __LINE__ );
1127 }
1128 if(AdaptiveHandler.total_steps < 2) {
1129 MACH3LOG_INFO("Had to shift diagonal {} time(s) to allow the covariance matrix to be decomposed", iAttempt);
1130 }
1131 //DB Resetting warning level
1132 gErrorIgnoreLevel = originalErrorWarning;
1133
1134 return;
1135}
