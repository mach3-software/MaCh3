279 void ReweightMCMC(const std::string& configFile, const std::string& inputFile)
280 {
281  MACH3LOG_INFO("File for reweighting: {} with config {}", inputFile, configFile);
282  // Load configuration
283  YAML::Node reweight_yaml = M3OpenConfig(configFile);
284  YAML::Node reweight_settings = reweight_yaml["ReweightMCMC"];
285 
286  // Parse all reweight configurations first
287  std::vector<ReweightConfig> reweightConfigs;
288 
289  LoadReweightingSettings(reweightConfigs, reweight_settings);
290 
291  // Create MCMCProcessor to get parameter information
292  auto processor = std::make_unique<MCMCProcessor>(inputFile);
293  processor->Initialise();
294 
295  // Validate that all required parameters exist in the chain
297  for (const auto& rwConfig : reweightConfigs) {
298  for (const auto& paramName : rwConfig.paramNames) {
299  int paramIndex = processor->GetParamIndexFromName(paramName);
300  if (paramIndex == M3::_BAD_INT_) {
301  MACH3LOG_ERROR("Parameter {} not found in MCMC chain", paramName);
302  throw MaCh3Exception(__FILE__, __LINE__);
303  }
304  MACH3LOG_INFO("Parameter {} found in chain", paramName);
305  }
306  }
307 
309  // Get the settings for the MCMC
310  auto TempFile = std::unique_ptr<TFile>(TFile::Open(inputFile.c_str(), "READ"));
311  if (!TempFile || TempFile->IsZombie()) {
312  MACH3LOG_ERROR("Cannot open MCMC file: {}", inputFile);
313  throw MaCh3Exception(__FILE__ , __LINE__ );
314  }
315  std::unique_ptr<TMacro> Config(TempFile->Get<TMacro>("MaCh3_Config"));
316  if (!Config) {
317  MACH3LOG_ERROR("Didn't find MaCh3_Config tree in MCMC file! {}", inputFile.c_str());
318  TempFile->ls();
319  throw MaCh3Exception(__FILE__ , __LINE__ );
320  }
321  MACH3LOG_INFO("Loading YAML config from MCMC chain");
322  YAML::Node Settings = TMacroToYAML(*Config);
323  bool asimovfit = GetFromManager<bool>(Settings["General"]["Asimov"], false);
324  if (asimovfit) {
325  MACH3LOG_WARN("MCMC chain was produced from an Asimov fit");
326  MACH3LOG_WARN("ReweightMCMC does not currently handle Asimov shifting, results may be incorrect!");
327  } else {
328  MACH3LOG_INFO("Not an Asimov fit, proceeding with reweighting");
329  }
330 
331  // Open input file and get tree
332  auto inFile = std::unique_ptr<TFile>(TFile::Open(inputFile.c_str(), "READ"));
333  if (!inFile || inFile->IsZombie()) {
334  MACH3LOG_ERROR("Cannot open input file: {}", inputFile);
335  throw MaCh3Exception(__FILE__, __LINE__);
336  }
337 
338  std::unique_ptr<TTree> inTree(inFile->Get<TTree>("posteriors"));
339  if (!inTree) {
340  MACH3LOG_ERROR("Cannot find 'posteriors' tree in input file");
341  throw MaCh3Exception(__FILE__, __LINE__);
342  }
343 
344  // Create output file
345  std::string configString = configFile.substr(configFile.find_last_of('/') + 1, configFile.find_last_of('.') - configFile.find_last_of('/') - 1);
346  std::string outputFile = inputFile.substr(0, inputFile.find_last_of('.')) + "_reweighted_" + configString + ".root";
347  auto outFile = std::unique_ptr<TFile>(TFile::Open(outputFile.c_str(), "RECREATE"));
348  if (!outFile || outFile->IsZombie()) {
349  MACH3LOG_ERROR("Cannot create output file: {}", outputFile);
350  throw MaCh3Exception(__FILE__, __LINE__);
351  }
352 
353  MACH3LOG_INFO("Output file will be: {}", outputFile);
354 
355  // Copy all the remaining objects into the out file (i.e. all but posteriors tree)
356  TIter next(inFile->GetListOfKeys());
357  while (TKey* key = dynamic_cast<TKey*>(next())) {
358  inFile->cd();
359  std::unique_ptr<TObject> obj(key->ReadObj());
360  if (obj->IsA()->InheritsFrom(TDirectory::Class())) {
361  // It's a folder, create and copy its contents
362  TDirectory* srcDir = static_cast<TDirectory*>(obj.get());
363  TDirectory* destDir = outFile->mkdir(srcDir->GetName());
364  TIter nextSubKey(srcDir->GetListOfKeys());
365  while (TKey* subKey = dynamic_cast<TKey*>(nextSubKey())) {
366  srcDir->cd();
367  std::unique_ptr<TObject> subObj(subKey->ReadObj());
368  destDir->cd();
369  subObj->Write();
370  }
371  } else if (std::string(key->GetName()) != "posteriors") {
372  // Regular object, skip "posteriors" tree
373  outFile->cd();
374  obj->Write();
375  }
376  }
377 
378  // Clone the tree structure
379  outFile->cd();
380  std::unique_ptr<TTree> outTree(inTree->CloneTree(0));
381 
382  // Set up parameter reading
383  std::map<std::string, double> paramValues;
384  for (const auto& rwConfig : reweightConfigs) {
385  for (const auto& paramName : rwConfig.paramNames) {
386  if (paramValues.find(paramName) == paramValues.end()) {
387  paramValues[paramName] = 0.0;
388  inTree->SetBranchAddress(paramName.c_str(), &paramValues[paramName]);
389  }
390  }
391  }
392 
393  // Add weight branches
394  std::map<std::string, double> weights;
395  std::map<std::string, TBranch*> weightBranches;
396 
397  for (const auto& rwConfig : reweightConfigs) {
398  weights[rwConfig.weightBranchName] = 1.0;
399  weightBranches[rwConfig.weightBranchName] = outTree->Branch(
400  rwConfig.weightBranchName.c_str(),
401  &weights[rwConfig.weightBranchName],
402  (rwConfig.weightBranchName + "/D").c_str()
403  );
404  MACH3LOG_INFO("Added weight branch: {}", rwConfig.weightBranchName);
405  }
406 
407  bool processMCMCreweighted=false;
408 
409  // If a given reweight is 1D Gaussian we can just let MCMCProcessor method do the reweight
410  for (const auto& rwConfig : reweightConfigs){
411  if (rwConfig.dimension == 1 && rwConfig.type == "Gaussian"){
412  // Extract the parameter names and convert priorValues to the format processor needs
413  const std::vector<std::string>& paramNames = rwConfig.paramNames;
414  std::vector<double> priorCentral;
415  std::vector<double> priorSigma;
416 
417  // Extract means and sigmas from the prior pairs
418  for (const auto& priorPair : rwConfig.priorValues) {
419  priorCentral.push_back(priorPair[0]); // mean
420  priorSigma.push_back(priorPair[1]); // sigma
421  }
422 
423  processor->ReweightPrior(paramNames, priorCentral, priorSigma);
424  MACH3LOG_INFO("Applied Gaussian reweighting for {} parameters", paramNames.size());
425  for (size_t i = 0; i < paramNames.size(); ++i) {
426  MACH3LOG_INFO(" {}: mean={}, sigma={}", paramNames[i], priorCentral[i], priorSigma[i]);
427  }
428  processMCMCreweighted=true;
429  }
430  }
431  // For 2D reweight and non-gaussian (ie TGraph) 1D reweight we need to do it ourselves
432  // Process all entries
433  Long64_t nEntries = inTree->GetEntries();
434  MACH3LOG_INFO("Processing {} entries", nEntries);
435 
436 
438 
439  if (processMCMCreweighted) {
440  MACH3LOG_INFO("MCMCProcessor has reweighted, skipping duplicate reweighting");
441  } else {
442  for (Long64_t i = 0; i < nEntries; ++i) {
443  if(i % (nEntries/20) == 0) MaCh3Utils::PrintProgressBar(i, nEntries);
444 
445  inTree->GetEntry(i);
446 
447  // Calculate weights for all configurations
448  for (const auto& rwConfig : reweightConfigs) {
449  double weight = 1.0;
450 
451  if (rwConfig.dimension == 1 && rwConfig.type != "Gaussian") {
452  if (rwConfig.type == "TGraph") {
453  double paramValue = paramValues[rwConfig.paramNames[0]];
454  weight = Graph_interpolate1D(rwConfig.graph_1D.get(), paramValue);
455  } else {
456  MACH3LOG_ERROR("Unsupported 1D reweight type: {} for {}", rwConfig.type, rwConfig.key);
457  }
458  } else if (rwConfig.dimension == 2) {
459  if (rwConfig.type == "TGraph2D") {
460  double dm32 = paramValues[rwConfig.paramNames[0]];
461  double theta13 = paramValues[rwConfig.paramNames[1]];
462  if (dm32 > 0) {
463  // Normal Ordering
464  if (rwConfig.graph_NO) {
465  weight = Graph_interpolateNO(rwConfig.graph_NO.get(), theta13, dm32);
466  } else {
467  MACH3LOG_ERROR("NO graph not available for {}", rwConfig.key);
468  weight = 0.0;
469  }
470  } else {
471  // Inverted Ordering
472  if (rwConfig.graph_IO) {
473  weight = Graph_interpolateIO(rwConfig.graph_IO.get(), theta13, dm32);
474  } else {
475  MACH3LOG_ERROR("IO graph not available for {}", rwConfig.key);
476  weight = 0.0;
477  }
478  }
479  }
480  }
481  weights[rwConfig.weightBranchName] = weight;
482  }
483  // Fill the output tree
484  outTree->Fill();
485  } // end loop over entries
486  }
487 
488  // Write and close
489  outFile->cd();
490  outTree->Write();
491 
492  // once we have finished the reweight save its configuration (reweightConfigNode) to the root file as a macro
493 
494  TMacro reweightMacro;
495  reweightMacro.SetName("Reweight_Config");
496  reweightMacro.SetTitle("ReweightMCMC configuration");
497  std::stringstream ss;
498  ss << reweight_settings;
499  reweightMacro.AddLine(ss.str().c_str());
500  reweightMacro.Write();
501 
502  if (processMCMCreweighted){
503  MACH3LOG_INFO("MCMCProcessor reweighting applied, Final reweighted file is: {}_reweighted.root", inputFile.substr(0, inputFile.find_last_of('.')));
504  // delete the file we just created since MCMCProcessor already did the reweighting and saved it to a file
505  outTree.reset(); // Release TTree before closing TFile
506  outFile->Close();
507  outFile.reset();
508 
509  if (std::remove(outputFile.c_str()) != 0) {
510  MACH3LOG_ERROR("Error deleting temporary file: {}", outputFile);
511  } else {
512  MACH3LOG_INFO("Deleted temporary file: {}", outputFile);
513  }
514  } else {
515  MACH3LOG_INFO("Reweighting completed successfully!");
516  MACH3LOG_INFO("Final reweighted file is: {}", outputFile);
517  }

520 double Graph_interpolateNO(TGraph2D* graph, double theta13, double dm32)
521 {
522  if (!graph) {

525  }
526

529  double ymin = graph->GetYmin();
530  double ymax = graph->GetYmax();
531 
532  double chiSquared, prior;
533

570 double Graph_interpolate1D(TGraph* graph, double theta13)
571 {
573  if (!graph) {
574  MACH3LOG_ERROR("Graph pointer is null");
575  throw MaCh3Exception(__FILE__, __LINE__);
576  }
577 
578  double xmax = -999999999;
579  double xmin = 999999999;
580 
581  for (int i = 0; i < graph->GetN(); i++) {
582  double x = graph->GetX()[i];
