196 double samplePDFBase::getTestStatLLH(const double data, const double mc) {
197 // ***************************************************************************
198  // Need some MC
199  if(mc == 0) return 0.;
200 
201  double negLogL = 0;
202  if(mc > 0 && data > 0)
203  {
204  //http://hyperphysics.phy-astr.gsu.edu/hbase/math/stirling.html
205  negLogL += (mc - data + data * std::log(data/mc));
206  }
207  else if(mc > 0 && data == 0) negLogL += mc;
208 
209  return negLogL;
210 }

213 // data is data, mc is mc, w2 is Sum(w_{i}^2) (sum of weights squared), which is sigma^2_{MC stats}
214 double samplePDFBase::getTestStatLLH(const double data, const double mc, const double w2) {
215 // *************************
216 
217  // Need some MC
218  if (mc == 0) return 0.0;
219 
220  // The MC used in the likeliihood calculation
221  // Is allowed to be changed by Barlow Beeston beta parameters
222  double newmc = mc;
223 
224  switch (fTestStatistic)
225  {
226  //CW: Not full Barlow-Beeston or what is referred to as "light": we're not introducing any more parameters
227  // Assume the MC has a Gaussian distribution around generated
228  // As in https://arxiv.org/abs/1103.0354 eq 10, 11
229  //CW: Calculate the Barlow-Beeston likelhood contribution from MC statistics
230  // Assumes the beta scaling parameters are Gaussian distributed
231  // Follows arXiv:1103.0354 section 5 and equation 8, 9, 10, 11 on page 4/5
232  // Essentially solves equation 11
233  case (kBarlowBeeston):
234  {
235  // The penalty from MC statistics using Conways approach (https://cds.cern.ch/record/1333496?)
236  double penalty = 0;
237  // Barlow-Beeston uses fractional uncertainty on MC, so sqrt(sum[w^2])/mc
238  const double fractional = std::sqrt(w2)/mc;
239  // b in quadratic equation
240  const double temp = mc*fractional*fractional-1;
241  // b^2 - 4ac in quadratic equation
242  const double temp2 = temp*temp + 4*data*fractional*fractional;
243  if (temp2 < 0) {
244  std::cerr << "Negative square root in Barlow Beeston coefficient calculation!" << std::endl;
245  std::cerr << __FILE__ << ":" << __LINE__ << std::endl;
246  throw;
247  }
248  // Solve for the positive beta
249  const double beta = (-1*temp+sqrt(temp2))/2.;
250  newmc = mc*beta;
251  // And penalise the movement in beta relative the mc uncertainty
252  if (fractional > 0) penalty = (beta-1)*(beta-1)/(2*fractional*fractional);
253  else penalty = 0;
254 
255  // Calculate the new Poisson likelihood
256  // For Barlow-Beeston newmc is modified, so can only calculate Poisson likelihood after Barlow-Beeston
257  // For the Poisson likelihood, this is just the usual calculation
258  // For IceCube likelihood, we calculate it later
259  double stat = 0;
260  // All likelihood calculations may use the bare Poisson likelihood, so calculate here
261  if (data == 0) stat = newmc;
262  else if (newmc > 0) stat = newmc-data+data*std::log(data/newmc);
263 
264  // Return the statistical contribution and penalty
265  return stat+penalty;
266  }
267  break;
268  //KS: Alternative calculation of Barlow-Beeston following Hans Dembinski and Ahmed Abdelmottele arXiv:2206.12346v2
269  case (kDembinskiAbdelmottele):
270  {
271  //KS: code follows authors implementation from:
272  //https://github.com/scikit-hep/iminuit/blob/059d06b00cae097ebf340b218b4eb57357111df8/src/iminuit/cost.py#L274-L300
273 
274  //the so-called effective count
275  const double k = mc*mc / w2;
276  //Calculate beta which is scaling factor between true and generated MC
277  const double beta = (data + k) / (mc + k);
278 
279  newmc = mc*beta;
280  // And penalise the movement in beta relative the mc uncertainty
281  const double penalty = k*beta-k+k*std::log(k/(k*beta));
282 
283  // Calculate the new Poisson likelihood
284  // For Barlow-Beeston newmc is modified, so can only calculate Poisson likelihood after Barlow-Beeston
285  // For the Poisson likelihood, this is just the usual calculation
286  // For IceCube likelihood, we calculate it later
287  double stat = 0;
288  // All likelihood calculations may use the bare Poisson likelihood, so calculate here
289  if (data == 0) stat = newmc;
290  else if (newmc > 0) stat = newmc-data+data*std::log(data/newmc);
291 
292  // Return the statistical contribution and penalty
293  return stat+penalty;
294  }
295  break;
296  //CW: Also try the IceCube likelihood
297  // It does not modify the MC content
298  // https://arxiv.org/abs/1901.04645
299  // Argelles, C.A., Schneider, A. & Yuan, T. J. High Energ. Phys. (2019) 2019: 30. https://doi.org/10.1007/JHEP06(2019)030
300  // We essentially construct eq 3.16 and take the logarithm
301  // in eq 3.16, mu is MC, sigma2 is w2, k is data
302  case (kIceCube):
303  {
304  double stat = 0.0;
305 
306  // If there for some reason is 0 mc uncertainty, return the Poisson LLH
307  if (w2 == 0)
308  {
309  // Calculate the new Poisson likelihood
310  if (data == 0) stat = newmc;
311  else if (newmc > 0) stat = newmc-data+data*std::log(data/newmc);
312 
313  return stat;
314  }
315  // Auxillary variables
316  const long double b = mc/w2;
317  const long double a = mc*b+1;
318  const long double k = data;
319  // Use C99's implementation of log of gamma function to not be C++11 dependent
320  stat = -1*(a * logl(b) + lgammal(k+a) - lgammal(k+(long double)1) - ((k+a)*log1pl(b)) - lgammal(a));
321 
322  // Return the statistical contribution and penalty
323  return stat;
324  }
325  break;
326  //KS: Pearson works on assumption that event distribution in each bin is described by a Gaussian which in our case is not fulfilled for all bins, hence use it at your own risk
327  case (kPearson):
328  {
329  //KS: 2 is because this function returns -LLH not -2LLH
330  const double stat = (data-mc)*(data-mc)/(2*mc);
331 
332  // Return the statistical
333  return stat;
334  }
335  break;
336  case (kPoisson):
337  {
338  //Just call getTestStatLLH which doesn't take in weights
339  //and is a poisson likelihood comparison.
340  return getTestStatLLH(data, mc);//stat;
341  }
342  break;
343 
344  default:
345  std::cerr << "Couldn't find TestStatistic " << fTestStatistic << " exiting!" << std::endl;
346  std::cerr << __FILE__ << ":" << __LINE__ << std::endl;
347  throw;
348  } // end switch
349 }
