1104 void MCMCProcessor::CacheSteps() {

1109 
1110  if(ParStep != nullptr)
1111  {

1116  }
1117 
1118  MACH3LOG_INFO("Caching input tree...");
1119  MACH3LOG_INFO("Allocating {:.2f} MB", (sizeof(double)*nDraw*nEntries)/1.E6);
1120  TStopwatch clock;
1121  clock.Start();
1122 
1123  ParStep = new double*[nDraw];
1124  StepNumber = new int[nEntries];
1125 
1126  hpost2D = new TH2D**[nDraw]();
1127 
1128  for (int i = 0; i < nDraw; ++i)
1129  {
1130  ParStep[i] = new double[nEntries];
1131  hpost2D[i] = new TH2D*[nDraw]();
1132 
1133  for (int j = 0; j < nEntries; ++j)
1134  {

1137  if(i == 0) StepNumber[j] = -999.99;
1138  }
1139  }
1140 
1141  // Set all the branches to off
1142  Chain->SetBranchStatus("*", false);
1143

1145  for (int i = 0; i < nDraw; ++i)
1146  {
1147  Chain->SetBranchStatus(BranchNames[i].Data(), true);
1148  }
1149  Chain->SetBranchStatus("step", true);
1150

1152  // Loop over the entries
1153  //KS: This is really a bottleneck right now, thus revisit with ROOT6 https://pep-root6.github.io/docs/analysis/parallell/root.html
1154  for (int j = 0; j < nEntries; ++j)
1155  {
1156  if (j % countwidth == 0)
1157  MaCh3Utils::PrintProgressBar(j, nEntries);
1158 
1159  Chain->SetBranchAddress("step", &StepNumber[j]);
1160  // Set the branch addresses for params
