18
19
20 if(delay_probability > 1.0){

22 delay_probability = 1.0;
23 }
24 if(delay_probability<=0.0){

79double DelayedMR2T2::AcceptanceProbability() {
80// *************************
81 // From https://arxiv.org/pdf/2010.04190 and DRAM Matlab implementation
82 const double numerator = std::max(0.0, std::exp(MinLogLikelihood - logLProp) - 1.0);
83 const double denominator = std::exp(MinLogLikelihood - logLCurr) - 1.0;
84 if (denominator <= 0.0) return 1.0;
85
86 // Large enough that the the minus 1 will make NO difference so the max term cancels
87 if(std::isinf(numerator) or std::isinf(denominator) ){
88 return MR2T2::AcceptanceProbability();
89 }
90
91 return std::min(numerator / denominator, 1.0);
92}

94// *************************
95void DelayedMR2T2::DoStep() {
96// *************************
97 // Set the initial rejection to false
98 StoreCurrentStep();
99
100 // Apply initial scale
101 ScaleSystematics(initial_scale);
102
103 bool accepted = false;
104 bool delay = false;
105 accepted_delayed = false;
106
107 MinLogLikelihood = M3::_LARGE_LOGL_;
108
109 for(int i=0; i<=max_rejections; ++i)
110 {
111 ProposeStep();
112
113 // Small hack to ensure we can "leapfrog"
114 for (auto &syst : systematics)
115 {
116 syst->AcceptStep();
117 }
118
119 if(out_of_bounds or logLProp>MinLogLikelihood){
120 continue;
121 }
122
123 if(i==0){
124 accProb = MR2T2::AcceptanceProbability();
125 }
126 else{
127 accProb = AcceptanceProbability();
128 delay = true;
129 }
130 if(IsStepAccepted(accProb)){
131 AcceptStep();
132 accepted = true;
133
134 // Keep track of if step is delayed + accepted for storage
135 accepted_delayed = delay;
136
137 // Update
138 break;
139 }
141 if(not out_of_bounds and delay_on_oob_only){
142 // If we are not out of bounds and only delaying on out of bounds steps
143 // We can skip the delay
144 break;
145 }
146
148 if(not ProbabilisticDelay()){
149 break;
150 }
151
152
153 // Temporary, means we can "leapfrog"
154 ScaleSystematics(decay_rate);
155 MinLogLikelihood = logLProp;
156 }
157
158 // If we reject we need to reset everything [this is probably a tad too slow...
159 // especially in PCA...
160 if(!accepted){
161 for(int i=0; i<static_cast<int>(systematics.size()); ++i){
162 for(int j=0; j<static_cast<int>(systematics[i]->GetParCurrVec().size()); ++j){
163 systematics[i]->SetParCurrProp(j, current_step_vals[i][j]);
164 }
165 }
166 }
167 ResetSystScale();
168}
