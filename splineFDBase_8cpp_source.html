238// function in SplineMonolith.cpp
239void splineFDBase::FindSplineSegment() {
240//****************************************
241 //HW okay let's try this, we delete+refill a new array which we'll fill with x-s for our segment
242 for (int iSyst = 0; iSyst < nUniqueSysts; ++iSyst) {
243 const int nPoints = UniqueSystNKnots[iSyst];
244 // KS: By reference to not copy memory like a fool
245 const std::vector<M3::float_t>& xArray = UniqueSystXPts[iSyst];
246
247 // Get the variation for this reconfigure for the ith parameter
248 const int GlobalIndex = UniqueSystIndices[iSyst];
249 const M3::float_t xvar = M3::float_t(xsec->getParProp(GlobalIndex));
250
251 xVarArray[iSyst] = xvar;
252
253 M3::int_t segment = 0;
254 M3::int_t kHigh = M3::int_t(nPoints - 1);
255
256 //KS: We expect new segment is very close to previous
257 const M3::int_t PreviousSegment = M3::int_t(UniqueSystCurrSegment[iSyst]);
258 //KS: It is quite probable the new segment is same as in previous step so try to avoid binary search
259 if( xArray[PreviousSegment+1] > xvar && xvar >= xArray[PreviousSegment] ) {
260 segment = PreviousSegment;
261 } else if (xvar <= xArray[0]) {
262 // If the variation is below the lowest saved spline point
263 segment = 0;
264 // If the variation is above the highest saved spline point
265 } else if (xvar >= xArray[nPoints-1]) {
266 //CW: Yes, the -2 is indeed correct, see TSpline.cxx:814 and //see: https://savannah.cern.ch/bugs/?71651
267 segment = kHigh;
268 //KS: It is quite probable the new segment is same as in previous step so try to avoid binary search
269 } else {
270 // The top point we've got
271 M3::int_t kHalf = 0;
272 // While there is still a difference in the points (we haven't yet found the segment)
273 // This is a binary search, incrementing segment and decrementing kHalf until we've found the segment
274 while (kHigh - segment > 1) {
275 // Increment the half-step
276 kHalf = M3::int_t((segment + kHigh)/2);
277 // If our variation is above the kHalf, set the segment to kHalf
278 if (xvar > xArray[kHalf]) {
279 segment = kHalf;
280 // Else move kHigh down
281 } else {
282 kHigh = kHalf;
283 }
284 } // End the while: we've now done our binary search
285 } // End the else: we've now found our point
286
287 if (segment >= nPoints-1 && nPoints > 1){segment = M3::int_t(nPoints-2);}
288 UniqueSystCurrSegment[iSyst] = segment;
289 } //end loop over params
290}

293void splineFDBase::CalcSplineWeights()
294//****************************************
295{
296 #ifdef MULTITHREAD
297 #pragma omp parallel for simd
298 #endif
299 for (size_t iCoeff = 0; iCoeff < uniquecoeffindices.size(); ++iCoeff)
300 {
301 const int iSpline = uniquecoeffindices[iCoeff];
302 const short int uniqueIndex = short(uniquesplinevec_Monolith[iSpline]);
303 const short int currentsegment = short(UniqueSystCurrSegment[uniqueIndex]);
304
305 const int segCoeff = coeffindexvec[iSpline]+currentsegment;
306 const int coeffOffset = segCoeff * 4;
307 // These are what we can extract from the TSpline3
308 const M3::float_t x = xcoeff_arr[segCoeff];
309 const M3::float_t y = manycoeff_arr[coeffOffset+kCoeffY];
310 const M3::float_t b = manycoeff_arr[coeffOffset+kCoeffB];
311 const M3::float_t c = manycoeff_arr[coeffOffset+kCoeffC];
312 const M3::float_t d = manycoeff_arr[coeffOffset+kCoeffD];
313
314 // Get the variation for this reconfigure for the ith parameter
315 const M3::float_t xvar = xVarArray[uniqueIndex];
316 // The Delta(x)
317 const M3::float_t dx = xvar - x;
318
319 //Speedy 1% time boost https://en.cppreference.com/w/c/numeric/math/fma (see ND code!)
320 M3::float_t weight = M3::fmaf_t(dx, M3::fmaf_t(dx, M3::fmaf_t(dx, d, c), b), y);
321 //This is the speedy version of writing dx^3+b*dx^2+c*dx+d
322
323 //ETA - do we need this? We check later for negative weights and I wonder if this is even
324 //possible with the fmaf line above?
325 if(weight < 0){weight = 0.;} //Stops is getting negative weights
326
327 weightvec_Monolith[iSpline] = weight;
328 }
329}

709// Rather work with spline coefficients in the splines, let's copy ND and use coefficient arrays
710void splineFDBase::getSplineCoeff_SepMany(int splineindex, M3::float_t* &xArray, M3::float_t* &manyArray){
711//****************************************
712 // Initialise all arrays to 1.0
713 int nPoints;
714 //No point evaluating a flat spline
715 nPoints = splinevec_Monolith[splineindex]->GetNp();
716
717 for (int i = 0; i < nPoints; i++) {
718 xArray[i] = 1.0;
719 for (int j = 0; j < 4; j++) {
720 manyArray[i*4+j] = 1.0;
721 }
722 }
723
724 for(int i=0; i<nPoints; i++) {
725 // Spline coefficients to be
726 // M3::float_t type is defined by the LOW_MEMORY_STRUCTS compiler flag
727 // so M3::float_t can be double or float depending on this
728 M3::float_t x = M3::float_t(-999.99);
729 M3::float_t y = M3::float_t(-999.99);
730 M3::float_t b = M3::float_t(-999.99);
731 M3::float_t c = M3::float_t(-999.99);
732 M3::float_t d = M3::float_t(-999.99);
733 splinevec_Monolith[splineindex]->GetCoeff(i, x, y, b, c, d);
734
735 // Store the coefficients for each knot contiguously in memory
736 // 4 because manyArray stores y,b,c,d
737 xArray[i] = x;
738 manyArray[i*4] = y;
739 manyArray[i*4+1] = b;
740 manyArray[i*4+2] = c;
741 manyArray[i*4+3] = d;
742 }
743
744 //We now clean up the splines!
745 delete splinevec_Monolith[splineindex];
746 splinevec_Monolith[splineindex] = nullptr;
747}
