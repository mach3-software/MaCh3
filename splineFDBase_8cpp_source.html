229void splineFDBase::CalcSplineWeights()
230//****************************************
231{
232 #ifdef MULTITHREAD
233 #pragma omp parallel for simd
234 #endif
235 for (size_t iCoeff = 0; iCoeff < uniquecoeffindices.size(); ++iCoeff)
236 {
237 const int iSpline = uniquecoeffindices[iCoeff];
238 const short int uniqueIndex = short(uniquesplinevec_Monolith[iSpline]);
239 const short int currentsegment = short(SplineSegments[uniqueIndex]);
240
241 const int segCoeff = coeffindexvec[iSpline]+currentsegment;
242 const int coeffOffset = segCoeff * 4;
243 // These are what we can extract from the TSpline3
244 const M3::float_t y = manycoeff_arr[coeffOffset+kCoeffY];
245 const M3::float_t b = manycoeff_arr[coeffOffset+kCoeffB];
246 const M3::float_t c = manycoeff_arr[coeffOffset+kCoeffC];
247 const M3::float_t d = manycoeff_arr[coeffOffset+kCoeffD];
248
249 // Get the variation for this reconfigure for the ith parameter
251 const M3::float_t xvar = (*SplineInfoArray[uniqueIndex].splineParsPointer);
252 // The Delta(x) = xvar - x
253 const M3::float_t dx = xvar - xcoeff_arr[segCoeff];
254
255 //Speedy 1% time boost https://en.cppreference.com/w/c/numeric/math/fma (see ND code!)
256 M3::float_t weight = M3::fmaf_t(dx, M3::fmaf_t(dx, M3::fmaf_t(dx, d, c), b), y);
257 //This is the speedy version of writing dx^3+b*dx^2+c*dx+d
258
259 //ETA - do we need this? We check later for negative weights and I wonder if this is even
260 //possible with the fmaf line above?
261 if(weight < 0){weight = 0.;} //Stops is getting negative weights
262
263 weightvec_Monolith[iSpline] = weight;
264 }
265}

553 ParamValues[iSpline] = 0.;

602//****************************************
603// Rather work with spline coefficients in the splines, let's copy ND and use coefficient arrays
604void splineFDBase::getSplineCoeff_SepMany(int splineindex, M3::float_t* &xArray, M3::float_t* &manyArray){
605//****************************************
606 //No point evaluating a flat spline
607 int nPoints = splinevec_Monolith[splineindex]->GetNp();
608
609 for (int i = 0; i < nPoints; i++) {
610 xArray[i] = 1.0;
611 for (int j = 0; j < 4; j++) {
612 manyArray[i*4+j] = 1.0;
613 }
614 }
615
616 for(int i=0; i<nPoints; i++) {
617 M3::float_t x = M3::float_t(-999.99);
618 M3::float_t y = M3::float_t(-999.99);
619 M3::float_t b = M3::float_t(-999.99);
620 M3::float_t c = M3::float_t(-999.99);
621 M3::float_t d = M3::float_t(-999.99);
622 splinevec_Monolith[splineindex]->GetCoeff(i, x, y, b, c, d);
623
624 // Store the coefficients for each knot contiguously in memory
625 // 4 because manyArray stores y,b,c,d
626 xArray[i] = x;
627 manyArray[i*4] = y;
628 manyArray[i*4+1] = b;
629 manyArray[i*4+2] = c;
630 manyArray[i*4+3] = d;
631 }
632
633 //We now clean up the splines!
634 delete splinevec_Monolith[splineindex];
635 splinevec_Monolith[splineindex] = nullptr;
636}

846void splineFDBase::FillSampleArray(std::string SampleName, std::vector<std::string> OscChanFileNames)
847{
848 int iSample = getSampleIndex(SampleName);
849 int nOscChannels = nOscChans[iSample];
850
851 for (int iOscChan = 0; iOscChan < nOscChannels; iOscChan++) {
852 MACH3LOG_INFO("Processing: {}", OscChanFileNames[iOscChan]);
853 TSpline3* mySpline = nullptr;
854 TSpline3_red* Spline = nullptr;
855 TString Syst, Mode;
856 int nKnots, SystNum, ModeNum, Var1Bin, Var2Bin, Var3Bin = M3::_BAD_INT_;
857 double x,y, Eval = M3::_BAD_DOUBLE_;
858 bool isFlat = true;
859
860 std::vector<std::string> SplineFileNames;
861
862 auto File = std::unique_ptr<TFile>(TFile::Open(OscChanFileNames[iOscChan].c_str()));
863
864 if (!File || File->IsZombie()) {
865 MACH3LOG_ERROR("File {} not found", OscChanFileNames[iOscChan]);
866 throw MaCh3Exception(__FILE__, __LINE__);
867 }
868
869 //This is the MC specific part of the code
870 //i.e. we always assume that the splines are just store in single TDirectory and they're all in there as single objects
871 for (auto k : *File->GetListOfKeys()) {
872 auto Key = static_cast<TKey*>(k);
873 TClass *Class = gROOT->GetClass(Key->GetClassName(), false);
874 if(!Class->InheritsFrom("TSpline3")) {
875 continue;
876 }
877
878 std::string FullSplineName = std::string(Key->GetName());
879
880 for (auto TestSplineFile : SplineFileNames) {
881 if (TestSplineFile == FullSplineName) {
882 MACH3LOG_CRITICAL("Skipping spline - Found a spline whose name has already been encountered before: {}", FullSplineName);
883 continue;
884 }
885 }
886 SplineFileNames.push_back(FullSplineName);
887
888 std::vector<std::string> Tokens = GetTokensFromSplineName(FullSplineName);
889
890 if (Tokens.size() != kNTokens) {
891 MACH3LOG_ERROR("Invalid tokens from spline name - Expected {} tokens. Check implementation in GetTokensFromSplineName()", static_cast<int>(kNTokens));
892 throw MaCh3Exception(__FILE__, __LINE__);
893 }
894
895 Syst = Tokens[kSystToken];
896 Mode = Tokens[kModeToken];
897 Var1Bin = std::stoi(Tokens[kVar1BinToken]);
898 Var2Bin = std::stoi(Tokens[kVar2BinToken]);
899 Var3Bin = std::stoi(Tokens[kVar3BinToken]);
900
901 SystNum = -1;
902 for (unsigned iSyst = 0; iSyst < SplineFileParPrefixNames[iSample].size(); iSyst++) {
903 if (Syst == SplineFileParPrefixNames[iSample][iSyst]) {
904 SystNum = iSyst;
905 break;
906 }
907 }
908
909 // If the syst doesn't match any of the spline names then skip it
910 if (SystNum == -1){
911 MACH3LOG_WARN("Couldn't match!!");
912 MACH3LOG_DEBUG("Couldn't Match any systematic name in xsec yaml with spline name: {}" , FullSplineName);
913 continue;
914 }
915
916 ModeNum = -1;
917 for (unsigned int iMode = 0; iMode < SplineModeVecs[iSample][SystNum].size(); iMode++) {
918 if (Mode == Modes->GetSplineSuffixFromMaCh3Mode(SplineModeVecs[iSample][SystNum][iMode])) {
919 ModeNum = iMode;
920 break;
921 }
922 }
923
924 if (ModeNum == -1) {
925 //DB - If you have splines in the root file that you don't want to use (e.g. removing a mode from a syst), this will cause a throw
926 // Therefore include as debug warning and continue instead
927 MACH3LOG_DEBUG("Couldn't find mode for {} in {}. Problem Spline is : {} ", Mode, Syst, FullSplineName);
928 continue;
929 }
930
931 mySpline = Key->ReadObject<TSpline3>();
932
933 if (isValidSplineIndex(SampleName, iOscChan, SystNum, ModeNum, Var1Bin, Var2Bin, Var3Bin)) { // loop over all the spline knots and check their value
934 MACH3LOG_DEBUG("Pushed back monolith for spline {}", FullSplineName);
935 // if the value is 1 then set the flat bool to false
936 nKnots = mySpline->GetNp();
937 isFlat = true;
938 for (int iKnot = 0; iKnot < nKnots; iKnot++) {
939 mySpline->GetKnot(iKnot, x, y);
940
941 Eval = mySpline->Eval(x);
942 if (Eval < 0.99999 || Eval > 1.00001)
943 {
944 isFlat = false;
945 break;
946 }
947 }
948
949 //Rather than keeping a mega vector of splines then converting, this should just keep everything nice in memory!
950 indexvec[iSample][iOscChan][SystNum][ModeNum][Var1Bin][Var2Bin][Var3Bin]=MonolithIndex;
951 coeffindexvec.push_back(CoeffIndex);
952 // Should save memory rather saving [x_i_0 ,... x_i_maxknots] for every spline!
953 if (isFlat) {
954 splinevec_Monolith.push_back(nullptr);
955 delete mySpline;
956 } else {
957 Spline = new TSpline3_red(mySpline, SplineInterpolationTypes[iSample][SystNum]);
958 delete mySpline;
959
960 splinevec_Monolith.push_back(Spline);
961 uniquecoeffindices.push_back(MonolithIndex); //So we can get the unique coefficients and skip flat splines later on!
962 CoeffIndex+=nKnots;
963 }
964 //Incrementing MonolithIndex to keep track of number of valid spline indices
965 MonolithIndex+=1;
966 } else {
967 //Potentially you are not a valid spline index
968 delete mySpline;
969 }
970 }//End of loop over all TKeys in file
971
972 //A bit of clean up
973 File->Delete("*");
974 File->Close();
975 } //End of oscillation channel loop
976}
