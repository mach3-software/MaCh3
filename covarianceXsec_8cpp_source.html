27void covarianceXsec::InitXsecFromConfig() {
28// ********************************************
29 _fSystToGlobalSystIndexMap.resize(SystType::kSystTypes);
30
31 _fParamType = std::vector<SystType>(_fNumPar);
32 _ParameterGroup = std::vector<std::string>(_fNumPar);
33
34 //KS: We know at most how params we expect so reserve memory for max possible params. Later we will shrink to size to not waste memory. Reserving means slightly faster loading and possible less memory fragmentation.
35 NormParams.reserve(_fNumPar);
36 SplineParams.reserve(_fNumPar);
37
38 int i = 0;
39 unsigned int ParamCounter[SystType::kSystTypes] = {0};
40 //ETA - read in the systematics. Would be good to add in some checks to make sure
41 //that there are the correct number of entries i.e. are the _fNumPars for Names,
42 //PreFitValues etc etc.
43 for (auto const &param : _fYAMLDoc["Systematics"])
44 {
45 _ParameterGroup[i] = Get<std::string>(param["Systematic"]["ParameterGroup"], __FILE__ , __LINE__);
46
47 //Fill the map to get the correlations later as well
48 auto ParamType = Get<std::string>(param["Systematic"]["Type"], __FILE__ , __LINE__);
49 //Now load in variables for spline systematics only
50 if (ParamType.find(SystType_ToString(SystType::kSpline)) != std::string::npos)
51 {
52 //Set param type
53 _fParamType[i] = SystType::kSpline;
54 // Fill Spline info
55 SplineParams.push_back(GetXsecSpline(param["Systematic"]));
56
57 if (param["Systematic"]["SplineInformation"]["SplineName"]) {
58 _fSplineNames.push_back(param["Systematic"]["SplineInformation"]["SplineName"].as<std::string>());
59 }
60
61 //Insert the mapping from the spline index i.e. the length of _fSplineNames etc
62 //to the Systematic index i.e. the counter for things like _fDetID and _fDetID
63 _fSystToGlobalSystIndexMap[SystType::kSpline].insert(std::make_pair(ParamCounter[SystType::kSpline], i));
64 ParamCounter[SystType::kSpline]++;
65 } else if(param["Systematic"]["Type"].as<std::string>() == SystType_ToString(SystType::kNorm)) {
66 _fParamType[i] = SystType::kNorm;
67 NormParams.push_back(GetXsecNorm(param["Systematic"], i));
68 _fSystToGlobalSystIndexMap[SystType::kNorm].insert(std::make_pair(ParamCounter[SystType::kNorm], i));
69 ParamCounter[SystType::kNorm]++;
70 } else if(param["Systematic"]["Type"].as<std::string>() == SystType_ToString(SystType::kFunc)){
71 _fParamType[i] = SystType::kFunc;
72 _fSystToGlobalSystIndexMap[SystType::kFunc].insert(std::make_pair(ParamCounter[SystType::kFunc], i));
73 ParamCounter[SystType::kFunc]++;
74 } else{
75 MACH3LOG_ERROR("Given unrecognised systematic type: {}", param["Systematic"]["Type"].as<std::string>());
76 std::string expectedTypes = "Expecting ";
77 for (int s = 0; s < SystType::kSystTypes; ++s) {
78 if (s > 0) expectedTypes += ", ";
79 expectedTypes += SystType_ToString(static_cast<SystType>(s)) + "\"";
80 }
81 expectedTypes += ".";
82 MACH3LOG_ERROR(expectedTypes);
83 throw MaCh3Exception(__FILE__, __LINE__);
84 }
85 i++;
86 } //end loop over params
87
88 //Add a sanity check,
89 if(_fSplineNames.size() != ParamCounter[SystType::kSpline]){
90 MACH3LOG_ERROR("_fSplineNames is of size {} but found {} spline parameters", _fSplineNames.size(), ParamCounter[SystType::kSpline]);
91 throw MaCh3Exception(__FILE__, __LINE__);
92 }
93 //KS We resized them above to all params to fight memory fragmentation, now let's resize to fit only allocated memory to save RAM
94 NormParams.shrink_to_fit();
95 SplineParams.shrink_to_fit();
96}

155// ********************************************
156 XsecNorms4 norm;
157 norm.name = GetParFancyName(Index);
158
159 // ETA Empty DummyVector can be used to specify no cut for mode, target and neutrino flavour
160 // ETA Has to be of size 0 to mean apply to all
161 std::vector<int> DummyModeVec;
162 //Ultimately all this information ends up in the NormParams vector
163
164 //Copy the mode information into an XsecNorms4 struct
165 norm.modes = GetFromManager<std::vector<int>>(param["Mode"], DummyModeVec, __FILE__ , __LINE__);
166 norm.pdgs = GetFromManager<std::vector<int>>(param["NeutrinoFlavour"], DummyModeVec, __FILE__ , __LINE__);
167 norm.preoscpdgs = GetFromManager<std::vector<int>>(param["NeutrinoFlavourUnosc"], DummyModeVec, __FILE__ , __LINE__);
168 norm.targets = GetFromManager<std::vector<int>>(param["TargetNuclei"], DummyModeVec, __FILE__ , __LINE__);
169
170 //ETA - I think this can go in the norm parameters only if statement above
171 int NumKinematicCuts = 0;
172 if(param["KinematicCuts"]){
173
174 NumKinematicCuts = int(param["KinematicCuts"].size());
175
176 std::vector<std::string> TempKinematicStrings;
177 std::vector<std::vector<double>> TempKinematicBounds;
178 //First element of TempKinematicBounds is always -999, and size is then 3
179 for(int KinVar_i = 0 ; KinVar_i < NumKinematicCuts ; ++KinVar_i){
180 //ETA: This is a bit messy, Kinematic cuts is a list of maps
181 for (YAML::const_iterator it = param["KinematicCuts"][KinVar_i].begin();it!=param["KinematicCuts"][KinVar_i].end();++it) {
182 TempKinematicStrings.push_back(it->first.as<std::string>());
183 TempKinematicBounds.push_back(it->second.as<std::vector<double>>());
184 }
185 if(TempKinematicStrings.size() == 0) {
186 MACH3LOG_ERROR("Recived a KinematicCuts node but couldn't read the contents (it's a list of single-element dictionaries (python) = map of pairs (C++))");
187 MACH3LOG_ERROR("For Param {}", norm.name);
188 throw MaCh3Exception(__FILE__, __LINE__);
189 }
190 }//KinVar_i
191 norm.KinematicVarStr = TempKinematicStrings;
192 norm.Selection = TempKinematicBounds;
193 }
194
195 //Next ones are kinematic bounds on where normalisation parameter should apply (at the moment only etrue but hope to add q2
196 //We set a bool to see if any bounds exist so we can short-circuit checking all of them every step
197 bool HasKinBounds = false;
198
199 if(norm.KinematicVarStr.size() > 0) HasKinBounds = true;
200
201 norm.hasKinBounds = HasKinBounds;
202 //End of kinematic bound checking
203
204 // Set the global parameter index of the normalisation parameter
205 norm.index = Index;
206
207 return norm;
208}

530// ********************************************
531 // KS: Lambda Function which simply checks if there are no duplicates in std::vector
532 auto CheckForDuplicates = [](const std::vector<std::string>& names, const std::string& nameType) {
533 std::unordered_map<std::string, size_t> seenStrings;
534 for (size_t i = 0; i < names.size(); ++i) {
535 const auto& name = names[i];
536 if (seenStrings.find(name) != seenStrings.end()) {
537 size_t firstIndex = seenStrings[name];
538 MACH3LOG_CRITICAL("There are two systematics with the same {} '{}', first at index {}, and again at index {}", nameType, name, firstIndex, i);
539 throw MaCh3Exception(__FILE__, __LINE__);
540 }
541 seenStrings[name] = i;
542 }
543 };

545 // KS: Checks if there are no duplicates in fancy names etc, this can happen if we merge configs etc
546 CheckForDuplicates(_fFancyNames, "_fFancyNames");
