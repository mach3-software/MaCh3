26void covarianceXsec::InitXsecFromConfig() {
27// ********************************************
28 _fSystToGlobalSystIndexMap.resize(SystType::kSystTypes);
29
30 _fDetID = std::vector<int>(_fNumPar);
31 _fParamType = std::vector<SystType>(_fNumPar);
32 _ParameterGroup = std::vector<std::string>(_fNumPar);
33
34 //KS: We know at most how params we expect so reserve memory for max possible params. Later we will shrink to size to not waste memory. Reserving means slightly faster loading and possible less memory fragmentation.
35 NormParams.reserve(_fNumPar);
36 SplineParams.reserve(_fNumPar);
37
38 int i = 0;
39 unsigned int ParamCounter[SystType::kSystTypes] = {0};
40 //ETA - read in the systematics. Would be good to add in some checks to make sure
41 //that there are the correct number of entries i.e. are the _fNumPars for Names,
42 //PreFitValues etc etc.
43 for (auto const &param : _fYAMLDoc["Systematics"])
44 {
45 _fDetID[i] = (param["Systematic"]["DetID"].as<int>());
46 _ParameterGroup[i] = (param["Systematic"]["ParameterGroup"].as<std::string>());
47
48 //Fill the map to get the correlations later as well
49 std::string ParamType = param["Systematic"]["Type"].as<std::string>();
50 //Now load in variables for spline systematics only
51 if (ParamType.find(SystType_ToString(SystType::kSpline)) != std::string::npos)
52 {
53 //Set param type
54 _fParamType[i] = SystType::kSpline;
55 // Fill Spline info
56 SplineParams.push_back(GetXsecSpline(param["Systematic"]));
57
58 if (param["Systematic"]["SplineInformation"]["SplineName"]) {
59 _fSplineNames.push_back(param["Systematic"]["SplineInformation"]["SplineName"].as<std::string>());
60 }
61
62 //Insert the mapping from the spline index i.e. the length of _fSplineNames etc
63 //to the Systematic index i.e. the counter for things like _fDetID and _fDetID
64 _fSystToGlobalSystIndexMap[SystType::kSpline].insert(std::make_pair(ParamCounter[SystType::kSpline], i));
65 ParamCounter[SystType::kSpline]++;
66 } else if(param["Systematic"]["Type"].as<std::string>() == SystType_ToString(SystType::kNorm)) {
67 _fParamType[i] = SystType::kNorm;
68 NormParams.push_back(GetXsecNorm(param["Systematic"], i));
69 _fSystToGlobalSystIndexMap[SystType::kNorm].insert(std::make_pair(ParamCounter[SystType::kNorm], i));
70 ParamCounter[SystType::kNorm]++;
71 }
72 else if(param["Systematic"]["Type"].as<std::string>() == SystType_ToString(SystType::kFunc)){
73 _fParamType[i] = SystType::kFunc;
74 _fSystToGlobalSystIndexMap[SystType::kFunc].insert(std::make_pair(ParamCounter[SystType::kFunc], i));
75 ParamCounter[SystType::kFunc]++;
76 }
77 else{
78 MACH3LOG_ERROR("Given unrecognised systematic type: {}", param["Systematic"]["Type"].as<std::string>());
79 std::string expectedTypes = "Expecting ";
80 for (int s = 0; s < SystType::kSystTypes; ++s) {
81 if (s > 0) expectedTypes += ", ";
82 expectedTypes += SystType_ToString(static_cast<SystType>(s)) + "\"";
83 }
84 expectedTypes += ".";
85 MACH3LOG_ERROR(expectedTypes);
86 throw MaCh3Exception(__FILE__, __LINE__);
87 }
88 i++;
89 }
90
91 //Add a sanity check,
92 if(_fSplineNames.size() != ParamCounter[SystType::kSpline]){
93 MACH3LOG_ERROR("_fSplineNames is of size {} but found {} spline parameters", _fSplineNames.size(), ParamCounter[SystType::kSpline]);
94 throw MaCh3Exception(__FILE__, __LINE__);
95 }
96 //KS We resized them above to all params to fight memory fragmentation, now let's resize to fit only allocated memory to save RAM
97 NormParams.shrink_to_fit();
98 SplineParams.shrink_to_fit();
99
100 return;
101}

482 auto CheckForDuplicates = [](const std::vector<std::string>& names, const std::string& nameType) {
483 std::unordered_map<std::string, int> seenStrings;
484 for (size_t i = 0; i < names.size(); ++i) {
485 const auto& name = names[i];
486 if (seenStrings.find(name) != seenStrings.end()) {
487 int firstIndex = seenStrings[name];
488 MACH3LOG_CRITICAL("There are two systematics with the same {} '{}', first at index {}, and again at index {}", nameType, name, firstIndex, i);
489 throw MaCh3Exception(__FILE__, __LINE__);
490 }
491 seenStrings[name] = i;
492 }
493 };
494

496 CheckForDuplicates(_fFancyNames, "_fFancyNames");
497 CheckForDuplicates(_fSplineNames, "_fSplineNames");
