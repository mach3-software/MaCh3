239inline std::string GetTF1(const SplineInterpolation i) {
240 // **************************************************
241 std::string Func = "";
242 switch(i) {
243 case SplineInterpolation::kLinearFunc:
244 Func = "([1]+[0]*x)";
245 break;
246 case SplineInterpolation::kTSpline3:
247 case SplineInterpolation::kLinear:
248 case SplineInterpolation::kMonotonic:
249 case SplineInterpolation::kAkima:
250 case SplineInterpolation::kSplineInterpolations:
251 MACH3LOG_ERROR("Interpolation type {} not supported for TF1!", static_cast<int>(i));
252 throw MaCh3Exception(__FILE__, __LINE__);
253 default:
254 MACH3LOG_ERROR("UNKNOWN SPLINE INTERPOLATION SPECIFIED!");
255 MACH3LOG_ERROR("You gave {}", static_cast<int>(i));
256 throw MaCh3Exception(__FILE__ , __LINE__ );
257 }
258 return Func;
259}

264inline RespFuncType SplineInterpolation_ToRespFuncType(const SplineInterpolation i) {
265// **************************************************
266 RespFuncType Type = kRespFuncTypes;
267 switch(i) {
268 case SplineInterpolation::kTSpline3:
269 case SplineInterpolation::kLinear:
270 case SplineInterpolation::kMonotonic:
271 case SplineInterpolation::kAkima:
272 Type = RespFuncType::kTSpline3_red;
273 break;
274 case SplineInterpolation::kLinearFunc:
275 Type = RespFuncType::kTF1_red;
276 break;
277 case SplineInterpolation::kSplineInterpolations:
278 MACH3LOG_ERROR("kSplineInterpolations is not a valid interpolation type!");
279 throw MaCh3Exception(__FILE__, __LINE__);
280 default:
281 MACH3LOG_ERROR("UNKNOWN SPLINE INTERPOLATION SPECIFIED!");
282 MACH3LOG_ERROR("You gave {}", static_cast<int>(i));
283 throw MaCh3Exception(__FILE__, __LINE__);
284 }
285 return Type;
286}

315 MACH3LOG_ERROR("UNKNOWN SPLINE INTERPOLATION SPECIFIED!");
