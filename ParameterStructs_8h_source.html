251inline std::string GetTF1(const SplineInterpolation i) {
252 // **************************************************
253 std::string Func = "";
254 switch(i) {
255 case SplineInterpolation::kLinearFunc:
256 Func = "([1]+[0]*x)";
257 break;
258 case SplineInterpolation::kTSpline3:
259 case SplineInterpolation::kLinear:
260 case SplineInterpolation::kMonotonic:
261 case SplineInterpolation::kAkima:
262 case SplineInterpolation::kSplineInterpolations:
263 MACH3LOG_ERROR("Interpolation type {} not supported for TF1!", static_cast<int>(i));
264 throw MaCh3Exception(__FILE__, __LINE__);
265 default:
266 MACH3LOG_ERROR("UNKNOWN SPLINE INTERPOLATION SPECIFIED!");
267 MACH3LOG_ERROR("You gave {}", static_cast<int>(i));
268 throw MaCh3Exception(__FILE__ , __LINE__ );
269 }
270 return Func;
271}

276inline RespFuncType SplineInterpolation_ToRespFuncType(const SplineInterpolation i) {
277// **************************************************
278 RespFuncType Type = kRespFuncTypes;
279 switch(i) {
280 case SplineInterpolation::kTSpline3:
281 case SplineInterpolation::kLinear:
282 case SplineInterpolation::kMonotonic:
283 case SplineInterpolation::kAkima:
284 Type = RespFuncType::kTSpline3_red;
285 break;
286 case SplineInterpolation::kLinearFunc:
287 Type = RespFuncType::kTF1_red;
288 break;
289 case SplineInterpolation::kSplineInterpolations:
290 MACH3LOG_ERROR("kSplineInterpolations is not a valid interpolation type!");
291 throw MaCh3Exception(__FILE__, __LINE__);
292 default:
293 MACH3LOG_ERROR("UNKNOWN SPLINE INTERPOLATION SPECIFIED!");
294 MACH3LOG_ERROR("You gave {}", static_cast<int>(i));
295 throw MaCh3Exception(__FILE__, __LINE__);
296 }
297 return Type;
298}
