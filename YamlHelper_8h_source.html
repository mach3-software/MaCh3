119inline std::string TMacroToString(const TMacro& macro) {
120// **********************
121 std::stringstream ss;
122
123 // Retrieve lines from TMacro
124 TList* linesList = macro.GetListOfLines();
125 if (!linesList) {
126 MACH3LOG_ERROR("Failed to retrieve lines from TMacro.");
127 return "";
128 }
129
130 TIter nextLine(linesList);
131 TObject *obj = nullptr;
132 while ((obj = nextLine())) {
133 TObjString* line = dynamic_cast<TObjString*>(obj);
134 if (!line) {
135 MACH3LOG_ERROR("Failed to cast object to TObjString.");
136 continue;
137 }
138 ss << line->GetString() << std::endl;
139 }
140
141 return ss.str();
142}

181inline bool compareYAMLNodes(const YAML::Node& node1, const YAML::Node& node2) {
182// **********************
183 // Check if the types of the nodes match
184 if (node1.Type() != node2.Type()) {
185 return false;
186 }
187
188 // Compare scalar types (like strings, numbers)
189 if (node1.IsScalar() && node2.IsScalar()) {
190 return node1.as<std::string>() == node2.as<std::string>();
191 }
192
193 // Compare sequences (like YAML lists)
194 if (node1.IsSequence() && node2.IsSequence()) {
195 if (node1.size() != node2.size()) {
196 return false;
197 }
198 for (std::size_t i = 0; i < node1.size(); ++i) {
199 if (!compareYAMLNodes(node1[i], node2[i])) {
200 return false;
201 }
202 }
203 return true;
204 }
205
206 // Compare maps (like YAML dictionaries)
207 if (node1.IsMap() && node2.IsMap()) {
208 if (node1.size() != node2.size()) {
209 return false;
210 }
211 for (auto it1 = node1.begin(); it1 != node1.end(); ++it1) {
212 auto key = it1->first.as<std::string>();
213 if (!node2[key] || !compareYAMLNodes(it1->second, node2[key])) {
214 return false;
215 }
216 }
217 return true;
218 }
219
220 // Default case: if it's neither scalar, sequence, nor map, consider it unequal
221 return false;
222}
