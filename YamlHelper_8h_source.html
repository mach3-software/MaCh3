118inline std::string TMacroToString(const TMacro& macro) {
119// **********************
120 std::stringstream ss;
121
122 // Retrieve lines from TMacro
123 TList* linesList = macro.GetListOfLines();
124 if (!linesList) {
125 std::cerr << "Error: Failed to retrieve lines from TMacro." << std::endl;
126 return "";
127 }
128
129 TIter nextLine(linesList);
130 TObject *obj = nullptr;
131 while ((obj = nextLine())) {
132 TObjString* line = dynamic_cast<TObjString*>(obj);
133 if (!line) {
134 std::cerr << "Error: Failed to cast object to TObjString." << std::endl;
135 continue;
136 }
137 ss << line->GetString() << std::endl;
138 }
139
140 return ss.str();
141}
