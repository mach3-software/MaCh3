121inline std::string TMacroToString(const TMacro& macro) {
122// **********************
123 std::stringstream ss;
124
125 // Retrieve lines from TMacro
126 TList* linesList = macro.GetListOfLines();
127 if (!linesList) {
128 std::cerr << "Error: Failed to retrieve lines from TMacro." << std::endl;
129 return "";
130 }
131
132 TIter nextLine(linesList);
133 TObject *obj = nullptr;
134 while ((obj = nextLine())) {
135 TObjString* line = dynamic_cast<TObjString*>(obj);
136 if (!line) {
137 std::cerr << "Error: Failed to cast object to TObjString." << std::endl;
138 continue;
139 }
140 ss << line->GetString() << std::endl;
141 }
142
143 return ss.str();
144}

183inline bool compareYAMLNodes(const YAML::Node& node1, const YAML::Node& node2) {
184// **********************
185 // Check if the types of the nodes match
186 if (node1.Type() != node2.Type()) {
187 return false;
188 }
189
190 // Compare scalar types (like strings, numbers)
191 if (node1.IsScalar() && node2.IsScalar()) {
192 return node1.as<std::string>() == node2.as<std::string>();
193 }
194
195 // Compare sequences (like YAML lists)
196 if (node1.IsSequence() && node2.IsSequence()) {
197 if (node1.size() != node2.size()) {
198 return false;
199 }
200 for (std::size_t i = 0; i < node1.size(); ++i) {
201 if (!compareYAMLNodes(node1[i], node2[i])) {
202 return false;
203 }
204 }
205 return true;
206 }
207
208 // Compare maps (like YAML dictionaries)
209 if (node1.IsMap() && node2.IsMap()) {
210 if (node1.size() != node2.size()) {
211 return false;
212 }
213 for (auto it1 = node1.begin(); it1 != node1.end(); ++it1) {
214 auto key = it1->first.as<std::string>();
215 if (!node2[key] || !compareYAMLNodes(it1->second, node2[key])) {
216 return false;
217 }
218 }
219 return true;
220 }
221
222 // Default case: if it's neither scalar, sequence, nor map, consider it unequal
223 return false;
224}
